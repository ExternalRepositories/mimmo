112 void
113 Apply::execute(){
114  if (getGeometry() == NULL) return;
115 
116  dvecarr3E vertex = getGeometry()->getVertexCoords();
117  long nv = getGeometry()->getNVertex();
118  nv = long(std::min(int(nv), int(m_input.size())));
119  livector1D & idmap = getGeometry()->getMapData();
120  for (long i=0; i<nv; i++){
121  vertex[i] += m_input[i];
122  getGeometry()->modifyVertex(vertex[i], idmap[i]);
123  }
124 
125  if(m_force){
126  getGeometry()->buildBvTree();
127  getGeometry()->buildKdTree();
128  }
129 
130  return;
131 };

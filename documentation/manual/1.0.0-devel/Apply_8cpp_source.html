94 void
95 Apply::execute(){
96  if (getGeometry() == NULL) return;
97 
98  dvecarr3E vertex = getGeometry()->getVertexCoords();
99  long nv = getGeometry()->getNVertex();
100  nv = long(std::min(int(nv), int(m_input.size())));
101  livector1D & idmap = getGeometry()->getMapData();
102  for (long i=0; i<nv; i++){
103  vertex[i] += m_input[i];
104  getGeometry()->modifyVertex(vertex[i], idmap[i]);
105  }
106 
107  if(m_force){
108  getGeometry()->buildBvTree();
109  getGeometry()->buildKdTree();
110  }
111 
112  return;
113 };

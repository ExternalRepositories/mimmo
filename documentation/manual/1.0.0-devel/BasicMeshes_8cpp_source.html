570  void UStructMesh::setShape(ShapeType type){
571  //create internal shape using unique_ptr member.
572  // unlink external shape eventually
573  if(m_shape){
574 
575  m_origin_temp = m_shape.get()->getOrigin();
576  m_span_temp = m_shape.get()->getSpan();
577  m_inflimits_temp = m_shape.get()->getInfLimits();
578  m_refsystem_temp = m_shape.get()->getRefSystem();
579  m_setorigin = true;
580  m_setspan = true;
581  m_setInfLimits = true;
582  m_setRefSys = true;
583 
584  m_shape.reset(nullptr);
585  }
586 
587  darray3E origin{{0,0,0}};
588  dmatrix33E spanMat;
589  spanMat[0].fill(1); spanMat[2].fill(1); spanMat[2].fill(1);
590  spanMat[1][1] = spanMat[2][1] = 2*M_PI;
591  spanMat[2][2] = M_PI;
592 
593  if(m_setorigin){origin = m_origin_temp; m_setorigin= false;}
594  if(m_setspan){
595  for(int i=0; i<3; ++i) spanMat[i] = m_span_temp;
596  m_setspan = false;
597  }
598 
599  switch(type){
600  case ShapeType::CYLINDER :
601  m_shape = std::unique_ptr<BasicShape>(new Cylinder(origin,spanMat[1]));
602  break;
603  case ShapeType::SPHERE :
604  m_shape = std::unique_ptr<BasicShape>(new Sphere(origin, spanMat[2]));
605  break;
606  default://CUBE
607  m_shape = std::unique_ptr<BasicShape>(new Cube(origin, spanMat[0]));
608  break;
609  }
610 
611  if(m_setInfLimits){
612  m_shape.get()->setInfLimits(m_inflimits_temp[0],0);
613  m_shape.get()->setInfLimits(m_inflimits_temp[1],1);
614  m_shape.get()->setInfLimits(m_inflimits_temp[2],2);
615  m_setInfLimits = false;
616  }
617 
618  if(m_setRefSys){
619  m_shape.get()->setRefSystem(m_refsystem_temp);
620  m_setRefSys = false;
621  }
622 
623  m_isBuild = false;
624 }

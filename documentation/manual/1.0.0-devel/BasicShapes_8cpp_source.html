40  * \brief Abstract Interface class for Elementary Shape Representation
41  *
42  * Interface class for Volumetric Core Element, suitable for interaction with Data Structure stored in a MimmoObject class.
43  * Object orientation in 3D space can be externally manipulated with dedicated transformation blocks. Class
44  * internally implement transformation to/from local sdr to/from world sdr, that can be used in derived objects from it.
45  * Class works with three reference systems:
46  * 1) Global Absolute SDR: is the external World reference system
47  * 2) Local Relative SDR: is the local reference system, not affected by Rigid Transformations as RotoTranslations or Scalings
48  * 3) basic SDR: local system remapping to unitary cube, not accounting of the shape type.
49  *
50  */

127 void BasicShape::setRefSystem(darray3E axis0, darray3E axis1, darray3E axis2){
128 
129  axis0 = axis0/norm2(axis0);
130  axis1 = axis1/norm2(axis1);
131  axis2 = axis2/norm2(axis2);
132 
133  double tol = 1.0e-12;
134  double check = std::abs(dotProduct(axis0,axis1)) + std::abs(dotProduct(axis1,axis2)) + std::abs(dotProduct(axis0,axis2));
135  if(check > tol) return;
136  m_sdr[0] = axis0;
137  m_sdr[1] = axis1;
138  m_sdr[2] = axis2;
139 }

145 void BasicShape::setRefSystem(int label, darray3E axis){
146 
147  if(label <0 || label >2 ) return;
148 
149  m_sdr[label] = axis/norm2(axis);
150  dvecarr3E point_mat(3,darray3E{0,0,0});
151  point_mat[0][0] = point_mat[1][1]= point_mat[2][2]=1.0;
152 
153  int next_label = (label + 1)%3;
154  int fin_label = (label + 2)%3;
155 
156  double pj = dotProduct(point_mat[next_label], m_sdr[label]);
157  m_sdr[next_label] = point_mat[next_label] - pj*m_sdr[label];
158  m_sdr[next_label] = m_sdr[next_label]/norm2(m_sdr[next_label]);
159 
160  m_sdr[fin_label] = crossProduct(m_sdr[label],m_sdr[next_label]);
161  m_sdr[fin_label] = m_sdr[fin_label]/norm2(m_sdr[fin_label]);
162 }

557 bool BasicShape::containShapeAABBox(darray3E bMin,darray3E bMax){
558  bool check = true;
559  dvecarr3E points(8);
560  points[0] = bMin;
561  points[1] = bMax;
562  points[2] = bMax; points[2][2] = bMin[2];
563  points[3] = bMin; points[3][2] = bMax[2];
564  points[4] = bMin; points[4][0] = bMax[0];
565  points[5] = bMax; points[5][0] = bMin[0];
566  points[6] = bMin; points[6][1] = bMax[1];
567  points[7] = bMax; points[7][1] = bMin[1];
568 
569  int counter=0;
570  while (check && counter<8) {
571  check = check && isPointIncluded(points[counter]);
572  ++counter;
573  }
574  return check;

607 void BasicShape::searchKdTreeMatches(bitpit::KdTree<3,bitpit::Vertex,long> & tree, int indexKdNode, int level, livector1D & result, int &counter ){
608 
609  //check indexKdNode admissible.
610  if(indexKdNode <0 || indexKdNode > tree.n_nodes) return;
611 
612  //1st step get data
613  bitpit::KdNode<bitpit::Vertex, long> & target = tree.nodes[indexKdNode];
614  //check target point is in the shape
615  if (bitpit::CGElem::intersectPointBox(target.object_->getCoords(), m_bbox[0], m_bbox[1], 3)){
616  if(isPointIncluded(target.object_->getCoords())){
617  result[counter] = target.label;
618  ++counter;
619  }
620  searchKdTreeMatches(tree, target.lchild_, level+1, result, counter);
621  searchKdTreeMatches(tree, target.rchild_, level+1, result, counter);
622  }else{
623 
624  uint32_t check = intersectShapePlane(level%3, target.object_->getCoords());
625  bool right = (check > 0);
626  bool left = (check%2 == 0);
627  if (left && target.lchild_ >= 0){
628  searchKdTreeMatches(tree, target.lchild_, level+1, result, counter);
629  }
630  if (right && target.rchild_ >= 0){
631  searchKdTreeMatches(tree, target.rchild_, level+1, result, counter);
632  }
633  }
634 
635  return;
636 };

648 void BasicShape::searchBvTreeMatches(mimmo::BvTree & tree, bitpit::PatchKernel * geo, int indexBvNode, livector1D & result, int &counter ){
649 
650  //check indexBvNode admissible.
651  if(indexBvNode <0 || indexBvNode > tree.m_nnodes) return;
652 
653  //1st step get data and control if it's leaf or if shape contain bounding box
654  if(containShapeAABBox(tree.m_nodes[indexBvNode].m_minPoint, tree.m_nodes[indexBvNode].m_maxPoint)){
655 
656  for(int i = tree.m_nodes[indexBvNode].m_element[0]; i<tree.m_nodes[indexBvNode].m_element[1]; ++i){
657  result[counter] = tree.m_elements[i].m_label;
658  ++counter;
659  }
660  return;
661  };
662 
663  if(tree.m_nodes[indexBvNode].m_leaf){
664  for(int i = tree.m_nodes[indexBvNode].m_element[0]; i<tree.m_nodes[indexBvNode].m_element[1]; ++i){
665  if(isSimplexIncluded(geo, tree.m_elements[i].m_label)){
666  result[counter] = tree.m_elements[i].m_label;
667  ++counter;
668  }
669  }
670  return;
671  }
672 
673  if(tree.m_nodes[indexBvNode].m_lchild >= 0) {
674  if( intersectShapeAABBox(tree.m_nodes[tree.m_nodes[indexBvNode].m_lchild].m_minPoint, tree.m_nodes[tree.m_nodes[indexBvNode].m_lchild].m_maxPoint) )
675  searchBvTreeMatches(tree, geo, tree.m_nodes[indexBvNode].m_lchild, result, counter);
676  }
677 
678  if(tree.m_nodes[indexBvNode].m_rchild >= 0) {
679  if( intersectShapeAABBox(tree.m_nodes[tree.m_nodes[indexBvNode].m_rchild].m_minPoint, tree.m_nodes[tree.m_nodes[indexBvNode].m_rchild].m_maxPoint) )
680  searchBvTreeMatches(tree, geo, tree.m_nodes[indexBvNode].m_rchild, result,counter);
681  }
682 
683  return;
684 };

819 darray3E Cube::toWorldCoord(darray3E point){
820 
821  darray3E work, work2;
822  //unscale your local point
823  for(int i =0; i<3; ++i){
824  work[i] = point[i]*m_scaling[i];
825  }
826 
827  //return to local xyz system
828  // -> cube, doing nothing
829 
830  //unapply change to local sdr transformation
831  linearalgebra::matmul(work, m_sdr, work2);
832 
833  //unapply origin translation
834  work = work2 + m_origin;
835  return(work);
836 };
842 darray3E Cube::toLocalCoord(darray3E point){
843 
844  darray3E work, work2;
845 
846  //unapply origin translation
847  work = point - m_origin;
848 
849  //apply change to local sdr transformation
850  dmatrix33E transp = linearalgebra::transpose(m_sdr);
851  linearalgebra::matmul(work, transp, work2);
852 
853  //get to proper local system
854  // -> cube, doing nothing
855 
856  //scale your local point
857  for(int i =0; i<3; ++i){
858  work[i] = work2[i]/m_scaling[i];
859  }
860 
861  return(work);
862 
863 };

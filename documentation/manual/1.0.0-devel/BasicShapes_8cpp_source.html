39  * \brief Abstract Interface class for Elementary Shape Representation
40  *
41  * Interface class for Volumetric Core Element, suitable for interaction with Data Structure stored in a MimmoObject class.
42  * Object orientation in 3D space can be externally manipulated with dedicated transformation blocks. Class
43  * internally implement transformation to/from local sdr to/from world sdr, that can be used in derived objects from it.
44  * Class works with three reference systems:
45  * 1) Global Absolute SDR: is the external World reference system
46  * 2) Local Relative SDR: is the local reference system, not affected by Rigid Transformations as RotoTranslations or Scalings
47  * 3) basic SDR: local system remapping to unitary cube, not accounting of the shape type.
48  *
49  */

133 void BasicShape::setRefSystem(darray3E axis0, darray3E axis1, darray3E axis2){
134 
135  axis0 = axis0/norm2(axis0);
136  axis1 = axis1/norm2(axis1);
137  axis2 = axis2/norm2(axis2);
138 
139  double tol = 1.0e-12;
140  double check = std::abs(dotProduct(axis0,axis1)) + std::abs(dotProduct(axis1,axis2)) + std::abs(dotProduct(axis0,axis2));
141  if(check > tol) return;
142  m_sdr[0] = axis0;
143  m_sdr[1] = axis1;
144  m_sdr[2] = axis2;
145 }

151 void BasicShape::setRefSystem(int label, darray3E axis){
152 
153  if(label <0 || label >2 ) return;
154 
155  m_sdr[label] = axis/norm2(axis);
156  dvecarr3E point_mat(3,darray3E{0,0,0});
157  point_mat[0][0] = point_mat[1][1]= point_mat[2][2]=1.0;
158 
159  int next_label = (label + 1)%3;
160  int fin_label = (label + 2)%3;
161 
162  double pj = dotProduct(point_mat[next_label], m_sdr[label]);
163  m_sdr[next_label] = point_mat[next_label] - pj*m_sdr[label];
164  m_sdr[next_label] = m_sdr[next_label]/norm2(m_sdr[next_label]);
165 
166  m_sdr[fin_label] = crossProduct(m_sdr[label],m_sdr[next_label]);
167  m_sdr[fin_label] = m_sdr[fin_label]/norm2(m_sdr[fin_label]);
168 }

563 bool BasicShape::containShapeAABBox(darray3E bMin,darray3E bMax){
564  bool check = true;
565  dvecarr3E points(8);
566  points[0] = bMin;
567  points[1] = bMax;
568  points[2] = bMax; points[2][2] = bMin[2];
569  points[3] = bMin; points[3][2] = bMax[2];
570  points[4] = bMin; points[4][0] = bMax[0];
571  points[5] = bMax; points[5][0] = bMin[0];
572  points[6] = bMin; points[6][1] = bMax[1];
573  points[7] = bMax; points[7][1] = bMin[1];
574 
575  int counter=0;
576  while (check && counter<8) {
577  check = check && isPointIncluded(points[counter]);
578  ++counter;
579  }
580  return check;

613 void BasicShape::searchKdTreeMatches(bitpit::KdTree<3,bitpit::Vertex,long> & tree, int indexKdNode, int level, livector1D & result, int &counter ){
614 
615  //check indexKdNode admissible.
616  if(indexKdNode <0 || indexKdNode > tree.n_nodes) return;
617 
618  //1st step get data
619  bitpit::KdNode<bitpit::Vertex, long> & target = tree.nodes[indexKdNode];
620  //check target point is in the shape
621  if (bitpit::CGElem::intersectPointBox(target.object_->getCoords(), m_bbox[0], m_bbox[1], 3)){
622  if(isPointIncluded(target.object_->getCoords())){
623  result[counter] = target.label;
624  ++counter;
625  }
626  searchKdTreeMatches(tree, target.lchild_, level+1, result, counter);
627  searchKdTreeMatches(tree, target.rchild_, level+1, result, counter);
628  }else{
629 
630  uint32_t check = intersectShapePlane(level%3, target.object_->getCoords());
631  bool right = (check > 0);
632  bool left = (check%2 == 0);
633  if (left && target.lchild_ >= 0){
634  searchKdTreeMatches(tree, target.lchild_, level+1, result, counter);
635  }
636  if (right && target.rchild_ >= 0){
637  searchKdTreeMatches(tree, target.rchild_, level+1, result, counter);
638  }
639  }
640 
641  return;
642 };

654 void BasicShape::searchBvTreeMatches(mimmo::BvTree & tree, bitpit::PatchKernel * geo, int indexBvNode, livector1D & result, int &counter ){
655 
656  //check indexBvNode admissible.
657  if(indexBvNode <0 || indexBvNode > tree.m_nnodes) return;
658 
659  //1st step get data and control if it's leaf or if shape contain bounding box
660  if(containShapeAABBox(tree.m_nodes[indexBvNode].m_minPoint, tree.m_nodes[indexBvNode].m_maxPoint)){
661 
662  for(int i = tree.m_nodes[indexBvNode].m_element[0]; i<tree.m_nodes[indexBvNode].m_element[1]; ++i){
663  result[counter] = tree.m_elements[i].m_label;
664  ++counter;
665  }
666  return;
667  };
668 
669  if(tree.m_nodes[indexBvNode].m_leaf){
670  for(int i = tree.m_nodes[indexBvNode].m_element[0]; i<tree.m_nodes[indexBvNode].m_element[1]; ++i){
671  if(isSimplexIncluded(geo, tree.m_elements[i].m_label)){
672  result[counter] = tree.m_elements[i].m_label;
673  ++counter;
674  }
675  }
676  return;
677  }
678 
679  if(tree.m_nodes[indexBvNode].m_lchild >= 0) {
680  if( intersectShapeAABBox(tree.m_nodes[tree.m_nodes[indexBvNode].m_lchild].m_minPoint, tree.m_nodes[tree.m_nodes[indexBvNode].m_lchild].m_maxPoint) )
681  searchBvTreeMatches(tree, geo, tree.m_nodes[indexBvNode].m_lchild, result, counter);
682  }
683

685  if( intersectShapeAABBox(tree.m_nodes[tree.m_nodes[indexBvNode].m_rchild].m_minPoint, tree.m_nodes[tree.m_nodes[indexBvNode].m_rchild].m_maxPoint) )

826 darray3E Cube::toWorldCoord(darray3E point){
827 
828  darray3E work, work2;
829  //unscale your local point
830  for(int i =0; i<3; ++i){
831  work[i] = point[i]*m_scaling[i];
832  }
833 
834  //return to local xyz system
835  // -> cube, doing nothing
836 
837  //unapply change to local sdr transformation
838  linearalgebra::matmul(work, m_sdr, work2);
839 
840  //unapply origin translation
841  work = work2 + m_origin;
842  return(work);
843 };
849 darray3E Cube::toLocalCoord(darray3E point){
850 
851  darray3E work, work2;
852 
853  //unapply origin translation
854  work = point - m_origin;
855 
856  //apply change to local sdr transformation
857  dmatrix33E transp = linearalgebra::transpose(m_sdr);
858  linearalgebra::matmul(work, transp, work2);
859 
860  //get to proper local system
861  // -> cube, doing nothing
862 
863  //scale your local point
864  for(int i =0; i<3; ++i){
865  work[i] = work2[i]/m_scaling[i];
866  }
867 
868  return(work);
869

189 void
190 Bend::execute(){
191 
192  //check coherence of degrees and coeffs;
193  for(int i=0; i<3; ++i){
194  for(int j=0; j<3; ++j){
195  m_coeffs[i][j].resize(m_degree[i][j]+1, 0.0);
196  }
197  }
198 
199  int ndispl = m_displ.size();
200  ndispl = std::min(ndispl, int(m_coords.size()));
201  for (int j=0; j<3; j++){
202  for (int i=0; i<ndispl; i++){
203  for (int z=0; z<3; z++){
204  if (m_degree[j][z] > 0){
205  for (int k=0; k<(int)m_degree[j][z]+1; k++){
206  m_displ[i][j] += pow(m_coords[i][z],(double)k)*m_coeffs[j][z][k];
207  }
208  }
209  }
210  }
211  }
212  return;
213 };

233 void Bend::absorbSectionXML(const bitpit::Config::Section & slotXML, std::string name){
234 
235  BITPIT_UNUSED(name);
236 
237  std::string input;
238 
239  if(slotXML.hasOption("Priority")){
240  input = slotXML.get("Priority");
241  int value =0;
242  if(!input.empty()){
243  std::stringstream ss(bitpit::utils::trim(input));
244  ss>>value;
245  }
246  setPriority(value);
247  };
248 
249  if(slotXML.hasSection("DegreesMatrix")){
250  auto & subslot = slotXML.getSection("DegreesMatrix");
251  umatrix33E temp;
252  temp.fill({{0,0,0}});
253 
254  if(subslot.hasOption("xDispl")){
255  input = subslot.get("xDispl");
256  std::stringstream ss(bitpit::utils::trim(input));
257  for(int i=0; i<3; ++i) ss>>temp[0][i];
258  }
259 
260  if(subslot.hasOption("yDispl")){
261  input = subslot.get("yDispl");
262  std::stringstream ss(bitpit::utils::trim(input));
263  for(int i=0; i<3; ++i) ss>>temp[1][i];
264  }
265 
266  if(subslot.hasOption("yDispl")){
267  input = subslot.get("yDispl");
268  std::stringstream ss(bitpit::utils::trim(input));
269  for(int i=0; i<3; ++i) ss>>temp[2][i];
270  }
271 
272  setDegree(temp);
273  };
274 
275  if(slotXML.hasSection("PolyCoefficients")){
276  auto & subslot = slotXML.getSection("PolyCoefficients");
277  dmat33Evec temp = getCoeffs();
278  std::string rootPoly = "Poly";
279  std::string locPoly;
280  int ik,jk;
281  for (int k=0; k<9; ++k){
282  locPoly = rootPoly + std::to_string(k);
283  if(subslot.hasOption(locPoly)){
284  input = subslot.get(locPoly);
285  std::stringstream ss(bitpit::utils::trim(input));
286  ik = (int)(k/3);
287  jk = k%3;
288  for(auto & val: temp[ik][jk]) ss>>val;
289  }
290  }
291  setCoeffs(temp);
292  };
293 };

331 
332  BITPIT_UNUSED(name);
333 
334  slotXML.set("ClassName", m_name);
335  slotXML.set("Priority", std::to_string(getPriority()));
336 
337  bitpit::Config::Section & degXML = slotXML.addSection("DegreesMatrix");
338 
339  svector1D d_str(3);
340  int counter = 0;
341  for(auto & val: m_degree){
342  std::stringstream ss;
343  for(auto & loc: val) ss<<loc<<'\t';
344  d_str[counter] = ss.str();
345  ++counter;
346  }
347 
348  degXML.set("xDispl", d_str[0]);
349  degXML.set("yDispl", d_str[1]);
350  degXML.set("zDispl", d_str[2]);
351 
352  bitpit::Config::Section & polyXML = slotXML.addSection("PolyCoefficients");
353  std::string rootPoly = "Poly";
354  std::string locPoly;
355  for(int i=0; i<3; i++){
356  for(int j =0; j<3; j++){
357  if(m_coeffs[i][j].empty()) continue;
358  locPoly = rootPoly + std::to_string(int(i*3+j));
359  std::stringstream ss;
360  for(auto & loc: m_coeffs[i][j]) ss<<loc<<'\t';
361  polyXML.set(locPoly, ss.str());
362  }
363  }

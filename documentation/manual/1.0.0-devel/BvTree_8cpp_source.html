240 void BvTree::buildTree()
241 {
242  if (m_patch_ == NULL) return;
243 
244  if (m_nelements == 0 || m_nodes.size() == 0) setup();
245  if (m_nelements == 0) return;
246 
247  int iel = 0;
248  long id;
249 
250  //fill ids and centroid

252  {
253  id = cell.getId();
254  m_elements[iel].m_label = id;
255  m_elements[iel].m_centroid = m_patch_->evalCellCentroid(id);
256  iel++;
257  }
258 
259  //node root

261  node.m_element[0] = 0;
262  node.m_element[1] = m_nelements;

264  m_nodes[m_nnodes] = node;

267  m_nodes[m_nnodes].m_leaf = true;
268  m_nleaf++;

275  }
276  decreaseStack();

283 void BvTree::fillTree(int iparent)
284 {
285 
286  int j;
287 
288  //compute Bounding Box of node
289  computeBoundingBox(iparent);
290 
291 
292  if ( m_nodes[iparent].m_nrange > m_maxsize )
293  {
294 
295  //split parent by mean of centroid in node
296  darray3E meanC = computeMeanPoint(m_nodes[iparent].m_element[0], m_nodes[iparent].m_element[1]);
297 
298  //split by plane normal to maximum dimension of the centroid box
299  darray3E minPC{{1.0e+18, 1.0e+18, 1.0e+18}};
300  darray3E maxPC{{-1.0e+18, -1.0e+18, -1.0e+18}};
301  int ic, ics = m_nodes[iparent].m_element[0], ice = m_nodes[iparent].m_element[1];
302  for ( ic=ics; ic<ice; ++ic )
303  {
304  for ( j=0; j<m_dim; ++j )
305  {
306  minPC[j] = std::min(minPC[j], m_elements[ic].m_centroid[j]);
307  maxPC[j] = std::max(maxPC[j], m_elements[ic].m_centroid[j]);
308  }
309  }
310 
311  int dir = 0;
312  double maxRange = maxPC[dir] - minPC[dir];
313  for ( int idim=1; idim<m_dim; ++idim )
314  {
315  if ((maxPC[idim] - minPC[idim]) > maxRange)
316  {
317  dir = idim;
318  maxRange = maxPC[dir] - minPC[dir];
319  }
320  }
321 
322  //Sort elements by coordinate dir.
323  std::vector<BvElement>::iterator itstart = m_elements.begin()+m_nodes[iparent].m_element[0];
324  std::vector<BvElement>::iterator itend = m_elements.begin()+(m_nodes[iparent].m_element[1]);
325 
326  //Old algorithm
327  // sort( itstart, itend, compareElements(dir) );
328  // //find first element with coords[dir] > of meanC[dir]
329  // int firstRight = findFirstGreater(iparent, meanC, dir);
330 
331 
332  int firstRight = pseudoSort( itstart, itend, meanC, dir );
333 
334  //insert lchild
335  if ( firstRight > m_nodes[iparent].m_element[0] )
336  {
337  m_nodes[iparent].m_lchild = m_nnodes;
338  if ( (int)m_nodes.size() <= m_nnodes+1) increaseStack();
339  BvNode node;
340  node.m_element[0] = m_nodes[iparent].m_element[0];
341  node.m_element[1] = firstRight;
342  node.m_nrange = node.m_element[1] - node.m_element[0];
343  m_nodes[m_nnodes] = node;
344  if ( node.m_nrange == 1 )
345  {
346  m_nodes[m_nnodes].m_leaf = true;
347  computeBoundingBox(m_nnodes);
348  m_nnodes++;
349  m_nleaf++;
350  }
351  else
352  {
353  m_nnodes++;
354  fillTree(m_nnodes-1);
355  }
356  }
357 
358  //insert rchild
359  if ( firstRight < m_nodes[iparent].m_element[1] )
360  {
361  m_nodes[iparent].m_rchild = m_nnodes;
362  if ( (int)m_nodes.size() <= m_nnodes+1) increaseStack();
363  BvNode node;
364  node.m_element[0] = firstRight;
365  node.m_element[1] = m_nodes[iparent].m_element[1];
366  node.m_nrange = node.m_element[1] - node.m_element[0];
367  m_nodes[m_nnodes] = node;
368  if (node.m_nrange == 1){
369  m_nodes[m_nnodes].m_leaf = true;
370  computeBoundingBox(m_nnodes);
371  m_nnodes++;
372  m_nleaf++;
373  }
374  else{
375  m_nnodes++;
376  fillTree(m_nnodes-1);
377  }
378  }
379 
380  }
381  else
382  {
383  //insert this node as leaf
384  m_nodes[iparent].m_leaf = true;
385  m_nleaf++;
386 
387  }
388 
389  return;
390 }

414 int BvTree::pseudoSort(std::vector<BvElement>::iterator itstart,
415  std::vector<BvElement>::iterator itend, darray3E meanC, int dir){
416 
417  double thres = meanC[dir];
418 
419  itend--;
420  while(itstart != itend){
421 
422  while (itstart->m_centroid[dir] <= thres && itstart != itend){
423  itstart++;
424  }
425  while (itend->m_centroid[dir] > thres && itend != itstart){
426  itend--;
427  }
428 
429  if (itstart != itend){
430  std::iter_swap(itstart, itend);
431  }
432  }
433 
434  return (std::max(0, int(distance(m_elements.begin(), itend))));

450 int BvTree::findFirstGreater(int inode, darray3E meanC, int dir)
451 {
452  double mean = meanC[dir];
453  int guess = (m_nodes[inode].m_element[1] - m_nodes[inode].m_element[0])/2;
454  int step = guess;
455 
456  while( step > 0 )
457  {
458  step /= 2;
459  guess += step * int(sign(mean - m_elements[guess].m_centroid[dir]));
460  guess = std::max(m_nodes[inode].m_element[0], guess);
461  guess = std::min(m_nodes[inode].m_element[1]-1, guess);
462  }
463 
464  while ( mean < m_elements[guess].m_centroid[dir] && guess > m_nodes[inode].m_element[0] )
465  {
466  guess--;
467  }
468  while ( mean > m_elements[guess].m_centroid[dir] && guess < m_nodes[inode].m_element[1]-1 )
469  {
470  guess++;
471  }
472  return guess;
473 }

484 void BvTree::computeBoundingBox(int inode)
485 {
486  BvNode &node(m_nodes[inode]);
487 
488  int is = node.m_element[0];
489  int ie = node.m_element[1];
490 
491  int i, j, iV, nV;
492 
493  node.m_minPoint = {{1.0e+18, 1.0e+18, 1.0e+18}};
494  node.m_maxPoint = {{-1.0e+18, -1.0e+18, -1.0e+18}};
495 
496  bitpit::Cell cell;
497  darray3E coords;
498 
499  for ( i=is; i<ie; i++ )
500  {
501  cell = m_patch_->getCell(m_elements[i].m_label);
502  nV = cell.getVertexCount();
503  for ( iV=0; iV<nV; ++iV )
504  {
505  coords = m_patch_->getVertexCoords(cell.getVertex(iV));
506  for ( j=0; j<m_dim; ++j )
507  {
508  node.m_minPoint[j] = std::min(node.m_minPoint[j], coords[j]);
509  node.m_maxPoint[j] = std::max(node.m_maxPoint[j], coords[j]);
510  }
511  }
512  }
513  node.m_minPoint -= m_tol;
514  node.m_maxPoint += m_tol;
515  return;

549 bool BvTree::SphereBoundingBox(darray3E *P_, BvNode *node_, double r)
550 {
551 
552  darray3E closestP;
553  for ( int i=0; i<m_dim; i++ )
554  {
555  closestP[i] = std::min(std::max((*P_)[i], node_->m_minPoint[i]), node_->m_maxPoint[i]);

558 
559  return(dist < r*r);

144 int
145 Chain::addObject(BaseManipulation* obj, int id_){
146  int id = id_;
147  if (id_ < 0){
148  id = m_objcounter;
149  m_objcounter++;
150  }
151  if (m_objects.size() == 0){
152  m_objects.push_back(obj);
153  m_idObjects.push_back(id);
154  return 0;
155  }
156  vector<BaseManipulation*>::iterator itchild = m_objects.end();
157  vector<BaseManipulation*>::iterator itparent;
158  int idxchild = m_objects.size();
159  ivector1D idxparent(obj->getNParent(), -1);
160  if (obj->getNChild()>0){
161  for (int i=0; i<obj->getNChild(); i++){
162  itchild = find(m_objects.begin(), m_objects.end(), obj->getChild(i));
163  if (itchild != m_objects.end()){
164  idxchild = min(idxchild, int(distance(m_objects.begin(), itchild)));
165  }
166  }
167  }
168  if (obj->getNParent()>0){
169  for (int i=0; i<obj->getNParent(); i++){
170  if (obj->getParent(i) != NULL){
171  itparent = find(m_objects.begin(), m_objects.end(), obj->getParent(i));
172  if (itparent != m_objects.end()){
173  idxparent[i] = distance(m_objects.begin(), itparent);
174  }
175  }
176  }
177  }
178  for (int i=0; i<obj->getNParent(); i++){
179  if (idxparent[i] == idxchild) return -1;
180  }
181 
182  bool parentsMinor = true;
183  for (int i=0; i<obj->getNParent(); i++){
184  if (idxparent[i] > idxchild){
185  parentsMinor = false;
186  }
187  }
188  if (!parentsMinor){
189  ivector1D idparent(obj->getNParent(), -1);
190  vector<BaseManipulation*> parent(obj->getNParent());
191  for (int i=0; i<obj->getNParent(); i++){
192  if (idxparent[i] > idxchild){
193  parent[i] = obj->getParent(i);
194  itparent = find(m_objects.begin(), m_objects.end(), parent[i]);
195  int idxtemp = distance(m_objects.begin(), itparent);
196  idparent[i] = m_idObjects[idxtemp];
197  m_objects.erase(itparent);
198  m_idObjects.erase(m_idObjects.begin()+idxtemp);
199  }
200  }
201  int idx = addObject(obj, id);
202  for (int i=0; i<obj->getNParent(); i++){
203  if (idxparent[i] > idxchild){
204  int idxp = addObject(parent[i], idparent[i]);
205  parent[i] = NULL;
206  }
207  }
208  return idx+1;
209  }else{
210  m_objects.insert(m_objects.begin()+idxchild, obj);
211  m_idObjects.insert(m_idObjects.begin()+idxchild, id);
212  return idxchild;
213  }
214  return 0;
215 }

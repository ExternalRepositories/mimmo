133 void
134 ClipGeometry::execute(){
135 
136  if(getGeometry() == NULL || getGeometry()->isEmpty()) return;
137 
138  m_patch.reset(nullptr);
139 
140  livector1D extracted = clipPlane();
141  if(extracted.empty()) return;
142 
143  //create your subpatch.
144  std::unique_ptr<MimmoObject> temp(new MimmoObject(getGeometry()->getType()));
145  bitpit::PatchKernel * tri = getGeometry()->getPatch();
146  bitpit::PiercedVector<bitpit::Vertex> & mapV = temp->getPatch()->getVertices();
147 
148  livector1D TT;
149 
150  long idV;
151  int sizeCC;
152  int counter=0;
153  bitpit::ElementInfo::Type eltype;
154 
155  for(auto && idCell : extracted){
156 
157  bitpit::Cell & cell = tri->getCell(idCell);
158  eltype = cell.getType();
159  sizeCC = cell.getVertexCount();
160  TT.resize(sizeCC);
161 
162  for(int i=0; i<sizeCC; ++i){
163  idV = cell.getVertex(i);
164  TT[i] = idV;
165 
166  if(!mapV.exists(idV)) temp->addVertex(tri->getVertexCoords(idV),idV);
167  }
168  temp->addConnectedCell(TT,eltype,idCell);
169  TT.clear();
170  counter++;
171  }
172 
173  m_patch = std::move(temp);
174  tri = NULL;
175  return;
176 };

182 livector1D ClipGeometry::clipPlane(){
183 
184  livector1D result;
185  darray3E norm;
186  double offset;
187  int counter;
188  double sig = 1.0 - 2.0*(int)isInsideOut();
189  long iD;
190 
191 
192  for(int i=0; i<3; ++i) norm[i] = m_plane[i];
193  offset = m_plane[3];
194 
195  double normPlane = norm2(norm);
196  if(normPlane < 1.E-18) return result;
197  norm /= normPlane;
198 
199  bitpit::PatchKernel * tri = getGeometry()->getPatch();
200 
201  if(getGeometry()->getType() != 1 && getGeometry()->getType() != 2 ){
202  counter = 0;
203  result.resize(tri->getVertexCount());
204  for(auto vert : tri->getVertices()){
205  iD = vert.getId();
206  if(sig*(dotProduct(norm, vert.getCoords()) + offset) >0) {
207  result[counter] = iD;
208  ++counter;
209  }
210  }
211 
212  }else{
213  counter = 0;
214  result.resize(tri->getCellCount());
215  for(auto cell : tri->getCells()){
216  iD = cell.getId();
217  if(sig*(dotProduct(norm, tri->evalCellCentroid(iD)) + offset) >0) {
218  result[counter] = iD;
219  ++counter;
220  }
221  }
222  }
223  result.resize(counter);
224  return result;
225 };

274 void
275 ControlDeformExtSurface::execute(){
276 
277  MimmoObject * geo = getGeometry();
278  if(geo->isEmpty()) return;
279 
280  int nDFS = m_defField.size();
281  m_defField.resize(geo->getNVertex(),darray3E{{0.0,0.0,0.0}});
282  m_violationField.resize(nDFS,-1.E+18);
283 
284  dvector1D violationField;
285  violationField.resize(nDFS);
286 
287  dvecarr3E points = geo->getVertexCoords();
288  dvecarr3E pointsOR = points;
289 
290  //evaluate deformable geometry barycenter************************
291  darray3E geoBary = {{0.0,0.0,0.0}};
292  double distBary = 0.0;
293 
294  for(auto & p: points) geoBary += p;
295 
296  geoBary /= (double)geo->getNVertex();
297 
298  for(auto & p: points){
299  distBary= std::fmax(distBary,norm2(p-geoBary));
300  }
301  //***************************************************************
302 
303  //adding deformation to points **********************************
304  points+= m_defField;
305  //***************************************************************
306 
307  //read external surfaces*****************************************
308  std::vector<std::unique_ptr<MimmoGeometry> > extgeo;
309  dvector1D tols;
310  readGeometries(extgeo, tols);
311  //***************************************************************
312 
313  if(extgeo.size() < 1) return;
314 
315  //evaluating bounding box of deformation*************************
316  darray3E bbMinDef, bbMaxDef, bbMin, bbMax;
317  {
318  int count2 = 0;
319  for(auto & p : points){
320  if(count2 == 0) {
321  bbMinDef = p;
322  bbMaxDef = p;
323  }else{
324  for(int i=0; i<3; ++i ){
325  bbMinDef[i] = std::fmin(bbMinDef[i], p[i]);
326  bbMaxDef[i] = std::fmax(bbMaxDef[i], p[i]);
327  }
328  }
329  count2++;
330  }
331  count2=0;
332  for(auto & p : pointsOR){
333  for(int i=0; i<3; ++i ){
334  bbMinDef[i] = std::fmin(bbMinDef[i], p[i]);
335  bbMaxDef[i] = std::fmax(bbMaxDef[i], p[i]);
336  }
337  count2++;
338  }
339  }
340  //***************************************************************
341  int counterExtGeo =0;
342  // start examining all external geometries***********************
343  for(auto &gg : extgeo){
344 
345  //check constraints properties ******************************
346  MimmoObject * local = gg->getGeometry();
347  if(!(local->isBvTreeBuilt())) local->buildBvTree();
348  bool checkOpen = local->isClosedLoop();
349 
350  double dist;
351  double radius, radius_old;
352  long id;
353  darray3E normal;
354  int count;
355  //************************************************************
356 
357  //add local constraints to the bounding box compute **********
358  bbMin = bbMinDef;
359  bbMax = bbMaxDef;
360 
361  for(auto & p : local->getVertexCoords()){
362  for(int i=0; i<3; ++i ){
363  bbMin[i] = std::fmin(bbMin[i], p[i]);
364  bbMax[i] = std::fmax(bbMax[i], p[i]);
365  }
366  }
367  //*************************************************************
368 
369  //Evaluate background grid dimension for Level set ************
370  //calculate dh as a tot part of bb diagonal *******************
371 
372  darray3E span = bbMax - bbMin;
373  bbMin += -1.0*(0.05*span);
374  span *= 1.1;
375  double dh = norm2(span)/(double)m_cellBackground;
376  iarray3E dim;
377  for(int i=0; i<3; ++i){
378  dim[i] = (int)(span[i]/dh + 0.5);
379  dim[i] = std::max(2, dim[i]);
380  }
381  //*************************************************************
382 
383  //check if its more convenient evaluate distances with a direct BvTree
384  //evaluation of distance on target deformation points or using a background grid +
385  //interpolation.
386  double nReq = double(dim[0]+1)*double(dim[1]+1)*double(dim[2]+1);
387  double nAva = 0.8*nDFS;
388 
389  if( nReq > nAva){
390 
391  //going to use direct evaluation.
392 
393  radius = distBary;
394  radius_old = radius;
395  dvector1D refsigns(nDFS, 1.0);
396 
397  //get the actual sign of distance of the undeformed cloud w.r.t constraints
398  if(checkOpen){
399  count = 0;
400  for(auto &p : pointsOR){
401  dist = evaluateSignedDistance(pointsOR[count], local, id, normal, radius);
402  if(radius < 1.0E-12) radius = radius_old;
403  else radius_old = radius;
404  if(dist < 0.0) refsigns[count] = -1.0;
405  count++;
406  }
407  }else{
408  dist = evaluateSignedDistance(geoBary, local, id, normal, radius);
409 
410  if(dist < 0.0) refsigns *= -1.0;
411  if(radius <1.0E-12) radius = radius_old;
412  }
413 
414  //evaluate distance of the deformation cloud w.r.t. constraints
415  count = 0;
416  for(auto &p : points){
417  dist = evaluateSignedDistance(p, local, id, normal, radius);
418  if(radius < 1.0E-12) radius = radius_old;
419  else radius_old = radius;
420  violationField[count] = -1.0*refsigns[count]*dist;
421  count++;
422  }
423 
424  }else{
425 
426  //going to use background grid to evaluate distances
427 
428  //instantiate a VolCartesian;
429  bitpit::VolCartesian * mesh = new bitpit::VolCartesian(0,3,bbMin,span, dim);
430  mesh->update();
431 
432  //calculate distance on point of background grid.
433  dvecarr3E background_points(mesh->getVertexCount());
434  int count = 0;
435  for(auto & v: mesh->getVertices()){
436  background_points[count] = v.getCoords();
437  count++;
438  }
439  dvector1D background_LS(background_points.size());
440 
441  radius = 0.5*norm2(span);
442  radius_old = radius;
443  count = 0;
444  for(auto &p : background_points){
445  dist = evaluateSignedDistance(p, local, id, normal, radius);
446  if(radius < 1.0E-12) radius = radius_old;
447  else radius_old = radius;
448  background_LS[count] = dist;
449  count++;
450  }
451 
452  //interpolate on background to obtain distances
453 
454  //evaluate sign of distances of the undeformed cloud w.r.t to constraints.
455  dvector1D refsigns(nDFS, 1.0);
456  if(checkOpen){
457  count= 0;
458  for(auto & p : pointsOR){
459  ivector1D stencil;
460  dvector1D weights;
461  int res = mesh->linearVertexInterpolation(p, stencil,weights);
462 
463  dist = 0.0;
464  for(int j=0; j<res; ++j){
465  dist += background_LS[stencil[j]]*weights[j];
466  }
467 
468  if(dist < 0.0) refsigns[count] = -1.0;
469  ++count;
470  }
471  }else{
472  dist = evaluateSignedDistance(geoBary, local, id, normal, radius);
473  if(dist< 0) refsigns *= -1.0;
474  if(radius < 1.0E-12) radius = radius_old;
475  }
476 
477  //evaluate violation field
478  count = 0;
479  for(auto & p : points){
480  ivector1D stencil;
481  dvector1D weights;
482 
483  int res = mesh->linearVertexInterpolation(p, stencil,weights);
484 
485  violationField[count] = 0.0;
486  for(int j=0; j<res; ++j){
487  violationField[count] += background_LS[stencil[j]]*weights[j];
488  }
489 
490  violationField[count] *= -1.0*refsigns[count];
491  ++count;
492 
493  }
494  delete mesh;
495  mesh = NULL;
496  }//end if else
497 
498  for(int i=0; i< nDFS; ++i){
499  m_violationField[i] = std::fmax(m_violationField[i], (violationField[i] + tols[counterExtGeo]));
500  }
501  ++counterExtGeo;
502  }
503  return;
504 };

721 double ControlDeformExtSurface::evaluateSignedDistance(darray3E &point, mimmo::MimmoObject * geo, long & id, darray3E & normal, double &initRadius){
722 
723  double dist = 1.0E+18;
724  double rate = 0.02;
725  int kmax = 1000;
726  int kiter = 0;
727  bool flag = true;
728 
729  if(!geo->isBvTreeBuilt()) geo->buildBvTree();
730 
731  while(flag && kiter < kmax){
732  dist = bvTreeUtils::signedDistance(&point, geo->getBvTree(), id, normal, initRadius);
733  flag = (dist == 1.0E+18);
734  if(flag) initRadius *= (1.0+ rate*((double)flag));
735  kiter++;
736  }
737 
738  return dist;
739 }

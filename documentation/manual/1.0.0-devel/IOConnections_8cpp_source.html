108 IOConnections_MIMMO::absorbConnections(bitpit::Config & slotXML, bool debug){
109 
110  if(slotXML.getSectionCount() == 0) {
111  if(debug) std::cout<<"IOConnections::absorbConnections does not found any connection to read in current XML slot."<<std::endl;
112  return;
113  }
114 
115  mimmo::BaseManipulation *sender = NULL, *receiver = NULL;
116  short int sPort, rPort;
117 
118 
119  for( auto & sect : slotXML.getSections()){
120 
121  std::string snd_str;
122  std::string rcv_str;
123  std::string sndP_str;
124  std::string rcvP_str;
125 
126  if(sect.second->hasOption("sender")) snd_str = sect.second->get("sender");
127  if(sect.second->hasOption("receiver")) rcv_str = sect.second->get("receiver");
128  if(sect.second->hasOption("senderPort")) sndP_str = sect.second->get("senderPort");
129  if(sect.second->hasOption("receiverPort")) rcvP_str = sect.second->get("receiverPort");
130 
131 
132  auto itSend = m_mapConn.find(snd_str);
133  auto itRece = m_mapConn.find(rcv_str);
134  auto itSPort = m_mapPorts.find(sndP_str);
135  auto itRPort = m_mapPorts.find(rcvP_str);
136 
137  if(itSend == m_mapConn.end() || itRece == m_mapConn.end() || itSPort == m_mapPorts.end() || itRPort == m_mapPorts.end()){
138  if(debug) std::cout<<"IOConnections::absorbConnections does not found any sender/receiver and/or port compatible with API."<<std::endl;
139  if(debug) std::cout<<"Please control connection data entries in your XML file"<<std::endl;
140  continue;
141  }
142 
143  bool check = addPin(itSend->second, itRece->second, itSPort->second, itRPort->second);
144 
145  if(debug && !check) std::cout<<"IOConnections::absorbConnections failed creating connection."<<std::endl;
146  if(debug && !check) std::cout<<"IOConnections::absorbConnections successfully created connection."<<std::endl;
147  }
148 
149  return;
150 };

172 IOConnections_MIMMO::flushConnections(bitpit::Config & slotXML, bool debug ){
173 
174  int counterGlob = 0;
175  for( auto object : m_mapConn){
176 
177  std::string sender = m_invMapConn[object.second]; //recover sender name
178  auto mapPOut = (object.second)->getPortsOut(); //get its slot of output port
179 
180 
181  for(auto & pOut : mapPOut ){
182 
183  std::string senderPort = m_invMapPorts[pOut.first]; //gets the Port Type of sender
184 
185  std::vector<BaseManipulation * > links = (pOut.second)->getLink(); //get all possible links to this specific port
186  std::vector<short int> lids = (pOut.second)->getPortLink();
187 
188  int counter = 0;
189  for( auto & val : links){
190  std::string receiver = m_invMapConn[val];
191  std::string receiverPort = m_invMapPorts[lids[counter]];
192 
193  std::string title = "conn" + std::to_string(counter+counterGlob);
194  bitpit::Config::Section & conn = slotXML.addSection(title);
195  conn.set("sender", sender);
196  conn.set("senderPort", senderPort);
197  conn.set("receiver", receiver);
198  conn.set("receiverPort", receiverPort);
199 
200  ++counter;
201  }//loop on possible links on the same port
202  counterGlob += counter;
203  }//loop on all the port of the single object
204  }//loop on all instantiated object.
205 
206  if(debug){
207  std::cout<<"IOConnections::flushConnections wrote "<< slotXML.getSectionCount()<< " active connections."<<std::endl;
208  }
209  return;
210 };

239 void Lattice::resizeMapDof(){
240  //reallocate your displacement node
241  m_intMapDOF.clear();
242  m_intMapDOF.resize((m_nx+1)*(m_ny+1)*(m_nz+1), -1);
243  ivector1D::iterator itMapBegin = m_intMapDOF.begin();
244  ivector1D::iterator itMap = itMapBegin;
245  ivector1D::iterator itMapEnd = m_intMapDOF.end();
246  bvector1D info;
247  m_np = reduceDimToDOF(m_nx+1,m_ny+1,m_nz+1, info);
248 
249  //set m_intMapDOF
250 
251  int target;
252  int index;
253  ivector1D dummy;
254 
255  int i0,i1,i2;
256  switch(getShapeType()){
257 
258  case ShapeType::CYLINDER :
259  target=0;
260  while(itMap != itMapEnd){
261 
262  *itMap = target;
263  index = std::distance(itMapBegin, itMap);
264  accessPointIndex(index,i0,i1,i2);
265 
266  if(info[0] && i0 == 0){
267  for(int k=0; k<=m_ny;++k){
268  m_intMapDOF[accessPointIndex(i0,k,i2)] = target;
269  }
270  }
271  if(info[1] && i1 == 0){
272  m_intMapDOF[accessPointIndex(i0,m_ny,i2)] = target;
273  }
274 
275  itMap = find(m_intMapDOF.begin(), itMapEnd,-1);
276  target++;
277  }
278  break;
279 
280  case ShapeType::SPHERE :
281 
282  target = 0;
283  while(itMap != itMapEnd){
284 
285  *itMap = target;
286  index = std::distance(itMapBegin, itMap);
287  accessPointIndex(index,i0,i1,i2);
288 
289  if(info[0] && i0 == 0){
290  for(int k1=0; k1<=m_ny;++k1){
291  for(int k2=0; k2<=m_nz; ++k2){
292  m_intMapDOF[accessPointIndex(i0,k1,k2)] = target;
293  }
294  }
295  }
296 
297  if(info[1] && i1 == 0){
298  m_intMapDOF[accessPointIndex(i0,m_ny-1,i2)] = target;
299  }
300 
301  if(info[2] && i2 == 0){
302  for(int k1=0; k1<=m_ny; ++k1){
303  m_intMapDOF[accessPointIndex(i0,k1,i2)] = target;
304  }
305  }
306 
307  if(info[3] && i2 == (m_nz)){
308  for(int k1=0; k1<=m_ny;++k1){
309  m_intMapDOF[accessPointIndex(i0,k1,i2)] = target;
310  }
311  }
312 
313  itMap = find(m_intMapDOF.begin(), itMapEnd,-1);
314  target++;
315  }
316  break;
317 
318 
319  case ShapeType::CUBE :
320  target = 0;
321  while(itMap != itMapEnd){
322 
323  *itMap = target;
324  itMap = find(m_intMapDOF.begin(), itMapEnd,-1);
325  target++;
326  }
327  break;
328 
329  default: //doing nothing
330  break;
331  }//end switch

353 int
354 Lattice::reduceDimToDOF(int nx, int ny, int nz, bvector1D & info){
355 
356  int delta = 0;
357  int dum = 0;
358  double dval;
359  switch(getShapeType()){
360 
361  case ShapeType::CYLINDER :
362  delta += nz;
363  nx--;
364  if(getCoordType(1) == CoordType::PERIODIC) ny--;
365 
366  info.push_back(true);
367  info.push_back(getCoordType(1) == CoordType::PERIODIC);
368  break;
369 
370  case ShapeType::SPHERE :
371  delta ++;
372  nx--;
373  if(getCoordType(1) == CoordType::PERIODIC) ny--;
374  dval = getInfLimits()[2];
375  if(dval == 0.0) {
376  nz--;
377  delta += nx;
378  }
379  if((dval + getLocalSpan()[2]) == M_PI){
380  nz--;
381  delta += nx;
382  }
383 
384  info.push_back(true);
385  info.push_back(getCoordType(1) == CoordType::PERIODIC);
386  info.push_back(dval==0.0);
387  info.push_back((dval + getLocalSpan()[2]) == M_PI);
388  break;
389 
390  default:
391  //doing nothing
392  break;
393  }
394 
395  int result = nx*ny*nz + delta;
396  return(result);

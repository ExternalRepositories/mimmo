271 ivector1D MRBF::checkDuplicatedNodes(double tol){
272  ivector1D marked;
273  int sizeEff = getTotalNodesCount();
274  if( sizeEff == 0 ) return marked;
275 
276  bvector1D check(sizeEff, false);
277 
278  darray3E target = m_node[0];
279  for(int i=1; i<sizeEff; ++i){

284  }
285  }

418 void MRBF::execute(){
419 
420  MimmoObject * container = getGeometry();
421  if(container->isEmpty() ) return;
422 
423  int size, sizeF = getDataCount();
424  for (int i=0; i<sizeF; i++){
425 
426  if(m_solver == MRBFSol::NONE) size = m_weight[i].size();
427  else size = m_value[i].size();
428 
429  if(size != getTotalNodesCount()){
430  std::cout << "MiMMO : WARNING : " << getName() << " has displacements of " << i << " field with size (" << size << ") that does not fit number of RBF nodes ("<< getTotalNodesCount() << ")" << std::endl;
431  fitDataToNodes(i);
432  }
433  }
434 
435  double bboxDiag;
436  {
437  darray3E pmin, pmax;
438  container->getPatch()->getBoundingBox(pmin, pmax);
439  bboxDiag= norm2(pmax - pmin);
440  }
441 
442  //Checking supportRadius.
443  double distance = 0.0;
444  if(m_SRRatio <=0.0){ //get maximum weight/value displ and assign support radius a 3 times this value.
445 
446  for(int i=0; i<size; ++i){
447 
448  dvector1D data(sizeF);
449 
450  for(int j=0; j<sizeF; ++j){
451  if(m_solver == MRBFSol::NONE) data[j] = m_weight[j][i];
452  else data[j] = m_value[j][i];
453  }
454 
455  distance = std::max(distance, norm2(data));
456  }
457 
458  distance *=3.0;
459 
460  }else{
461  if (m_supRIsValue){
462  distance = m_SRRatio;
463  }
464  else{
465  distance = m_SRRatio * bboxDiag;
466  }
467  }
468 
469  //TODO remove it (I can't use a support radius 0....why?)
470  if(distance <=1.E-18){ //checkSupportRadius if too small, set it to the semidiagonal value of the geometry AABB
471  distance = 0.5*bboxDiag;
472  }
473 
474  const double radius = distance;
475  RBF::setSupportRadius(radius);
476 
477  if (m_solver == MRBFSol::WHOLE) solve();
478  if (m_solver == MRBFSol::GREEDY) greedy(m_tol);
479 
480  int nv = container->getNVertex();
481  dvecarr3E vertex = container->getVertexCoords();
482 
483  m_displ.resize(nv, darray3E{0,0,0});
484  dvector1D displ;
485  for(int i=0; i<nv; ++i){
486  displ = RBF::evalRBF(vertex[i]);
487  for (int j=0; j<3; j++) m_displ[i][j] = displ[j];
488  }
489 
490  //if m_filter is active;
491  if(m_bfilter){
492  m_filter.resize(nv,0.0);
493  int counter = 0;
494  for (auto && vec : m_displ){
495  vec = vec * m_filter[counter];
496  ++counter;
497  }
498  }
499 
500  return;
501 };

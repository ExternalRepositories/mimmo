201 void
202 Mask::execute(){
203  int ndispl = m_displ.size();
204  ndispl = std::min(ndispl, int(m_coords.size()));
205  for (int i=0; i<ndispl; i++){
206  if (m_coords[i][0]>m_thres[0][0] && m_coords[i][0]<m_thres[0][1] &&
207  m_coords[i][1]>m_thres[1][0] && m_coords[i][1]<m_thres[1][1] &&
208  m_coords[i][2]>m_thres[2][0] && m_coords[i][2]<m_thres[2][1]){
209  for (int j=0; j<3; j++){
210  m_displ[i][j] = (1-m_inside[j])*m_displ[i][j];
211  }
212  }
213  else{
214  for (int j=0; j<3; j++){
215  m_displ[i][j] = (m_inside[j])*m_displ[i][j];
216  }
217  }
218  }
219  return;
220 };

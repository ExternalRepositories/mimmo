66 MimmoObject::MimmoObject(int type, dvecarr3E & vertex, ivector2D * connectivity){
67  m_type = max(1,type);
68  m_internalPatch = true;
69  const int id = 0;
70  if (m_type == 2){
71  m_patch = new VolUnstructured(id, 3);
72  dynamic_cast<VolUnstructured*> (m_patch)->setExpert(true);
73  }else{
74  m_patch = new SurfUnstructured(id);
75  dynamic_cast<SurfUnstructured*> (m_patch)->setExpert(true);
76  }
77 
78  bitpit::ElementInfo::Type eltype = bitpit::ElementInfo::UNDEFINED;
79  int sizeVert, sizeCell;
80 
81  sizeVert = vertex.size();
82  m_patch->reserveVertices(sizeVert);
83 
84  for(auto & vv : vertex) addVertex(vv);
85 
86  if(connectivity == NULL) m_type = 3;
87  if (m_type != 3){
88  int sizeConn = (*connectivity)[0].size();
89  sizeCell = connectivity->size();
90 
91  switch(m_type){
92  case 1:
93  if(sizeConn == 3) eltype = bitpit::ElementInfo::TRIANGLE;
94  if(sizeConn == 4) eltype = bitpit::ElementInfo::QUAD;
95  break;
96  case 2:
97  if(sizeConn == 4) eltype = bitpit::ElementInfo::TETRA;
98  if(sizeConn == 8) eltype = bitpit::ElementInfo::HEXAHEDRON;
99  break;
100  case 4:
101  if(sizeConn == 2) eltype = bitpit::ElementInfo::LINE;
102  break;
103  default:
104  // never been reached
105  break;
106  }
107 
108  if(eltype != bitpit::ElementInfo::UNDEFINED){
109 
110  m_patch->reserveCells(sizeCell);
111 
112  for(auto & cc : *connectivity){
113 
114  livector1D temp(cc.size());
115  int counter=0;
116 
117  for(auto && val : cc) {
118  temp[counter] = val;
119  ++counter;
120  }
121 
122  addConnectedCell(temp, eltype);
123  }
124 
125  m_pidsType.insert(0);
126 
127  }else{
128  std::cout<<"Not supported connectivity found for MimmoObject"<<std::endl;
129  std::cout<<"Proceeding as Point Cloud geometry"<<std::endl;
130  }
131  }
132  m_bvTree.setPatch(m_patch);
133  m_bvTreeBuilt = false;
134  m_kdTreeBuilt = false;
135  m_bvTreeSupported = (m_type != 3);
136  m_retrackBvTree = false;
137  m_retrackKdTree = false;
138 };

647 bool
648 MimmoObject::setCells(const bitpit::PiercedVector<Cell> & cells){
649  if (isEmpty() || cells.size()==0 || !m_bvTreeSupported) return false;
650 
651  m_mapCell.clear();
652  m_mapCellInv.clear();
653  m_pidsType.clear();
654 
655  getPatch()->resetCells();
656 
657  int sizeCell = cells.size();
658  getPatch()->reserveCells(sizeCell);
659 
660  long idc;
661  int nVert;
662  ElementInfo::Type eltype;
663 
664  for (const auto & cell : cells){
665 
666  // get ID
667  idc = cell.getId();
668  //check on element type
669  nVert = checkCellType(cell.getType());
670  //pass info to my local cell
671  if(nVert > 0) m_patch->addCell(cell, idc);
672 
673  short pid = (short) cell.getPID();
674  m_pidsType.insert(pid);
675  m_patch->getCell(idc).setPID((int)pid);
676  }
677 
678  //create inverse map of cells
679  setMapCell();
680  m_bvTreeBuilt = false;
681  return true;
682 };

696 
697  if (isEmpty() || conn.empty() || !m_bvTreeSupported) return false;
698  if(idtag != bitpit::Cell::NULL_ID && m_patch->getCells().exists(idtag)) return false;

711  checkedID = it->getId();

713  it = m_patch->addCell(type, true,conn_dum, idtag);
714  checkedID = idtag;

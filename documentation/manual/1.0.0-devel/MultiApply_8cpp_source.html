137 void
138 MultiApply::execute(){
139 
140 
141  if (m_input.empty()) return;
142 
143 
144  for(auto val: m_input){
145 
146 
147 
148  dvecarr3E vertex = val.first->getVertexCoords();
149  long nv = val.first->getNVertex();
150  nv = long(std::min(int(nv), int(val.second->size())));
151 
152  livector1D & idmap = val.first->getMapData();
153  for (long i=0; i<nv; i++){
154 
155  vertex[i] += (*(val.second))[i];
156  val.first->modifyVertex(vertex[i], idmap[i]);
157  }
158 
159  if(m_force){
160  val.first->buildBvTree();
161  val.first->buildKdTree();
162  }
163  }
164  return;
165 };

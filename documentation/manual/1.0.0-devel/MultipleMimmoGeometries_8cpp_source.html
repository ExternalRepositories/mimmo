551 bool
552 MultipleMimmoGeometries::write(){
553  if (m_extgeo.empty() || m_winfo.empty()) return false;
554 
555  int totGeo = m_extgeo.size();
556  int totInfo = m_winfo.size();
557 
558  if(totGeo != totInfo) {
559  std::cout<<"WARNING:Not enough output info or some invalid geoemetries found writing in class "<<m_name<<std::endl;
560  }
561 
562  totGeo = std::min(totGeo, totInfo);
563 
564  for(int k=0; k<totGeo; ++k){
565 
566  std::unique_ptr<MimmoGeometry> writer(new MimmoGeometry());
567  writer->setWrite(true);
568  writer->setWriteDir(m_winfo[k].fdir);
569  writer->setWriteFilename(m_winfo[k].fname);
570  writer->setWriteFileType(m_winfo[k].ftype);
571  writer->setCodex(m_codex);
572  writer->setFormatNAS(m_wformat);
573  writer->setGeometry(m_extgeo[k]);
574  writer->execute();
575  }
576  return true;
577 };

583 bool
584 MultipleMimmoGeometries::read(){
585  if(!m_isInternal || m_rinfo.empty()) return false;
586 
587  setGeometry();
588 
589  //absorbing Geometries.
590  int counter = 0;
591  m_intgeo.resize(m_rinfo.size());
592  for(auto & data: m_rinfo){
593  std::unique_ptr<MimmoGeometry> geo(new MimmoGeometry());
594  std::unique_ptr<MimmoObject> subData(new MimmoObject());
595  geo->setRead(true);
596  geo->setReadDir(data.fdir);
597  geo->setReadFilename(data.fname);
598  geo->setReadFileType(data.ftype);
599  geo->execute();
600 
601  subData->setHARDCopy(geo->getGeometry());
602  m_intgeo[counter] = std::move(subData);
603  counter++;
604  }
605  if(counter == 0) return false;
606  m_intgeo.resize(counter);
607 
608  for(auto & val: m_intgeo){
609  val->cleanGeometry();
610  if(m_buildBvTree) val->buildBvTree();
611  if(m_buildKdTree) val->buildKdTree();
612  }
613  return true;
614 };

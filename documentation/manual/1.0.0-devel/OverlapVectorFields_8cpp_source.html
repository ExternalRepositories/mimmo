295 void OverlapVectorFields::plotData(std::string dir, std::string name, bool flag, int counter, mimmo::MimmoObject * geo){
296 
297  if(m_originals.count(geo) == 0) return;
298  dvecarr3E field;
299  bool flagCalc = (m_results.count(geo) > 0);
300 
301 
302  dvecarr3E points = geo->getVertexCoords();
303  ivector2D connectivity;
304  bitpit::VTKElementType cellType = geo->desumeElement();
305 
306  if (geo->getType() != 3){
307  connectivity = geo->getCompactConnectivity();
308  }
309  else{
310  int np = points.size();
311  connectivity.resize(np);
312  for (int i=0; i<np; i++){
313  connectivity[i].resize(1);
314  connectivity[i][0] = i;
315  }
316  }
317  bitpit::VTKUnstructuredGrid output(dir,name,cellType);
318  output.setGeomData( bitpit::VTKUnstructuredField::POINTS, points);
319  output.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, connectivity);
320  output.setDimensions(connectivity.size(), points.size());
321 
322  if(flagCalc) field = m_results[geo];
323  else field.resize(points.size(), {{0.0,0.0,0.0}});
324 
325  output.addData("vectorfield", bitpit::VTKFieldType::VECTOR, bitpit::VTKLocation::POINT, field);
326 
327  std::vector<long> ids(points.size());
328  long ID;
329  for (auto vertex : geo->getVertices()){
330  ID = vertex.getId();
331  ids[geo->getMapDataInv(ID)] = ID;
332  }
333 
334  output.addData("ID", bitpit::VTKFieldType::SCALAR, bitpit::VTKLocation::POINT, ids);
335 
336  output.setCounter(counter);
337  output.setCodex(bitpit::VTKFormat::APPENDED);
338  if(!flag) output.setCodex(bitpit::VTKFormat::ASCII);
339 
340  output.write();
341 };

363 void OverlapVectorFields::execute(){
364 
365  if(m_originals.empty()) return;
366  m_results.clear();
367  int size, listsize;
368  dvecarr3E temp;
369  dvecarr3E results;
370  for(auto & obj : m_originals){
371 
372  size = obj.first->getPatch()->getVertexCount();
373  //careful resizing
374  results.resize(size, {{0.0,0.0,0.0}});
375  for(auto &vec : obj.second) vec->resize(size,{{0.0,0.0,0.0}});
376 
377  listsize = obj.second.size();
378  temp.resize(listsize);
379 
380  for(int i=0; i<size; ++i){
381  for(int j=0; j<listsize; j++) temp[j] = (*(obj.second[j]))[i];
382  results[i] = overlapFields(temp);
383  }
384 
385  m_results[obj.first] = results;
386  }
387  return;
388 }

408 darray3E OverlapVectorFields::overlapFields(dvecarr3E & locField){
409  int size = locField.size();
410  if(size < 1) return {{0.0,0.0,0.0}};
411 
412  if(size ==1 )return locField[0];
413  darray3E value;
414  darray3E dir; dir.fill(0.0);
415  double match;
416 
417 
418 
419 
420  switch(m_overlapCriterium){
421  case OverlapMethod::MAX :
422  value = {{0.0,0.0,0.0}};
423  for(auto && loc : locField){
424  dir += loc;
425  }
426 
427  dir /= norm2(dir);
428 
429  match = 1.e-18;
430  for(auto && loc : locField){
431  double dummy = dotProduct(loc, dir);
432  match = std::fmax(match,dummy);
433  }
434 
435  value = match*dir;
436  break;
437 
438  case OverlapMethod::MIN :
439 
440  value = {{0.0,0.0,0.0}};
441  for(auto && loc : locField){
442  dir += loc;
443  }
444 
445  dir /= norm2(dir);
446 
447  match = 1.e18;
448  for(auto && loc : locField){
449  double dummy = dotProduct(loc, dir);
450  match = std::fmin(match,dummy);
451  }
452 
453  value = match*dir;
454  break;
455 
456  case OverlapMethod::AVERAGE :
457  value = {{0.0,0.0,0.0}};
458  for(auto && loc : locField){
459  value += loc/double(size);
460  }
461  break;
462 
463  case OverlapMethod::SUM :
464  value = {{0.0,0.0,0.0}};
465  for(auto && loc : locField){
466  value += loc;
467  }
468  break;
469  default : //never been reached
470  break;
471  }
472 
473  return value;
474 };

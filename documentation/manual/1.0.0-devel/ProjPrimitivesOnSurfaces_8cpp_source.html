403 void
404 ProjSegmentOnSurface::projection(){
405 
406  if(getGeometry()->isEmpty()) return;
407  int counter = 0;
408  std::unique_ptr<MimmoObject> dum(new MimmoObject(4));
409  //reserving memory
410  dum->getPatch()->reserveVertices(m_nC+1);
411  dum->getPatch()->reserveCells(m_nC);
412 
413  //start filling connectivity of your object.
414  bitpit::ElementInfo::Type eltype = bitpit::ElementInfo::Type::LINE;
415 
416  for(int i=0; i<m_nC; ++i){
417  livector1D conn(2);
418  conn[0] = i;
419  conn[1] = i+1;
420  dum->addConnectedCell(conn, eltype);
421  }
422 
423  //create points ...
424  dvecarr3E verts((m_nC + 1)), projs;
425  {
426  //create the vertices array, ordered from pointA to pointB
427  auto dx = (m_pointB - m_pointA);
428  dx /= double(m_nC);
429  counter = 0;
430  for( auto && ele : verts){
431  ele = m_pointA + double(counter)*dx;
432  ++counter;
433  }
434  }
435 
436  //...and projecting them onto target surface
437  if(!getGeometry()->isBvTreeBuilt()) getGeometry()->buildBvTree();
438  counter = 0;
439  projs.resize(verts.size());
440  for(auto &val : verts){
441  projs[counter]= bvTreeUtils::projectPoint(&val, getGeometry()->getBvTree());
442  ++counter;
443  }
444 
445  //storing the projected points in the MImmoObject:
446  long idS = 0;
447  for(auto vv : projs){
448  dum->addVertex(vv, idS);
449  ++idS;
450  }
451 
452  dum->cleanGeometry();
453  m_patch = std::move(dum);
454  return;
455 };

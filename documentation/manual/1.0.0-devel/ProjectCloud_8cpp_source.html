101 void
102 ProjectCloud::execute(){
103 
104  if(getGeometry() == NULL || getGeometry()->isEmpty()) return;
105 
106  if(!getGeometry()->isBvTreeBuilt()) getGeometry()->buildBvTree();
107 
108  //project points on surface.
109  int counter = 0;
110  m_proj.resize(m_points.size());
111  for(auto &val : m_points){
112  m_proj[counter]= bvTreeUtils::projectPoint(&val, getGeometry()->getBvTree());
113  ++counter;
114  }
115  return;
116 };

122 void ProjectCloud::plotOptionalResults(){
123  bitpit::VTKFormat codex = bitpit::VTKFormat::APPENDED;
124 
125  int size = m_proj.size();
126  ivector1D conn(size);
127  for(int i=0; i<size; i++){
128  conn[i] = i;
129  }
130  std::string dir = "./";
131  std::string file = m_name + "_" + std::to_string(getClassCounter());
132 
133  bitpit::VTKUnstructuredGrid vtk(dir, file, bitpit::VTKElementType::VERTEX);
134  vtk.setGeomData( bitpit::VTKUnstructuredField::POINTS, m_proj) ;
135  vtk.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, conn) ;
136  vtk.setDimensions(size, size);
137  vtk.setCodex(codex);
138 
139  vtk.write();
140 }

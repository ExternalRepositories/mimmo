151 void RBFBox::execute(){
152 
153  int np = m_nodes.size();
154 
155  darray3E min, max;
156  min.fill(1.0e+18);
157  max.fill(-1.0e+18);
158 
159  for(int i=0; i<np; i++){
160  for(int j=0; i<3; j++){
161  if (m_nodes[i][j] - m_suppR*1.01 < min[j]) min[j] = m_nodes[i][j];
162  if (m_nodes[i][j] + m_suppR*1.01 > max[j]) max[j] = m_nodes[i][j];
163  }
164  }
165 
166  m_span = max - min;
167  m_origin = min + m_span * 0.5;
168 
169 };

144 void
145 RotationBox::execute(){
146 
147  //Rotation of origin
148  m_rotax_origin = {{0,0,0}};
149  m_axes_origin -= m_origin;
150  //rodrigues formula
151  m_rotax_origin = m_axes_origin * cos(m_alpha) +
152  dotProduct(m_direction, m_axes_origin) * (1 - cos(m_alpha)) * m_direction +
153  crossProduct(m_direction, m_axes_origin) * sin(m_alpha);
154 
155  m_rotax_origin += m_origin;
156  m_axes_origin += m_origin;
157 
158  //rotation of axes
159  m_rotax.fill(darray3E{{0,0,0}});
160  //rodrigues formula
161  for (int i=0; i<3; i++){
162  m_rotax[i] = m_axes[i] * cos(m_alpha) +
163  dotProduct(m_direction, m_axes[i]) * (1 - cos(m_alpha)) * m_direction +
164  crossProduct(m_direction, m_axes[i]) * sin(m_alpha);
165  }
166 
167  return;
168 };

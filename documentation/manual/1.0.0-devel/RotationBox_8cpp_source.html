166 void
167 RotationBox::execute(){
168 
169  //Rotation of origin
170  m_rotax_origin = {{0,0,0}};
171  m_axes_origin -= m_origin;
172  //rodrigues formula
173  m_rotax_origin = m_axes_origin * cos(m_alpha) +
174  dotProduct(m_direction, m_axes_origin) * (1 - cos(m_alpha)) * m_direction +
175  crossProduct(m_direction, m_axes_origin) * sin(m_alpha);
176 
177  m_rotax_origin += m_origin;
178  m_axes_origin += m_origin;
179 
180  //rotation of axes
181  m_rotax.fill(darray3E{{0,0,0}});
182  //rodrigues formula
183  for (int i=0; i<3; i++){
184  m_rotax[i] = m_axes[i] * cos(m_alpha) +
185  dotProduct(m_direction, m_axes[i]) * (1 - cos(m_alpha)) * m_direction +
186  crossProduct(m_direction, m_axes[i]) * sin(m_alpha);
187  }
188 
189  return;
190 };

151 void
152 RotationGeometry::execute(){
153 
154  m_displ.resize(m_geometry->getNVertex());
155 
156  //compute coefficients and constant vectors of rodriguez formula
157  double a = cos(m_alpha);
158  darray3E b = (1 - cos(m_alpha)) * m_direction;
159  double c = sin(m_alpha);
160 
161  darray3E point, rotated;
162  long ID;
163  int idx;
164  liimap mapID = m_geometry->getMapDataInv();
165 
166  for (auto vertex : m_geometry->getVertices()){
167  point = vertex.getCoords();
168  ID = vertex.getId();
169  idx = mapID[ID];
170 
171  point -= m_origin;
172  //rodrigues formula
173  rotated = a * point +
174  b * dotProduct(m_direction, point) +
175  c * crossProduct(m_direction, point);
176 
177  rotated += m_origin;
178  point += m_origin;
179 
180  m_displ[idx] = (rotated-point)*m_filter[idx];
181 
182  }
183 
184  return;
185 };

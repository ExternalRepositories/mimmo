226 
227  if(slotXML.hasOption("PlotInExecution")){
228  std::string input = slotXML.get("PlotInExecution");

230  bool value = false;

232  std::stringstream ss(input);
233  ss >> value;
234  }
235  setPlotInExecution(value);
236  }
237 
238  if(slotXML.hasOption("OutputPlot")){
239  std::string input = slotXML.get("OutputPlot");
240  input = bitpit::utils::trim(input);
241  std::string temp = ".";

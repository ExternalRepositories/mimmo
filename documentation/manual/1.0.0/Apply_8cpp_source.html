116 void
117 Apply::execute(){
118  if (getGeometry() == NULL) return;
119 
120  dvecarr3E vertex = getGeometry()->getVertexCoords();
121  long nv = getGeometry()->getNVertex();
122  nv = long(std::min(int(nv), int(m_input.size())));
123  livector1D & idmap = getGeometry()->getMapData();
124  for (long i=0; i<nv; i++){
125  vertex[i] += m_input[i];
126  getGeometry()->modifyVertex(vertex[i], idmap[i]);
127  }
128 
129  if(m_force){
130  getGeometry()->buildBvTree();
131  getGeometry()->buildKdTree();
132  }
133 
134 };

38 * \brief Class for 3D uniform structured mesh --> based on ucartmesh scheme of Bitpit
39 *
40 * Interface class for uniform structured grids, suitable for derivation of meshes on pure cartesian, cylindrical or spherical
41 * coordinates system. The class retains internal members of Class BasicShape who determine the shape of your current grid.
42 * The mesh works and its nodal structures are defined in the its local reference frame, that is, the local reference frame
43 * retained by its core BasicShape object.
44 */

96  m_xedge = other.m_xedge;
97  m_yedge = other.m_yedge;
98  m_zedge = other.m_zedge;
99 
100  if(other.m_shape){
101  switch(other.getShape()->getShapeType()){
102  case ShapeType::CUBE :
103  m_shape = std::unique_ptr<BasicShape>(new Cube(*(dynamic_cast<const Cube *> (other.getShape()))));
104  break;
105  case ShapeType::CYLINDER :
106  m_shape = std::unique_ptr<BasicShape>(new Cylinder(*(dynamic_cast<const Cylinder*> (other.getShape()))));
107  break;
108  case ShapeType::SPHERE :
109  m_shape = std::unique_ptr<BasicShape>(new Sphere(*(dynamic_cast<const Sphere*> (other.getShape()))));
110  break;
111  default:
112  //never been reached
113  break;
114  }
115  }
116 
117  //copy temporary members
118  m_origin_temp = other.m_origin_temp;
119  m_span_temp = other.m_span_temp;
120  m_inflimits_temp = other.m_inflimits_temp;
121  m_refsystem_temp = other.m_refsystem_temp;

126  m_setRefSys = other.m_setRefSys;
127 
128  return(*this);

580  void UStructMesh::setShape(ShapeType type){
581  //create internal shape using unique_ptr member.
582  // unlink external shape eventually
583  if(m_shape){
584 
585  m_origin_temp = m_shape.get()->getOrigin();
586  m_span_temp = m_shape.get()->getSpan();
587  m_inflimits_temp = m_shape.get()->getInfLimits();
588  m_refsystem_temp = m_shape.get()->getRefSystem();
589  m_setorigin = true;
590  m_setspan = true;
591  m_setInfLimits = true;
592  m_setRefSys = true;
593 
594  m_shape.reset(nullptr);
595  }
596 
597  darray3E origin{{0,0,0}};
598  dmatrix33E spanMat;
599  spanMat[0].fill(1); spanMat[2].fill(1); spanMat[2].fill(1);
600  spanMat[1][1] = spanMat[2][1] = 2*M_PI;
601  spanMat[2][2] = M_PI;
602 
603  if(m_setorigin){origin = m_origin_temp; m_setorigin= false;}
604  if(m_setspan){
605  for(int i=0; i<3; ++i) spanMat[i] = m_span_temp;
606  m_setspan = false;
607  }
608 
609  switch(type){
610  case ShapeType::CYLINDER :
611  m_shape = std::unique_ptr<BasicShape>(new Cylinder(origin,spanMat[1]));
612  break;
613  case ShapeType::SPHERE :
614  m_shape = std::unique_ptr<BasicShape>(new Sphere(origin, spanMat[2]));
615  break;
616  default://CUBE
617  m_shape = std::unique_ptr<BasicShape>(new Cube(origin, spanMat[0]));
618  break;
619  }
620 
621  if(m_setInfLimits){
622  m_shape.get()->setInfLimits(m_inflimits_temp[0],0);
623  m_shape.get()->setInfLimits(m_inflimits_temp[1],1);
624  m_shape.get()->setInfLimits(m_inflimits_temp[2],2);
625  m_setInfLimits = false;
626  }
627 
628  if(m_setRefSys){
629  m_shape.get()->setRefSystem(m_refsystem_temp);
630  m_setRefSys = false;
631  }
632 
633  m_isBuild = false;
634 }

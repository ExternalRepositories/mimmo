222 void
223 BendGeometry::execute(){
224 
225  if (getGeometry() == NULL) return;
226 
227  //check coherence of degrees and coeffs;
228  for(int i=0; i<3; ++i){
229  for(int j=0; j<3; ++j){
230  m_coeffs[i][j].resize(m_degree[i][j]+1, 0.0);
231  }
232  }
233 
234  int nV = m_geometry->getNVertex();
235  m_displ.resize(nV);
236  m_filter.resize(nV, 1.0);
237  long ID;
238  int idx;
239  liimap mapID = m_geometry->getMapDataInv();
240 
241  darray3E point;
242  for (auto vertex : m_geometry->getVertices()){
243  point = vertex.getCoords();
244  if (m_local) point = toLocalCoord(point);
245  ID = vertex.getId();
246  idx = mapID[ID];
247  for (int j=0; j<3; j++){
248  for (int z=0; z<3; z++){
249  if (m_degree[j][z] > 0){
250  for (int k=0; k<(int)m_degree[j][z]+1; k++){
251  m_displ[idx][j] += pow(point[z],(double)k)*m_coeffs[j][z][k];
252  }
253  }
254  }
255  }
256  }
257  return;
258 };

266 BendGeometry::toLocalCoord(darray3E point){
267  darray3E work, work2;
268  //unapply origin translation
269  work = point - m_origin;
270  //apply change to local sdr transformation
271  dmatrix33E transp = linearalgebra::transpose(m_system);
272  linearalgebra::matmul(work, transp, work2);
273  return(work);

282 BendGeometry::absorbSectionXML(const bitpit::Config::Section & slotXML, std::string name){
283 
284  BITPIT_UNUSED(name);
285 
286  std::string input;
287 
288  if(slotXML.hasOption("Priority")){
289  input = slotXML.get("Priority");
290  int value =0;
291  if(!input.empty()){
292  std::stringstream ss(bitpit::utils::trim(input));
293  ss>>value;
294  }
295  setPriority(value);
296  };
297 
298  if(slotXML.hasSection("DegreesMatrix")){
299  auto & subslot = slotXML.getSection("DegreesMatrix");
300  umatrix33E temp;
301  temp.fill({{0,0,0}});
302 
303  if(subslot.hasOption("xDispl")){
304  input = subslot.get("xDispl");
305  std::stringstream ss(bitpit::utils::trim(input));
306  for(int i=0; i<3; ++i) ss>>temp[0][i];
307  }
308 
309  if(subslot.hasOption("yDispl")){
310  input = subslot.get("yDispl");
311  std::stringstream ss(bitpit::utils::trim(input));
312  for(int i=0; i<3; ++i) ss>>temp[1][i];
313  }
314 
315  if(subslot.hasOption("zDispl")){
316  input = subslot.get("zDispl");
317  std::stringstream ss(bitpit::utils::trim(input));
318  for(int i=0; i<3; ++i) ss>>temp[2][i];
319  }
320 
321  setDegree(temp);
322  };
323 
324  if(slotXML.hasSection("PolyCoefficients")){
325  auto & subslot = slotXML.getSection("PolyCoefficients");
326  dmat33Evec temp = getCoeffs();
327  std::string rootPoly = "Poly";
328  std::string locPoly;
329  int ik,jk;
330  for (int k=0; k<9; ++k){
331  locPoly = rootPoly + std::to_string(k);
332  if(subslot.hasOption(locPoly)){
333  input = subslot.get(locPoly);
334  std::stringstream ss(bitpit::utils::trim(input));
335  ik = (int)(k/3);
336  jk = k%3;
337  for(auto & val: temp[ik][jk]) ss>>val;
338  }
339  }
340  setCoeffs(temp);
341  };
342 
343  if(slotXML.hasOption("Origin")){
344  std::string input = slotXML.get("Origin");
345  input = bitpit::utils::trim(input);
346  darray3E temp = {{0.0,0.0,0.0}};
347  if(!input.empty()){
348  std::stringstream ss(input);
349  for(auto &val : temp) ss>>val;
350  }
351  setOrigin(temp);
352  };
353 
354  if(slotXML.hasSection("RefSystem")){
355  const bitpit::Config::Section & rfXML = slotXML.getSection("RefSystem");
356  std::string rootAxis = "axis";
357  std::string axis;
358  dmatrix33E temp;
359  temp[0].fill(0.0); temp[0][0] = 1.0;
360  temp[1].fill(0.0); temp[1][1] = 1.0;
361  temp[2].fill(0.0); temp[2][2] = 1.0;
362  for(int i=0; i<3; ++i){
363  axis = rootAxis + std::to_string(i);
364  std::string input = rfXML.get(axis);
365  input = bitpit::utils::trim(input);
366  if(!input.empty()){
367  std::stringstream ss(input);
368  for(auto &val : temp[i]) ss>>val;
369  }
370  }
371  setRefSystem(temp);
372  };

405  std::string rootPoly = "Poly";
406  std::string locPoly;
407  for(int i=0; i<3; i++){

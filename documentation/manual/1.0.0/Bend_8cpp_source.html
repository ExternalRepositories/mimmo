163 void
164 Bend::execute(){
165  int ndispl = m_displ.size();
166  ndispl = std::min(ndispl, int(m_coords.size()));
167  for (int j=0; j<3; j++){
168  for (int i=0; i<ndispl; i++){
169  for (int z=0; z<3; z++){
170  if (m_degree[j][z] > 0){
171  for (int k=0; k<m_degree[j][z]+1; k++){
172  m_displ[i][j] += pow(m_coords[i][z],(double)k)*m_coeffs[j][z][k];
173  }
174  }
175  }
176  }
177  }
178  return;
179 };

159 Chain::addObject(BaseManipulation* obj, int id_){
160  int id = id_;
161  if (id_ < 0){
162  id = m_objcounter;
163  m_objcounter++;
164  }
165  if (m_objects.size() == 0){
166  m_objects.push_back(obj);
167  m_idObjects.push_back(id);
168  return 0;
169  }
170  vector<BaseManipulation*>::iterator itchild = m_objects.end();
171  vector<BaseManipulation*>::iterator itparent;
172  int idxchild = m_objects.size();
173  ivector1D idxparent(obj->getNParent(), -1);
174  if (obj->getNChild()>0){
175  for (int i=0; i<obj->getNChild(); i++){
176  itchild = find(m_objects.begin(), m_objects.end(), obj->getChild(i));
177  if (itchild != m_objects.end()){
178  idxchild = min(idxchild, int(distance(m_objects.begin(), itchild)));
179  }
180  }
181  }
182  if (obj->getNParent()>0){
183  for (int i=0; i<obj->getNParent(); i++){
184  if (obj->getParent(i) != NULL){
185  itparent = find(m_objects.begin(), m_objects.end(), obj->getParent(i));
186  if (itparent != m_objects.end()){
187  idxparent[i] = distance(m_objects.begin(), itparent);
188  }
189  }
190  }
191  }
192  for (int i=0; i<obj->getNParent(); i++){
193  if (idxparent[i] == idxchild) return -1;
194  }
195 
196  bool parentsMinor = true;
197  for (int i=0; i<obj->getNParent(); i++){
198  if (idxparent[i] > idxchild){
199  parentsMinor = false;
200  }
201  }
202  if (!parentsMinor){
203  ivector1D idparent(obj->getNParent(), -1);
204  vector<BaseManipulation*> parent(obj->getNParent());
205  for (int i=0; i<obj->getNParent(); i++){
206  if (idxparent[i] > idxchild){
207  parent[i] = obj->getParent(i);
208  itparent = find(m_objects.begin(), m_objects.end(), parent[i]);
209  int idxtemp = distance(m_objects.begin(), itparent);
210  idparent[i] = m_idObjects[idxtemp];
211  m_objects.erase(itparent);
212  m_idObjects.erase(m_idObjects.begin()+idxtemp);
213  }
214  }
215  int idx = addObject(obj, id);
216  for (int i=0; i<obj->getNParent(); i++){
217  if (idxparent[i] > idxchild){
218  addObject(parent[i], idparent[i]);
219  parent[i] = NULL;
220  }
221  }
222  return idx+1;
223  }else{
224  m_objects.insert(m_objects.begin()+idxchild, obj);
225  m_idObjects.insert(m_idObjects.begin()+idxchild, id);
226  return idxchild;
227  }
228  return 0;
229 }

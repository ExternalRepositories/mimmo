155 Chain::addObject(BaseManipulation* obj, int id_){
156  int id = id_;
157  if (id_ < 0){
158  id = m_objcounter;
159  m_objcounter++;
160  }
161  if (m_objects.size() == 0){
162  m_objects.push_back(obj);
163  m_idObjects.push_back(id);
164  return 0;
165  }
166  vector<BaseManipulation*>::iterator itchild = m_objects.end();
167  vector<BaseManipulation*>::iterator itparent;
168  int idxchild = m_objects.size();
169  ivector1D idxparent(obj->getNParent(), -1);
170  if (obj->getNChild()>0){
171  for (int i=0; i<obj->getNChild(); i++){
172  itchild = find(m_objects.begin(), m_objects.end(), obj->getChild(i));
173  if (itchild != m_objects.end()){
174  idxchild = min(idxchild, int(distance(m_objects.begin(), itchild)));
175  }
176  }
177  }
178  if (obj->getNParent()>0){
179  for (int i=0; i<obj->getNParent(); i++){
180  if (obj->getParent(i) != NULL){
181  itparent = find(m_objects.begin(), m_objects.end(), obj->getParent(i));
182  if (itparent != m_objects.end()){
183  idxparent[i] = distance(m_objects.begin(), itparent);
184  }
185  }
186  }
187  }
188  for (int i=0; i<obj->getNParent(); i++){
189  if (idxparent[i] == idxchild) return -1;
190  }
191 
192  bool parentsMinor = true;
193  for (int i=0; i<obj->getNParent(); i++){
194  if (idxparent[i] > idxchild){
195  parentsMinor = false;
196  }
197  }
198  if (!parentsMinor){
199  ivector1D idparent(obj->getNParent(), -1);
200  vector<BaseManipulation*> parent(obj->getNParent());
201  for (int i=0; i<obj->getNParent(); i++){
202  if (idxparent[i] > idxchild){
203  parent[i] = obj->getParent(i);
204  itparent = find(m_objects.begin(), m_objects.end(), parent[i]);
205  int idxtemp = distance(m_objects.begin(), itparent);
206  idparent[i] = m_idObjects[idxtemp];
207  m_objects.erase(itparent);
208  m_idObjects.erase(m_idObjects.begin()+idxtemp);
209  }
210  }
211  int idx = addObject(obj, id);
212  for (int i=0; i<obj->getNParent(); i++){
213  if (idxparent[i] > idxchild){
214  addObject(parent[i], idparent[i]);
215  parent[i] = NULL;
216  }
217  }
218  return idx+1;
219  }else{
220  m_objects.insert(m_objects.begin()+idxchild, obj);
221  m_idObjects.insert(m_idObjects.begin()+idxchild, id);
222  return idxchild;
223  }
224  return 0;

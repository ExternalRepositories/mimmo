197 void
198 ClipGeometry::execute(){
199 
200  if(getGeometry() == NULL || getGeometry()->isEmpty()) return;
201 
202  /* If an implicit definition is not present it has to be computed
203  * by using origin and normal.
204  */
205  if (!m_implicit) setClipPlane(m_origin, m_normal);
206 
207  m_patch.reset(nullptr);
208 
209  livector1D extracted = clipPlane();
210  if(extracted.empty()) return;
211 
212  /* Create subpatch.*/
213  std::unique_ptr<MimmoObject> temp(new MimmoObject(getGeometry()->getType()));
214  bitpit::PatchKernel * tri = getGeometry()->getPatch();
215  bitpit::PiercedVector<bitpit::Vertex> & mapV = temp->getPatch()->getVertices();
216 
217  livector1D TT;
218 
219  long idV;
220  int sizeCC;
221  bitpit::ElementInfo::Type eltype;
222  int PID;
223 
224  if (getGeometry()->getType() != 3){
225  for(auto && idCell : extracted){
226 
227  bitpit::Cell & cell = tri->getCell(idCell);
228  eltype = cell.getType();
229  sizeCC = cell.getVertexCount();
230  PID = cell.getPID();
231  TT.resize(sizeCC);
232 
233  for(int i=0; i<sizeCC; ++i){
234  idV = cell.getVertex(i);
235  TT[i] = idV;
236 
237  if(!mapV.exists(idV))temp->addVertex(tri->getVertexCoords(idV),idV);
238  }
239  temp->addConnectedCell(TT,eltype,short(PID), idCell);
240  TT.clear();
241  }
242  }
243  else{
244  for(auto && idV : extracted){
245  temp->addVertex(tri->getVertexCoords(idV),idV);
246  }
247  }
248 
249  m_patch = std::move(temp);
250  tri = NULL;
251 };

259 livector1D
260 ClipGeometry::clipPlane(){
261 
262  livector1D result;
263  darray3E norm;
264  double offset;
265  int counter;
266  double sig = 1.0 - 2.0*(int)isInsideOut();
267  long iD;
268 
269 
270  for(int i=0; i<3; ++i)norm[i] = m_plane[i];
271  offset = m_plane[3];
272 
273  double normPlane = norm2(norm);
274  if(normPlane < 1.E-18)return result;
275  norm /= normPlane;
276 
277  bitpit::PatchKernel * tri = getGeometry()->getPatch();
278 
279  if(getGeometry()->getType() == 3){
280  counter = 0;
281  result.resize(tri->getVertexCount());
282  for(auto vert : tri->getVertices()){
283  iD = vert.getId();
284  if(sig*(dotProduct(norm, vert.getCoords()) + offset) >0){
285  result[counter] = iD;
286  ++counter;
287  }
288  }
289 
290  }else{
291  counter = 0;
292  result.resize(tri->getCellCount());
293  for(auto cell : tri->getCells()){
294  iD = cell.getId();
295  if(sig*(dotProduct(norm, tri->evalCellCentroid(iD)) + offset) >0){
296  result[counter] = iD;
297  ++counter;
298  }
299  }
300  }
301  result.resize(counter);
302  return result;
303 };

310 ClipGeometry::plotOptionalResults(){
311  if(getClippedPatch() == NULL) return;
312  if(getClippedPatch()->isEmpty()) return;
313 
314  dvecarr3E points = getClippedPatch()->getVertexCoords();
315  ivector2D connectivity;
316  bitpit::VTKElementType cellType;
317 
318  std::string dir = m_outputPlot;
319  std::string name = m_name + "_Patch";
320 
321 
322  if (getClippedPatch()->getType() != 3){
323  connectivity = getClippedPatch()->getCompactConnectivity();
324  }
325  else{
326  int np = points.size();
327  connectivity.resize(np);
328  for (int i=0; i<np; i++){
329  connectivity[i].resize(1);
330  connectivity[i][0] = i;
331 
332  }
333  }
334  cellType = getClippedPatch()->desumeElement();
335 
336 
337  bitpit::VTKUnstructuredGrid output(dir,name,cellType);
338  output.setGeomData( bitpit::VTKUnstructuredField::POINTS, points);
339  output.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, connectivity);
340  output.setDimensions(connectivity.size(), points.size());
341 
342  auto pids = getClippedPatch()->getCompactPID();
343  if(pids.size() > 0) output.addData("PID", bitpit::VTKFieldType::SCALAR, bitpit::VTKLocation::CELL, pids);
344 
345  output.setCounter(getClassCounter());
346  output.setCodex(bitpit::VTKFormat::APPENDED);
347

434 ClipGeometry::flushSectionXML(bitpit::Config::Section & slotXML, std::string name){
435 
436  BITPIT_UNUSED(name);
437 
438  slotXML.set("ClassName", m_name);
439  slotXML.set("Priority", std::to_string(getPriority()));
440 
441  int value = m_insideout;
442  slotXML.set("InsideOut", std::to_string(value));
443 
444  {
445  darray4E org = getClipPlane();
446  darray3E normal;
447  darray3E point = {{0.0,0.0,0.0}};
448  int imax = -1;
449  double dum = 0.0;
450  for(int i=0; i<3; ++i){
451  normal[i] =org[i];
452  if(abs(normal[i]) > dum) {
453  imax = i;
454  dum = abs(normal[i]);
455  }
456  }
457  if(imax != -1)point[imax] = -1.0*org[3]/normal[imax];
458 
459  std::stringstream ss1, ss2;
460  ss1<<std::scientific<<point[0]<<'\t'<<point[1]<<'\t'<<point[2];
461  ss2<<std::scientific<<normal[0]<<'\t'<<normal[1]<<'\t'<<normal[2];
462 
463  bitpit::Config::Section & planeXML = slotXML.addSection("ClipPlane");
464  planeXML.set("Point",ss1.str());
465  planeXML.set("Normal",ss2.str());
466  }
467 
468  if(isPlotInExecution()){
469  slotXML.set("PlotInExecution", std::to_string(1));
470  }
471 
472  if(m_outputPlot != "."){
473  slotXML.set("OutputPlot", m_outputPlot);
474  }

115 void
116 ClipGeometry::execute(){
117 
118  if(getGeometry() == NULL || getGeometry()->isEmpty()) return;
119 
120  m_patch.reset(nullptr);
121 
122  livector1D extracted = clipPlane();
123  if(extracted.empty()) return;
124 
125  //create your subpatch.
126  std::unique_ptr<MimmoObject> temp(new MimmoObject(getGeometry()->getType()));
127  bitpit::PatchKernel * tri = getGeometry()->getPatch();
128  bitpit::PiercedVector<bitpit::Vertex> & mapV = temp->getPatch()->getVertices();
129 
130  livector1D TT;
131 
132  long idV;
133  int sizeCC;
134  int counter=0;
135  bitpit::ElementInfo::Type eltype;
136 
137  for(auto && idCell : extracted){
138 
139  bitpit::Cell & cell = tri->getCell(idCell);
140  eltype = cell.getType();
141  sizeCC = cell.getVertexCount();
142  TT.resize(sizeCC);
143 
144  for(int i=0; i<sizeCC; ++i){
145  idV = cell.getVertex(i);
146  TT[i] = idV;
147 
148  if(!mapV.exists(idV)) temp->addVertex(tri->getVertexCoords(idV),idV);
149  }
150  temp->addConnectedCell(TT,eltype,idCell);
151  TT.clear();
152  counter++;
153  }
154 
155  m_patch = std::move(temp);
156  tri = NULL;
157  return;
158 };

164 livector1D ClipGeometry::clipPlane(){
165 
166  livector1D result;
167  darray3E norm;
168  double offset;
169  int counter;
170  double sig = 1.0 - 2.0*(int)isInsideOut();
171  long iD;
172 
173 
174  for(int i=0; i<3; ++i) norm[i] = m_plane[i];
175  offset = m_plane[3];
176 
177  double normPlane = norm2(norm);
178  if(normPlane < 1.E-18) return result;
179  norm /= normPlane;
180 
181  bitpit::PatchKernel * tri = getGeometry()->getPatch();
182 
183  if(getGeometry()->getType() != 1 && getGeometry()->getType() != 2 ){
184  counter = 0;
185  result.resize(tri->getVertexCount());
186  for(auto vert : tri->getVertices()){
187  iD = vert.getId();
188  if(sig*(dotProduct(norm, vert.getCoords()) + offset) >0) {
189  result[counter] = iD;
190  ++counter;
191  }
192  }
193 
194  }else{
195  counter = 0;
196  result.resize(tri->getCellCount());
197  for(auto cell : tri->getCells()){
198  iD = cell.getId();
199  if(sig*(dotProduct(norm, tri->evalCellCentroid(iD)) + offset) >0) {
200  result[counter] = iD;
201  ++counter;
202  }
203  }
204  }
205  result.resize(counter);
206  return result;
207 };

269 void
270 ControlDeformExtSurface::execute(){
271 
272  MimmoObject * geo = getGeometry();
273  if(geo == NULL || geo->isEmpty()) return;
274 
275  int nDFS = m_defField.size();
276  m_defField.resize(geo->getNVertex(),darray3E{{0.0,0.0,0.0}});
277  m_violationField.resize(nDFS,-1.E+18);
278 
279  dvector1D violationField;
280  violationField.resize(nDFS);
281 
282  dvecarr3E points = geo->getVertexCoords();
283  dvecarr3E pointsOR = points;
284 
285  //evaluate deformable geometry barycenter************************
286  darray3E geoBary = {{0.0,0.0,0.0}};
287  double distBary = 0.0;
288 
289  for(auto & p: points) geoBary += p;
290 
291  geoBary /= (double)geo->getNVertex();
292 
293  for(auto & p: points){
294  distBary= std::fmax(distBary,norm2(p-geoBary));
295  }
296  //***************************************************************
297 
298  //adding deformation to points **********************************
299  points+= m_defField;
300  //***************************************************************
301 
302  //read external surfaces*****************************************
303  std::vector<std::unique_ptr<MimmoGeometry> > extgeo;
304  readGeometries(extgeo);
305  //***************************************************************
306 
307  if(extgeo.size() < 1) return;
308 
309  //evaluating bounding box of deformation*************************
310  darray3E bbMinDef, bbMaxDef, bbMin, bbMax;
311  {
312  int count2 = 0;
313  for(auto & p : points){
314  if(count2 == 0) {
315  bbMinDef = p;
316  bbMaxDef = p;
317  }else{
318  for(int i=0; i<3; ++i ){
319  bbMinDef[i] = std::fmin(bbMinDef[i], p[i]);
320  bbMaxDef[i] = std::fmax(bbMaxDef[i], p[i]);
321  }
322  }
323  count2++;
324  }
325  count2=0;
326  for(auto & p : pointsOR){
327  for(int i=0; i<3; ++i ){
328  bbMinDef[i] = std::fmin(bbMinDef[i], p[i]);
329  bbMaxDef[i] = std::fmax(bbMaxDef[i], p[i]);
330  }
331  count2++;
332  }
333  }
334  //***************************************************************
335 
336  // start examining all external geometries***********************
337  for(auto &gg : extgeo){
338 
339  //check constraints properties ******************************
340  MimmoObject * local = gg->getGeometry();
341  if(!(local->isBvTreeBuilt())) local->buildBvTree();
342  bool checkOpen = local->isClosedLoop();
343 
344  double dist;
345  double radius, radius_old;
346  long id;
347  darray3E normal;
348  int count;
349  //************************************************************
350 
351  //add local constraints to the bounding box compute **********
352  bbMin = bbMinDef;
353  bbMax = bbMaxDef;
354 
355  for(auto & p : local->getVertexCoords()){
356  for(int i=0; i<3; ++i ){
357  bbMin[i] = std::fmin(bbMin[i], p[i]);
358  bbMax[i] = std::fmax(bbMax[i], p[i]);
359  }
360  }
361  //*************************************************************
362 
363  //Evaluate background grid dimension for Level set ************
364  //calculate dh as a tot part of bb diagonal *******************
365 
366  darray3E span = bbMax - bbMin;
367  bbMin += -1.0*(0.05*span);
368  span *= 1.1;
369  double dh = norm2(span)/(double)m_cellBackground;
370  iarray3E dim;
371  for(int i=0; i<3; ++i){
372  dim[i] = (int)(span[i]/dh + 0.5);
373  dim[i] = std::max(2, dim[i]);
374  }
375  //*************************************************************
376 
377  //check if its more convenient evaluate distances with a direct BvTree
378  //evaluation of distance on target deformation points or using a background grid +
379  //interpolation.
380  double nReq = double(dim[0]+1)*double(dim[1]+1)*double(dim[2]+1);
381  double nAva = 0.8*nDFS;
382 
383  if( nReq > nAva){
384 
385  //going to use direct evaluation.
386 
387  radius = distBary;
388  radius_old = radius;
389  dvector1D refsigns(nDFS, 1.0);
390 
391  //get the actual sign of distance of the undeformed cloud w.r.t constraints
392  if(checkOpen){
393  count = 0;
394  for(auto &p : pointsOR){
395  dist = evaluateSignedDistance(pointsOR[count], local, id, normal, radius);
396  if(radius < 1.0E-12) radius = radius_old;
397  else radius_old = radius;
398  if(dist < 0.0) refsigns[count] = -1.0;
399  count++;
400  }
401  }else{
402  dist = evaluateSignedDistance(geoBary, local, id, normal, radius);
403 
404  if(dist < 0.0) refsigns *= -1.0;
405  if(radius <1.0E-12) radius = radius_old;
406  }
407 
408  //evaluate distance of the deformation cloud w.r.t. constraints
409  count = 0;
410  for(auto &p : points){
411  dist = evaluateSignedDistance(p, local, id, normal, radius);
412  if(radius < 1.0E-12) radius = radius_old;
413  else radius_old = radius;
414  violationField[count] = -1.0*refsigns[count]*dist;
415  count++;
416  }
417 
418  }else{
419 
420  //going to use background grid to evaluate distances
421 
422  //instantiate a VolCartesian;
423  bitpit::VolCartesian * mesh = new bitpit::VolCartesian(0,3,bbMin,span, dim);
424  mesh->update();
425 
426  //calculate distance on point of background grid.
427  dvecarr3E background_points(mesh->getVertexCount());
428  int count = 0;
429  for(auto & v: mesh->getVertices()){
430  background_points[count] = v.getCoords();
431  count++;
432  }
433  dvector1D background_LS(background_points.size());
434 
435  radius = 0.5*norm2(span);
436  radius_old = radius;
437  count = 0;
438  for(auto &p : background_points){
439  dist = evaluateSignedDistance(p, local, id, normal, radius);
440  if(radius < 1.0E-12) radius = radius_old;
441  else radius_old = radius;
442  background_LS[count] = dist;
443  count++;
444  }
445 
446  //interpolate on background to obtain distances
447 
448  //evaluate sign of distances of the undeformed cloud w.r.t to constraints.
449  dvector1D refsigns(nDFS, 1.0);
450  if(checkOpen){
451  count= 0;
452  for(auto & p : pointsOR){
453  ivector1D stencil;
454  dvector1D weights;
455  int res = mesh->linearVertexInterpolation(p, stencil,weights);
456 
457  dist = 0.0;
458  for(int j=0; j<res; ++j){
459  dist += background_LS[stencil[j]]*weights[j];
460  }
461 
462  if(dist < 0.0) refsigns[count] = -1.0;
463  ++count;
464  }
465  }else{
466  dist = evaluateSignedDistance(geoBary, local, id, normal, radius);
467  if(dist< 0) refsigns *= -1.0;
468  if(radius < 1.0E-12) radius = radius_old;
469  }
470 
471  //evaluate violation field
472  count = 0;
473  for(auto & p : points){
474  ivector1D stencil;
475  dvector1D weights;
476 
477  int res = mesh->linearVertexInterpolation(p, stencil,weights);
478 
479  violationField[count] = 0.0;
480  for(int j=0; j<res; ++j){
481  violationField[count] += background_LS[stencil[j]]*weights[j];
482  }
483 
484  violationField[count] *= -1.0*refsigns[count];
485  ++count;
486 
487  }
488  delete mesh;
489  mesh = NULL;
490  }//end if else
491 
492  for(int i=0; i< nDFS; ++i){
493  m_violationField[i] = std::fmax(m_violationField[i], (violationField[i] + m_tolerance));
494  }
495  }
496  return;
497 };

706 double ControlDeformExtSurface::evaluateSignedDistance(darray3E &point, mimmo::MimmoObject * geo, long & id, darray3E & normal, double &initRadius){
707 
708  double dist = 1.0E+18;
709  double rate = 0.02;
710  int kmax = 1000;
711  int kiter = 0;
712  bool flag = true;
713 
714  if(!geo->isBvTreeBuilt()) geo->buildBvTree();
715 
716  while(flag && kiter < kmax){
717  dist = bvTreeUtils::signedDistance(&point, geo->getBvTree(), id, normal, initRadius);
718  flag = (dist == 1.0E+18);
719  if(flag) initRadius *= (1.0+ rate*((double)flag));
720  kiter++;
721  }
722 
723  return dist;
724 }

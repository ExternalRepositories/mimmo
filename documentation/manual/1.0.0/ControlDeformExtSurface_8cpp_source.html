305 void
306 ControlDeformExtSurface::execute(){
307 
308  MimmoObject * geo = getGeometry();
309  if(geo->isEmpty()) return;
310 
311  int nDFS = m_defField.size();
312  m_defField.resize(geo->getNVertex(),darray3E{{0.0,0.0,0.0}});
313  m_violationField.resize(nDFS,-1.E+18);
314 
315  dvector1D violationField;
316  violationField.resize(nDFS);
317 
318  dvecarr3E points = geo->getVertexCoords();
319  dvecarr3E pointsOR = points;
320 
321  //evaluate deformable geometry barycenter************************
322  darray3E geoBary = {{0.0,0.0,0.0}};
323  double distBary = 0.0;
324 
325  for(auto & p: points) geoBary += p;
326 
327  geoBary /= (double)geo->getNVertex();
328 
329  for(auto & p: points){
330  distBary= std::fmax(distBary,norm2(p-geoBary));
331  }
332  //***************************************************************
333 
334  //adding deformation to points **********************************
335  points+= m_defField;
336  //***************************************************************
337 
338  //read external surfaces*****************************************
339  std::vector<std::unique_ptr<MimmoGeometry> > extgeo;
340  dvector1D tols;
341  readGeometries(extgeo, tols);
342  //***************************************************************
343 
344  if(extgeo.size() < 1) return;
345 
346  //evaluating bounding box of deformation*************************
347  darray3E bbMinDef, bbMaxDef, bbMin, bbMax;
348  {
349  int count2 = 0;
350  for(auto & p : points){
351  if(count2 == 0) {
352  bbMinDef = p;
353  bbMaxDef = p;
354  }else{
355  for(int i=0; i<3; ++i ){
356  bbMinDef[i] = std::fmin(bbMinDef[i], p[i]);
357  bbMaxDef[i] = std::fmax(bbMaxDef[i], p[i]);
358  }
359  }
360  count2++;
361  }
362  count2=0;
363  for(auto & p : pointsOR){
364  for(int i=0; i<3; ++i ){
365  bbMinDef[i] = std::fmin(bbMinDef[i], p[i]);
366  bbMaxDef[i] = std::fmax(bbMaxDef[i], p[i]);
367  }
368  count2++;
369  }
370  }
371  //***************************************************************
372  int counterExtGeo =0;
373  // start examining all external geometries***********************
374  for(auto &gg : extgeo){
375 
376  //check constraints properties ******************************
377  MimmoObject * local = gg->getGeometry();
378  if(!(local->isBvTreeBuilt())) local->buildBvTree();
379  bool checkOpen = local->isClosedLoop();
380 
381  double dist;
382  double radius, radius_old;
383  long id;
384  darray3E normal;
385  int count;
386  //************************************************************
387 
388  //add local constraints to the bounding box compute **********
389  bbMin = bbMinDef;
390  bbMax = bbMaxDef;
391 
392  for(auto & p : local->getVertexCoords()){
393  for(int i=0; i<3; ++i ){
394  bbMin[i] = std::fmin(bbMin[i], p[i]);
395  bbMax[i] = std::fmax(bbMax[i], p[i]);
396  }
397  }
398  //*************************************************************
399 
400  //Evaluate background grid dimension for Level set ************
401  //calculate dh as a tot part of bb diagonal *******************
402 
403  darray3E span = bbMax - bbMin;
404  bbMin += -1.0*(0.05*span);
405  span *= 1.1;
406  double dh = norm2(span)/(double)m_cellBackground;
407  iarray3E dim;
408  for(int i=0; i<3; ++i){
409  dim[i] = (int)(span[i]/dh + 0.5);
410  dim[i] = std::max(2, dim[i]);
411  }
412  //*************************************************************
413 
414  //check if its more convenient evaluate distances with a direct BvTree
415  //evaluation of distance on target deformation points or using a background grid +
416  //interpolation.
417  double nReq = double(dim[0]+1)*double(dim[1]+1)*double(dim[2]+1);
418  double nAva = 0.8*nDFS;
419 
420  if( nReq > nAva){
421 
422  //going to use direct evaluation.
423 
424  radius = distBary;
425  radius_old = radius;
426  dvector1D refsigns(nDFS, 1.0);
427 
428  //get the actual sign of distance of the undeformed cloud w.r.t constraints
429  if(checkOpen){
430  count = 0;
431  for(auto &p : pointsOR){
432  dist = evaluateSignedDistance(p, local, id, normal, radius);
433  if(radius < 1.0E-12) radius = radius_old;
434  else radius_old = radius;
435  if(dist < 0.0) refsigns[count] = -1.0;
436  count++;
437  }
438  }else{
439  dist = evaluateSignedDistance(geoBary, local, id, normal, radius);
440 
441  if(dist < 0.0) refsigns *= -1.0;
442  if(radius <1.0E-12) radius = radius_old;
443  }
444 
445  //evaluate distance of the deformation cloud w.r.t. constraints
446  count = 0;
447  for(auto &p : points){
448  dist = evaluateSignedDistance(p, local, id, normal, radius);
449  if(radius < 1.0E-12) radius = radius_old;
450  else radius_old = radius;
451  violationField[count] = -1.0*refsigns[count]*dist;
452  count++;
453  }
454 
455  }else{
456 
457  //going to use background grid to evaluate distances
458 
459  //instantiate a VolCartesian;
460  bitpit::VolCartesian * mesh = new bitpit::VolCartesian(0,3,bbMin,span, dim);
461  mesh->update();
462 
463  //calculate distance on point of background grid.
464  dvecarr3E background_points(mesh->getVertexCount());
465  int count = 0;
466  for(auto & v: mesh->getVertices()){
467  background_points[count] = v.getCoords();
468  count++;
469  }
470  dvector1D background_LS(background_points.size());
471 
472  radius = 0.5*norm2(span);
473  radius_old = radius;
474  count = 0;
475  for(auto &p : background_points){
476  dist = evaluateSignedDistance(p, local, id, normal, radius);
477  if(radius < 1.0E-12) radius = radius_old;
478  else radius_old = radius;
479  background_LS[count] = dist;
480  count++;
481  }
482 
483  //interpolate on background to obtain distances
484 
485  //evaluate sign of distances of the undeformed cloud w.r.t to constraints.
486  dvector1D refsigns(nDFS, 1.0);
487  if(checkOpen){
488  count= 0;
489  for(auto & p : pointsOR){
490  ivector1D stencil;
491  dvector1D weights;
492  int res = mesh->linearVertexInterpolation(p, stencil,weights);
493 
494  dist = 0.0;
495  for(int j=0; j<res; ++j){
496  dist += background_LS[stencil[j]]*weights[j];
497  }
498 
499  if(dist < 0.0) refsigns[count] = -1.0;
500  ++count;
501  }
502  }else{
503  dist = evaluateSignedDistance(geoBary, local, id, normal, radius);
504  if(dist< 0) refsigns *= -1.0;
505  if(radius < 1.0E-12) radius = radius_old;
506  }
507 
508  //evaluate violation field
509  count = 0;
510  for(auto & p : points){
511  ivector1D stencil;
512  dvector1D weights;
513 
514  int res = mesh->linearVertexInterpolation(p, stencil,weights);
515 
516  violationField[count] = 0.0;
517  for(int j=0; j<res; ++j){
518  violationField[count] += background_LS[stencil[j]]*weights[j];
519  }
520 
521  violationField[count] *= -1.0*refsigns[count];
522  ++count;
523 
524  }
525  delete mesh;
526  mesh = NULL;
527  }//end if else
528 
529  for(int i=0; i< nDFS; ++i){
530  m_violationField[i] = std::fmax(m_violationField[i], (violationField[i] + tols[counterExtGeo]));
531  }
532  ++counterExtGeo;
533  }
534 };

754 double
755 ControlDeformExtSurface::evaluateSignedDistance(darray3E &point, mimmo::MimmoObject * geo, long & id, darray3E & normal, double &initRadius){
756 
757  double dist = 1.0E+18;
758  double rate = 0.02;
759  int kmax = 1000;
760  int kiter = 0;
761  bool flag = true;
762 
763  if(!geo->isBvTreeBuilt()) geo->buildBvTree();
764 
765  while(flag && kiter < kmax){
766  dist = bvTreeUtils::signedDistance(&point, geo->getBvTree(), id, normal, initRadius);
767  flag = (dist == 1.0E+18);
768  if(flag) initRadius *= (1.0+ rate*((double)flag));
769  kiter++;
770  }
771 
772  return dist;
773 }

779 void
780 ControlDeformExtSurface::plotOptionalResults(){
781  if(getGeometry()->isEmpty()) return;
782 
783  dvecarr3E points = getGeometry()->getVertexCoords();
784  m_defField.resize(points.size());
785  points+=m_defField;
786  ivector2D connectivity = getGeometry()->getCompactConnectivity();
787 
788  bitpit::VTKElementType elDM = bitpit::VTKElementType::TRIANGLE;
789 
790  std::string name = m_name +std::to_string(getClassCounter())+ "_ViolationField";

792  output.setGeomData( bitpit::VTKUnstructuredField::POINTS, points) ;
793  output.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, connectivity) ;
794  output.setDimensions(connectivity.size(), points.size());
795 
796

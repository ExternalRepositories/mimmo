331 CreateSeedsOnSurface::solve(bool debug){
332 
333  if(getGeometry() == NULL || m_nPoints< 1){
334  if(debug) std::cout<<"No geometry linked, or not enough total point set in "<<m_name<<" object. Doing Nothing"<<std::endl;

338  bbox->execute();
339  if(m_seedbaricenter) m_seed = bbox->getOrigin();
340 
341  switch(m_engine){
342  case CSeedSurf::RANDOM :
343  solveRandom(debug);

346  case CSeedSurf::LEVELSET :
347  solveLSet(debug);
348  break;
349 
350  case CSeedSurf::CARTESIANGRID :
351  solveGrid(debug);
352  break;
353 
354  default: //never been reached
355  break;
356  }

365 void
366 CreateSeedsOnSurface::solveLSet(bool debug){
367 
368  if(debug) std::cout<<m_name<<" : started LevelSet engine"<<std::endl;
369  dvecarr3E initList;
370  m_deads.reserve(m_nPoints);
371 
372  getGeometry()->getPatch()->buildAdjacencies();
373  //find the nearest point of triagulation to the seed
374  if(!(getGeometry()->isKdTreeBuilt())) getGeometry()->buildKdTree();
375 
376  bitpit::SurfUnstructured * tri = static_cast<bitpit::SurfUnstructured * >(getGeometry()->getPatch());
377  auto map = getGeometry()->getMapData();
378 
379  double distance = 0.0;
380  for(auto &cell : tri->getCells()){
381  distance += tri->evalCellArea(cell.getId());
382  }
383  distance /= double(tri->getCellCount());
384  distance = std::pow(distance, 0.5);
385 
386  livector1D neighs, excl;
387  darray3E projSeed = bvTreeUtils::projectPoint(&m_seed, getGeometry()->getBvTree());
388  bitpit::Vertex vertSeed(0, projSeed);
389  int nSize = 0;
390  while( nSize < 1){
391  getGeometry()->getKdTree()->hNeighbors(&vertSeed, distance, &neighs, & excl );
392  nSize = neighs.size();
393  distance *= 1.1;
394  }
395 
396  long candidate = neighs[0];
397  double minDist = norm2(projSeed - tri->getVertexCoords(candidate));
398  for(int i=1; i<nSize; ++i){
399  double val = norm2(projSeed - tri->getVertexCoords(neighs[i]));
400  if(val < minDist){
401  candidate = neighs[i];
402  minDist = val;
403  }
404  }
405 
406  m_deads.push_back(getGeometry()->getMapDataInv(candidate));
407  int deadSize = m_deads.size();
408  if(debug) std::cout<<m_name<<" : projected seed point"<<std::endl;
409 
410  std::unordered_map<long,long> invConn = getInverseConn();
411  if(debug) std::cout<<m_name<<" : created geometry inverse connectivity"<<std::endl;
412 
413  while(deadSize < m_nPoints){
414 
415  dvector1D field(tri->getVertexCount(), 1.0E18);
416  for(auto & dd : m_deads) field[dd] = 0.0;
417 
418  solveEikonal(1.0,1.0, invConn, field);
419 
420  double maxField;
421  maxval(field, maxField);
422  dvector1D::iterator itF = std::find(field.begin(), field.end(), maxField);
423 
424  m_deads.push_back(std::distance(field.begin(), itF));
425 
426  deadSize = m_deads.size();
427  if(debug) std::cout<<m_name<<" : geodesic distance field for point "<<deadSize-1<<" found"<<std::endl;
428  }
429 
430  //store result in m_points.
431  m_points.reserve(deadSize);
432  for(auto val: m_deads){
433  m_points.push_back(tri->getVertexCoords(map[val]));
434  }
435 
436  m_minDist = 1.E18;
437 
438  for(int i=0; i<deadSize; ++i){
439  for(int j=i+1; j<deadSize; ++j){
440  m_minDist = std::fmin(m_minDist,norm2(m_points[i] - m_points[j]));
441  }
442  }
443 
444  m_deads.clear();
445  if(debug) std::cout<<m_name<<" : distribution of point successfully found w/ LevelSet engine "<<std::endl;
446 };

454 void
455 CreateSeedsOnSurface::solveGrid(bool debug){
456 
457  if(debug) std::cout<<m_name<<" : started CartesianGrid engine"<<std::endl;
458  iarray3E dim;
459 
460  //get the seed and project it on surface
461  darray3E projSeed = bvTreeUtils::projectPoint(&m_seed, getGeometry()->getBvTree());
462  if(debug) std::cout<<m_name<<" : projected seed point"<<std::endl;
463  if (m_nPoints == 1) {
464  m_points.clear();
465  m_points.push_back(projSeed);
466  return;
467  }
468 
469  m_minDist = norm2(bbox->getSpan());
470  double dx = m_minDist/int(std::pow(double(m_nPoints),0.5)+ 0.5);
471  {
472  //check dimension
473  std::vector<std::pair<double,int> > mapDimension(3);
474  std::vector<std::pair<double,int> >::iterator itm;
475  std::vector<std::pair<double,int> >::reverse_iterator ritm;
476 
477  mapDimension[0] = std::make_pair(bbox->getSpan()[0]/dx, 0);
478  mapDimension[1] = std::make_pair(bbox->getSpan()[1]/dx, 1);
479  mapDimension[2] = std::make_pair(bbox->getSpan()[2]/dx, 2);
480 
481  std::sort(mapDimension.begin(), mapDimension.end());
482 
483  for(itm = mapDimension.begin(); itm !=mapDimension.end(); ++itm){
484  dim[itm->second] = std::max(int(itm->first + 0.5),1);
485  }
486 
487  int cumCells = dim[0]*dim[1]*dim[2];
488  ritm=mapDimension.rbegin();
489  while (cumCells < m_nPoints){
490 
491  dim[ritm->second] += 1;
492  ritm++;
493  if(ritm == mapDimension.rend()) ritm = mapDimension.rbegin();
494 
495  cumCells = dim[0]*dim[1]*dim[2];
496  }
497  }
498 
499  //raise dimension to number of effective nodes for each direction.
500  dim[0] +=1;
501  dim[1] +=1;
502  dim[2] +=1;
503 
504  //create a lattice on it
505  mimmo::Lattice * grid = new Lattice();
506  grid->setShape(mimmo::ShapeType::CUBE);
507  grid->setOrigin(bbox->getOrigin());
508  grid->setSpan(bbox->getSpan());
509  grid->setRefSystem(bbox->getAxes());
510  grid->setDimension(dim);
511  grid->execute();
512  //grid->plotGrid("./", "lattice",0,false);
513 
514  if(debug) std::cout<<m_name<<" : build volume cartesian grid wrapping 3D surface"<<std::endl;
515  //find narrow band cells and extracting their centroids
516  dvecarr3E centroids = grid->getGlobalCellCentroids();
517 
518  dvecarr3E initList;
519  initList.reserve(centroids.size());
520 
521  double distR = 0.5*norm2(grid->getSpacing());
522  double dist, dummy;
523  long id;
524  darray3E normal;
525  for(auto & p : centroids){
526  dummy=distR;
527  dist = mimmo::bvTreeUtils::signedDistance(&p,getGeometry()->getBvTree(),id, normal, dummy);
528  if(std::abs(dist) < distR){
529  initList.push_back(p-dist*normal);
530  }
531  normal.fill(0.0);
532  }
533  if(debug) std::cout<<m_name<<" : found grid cell centers in the narrow band of 3D surface and projected them on it "<<std::endl;
534 
535 
536  //rearrange the list, putting the most nearest point to the seed on top
537  // and decimate points up to desired value.
538  int initListSize = initList.size();
539  if( initListSize > m_nPoints){
540 
541  dvecarr3E secondList;
542  secondList.reserve(initList.size());
543  {
544  dvecarr3E::iterator it, itMaxNorm = initList.begin();
545  double normPoint, minValue = 1.E18;
546 
547  for(it= initList.begin(); it !=initList.end(); ++it){
548  normPoint= norm2(*it - m_seed);
549  if(normPoint < minValue){
550  minValue = normPoint;
551  itMaxNorm= it;
552  }
553  }
554 
555  darray3E temp = *itMaxNorm;
556  initList.erase(itMaxNorm);
557  secondList.push_back(temp);
558  secondList.insert(secondList.end(),initList.begin(), initList.end());
559  initList.clear();
560  }
561 
562  initList = decimatePoints(secondList);
563  if(debug) std::cout<<m_name<<" : candidates decimated "<<std::endl;
564  }
565  //store result in m_points.
566  m_points = initList;
567  m_nPoints = (int)m_points.size();
568  if(debug) std::cout<<m_name<<" : distribution of point successfully found w/ CartesianGrid engine "<<std::endl;
569  delete grid; grid = NULL;
570 };

578 void
579 CreateSeedsOnSurface::solveRandom(bool debug){
580 
581  if(debug) std::cout<<m_name<<" : started Random engine"<<std::endl;
582  dvecarr3E initList(getNPoints());
583 
584  //get the seed and project it on surface
585  initList[0] = bvTreeUtils::projectPoint(&m_seed, getGeometry()->getBvTree());
586  if(debug) std::cout<<m_name<<" : projected seed point"<<std::endl;
587  if (m_nPoints == 1) {
588  m_points.clear();
589  m_points = initList;
590  return;
591  }
592 
593  m_minDist = norm2(bbox->getSpan()) / 2.0;
594 
595  //fill the oriented bounding box of the figure randomly with max of 100 and 5*m_nPoints
596  dvecarr3E tentative;
597  {
598  darray3E span = bbox->getSpan();
599  dmatrix33E axes = bbox->getAxes();
600  darray3E minP = bbox->getOrigin();
601  for(int i=0; i<3; ++i) {minP += - 0.5*span[i]*axes[i];}
602 
603 
604  if (m_randomFixed <0 ){
605  m_randomFixed = (unsigned int)time(NULL);
606  }
607  srand(static_cast<unsigned int>(m_randomFixed));
608 
609  int nTent = std::max(100,5*m_nPoints);
610  tentative.resize(nTent+1);
611  tentative[0] = initList[0];
612 
613  for(int i = 0; i<nTent; ++i){
614  tentative[i+1] = minP;
615  for(int j=0; j<3; ++j){
616 
617  double valrand = (std::rand()%100)/99.0;
618  tentative[i+1] += valrand * span[j]*axes[j];
619  }
620  }
621 
622  //project tentative points on surface.
623  for(int i = 0; i<nTent; ++i){
624  minP = bvTreeUtils::projectPoint(&tentative[i+1], getGeometry()->getBvTree());
625  tentative[i+1] = minP;
626  }
627  }
628 
629 
630  if(debug) std::cout<<m_name<<" : found random points"<<std::endl;
631  //decimate points up to desired value
632  initList = decimatePoints(tentative);
633  if(debug) std::cout<<m_name<<" : decimated random points"<<std::endl;
634  //store result in m_points.
635  m_points = initList;
636  m_nPoints = (int)m_points.size();
637  if(debug) std::cout<<m_name<<" : distribution of point successfully found w/ Random engine "<<std::endl;
638 };

676 dvecarr3E
677 CreateSeedsOnSurface::decimatePoints(dvecarr3E & list){
678 
679  dvecarr3E result(m_nPoints);
680  int listS = list.size();
681 
682  bitpit::KdTree<3,darray3E,long> kdT;
683  //build a kdtree of points
684  kdT.nodes.resize(list.size() + kdT.MAXSTK);
685  long label=0;
686  for(auto & val : list ){
687  kdT.insert(&val, label);
688  ++label;
689  }
690 
691  long candidate = 0; //starting seed;
692 
693  //use kdTree to search points within a radius of m_minDist from the seed.
694  // if no matches are found, choose randomly the next candidate
695  livector1D neighs, excl,effective;
696  std::set<long> visited;
697  std::map<double, long> chooseCand;
698  livector1D finalCandidates;
699 
700  finalCandidates.reserve(m_nPoints);
701  int candSize = finalCandidates.size();
702  while( candSize < m_nPoints){
703 
704  finalCandidates.push_back(candidate);
705  candSize++;
706  visited.insert(candidate);
707  excl.insert(excl.end(), visited.begin(), visited.end());
708  kdT.hNeighbors(&list[candidate], m_minDist, &neighs, & excl );
709 
710  visited.insert(neighs.begin(), neighs.end());
711 
712  int sizeN = visited.size();
713  neighs.clear();
714  excl.clear();
715 
716 
717  while(sizeN == listS){
718  m_minDist *= 0.9;
719  visited.clear();
720  visited.insert(finalCandidates.begin(), finalCandidates.end());
721  for(auto ind : finalCandidates){
722  excl.insert(excl.end(), visited.begin(), visited.end());
723  kdT.hNeighbors(&list[ind], m_minDist, &neighs, & excl );
724  visited.insert(neighs.begin(), neighs.end());
725  }
726  sizeN = visited.size();
727  excl.clear();
728  neighs.clear();
729  }
730 
731  effective.reserve(list.size() - visited.size());
732  std::set<long>::iterator it1 = visited.begin();
733  for(int i=0; i< listS; ++i){
734  if(it1 != visited.end() || i != *it1 ){
735  effective.push_back(i);
736 
737  }else{ ++it1;}
738  }
739 
740  for(auto & ind : effective){
741  dvector1D norms(finalCandidates.size());
742  int countNorms=0;
743  for(auto indEx: finalCandidates){
744  norms[countNorms] = norm2(list[ind] - list[indEx]);
745  ++countNorms;
746  }
747 
748  double norm = 0.0, variance=0.0;
749  for(auto &val : norms) norm += val;
750  norm /= double(norms.size());
751  for(auto &val : norms) val = std::abs(val/norm - 1.0);
752  maxval(norms, variance);
753  norm /= std::pow((variance+1.0),2);
754 
755  chooseCand[norm] = ind;
756  }
757 
758  candidate = (chooseCand.rbegin())->second;
759 
760  effective.clear();
761  chooseCand.clear();
762  }
763 
764  int counter = 0;
765  for(auto index : finalCandidates){
766  result[counter] = list[index];
767  ++counter;
768  }
769 
770  return result;
771 };

786 double
787 CreateSeedsOnSurface::updateEikonal(double g, double s, long tVert,long tCell, std::unordered_map<long int, short int> &flag, dvector1D & field){
788 
789  BITPIT_UNUSED(s);
790  //get the pointer to reference geometry
791  liimap & vmap = getGeometry()->getMapDataInv();
792  bitpit::PatchKernel * tri = getGeometry()->getPatch();
793 
794  //todo get dimension? of Simplex. Get a check for non triangular cell or generalize it
795 
796  livector1D oneRing;
797  long I, U, V, W;
798 
799  int k, m;
800 
801  int select = -1;
802  double value(1.0e+18);
803  double dVU, dVW, dWU, dVP;
804  std::array<double,3> eVU, eVW, eWU, eVP, P;
805 
806 
807  double a, b, c, A, B, C, K, discr ;
808  double phi_U, phi_W, phi_P;
809  int discrType;
810 
811  double xi1, xi2;
812  double tempVal1, tempVal2;
813 
814  //get current field value of the node;
815  value = std::abs(field[vmap[tVert]]);
816 
817  V = tVert;
818

820  // find 1-Ring cells around target node
821  bitpit::Cell & targetCell = tri->getCell(tCell);
822  int locVert = targetCell.findVertex(tVert);
823  if(locVert == bitpit::Vertex::NULL_ID) return value;
824  oneRing = tri->findCellVertexOneRing(tCell, locVert);
825  }
826 
827  // Loop over cells in the 1-Ring --------------------------------------------------- //
828  for (auto && oneIndex : oneRing) {
829 
830  // Cell data get id of vertex composing triangular cell
831  I = oneIndex;
832  bitpit::Cell & cellI = tri->getCell(I);
833 
834  k = cellI.findVertex(V);
835  k = (k + 1) % cellI.getVertexCount();
836  U = cellI.getVertex(k);
837  m = (k + 1) % cellI.getVertexCount();
838  W = cellI.getVertex(m);
839 
840  // discriminate case, according to flag vector of deads, alives and far-aways
841  if ((flag[U] == 0) && (flag[W] == 0)) {
842  select = 2;
843  }
844  else {
845  if ((flag[U] == 0) || (flag[W] == 0)) {
846  select = 1;
847  if (flag[W] == 0) {
848  U = W;
849  }
850  }
851  else {
852  select = 0;
853  }
854  }
855 
856  //Compute solution to the 2D Eikonal equation
857  switch (select){
858 
859  case 1 : // with 1 dead node
860  eVU = tri->getVertexCoords(V) - tri->getVertexCoords(U);
861  dVU = norm2(eVU);
862  value = std::min(value, std::abs(field[vmap[U]]) + g*dVU);
863  break;
864 
865  case 2 : // with 2 dead nodes
866 
867  eVW = tri->getVertexCoords(V) - tri->getVertexCoords(W);
868  dVW = norm2(eVW);
869  eVW = eVW/dVW;
870  eVU = tri->getVertexCoords(V) - tri->getVertexCoords(U);
871  dVU = norm2(eVU);
872  eVU = eVU/dVU;
873  eWU = tri->getVertexCoords(W) - tri->getVertexCoords(U);
874  dWU = norm2(eWU);
875  eWU = eWU/dWU;
876 
877  // Coeffs -------------------------------------------------------------------- //
878  phi_U = std::abs(field[vmap[U]]);
879  phi_W = std::abs(field[vmap[W]]);
880  K = phi_W - phi_U;
881  a = pow(dWU, 2);
882  b = -dWU*dVU*dotProduct(eWU, eVU);
883  c = pow(dVU, 2);
884  A = a*(pow(K, 2) - a);
885  B = b*(pow(K, 2) - a);
886  C = (pow(K, 2)*c - pow(b, 2));
887  discr = pow(B, 2) - A*C;
888 
889  // Find optimal solution ----------------------------------------------------- //
890  discrType = (discr < -1.0e-12) + 2*(std::abs(A) > 1.0e-12) +3*((std::abs(A) < 1.0e-12) && (std::abs(A) >= 0.0)) -1;
891 
892  switch(discrType){
893  case 1: //2 distinct solutions
894  discr = std::abs(discr);
895 
896  xi1 = (-B - sqrt(discr))/A;
897  xi2 = (-B + sqrt(discr))/A;
898 
899  // Restriction of solutions onto [0, 1]
900  xi1 = std::min(1.0, std::max(0.0, xi1));
901  xi2 = std::min(1.0, std::max(0.0, xi2));
902 
903  // Solution #1
904  P = (1.0 - xi1) * tri->getVertexCoords(U) + xi1 * tri->getVertexCoords(W);
905  eVP = tri->getVertexCoords(V) - P;
906  dVP = norm2(eVP);
907  eVP = eVP/dVP;
908  phi_P = (1.0 - xi1) * phi_U + xi1 * phi_W;
909  tempVal1 = phi_P + g * dVP;
910 
911  // Solution #2
912  P = (1.0 - xi2) * tri->getVertexCoords(U) + xi2 * tri->getVertexCoords(W);
913  eVP = tri->getVertexCoords(V) - P;
914  dVP = norm2(eVP);
915  eVP = eVP/dVP;
916  phi_P = (1.0 - xi2) * phi_U + xi2 * phi_W;
917  tempVal2 = phi_P + g * dVP;
918 
919  break;
920 
921  case 2: // coincident solutions
922  discr = std::abs(discr);
923 
924  // Solution #1
925  P = tri->getVertexCoords(U);
926  eVP = tri->getVertexCoords(V) - P;
927  dVP = norm2(eVP);
928  eVP = eVP/dVP;
929  phi_P = phi_U;
930  tempVal1 = phi_P + g*dVP;
931 
932  // Solution #2
933  P = tri->getVertexCoords(W);
934  eVP = tri->getVertexCoords(V) - P;
935  dVP = norm2(eVP);
936  eVP = eVP/dVP;
937  phi_P = phi_W;
938  tempVal2 = phi_P + g*dVP;
939 
940  break;
941 
942  default: //no real solutions indeed
943  tempVal1 = value;
944  tempVal2 = value;
945  break;
946  }//end on switch discrType
947 
948  // Update solution for case 2:
949  value = std::min(value, std::min(tempVal1, tempVal2));
950  break;
951 
952  default: //doing really nothing. No dead nodes to hang out.
953  break;
954  }//end switch select
955 
956  } //loop on oneRing
957 
958  tri = NULL;
959 
960  return(value);
961 };

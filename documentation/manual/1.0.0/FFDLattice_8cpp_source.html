704 darray3E
705 FFDLattice::nurbsEvaluator(darray3E & pointOr){
706 
707  darray3E point = transfToLocal(pointOr);
708  darray3E scaling = getShape()->getScaling();
709 
710  ivector1D knotInterval(3,0);
711  dvector2D BSbasis(3);
712  dvector1D valH(4,0), temp1(4,0),temp2(4,0), zeros(4,0);
713  int uind, vind, wind, index;
714 
715  dvecarr3E* displ = getDisplacements();
716 
717  int i0 = m_mapdeg[0];
718  int i1 = m_mapdeg[1];
719  int i2 = m_mapdeg[2];
720 
721  iarray3E mappedIndex;
722 
723  for(int i=0; i<3; i++){
724  knotInterval[i] = getKnotInterval(point[i],i);
725  BSbasis[i] = basisITS0(knotInterval[i], i, point[i]);
726  }
727 
728  uind = knotInterval[i0] - m_deg[i0];
729  vind = knotInterval[i1] - m_deg[i1];
730  wind = knotInterval[i2] - m_deg[i2];
731 
732  for(int i=0; i<=m_deg[i0]; ++i){
733 
734  mappedIndex[0] = uind+i;
735  temp1 = zeros;
736 
737  for(int j=0; j<=m_deg[i1]; ++j){
738 
739  mappedIndex[1] = vind+j;
740  temp2 = zeros;
741 
742  for(int k=0; k<=m_deg[i2]; ++k){
743 
744  mappedIndex[2] = wind+k;
745  index = accessMapNodes(mappedIndex[0],mappedIndex[1],mappedIndex[2]);
746 
747  for(int intv=0; intv<3; ++intv){
748  temp2[intv] += BSbasis[i2][k]*m_weights[m_intMapDOF[index]]*(*displ)[m_intMapDOF[index]][intv];
749  }
750  temp2[3] += BSbasis[i2][k]*m_weights[index];
751  }
752  for(int intv=0; intv<4; ++intv){
753  temp1[intv] += BSbasis[i1][j]*temp2[intv];
754  }
755 
756  }
757  for(int intv=0; intv<4; ++intv){
758  valH[intv] += BSbasis[i0][i]*temp1[intv];
759  }
760  }
761 
762  darray3E outres;
763  if(isDisplGlobal()){
764  for(int i=0; i<3; ++i){
765  outres[i] = valH[i]/valH[3];
766  }
767 
768  }else{
769  //summing scaled displ in local ref frame;
770  for(int i=0; i<3; ++i){
771  point[i] += valH[i]/(valH[3]*scaling[i]);
772  }
773 
774  //get final displ in global ref frame:
775  outres = transfToGlobal(point) - pointOr;
776  }
777  return(outres);
778 
779 };

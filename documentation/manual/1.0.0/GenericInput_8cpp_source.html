192 
193  BITPIT_UNUSED(name);
194 
195  std::string input;
196 
197  BaseManipulation::absorbSectionXML(slotXML, name);
198 
199  if(slotXML.hasOption("ReadFromFile")){
200  std::string input = slotXML.get("ReadFromFile");
201  input = bitpit::utils::trim(input);
202  bool temp = false;
203  if(!input.empty()){
204  std::stringstream ss(input);
205  ss>>temp;
206  }
207  setReadFromFile(temp);
208  };
209 
210  if(slotXML.hasOption("CSV")){
211  std::string input = slotXML.get("CSV");
212  input = bitpit::utils::trim(input);
213  bool temp = false;
214  if(!input.empty()){
215  std::stringstream ss(input);
216  ss>>temp;
217  }
218  setCSV(temp);
219  };
220 
221  if(slotXML.hasOption("Filename") && m_readFromFile){
222  std::string input = slotXML.get("Filename");
223  input = bitpit::utils::trim(input);
224  setFilename(input);
225  };
226 
227  if(slotXML.hasOption("ReadDir") && m_readFromFile){
228  std::string input = slotXML.get("ReadDir");
229  input = bitpit::utils::trim(input);
230  setReadDir(input);

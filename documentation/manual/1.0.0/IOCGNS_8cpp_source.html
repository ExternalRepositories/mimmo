318 bool
319 IOCGNS::read(){
320 
321  std::string file = m_rdir+"/"+m_rfilename+".cgns";
322  std::string error_string = "read CGNS grid: " + file;
323 
324  long nVertices;
325  // long nCells, nBoundVertices;
326  int nCoords; //, nConnSections;
327  std::array< std::vector<double>,3 > coords;
328  std::vector<CG_ElementType_t> orderedConns;
329  std::unordered_map<int, ivector1D> conns;
330  std::unordered_map<int, ivector1D > bcLists;
331 
332  //Open cgns file
333  int indexfile;
334  if(cg_open(file.c_str(), CG_MODE_READ, &indexfile) != CG_OK){
335  return false;
336  }
337 
338  //Read number of bases
339  int nbases;
340  if(cg_nbases(indexfile, &nbases) != CG_OK){
341  return false;
342  }
343  if(nbases > 1){
344  std::cout << "More than one base found in grid file -> only the first zone will be read" << std::endl;
345  }
346 
347  //Read name of basis and physical dimension
348  char basename[33];
349  int physdim, celldim;
350  if(cg_base_read(indexfile, 1, basename, &celldim, &physdim) != CG_OK){
351  return false;
352  }
353  if(celldim != 3 || physdim !=3){
354  //Only volume mesh supported
355  //TODO NO SURFACE MESH (ONLY SURFACE WITHOUT VOLUME)?
356  return false;
357  }
358 
359  //Read number of zone in basis, check if only one.
360  int nzones;
361  if(cg_nzones(indexfile, 1, &nzones)!=CG_OK){
362  return false;
363  }
364  if(nzones > 1){
365  //Only single volume zone supported for now
366  return false;
367  }
368 
369  //Read type mesh of zone
370  CG_ZoneType_t zoneType;
371  int index_dim;
372  if(cg_zone_type(indexfile,1,1, &zoneType)!= CG_OK){
373  return false;
374  }
375  if(cg_index_dim(indexfile,1,1, &index_dim)!= CG_OK){
376  return false;
377  }
378  if(zoneType != CG_ZoneType_t::CG_Unstructured || index_dim != 1){
379  //Only unstructured mesh supported for now (index_dim == 1 for unstructured)
380  return false;
381  }
382 
383  //Read size of zone (n nodes, n cells, n boundary nodes)
384  std::vector<cgsize_t> sizeG(3);
385  char zonename[33];
386  if(cg_zone_read(indexfile,1,1, zonename, sizeG.data()) != CG_OK ){
387  return false;
388  }
389 
390 
391  nVertices = sizeG[0];
392  // nCells = sizeG[1];
393  // nBoundVertices = sizeG[2];
394 
395  //Read Vertices.
396  if(cg_ncoords(indexfile,1,1, &nCoords)!= CG_OK){
397  return false;
398  }
399  for(int i = 1; i <= nCoords; ++i){
400  cgsize_t startIndex = 1;
401  cgsize_t finishIndex = sizeG[0];
402  CG_DataType_t datatype;
403  char name[33];
404  coords[i-1].resize(nVertices);
405  if(cg_coord_info(indexfile,1,1,i, &datatype, name)!=CG_OK){
406  return false;
407  }
408  if(cg_coord_read(indexfile,1,1,name, datatype, &startIndex, &finishIndex, coords[i-1].data() )!=CG_OK){
409  return false;
410  }
411  }
412 
413  //Read connectivities.
414  //They are read starting from 1, fortran style. When matching up w/ coords
415  //vector positions remember to diminish conn value of 1.
416  //Read number of sections of connectivity structure
417  int nSections;
418  if(cg_nsections(indexfile,1,1, &nSections)!= CG_OK){
419  return false;
420  }
421 
422 
423  for(int sec = 1; sec <= nSections; ++sec){
424 
425  //Read section sec
426  char elementname[33];
427  CG_ElementType_t type;
428  cgsize_t eBeg, eEnd, enBdry;
429  int parent_flag;
430  std::vector<cgsize_t> connlocal;
431  cgsize_t size;
432 
433  //Read elements name, type and range
434  if(cg_section_read(indexfile,1,1, sec, elementname, &type, &eBeg, &eEnd,&enBdry, & parent_flag)!= CG_OK){
435  return false;
436  }
437 
438  //Read size of connectivity data
439  if(cg_ElementDataSize(indexfile,1,1, sec,&size)!= CG_OK){
440  return false;
441  }
442  //cgsize_t nelements = eEnd-eBeg + 1;
443 
444  connlocal.resize((size_t) size);
445  int *ptr = NULL;
446  if(cg_elements_read(indexfile,1,1,sec, connlocal.data(),ptr) !=CG_OK){
447  return false;
448  }
449 
450  conns[type].resize(connlocal.size());
451  int count=0;
452  for(auto &val: connlocal){
453  conns[type][count] = (int)val;
454  ++count;
455  }
456 
457  orderedConns.push_back(type);
458  }
459 
460  //explore superficial boundary conditions definition, for boundary surface extraction.
461  int nBcs;
462  if(cg_nbocos(indexfile,1,1, &nBcs)!= CG_OK){
463  return false;
464  }
465 
466  //up to now i'm not able to encoding all the possible variants of bc node structure.
467  // i rely on the element list . Need to study better the cgns docs.
468  // TODO ????
469  for(int bc=1; bc<=nBcs; ++bc){
470 
471  char name[33];
472  CG_BCType_t bocotype;
473  CG_PointSetType_t ptset_type;
474  std::vector<cgsize_t> nBCElements(2);
475  int normalIndex;
476  cgsize_t normalListSize;
477  CG_DataType_t normalDataType;
478  int ndataset;
479  CG_GridLocation_t location;
480 
481  if(cg_boco_gridlocation_read(indexfile,1,1,bc, &location) != CG_OK){
482  return false;
483  }
484 
485  if(cg_boco_info(indexfile,1,1,bc, name, &bocotype, &ptset_type, nBCElements.data(),
486  &normalIndex, &normalListSize, &normalDataType, &ndataset) != CG_OK){
487  return false;
488  }
489 
490  if(ptset_type == CG_PointSetType_t::CG_ElementList || ptset_type == CG_PointSetType_t::CG_ElementRange){
491 
492  cgsize_t dim;
493  if(ptset_type == CG_PointSetType_t::CG_ElementList){
494  dim = nBCElements[0];
495  }else{
496  dim = nBCElements[1] - nBCElements[0] +1;
497  }
498 
499  std::vector<cgsize_t> localbc((size_t) dim);
500  int *ptr = NULL;
501 
502  if(cg_boco_read(indexfile,1,1,bc, localbc.data(), ptr )!= CG_OK){
503  return false;
504  }
505 
506  bcLists[bc-1].resize(localbc.size());
507  int count=0;
508  for(auto &val: localbc){
509  // WHY INT AND NOT LONG INT (IT IS A BITPIT PID) ??
510  bcLists[bc-1][count] = (int)val;
511  ++count;
512  }
513 
514  }
515 
516  m_storedBC->mcg_pidtobc[bc] = bocotype;
517  m_storedBC->mcg_pidtoname[bc] = name;
518 
519  }
520 
521  m_storedBC->mcg_pidtobc[0] = CG_BCTypeNull;
522  m_storedBC->mcg_pidtoname[0] = "undefined";
523 
524  //Finish reading CGNS file
525  cg_close(indexfile);
526 
527  //Reverse info in your grids.
528  std::unique_ptr<MimmoObject> patchVol(new MimmoObject(2));
529  std::unique_ptr<MimmoObject> patchBnd(new MimmoObject(1));
530  long id;
531  darray3E temp;
532 
533  // Stock vertices in volume grid
534  for(int i=0; i<nVertices; ++i){
535 
536  for(int j=0; j<3; ++j) temp[j] = coords[j][i];
537  id = i+1; //not C indexing, labeling coherent with connectivity.
538  patchVol->addVertex(temp,id);
539  }
540 
541  //debug only;
542  //TODO ????
543 
544  id = 1;
545  //Unpack 3D elements connectivities and store it in volume grid. Label same species elements w PID.
546  for(auto & val: orderedConns){
547 
548  livector1D lConn;
549  bitpit::ElementInfo::Type btype;
550  short PID = 0;
551  int size = conns[(int)val].size();
552 
553  switch(val){
554  case CGNS_ENUMV(MIXED):
555 
556  unpack3DElementsMixedConns(patchVol.get(), patchBnd.get(), conns[(int)val], id);
557  break;
558 
559  case CGNS_ENUMV(TETRA_4):
560  case CGNS_ENUMV(TETRA_10):
561 
562  btype = bitpit::ElementInfo::Type::TETRA;
563  lConn.resize(4);
564  for(int i=0; i<size; i+=4){
565  for(int j=0; j<4; ++j){
566  lConn[j] = conns[(int)val][i+j];
567  }
568  patchVol->addConnectedCell(lConn, btype, id );
569  patchVol->setPIDCell(id,PID);
570  id++;
571  }
572  break;
573 
574  case CGNS_ENUMV(PYRA_5):
575  case CGNS_ENUMV(PYRA_14):
576 
577  btype = bitpit::ElementInfo::Type::PYRAMID;
578  lConn.resize(5);
579  for(int i=0; i<size; i+=5){
580  for(int j=0; j<5; ++j){
581  lConn[j] = conns[(int)val][i+j];
582  }
583  patchVol->addConnectedCell(lConn, btype, id );
584  patchVol->setPIDCell(id,PID);
585  id++;
586  }
587  break;
588 
589  case CGNS_ENUMV(PENTA_6):
590  case CGNS_ENUMV(PENTA_15):
591  case CGNS_ENUMV(PENTA_18):
592 
593  btype = bitpit::ElementInfo::Type::WEDGE;
594  lConn.resize(6);
595  for(int i=0; i<size; i+=6){
596  for(int j=0; j<6; ++j){
597  lConn[j] = conns[(int)val][i+j];
598  }
599  patchVol->addConnectedCell(lConn, btype, id );
600  patchVol->setPIDCell(id,PID);
601  id++;
602  }
603 
604  break;
605 
606  case CGNS_ENUMV(HEXA_8):
607  case CGNS_ENUMV(HEXA_20):
608  case CGNS_ENUMV(HEXA_27):
609 
610  btype = bitpit::ElementInfo::Type::HEXAHEDRON;
611  lConn.resize(8);
612  for(int i=0; i<size; i+=8){
613  for(int j=0; j<8; ++j){
614  lConn[j] = conns[(int)val][i+j];
615  }
616  patchVol->addConnectedCell(lConn, btype, id );
617  patchVol->setPIDCell(id,PID);
618  id++;
619  }
620  break;
621 
622  case CGNS_ENUMV(TRI_3):
623  case CGNS_ENUMV(TRI_6):
624 
625  btype = bitpit::ElementInfo::Type::TRIANGLE;
626  lConn.resize(3);
627 
628  for(int i=0; i<size; i+=3){
629  for(int j=0; j<3; ++j){
630  lConn[j] = conns[(int)val][i+j];
631  }
632  patchBnd->addConnectedCell(lConn, btype, id );
633  patchBnd->setPIDCell(id,PID);
634  id++;
635  }
636  break;
637 
638  case CGNS_ENUMV(QUAD_4):
639  case CGNS_ENUMV(QUAD_8):
640  case CGNS_ENUMV(QUAD_9):
641 
642  btype = bitpit::ElementInfo::Type::QUAD;
643  lConn.resize(4);
644  for(int i=0; i<size; i+=4){
645  for(int j=0; j<4; ++j){
646  lConn[j] = conns[(int)val][i+j];
647  }
648  patchBnd->addConnectedCell(lConn, btype, id );
649  patchBnd->setPIDCell(id,PID);
650  id++;
651  }
652  break;
653 
654  default:
655  //do nothing
656  break;
657  }
658  }
659 
660  //divide patchBnd in subpatch if any bc is present.
661  if(bcLists.size() > 0) {
662  for(auto & sel: bcLists){
663  if(sel.second.size() > 0){
664  for(auto &ind : sel.second){
665  id = ind;
666  short PID = sel.first + 1;
667  patchBnd->setPIDCell(id,PID);
668  }
669  }
670  }//end for
671 
672  }
673 
674 
675  //adding vstand alone vertices to boundary patches and release all structures
676  {
677  livector2D tempConn = patchBnd->getConnectivity();
678  std::set<long> ordIndex;
679 
680  for(auto & val: tempConn){
681  ordIndex.insert(val.begin(), val.end());
682  }
683 
684  for(auto & val: ordIndex){
685  temp = patchVol->getVertexCoords(val);
686  patchBnd->addVertex(temp,val);
687  }
688  }
689 
690  //release the meshes
691  m_volmesh = std::move(patchVol);
692  m_surfmesh = std::move(patchBnd);
693 
694  return true;
695 }

701 bool
702 IOCGNS::write(){
703 
704  /* Clear old Info if stored. */
705  m_storedInfo->mcg_number.clear();
706 
707  std::string file = m_wdir+"/"+m_wfilename+".cgns";
708  std::string error_string = "write CGNS grid: " + file;
709 
710  long nVertices, nCells, nBoundVertices;
711  //int nCoords;
712  std::array< std::vector<double>,3 > coords;
713 
714  /* Fill this structures (surface boundary connectivity has
715  * to be filled with same vertex indices of volume mesh) */
716  MimmoObject * vol = getGeometry();
717  MimmoObject * bnd = getSurfaceBoundary();
718 
719  if( vol == NULL && bnd == NULL ) return false;
720  if( vol->isEmpty() && bnd->isEmpty() ) return false;
721 
722  /* Verify if a surface boundary mesh exists. */
723  //bool flagBnd = (bnd != NULL);
724 
725  /* Check number of volume cells and global vertices. */
726  nVertices = vol->getNVertex();
727  nCells = vol->getNCells();
728  nBoundVertices=0;
729 
730  //fill coordinates;

998 void
999 IOCGNS::recoverCGNSInfo(){
1000 
1001  MimmoObject * vol = getGeometry();
1002  MimmoObject * bnd = getSurfaceBoundary();
1003 
1004  /* Verifiy if a surface mesh (boundary mesh) is set. */
1005  bool flagBnd = (bnd != NULL);
1006 
1007  if( vol == NULL && bnd == NULL ) return;
1008  if( vol->isEmpty() && bnd->isEmpty() ) return;
1009 
1010  int nVolElements = 0;
1011 
1012  /* Fill volume map info. */
1013  CG_ElementType_t cgtype;
1014  long int ID;
1015  for (auto & cell : vol->getCells()){
1016  cgtype = m_storedInfo->mcg_typeconverter[cell.getType()];
1017  ID = cell.getId();
1018  m_storedInfo->mcg_number[cgtype]++;
1019  m_storedInfo->mcg_typetoid[cgtype].push_back(ID);
1020  long int * conn = cell.getConnect();
1021  for (int iV=0; iV<cell.getVertexCount(); iV++){
1022  m_storedInfo->mcg_typetoconn[cgtype].push_back(conn[iV]);
1023  }
1024  }
1025 
1026  /*Iterate map volume info and fill local CGNS index info
1027  * (volume elements will be appended before boundary elements)
1028  */
1029  std::map<CG_ElementType_t, std::vector<long int> >::iterator it, itend;
1030  std::vector<long int>::iterator itV, itVend;
1031  itend = m_storedInfo->mcg_typetoid.end();
1032  int cgnsidx = 0;
1033  for ( it = m_storedInfo->mcg_typetoid.begin(); it != itend; ++it ){
1034  itVend = it->second.end();
1035  for ( itV = it->second.begin(); itV != itVend; ++itV ){
1036  cgnsidx++;
1037  ID = (*itV);
1038  m_storedInfo->mcg_idtoindex[ID] = cgnsidx;
1039  m_storedInfo->mcg_indextoid.push_back(ID);
1040  }
1041  }
1042  nVolElements = cgnsidx;
1043 
1044 
1045  if (flagBnd){
1046  /* Fill surface mesh (boundary) map info. */
1047  bitpit::PatchKernel* bndpatch = bnd->getPatch();
1048  CG_ElementType_t cgtype;
1049  long int ID;
1050  for (auto & cell : bnd->getCells()){
1051  cgtype = m_storedInfo->mcg_typeconverter[cell.getType()];
1052  ID = cell.getId();
1053  m_storedInfo->mcg_bndnumber[cgtype]++;
1054  m_storedInfo->mcg_bndtypetoid[cgtype].push_back(ID);
1055  long int * conn = cell.getConnect();
1056  for (int iV=0; iV<cell.getVertexCount(); iV++){
1057  m_storedInfo->mcg_bndtypetoconn[cgtype].push_back(conn[iV]);
1058  }
1059  }
1060 
1061  /*Iterate map surface (boundary) info and fill local CGNS index info
1062  * (surface elements will be appended after volume elements)
1063  */
1064  std::map<CG_ElementType_t, std::vector<long int> >::iterator it, itend;
1065  std::vector<long int>::iterator itV, itVend;
1066  itend = m_storedInfo->mcg_bndtypetoid.end();
1067  int bndcgnsidx = 0;
1068  for ( it = m_storedInfo->mcg_bndtypetoid.begin(); it != itend; ++it ){
1069  itVend = it->second.end();
1070  for ( itV = it->second.begin(); itV != itVend; ++itV ){
1071  bndcgnsidx++;
1072  ID = (*itV);
1073  m_storedInfo->mcg_bndidtoindex[ID] = nVolElements + bndcgnsidx;
1074  m_storedInfo->mcg_bndindextoid.push_back(ID);
1075  m_storedBC->mcg_bndpidtoindex[bndpatch->getCell(ID).getPID()].push_back(bndcgnsidx + nVolElements);
1076  }
1077  }
1078  }

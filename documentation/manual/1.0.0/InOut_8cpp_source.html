977 void
978 mimmo::PortOut::exec(){
979  if (m_objLink.size() > 0){
980  writeBuffer();
981  bitpit::IBinaryStream input(m_obuffer.data());
982  cleanBuffer();
983  for (int j=0; j<(int)m_objLink.size(); j++){
984  if (m_objLink[j] != NULL){
985  m_objLink[j]->setBufferIn(m_portLink[j], input);
986  m_objLink[j]->readBufferIn(m_portLink[j]);
987  m_objLink[j]->cleanBufferIn(m_portLink[j]);
988  }
989  }
990  }

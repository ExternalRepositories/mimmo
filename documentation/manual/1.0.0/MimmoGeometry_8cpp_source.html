608 bool
609 MimmoGeometry::read(){
610  if(!m_isInternal) return false;
611 
612  switch(FileType::_from_integral(m_rinfo.ftype)){
613 
614  //Import STL
615  case FileType::STL :
616  {
617  setGeometry(1);
618  string name;
619 
620  {
621  std::ifstream infile(m_rinfo.fdir+"/"+m_rinfo.fname+".stl");
622  bool check = infile.good();
623  name = m_rinfo.fdir+"/"+m_rinfo.fname+".stl";
624  if (!check){
625  infile.open(m_rinfo.fdir+m_rinfo.fname+".STL");
626  check = infile.good();
627  name = m_rinfo.fdir+"/"+m_rinfo.fname+".STL";
628  if (!check) return false;
629  }
630  }
631 
632  ifstream in(name);
633  string ss, sstype;
634  getline(in,ss);
635  stringstream ins;
636  ins << ss;
637  ins >> sstype;
638  bool binary = true;
639  if (sstype == "solid" || sstype == "SOLID") binary = false;
640  in.close();
641 
642  dynamic_cast<SurfUnstructured*>(getGeometry()->getPatch())->importSTL(name, binary);
643  getGeometry()->cleanGeometry();
644 
645  //count PID if multi-solid
646  auto & map = getGeometry()->getPIDTypeList();
647  for(auto & cell : getGeometry()->getCells() ){
648  map.insert(cell.getPID());
649  }
650  }
651  break;
652 
653  case FileType::STVTU :
654  //Import Triangulation Surface VTU
655  {
656  std::ifstream infile(m_rinfo.fdir+"/"+m_rinfo.fname+".vtu");
657  bool check = infile.good();

661  ivector2D Iconnectivity ;
662  shivector1D pids;

665  vtk.setGeomData( bitpit::VTKUnstructuredField::POINTS, Ipoints) ;
666  vtk.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, Iconnectivity) ;
667  vtk.addData("PID", pids);

672

674

677  sizeC = Iconnectivity.size();
678  sizeP = pids.size();

680  m_intgeo->getPatch()->reserveCells(sizeC);
681 
682  for(auto & vv : Ipoints) m_intgeo->addVertex(vv);
683 
684  int ccell = 0;
685  for(auto & cc : Iconnectivity) {
686  livector1D temp(cc.size());

691  }
692  if(sizeP == 0){
693  m_intgeo->addConnectedCell(temp, eltype);
694  }else{
695  m_intgeo->addConnectedCell(temp, eltype, pids[ccell]);
696  }
697  ccell++;
698  }
699

701  }
702  break;

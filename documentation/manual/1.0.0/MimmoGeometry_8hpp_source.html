223  template<class Type>
224  void writeValue (Type& value, std::ofstream& os){
225 
226  int offset = 5;
227  if (wformat == Long) offset = 13;
228 
229  std::stringstream manip;
230  manip << value;
231  std::string manips = manip.str();
232  std::string mantissa, expon;
233  int pos = manips.find("E");
234  if (pos < manips.size()){
235  mantissa = manips.substr(0,std::min(offset,pos));
236  expon = manips.substr(pos+1,manips.size());
237  manips = mantissa + expon;
238  }
239  pos = manips.find("e");
240  if (pos < manips.size()){
241  mantissa = manips.substr(0,std::min(offset,pos));
242  expon = manips.substr(pos+1,manips.size());
243  manips = mantissa + expon;
244  }
245  manips = manips.substr(0, offset+3);
246 
247  switch (wformat)
248  {
249  case Short:
250  {
251  os << std::setw(8) << manips;
252  break;
253  }
254  case Long:
255  {
256  os << std::setw(16) << manips;
257  break;
258  }
259  }
260  }
261 };

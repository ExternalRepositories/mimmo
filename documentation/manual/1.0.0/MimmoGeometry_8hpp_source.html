245  template<class Type>
246  void writeValue (Type& value, std::ofstream& os){
247 
248  int offset = 5;
249  if (wformat == Long) offset = 13;
250 
251  std::stringstream manip;
252  manip << value;
253  std::string manips = manip.str();
254  std::string mantissa, expon;
255  int pos = manips.find("E");
256  if (pos < manips.size()){
257  mantissa = manips.substr(0,std::min(offset,pos));
258  expon = manips.substr(pos+1,manips.size());
259  manips = mantissa + expon;
260  }
261  pos = manips.find("e");
262  if (pos < manips.size()){
263  mantissa = manips.substr(0,std::min(offset,pos));
264  expon = manips.substr(pos+1,manips.size());
265  manips = mantissa + expon;
266  }
267  manips = manips.substr(0, offset+3);
268 
269  switch (wformat)
270  {
271  case Short:
272  {
273  os << std::setw(8) << manips;
274  break;
275  }
276  case Long:
277  {
278  os << std::setw(16) << manips;
279  break;
280  }
281  }
282  }
283 };

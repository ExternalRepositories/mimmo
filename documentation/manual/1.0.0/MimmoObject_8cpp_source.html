83 MimmoObject::MimmoObject(int type, dvecarr3E & vertex, ivector2D * connectivity){
84  m_type = max(1,type);
85  m_internalPatch = true;
86  const int id = 0;
87  if (m_type == 2){
88  m_patch = new VolUnstructured(id, 3);
89  dynamic_cast<VolUnstructured*> (m_patch)->setExpert(true);
90  }else{
91  m_patch = new SurfUnstructured(id);
92  dynamic_cast<SurfUnstructured*> (m_patch)->setExpert(true);
93  }
94 
95  bitpit::ElementInfo::Type eltype = bitpit::ElementInfo::UNDEFINED;
96  int sizeVert, sizeCell;
97 
98  sizeVert = vertex.size();
99  m_patch->reserveVertices(sizeVert);
100 
101  for(auto & vv : vertex) addVertex(vv);
102 
103  if(connectivity == NULL) m_type = 3;
104  if (m_type != 3){
105  int sizeConn = (*connectivity)[0].size();
106  sizeCell = connectivity->size();
107 
108  switch(m_type){
109  case 1:
110  if(sizeConn == 3) eltype = bitpit::ElementInfo::TRIANGLE;
111  if(sizeConn == 4) eltype = bitpit::ElementInfo::QUAD;
112  break;
113  case 2:
114  if(sizeConn == 4) eltype = bitpit::ElementInfo::TETRA;
115  if(sizeConn == 8) eltype = bitpit::ElementInfo::HEXAHEDRON;
116  break;
117  case 4:
118  if(sizeConn == 2) eltype = bitpit::ElementInfo::LINE;
119  break;
120  default:
121  // never been reached
122  break;
123  }
124 
125  if(eltype != bitpit::ElementInfo::UNDEFINED){
126 
127  m_patch->reserveCells(sizeCell);
128 
129  for(auto & cc : *connectivity){
130 
131  livector1D temp(cc.size());
132  int counter=0;
133 
134  for(auto && val : cc) {
135  temp[counter] = val;
136  ++counter;
137  }
138 
139  addConnectedCell(temp, eltype);
140  }
141 
142  m_pidsType.insert(0);
143 
144  }else{
145  (*m_log)<<"Not supported connectivity found for MimmoObject"<<std::endl;
146  (*m_log)<<"Proceeding as Point Cloud geometry"<<std::endl;
147  }
148  }
149  m_bvTree.setPatch(m_patch);
150  m_bvTreeBuilt = false;
151  m_kdTreeBuilt = false;
152  m_bvTreeSupported = (m_type != 3);
153  m_bvTreeSync = false;
154  m_kdTreeSync = false;
155  m_AdjBuilt = false;
156  m_log = &bitpit::log::cout(MIMMO_LOG_FILE);
157 };

736 bool
737 MimmoObject::setCells(const bitpit::PiercedVector<Cell> & cells){
738 
739  if (cells.empty() || !m_bvTreeSupported) return false;
740 
741  m_mapCell.clear();
742  m_mapCellInv.clear();
743  m_pidsType.clear();
744 
745  getPatch()->resetCells();
746 
747  int sizeCell = cells.size();
748  getPatch()->reserveCells(sizeCell);
749 
750  long idc;
751  int nVert;
752  short pid;
753  bitpit::ElementInfo::Type eltype;
754  bool checkTot = true;
755  livector1D connectivity;
756 
757  for (const auto & cell : cells){
758  // get ID
759  idc = cell.getId();
760  //check on element type
761  eltype = cell.getType();
762  //check info PID
763  pid = (short)cell.getPID();
764  nVert = cell.getVertexCount();
765  connectivity.resize(nVert);
766  auto conn = cell.getConnect();
767  for(int i=0; i<nVert; ++i){
768  connectivity[i] = conn[i];
769  }
770  checkTot = checkTot && addConnectedCell(connectivity, eltype, pid, idc);
771  }
772 
773  return checkTot;

793 
794  if (isEmpty() || conn.empty() || !m_bvTreeSupported) return false;
795  if(idtag != bitpit::Cell::NULL_ID && m_patch->getCells().exists(idtag)) return false;
796

799 
800  bitpit::PatchKernel::CellIterator it;

803  conn_dum.resize(sizeElement, 0);
804 
805  long checkedID;
806  if(idtag == bitpit::Cell::NULL_ID){
807  it = m_patch->addCell(type, true, conn_dum);
808  checkedID = it->getId();
809  }else{
810  it = m_patch->addCell(type, true,conn_dum, idtag);
811  checkedID = idtag;
812  }
813 
814  m_pidsType.insert(0);
815 
816  //create inverse map of cells
817  m_mapCell.push_back(checkedID);
818  m_mapCellInv[checkedID] = m_mapCell.size()-1;
819  m_bvTreeBuilt = false;
820  m_bvTreeSync = false;

822  return true;

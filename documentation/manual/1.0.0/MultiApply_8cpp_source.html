120 void
121 MultiApply::execute(){
122 
123 
124  if (m_input.empty()) return;
125 
126 
127  for(auto val: m_input){
128 
129 
130 
131  dvecarr3E vertex = val.first->getVertexCoords();
132  long nv = val.first->getNVertex();
133  nv = long(std::min(int(nv), int(val.second->size())));
134 
135  livector1D & idmap = val.first->getMapData();
136  for (long i=0; i<nv; i++){
137 
138  vertex[i] += (*(val.second))[i];
139  val.first->modifyVertex(vertex[i], idmap[i]);
140  }
141 
142  }
143  return;
144 };

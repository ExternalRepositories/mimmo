221 void
222 OBBox::execute(){
223  if(m_geometry == NULL) return;
224 
225  dmatrix33E covariance;
226  darray3E spectrum;
227  darray3E etaPoint;
228 
229  evaluateCovarianceMatrix(covariance, etaPoint);
230 
231  m_axes = eigenVectors(covariance, spectrum);
232 
233  adjustBasis(m_axes, spectrum);
234 
235  darray3E pmin, pmax;
236  pmin.fill(1.e18);
237  pmax.fill(-1.e18);
238  double val;
239 
240  dmatrix33E trasp = bitpit::linearalgebra::transpose(m_axes);
241 
242  for(auto & vert: getGeometry()->getVertices()){
243  darray3E coord = vert.getCoords();
244  for(int i=0;i<3; ++i){
245  val = dotProduct(coord, m_axes[i]);
246  pmin[i] = std::fmin(pmin[i], val);
247  pmax[i] = std::fmax(pmax[i], val);
248  }
249  }
250 
251  m_span = pmax - pmin;
252  //check if one of the span goes to 0;
253  double avg_span = 0.0;
254  for(auto & val: m_span) avg_span+=val;
255  avg_span /= 3.0;
256 
257  for(auto &val : m_span) {
258  val = std::fmax(val, 1.E-04*avg_span);
259  }
260 
261  darray3E originLoc = 0.5*(pmin+pmax);
262  for(int i=0; i<3; ++i){
263  m_origin[i] = dotProduct(originLoc, trasp[i]);
264  }
265 
266  double volOBB = m_span[0]*m_span[1]*m_span[2];
267 
268 
269  //check the axis aligned bounding box if volume is lesser then obb
270  //take it instead of the oriented.
271 
272  pmin.fill(1.e18);
273  pmax.fill(-1.e18);
274  for(int i=0; i<3; ++i){
275  trasp[i].fill(0.0);
276  trasp[i][i] = 1.0;
277  }
278 
279  for(auto & vert: getGeometry()->getVertices()){
280  darray3E coord = vert.getCoords();
281  for(int i=0;i<3; ++i){
282  val = dotProduct(coord, trasp[i]);
283  pmin[i] = std::fmin(pmin[i], val);
284  pmax[i] = std::fmax(pmax[i], val);
285  }
286  }
287 
288  darray3E span2 = pmax - pmin;
289  darray3E orig = 0.5*(pmin+pmax);
290  //check if one of the span goes to 0;
291  avg_span = 0.0;
292  for(auto & val: span2) avg_span+=val;
293  avg_span /= 3.0;
294 
295  for(auto &val : span2) {
296  val = std::fmax(val, 1.E-04*avg_span);
297  }
298 
299  double volAAA =span2[0]*span2[1]*span2[2];
300 
301 
302  if(volAAA <= volOBB){
303  for(int i=0; i<3; ++i) m_axes[i] = trasp[i];
304  m_span = span2;
305  m_origin = orig;
306  }

352 dmatrix33E
353 OBBox::eigenVectors( dmatrix33E & matrix, darray3E & eigenvalues){
354 
355  dmatrix33E result;
356  double * a = new double [9];
357  double * u = new double [9];
358  double * vt = new double [9];
359  double * s = new double[3];
360  double * superb = new double[3];
361 
362  int k=0;
363  for(int i=0; i<3; i++){
364  for(int j=0; j<3; j++){
365  //transpose
366  a[k] = matrix[j][i];
367  k++;
368  }
369  }
370 
371  LAPACKE_dgesvd( LAPACK_COL_MAJOR, 'N','S', 3, 3, a, 3, s, u, 3, vt, 3, superb);
372 
373  //solution norm
374  for (int i=0; i<9; i++){
375  result[i/3][i%3] = vt[i];
376  }
377  for(int i=0; i<2; ++i) {
378  result[i] /= norm2(result[i]);
379  eigenvalues[i] = s[i];
380  }
381  result[2] = crossProduct(result[0],result[1]);
382  eigenvalues[2] = s[2];
383 
384  delete [] a; a = NULL;
385  delete [] u; u = NULL;
386  delete [] vt; vt = NULL;
387  delete [] s; s = NULL;
388  delete [] superb; superb = NULL;
389 
390  return result;
391 }

400 void
401 OBBox::evaluateCovarianceMatrix( dmatrix33E & covariance, darray3E & eta){
402 
403  if(getGeometry()==NULL) {return;}
404 
405  eta.fill(0.0);
406  for(auto & val:covariance) val.fill(0.0);
407  int size, counter;
408  dvector1D area;
409  darray3E temp;
410  double areaTot = 0.0;
411  double maxVal= 1.e-18;
412 
413  if(getGeometry()->getType() == 3){
414  size = getGeometry()->getNVertex();
415  //evaluate eta;
416  for(auto & vert: getGeometry()->getVertices()){
417  eta += vert.getCoords();
418  }
419  eta /= (double)size;
420 
421  //evaluate covariance
422  for(auto & vert: getGeometry()->getVertices()){
423  temp = vert.getCoords() - eta;
424 
425  for(int j=0; j<3; ++j){
426  for(int k=j; k<3; ++k){
427  covariance[j][k] += temp[j]*temp[k];
428  }
429  }
430 
431  for(int j=0; j<3; ++j){
432  for(int k=j; k<3; ++k){
433  covariance[j][k] /= (double) size;
434  maxVal = std::fmax(maxVal, std::abs(covariance[j][k]));
435  }
436  }
437 
438  }
439 
440  }else{
441  size = getGeometry()->getNCells();
442 
443  bitpit::SurfUnstructured * tri = static_cast<bitpit::SurfUnstructured * >(getGeometry()->getPatch());
444  area.resize(size);
445  counter = 0;
446  for(auto &cell : tri->getCells()){
447  area[counter] = tri->evalCellArea(cell.getId());
448  areaTot +=area[counter];
449  ++counter;
450  }
451  for(auto & val: area) val /= areaTot;
452 
453  //evaluate eta;
454  counter = 0;
455  darray3E pp;
456  long vCount;
457  for(auto & cell: tri->getCells()){
458  vCount = cell.getVertexCount();
459  pp.fill(0.0);
460  for(int kk=0; kk<vCount; ++kk){
461  pp += tri->getVertexCoords(cell.getVertex(kk));
462  }
463 
464  eta += pp*area[counter];
465  ++counter;
466  }
467 
468  eta /= (3.0);
469 
470 
471  counter = 0;
472  dvecarr3E p2;
473  for(auto & cell: tri->getCells()){
474  vCount = cell.getVertexCount();
475  p2.resize(vCount);
476  for(int kk=0; kk<vCount; ++kk){
477  p2[kk] = tri->getVertexCoords(cell.getVertex(kk)) - eta;
478  }
479 
480  dmatrix33E tempM = evalCovTriangle(p2);
481  for(int j=0; j<3; ++j){
482  for(int k=j; k<3; ++k){
483  covariance[j][k] += area[counter]*tempM[j][k];
484  }
485  }
486 
487  ++counter;
488  }
489 
490 
491  for(int j=0; j<3; ++j){
492  for(int k=j; k<3; ++k){
493  covariance[j][k] /= (12.0*size);
494  maxVal = std::fmax(maxVal, std::abs(covariance[j][k]));
495  }
496  }
497  }
498 
499  for(int j=0; j<3; ++j){
500  for(int k=j; k<3; ++k){
501  covariance[j][k] /= maxVal;
502  }
503  }
504 
505  covariance[1][0] = covariance[0][1];
506  covariance[2][0] = covariance[0][2];
507  covariance[2][1] = covariance[1][2];
508 
509 };

568 void
569 OBBox::adjustBasis(dmatrix33E & eigVec, darray3E & eigVal){
570 
571  darray3E diff;
572  double tol = 1.0E-3;
573 
574  diff[0] = std::abs(eigVal[1]-eigVal[0]);
575  diff[1] = std::abs(eigVal[2]-eigVal[1]);
576  diff[2] = std::abs(eigVal[0]-eigVal[2]);
577 
578  if(diff[0]>tol && diff[1]>tol && diff[2]>tol) return;
579  if(diff[0]<=tol && diff[1]<=tol && diff[2]<= tol){
580  int counter=0;
581  for(auto & val : eigVec){
582  val.fill(0.0);
583  val[counter] = 1.0;
584  ++counter;
585  }
586  return;
587  }
588 
589  int guess = 0, third =1, stable=2;
590 
591  if(diff[1] <= tol) {
592  guess = 1;
593  third = 2;
594  stable= 0;
595  }
596 
597  if(diff[2] <= tol) {
598  guess = 2;
599  third = 0;
600  stable= 1;
601  }
602 
603  dmatrix33E axes, trasp;
604  int counter=0;
605  for(auto & val: eigVec){
606  axes[counter] = val;
607  ++counter;
608  }
609 
610  darray3E guessVec = axes[guess];
611  darray3E refVec = axes[stable];
612 
613  int nstage = 15;
614  int niterations = 8;
615 
616  double distance = M_PI/(2.0*(nstage));
617  double volume, theta, val;
618  std::map<double, double> mapval;
619  darray3E pmin, pmax, temp;
620 
621  for(int i=0; i<nstage; ++i){
622 
623  theta = i*distance;
624 
625  axes[guess] = guessVec*std::cos(theta) + std::sin(theta)*crossProduct(refVec, guessVec);
626  axes[third] = crossProduct(axes[stable],axes[guess]);
627 
628  pmin.fill(1.e18);
629  pmax.fill(-1.e18);
630 
631  //trasp = bitpit::linearalgebra::transpose(axes);
632 
633  for(auto & vert: getGeometry()->getVertices()){
634  darray3E coord = vert.getCoords();
635  for(int i=0;i<3; ++i){
636  val = dotProduct(coord, axes[i]); //trasp[i]);
637  pmin[i] = std::fmin(pmin[i], val);
638  pmax[i] = std::fmax(pmax[i], val);
639  }
640  }
641 
642  temp = pmax - pmin;
643  volume = temp[0]*temp[1]*temp[2];
644 
645  mapval[volume] = theta;
646  }
647 
648  int it =0;
649  while (it < niterations){
650 
651 
652  distance /= 2.0;
653 
654  double thetadx = mapval.begin()->second + distance;
655  if(thetadx >= M_PI/2.0) thetadx += -1.0*M_PI/2.0;
656 
657  double thetasx = mapval.begin()->second - distance;
658  if(thetasx <= 0.0) thetasx += M_PI/2.0;
659 
660  //evaluate on the right
661  axes[guess] = guessVec*std::cos(thetadx) + std::sin(thetadx)*crossProduct(refVec, guessVec);
662  axes[third] = crossProduct(axes[stable],axes[guess]);
663 
664 
665  //trasp = bitpit::linearalgebra::transpose(axes);
666  pmin.fill(1.e18);
667  pmax.fill(-1.e18);
668  for(auto & vert: getGeometry()->getVertices()){
669  darray3E coord = vert.getCoords();
670  for(int i=0;i<3; ++i){
671  val = dotProduct(coord,axes[i]); // trasp[i]);
672  pmin[i] = std::fmin(pmin[i], val);
673  pmax[i] = std::fmax(pmax[i], val);
674  }
675  }
676 
677  temp = pmax - pmin;
678  volume = temp[0]*temp[1]*temp[2];
679 
680  mapval[volume] = thetadx;
681 
682  //evaluate on the left
683  axes[guess] = guessVec*std::cos(thetasx) + std::sin(thetasx)*crossProduct(refVec, guessVec);
684  axes[third] = crossProduct(axes[stable],axes[guess]);
685 
686  trasp = bitpit::linearalgebra::transpose(axes);
687  pmin.fill(1.e18);
688  pmax.fill(-1.e18);
689  for(auto & vert: getGeometry()->getVertices()){
690  darray3E coord = vert.getCoords();
691  for(int i=0;i<3; ++i){
692  val = dotProduct(coord,axes[i]); // trasp[i]);
693  pmin[i] = std::fmin(pmin[i], val);
694  pmax[i] = std::fmax(pmax[i], val);
695  }
696  }
697 
698  temp = pmax - pmin;
699  volume = temp[0]*temp[1]*temp[2];
700 
701  mapval[volume] = thetasx;
702  ++it;
703  }
704 
705  theta = mapval.begin()->second;
706  eigVec[guess] = guessVec*std::cos(theta) + std::sin(theta)*crossProduct(refVec, guessVec);
707  eigVec[third] = crossProduct(eigVec[stable],eigVec[guess]);
708 
709 };

190 void
191 RBFBox::plot(std::string directory, std::string filename,int counter, bool binary){
192 
193 
194  dvecarr3E activeP(8);
195 
196  activeP[0] = - 0.5 * m_span;
197  activeP[6] = 0.5 * m_span;
198 
199  activeP[1] = activeP[0]; activeP[1][0] += m_span[0];
200  activeP[3] = activeP[0]; activeP[3][1] += m_span[1];
201  activeP[2] = activeP[6]; activeP[2][2] += -1.0*m_span[2];
202 
203  activeP[7] = activeP[6]; activeP[7][0] += -1.0*m_span[0];
204  activeP[5] = activeP[6]; activeP[5][1] += -1.0*m_span[1];
205  activeP[4] = activeP[0]; activeP[4][2] += m_span[2];
206 
207  darray3E temp;
208  dmatrix33E trasp = bitpit::linearalgebra::transpose(m_axes);
209  for(auto &val : activeP){
210 
211 
212  for(int i=0; i<3; ++i){
213  temp[i] = dotProduct(val, trasp[i]);
214  }
215  val = temp + m_origin;
216  }
217 
218  ivector2D activeConn(1);
219  for(int i=0; i<8; ++i) activeConn[0].push_back(i);
220 
221  bitpit::VTKFormat codex = bitpit::VTKFormat::ASCII;
222  if(binary){codex=bitpit::VTKFormat::APPENDED;}
223  bitpit::VTKElementType elDM = bitpit::VTKElementType::HEXAHEDRON;
224  bitpit::VTKUnstructuredGrid vtk(directory, filename, elDM);
225  vtk.setGeomData( bitpit::VTKUnstructuredField::POINTS, activeP) ;
226  vtk.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, activeConn) ;
227  vtk.setDimensions(1, 8);
228  vtk.setCodex(codex);
229  if(counter>=0){vtk.setCounter(counter);}
230 
231  vtk.write();
232 };

239 void
240 RBFBox::execute(){
241 
242  int np = m_nodes.size();
243 
244  darray3E min, max;
245  min.fill(1.0e+18);
246  max.fill(-1.0e+18);
247 
248  for(int i=0; i<np; i++){
249  for(int j=0; i<3; j++){
250  if (m_nodes[i][j] - m_suppR*1.01 < min[j]) min[j] = m_nodes[i][j];
251  if (m_nodes[i][j] + m_suppR*1.01 > max[j]) max[j] = m_nodes[i][j];
252  }
253  }
254 
255  m_span = max - min;
256  m_origin = min + m_span * 0.5;
257 
258 };

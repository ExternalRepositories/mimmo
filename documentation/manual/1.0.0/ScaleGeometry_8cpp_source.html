157 void
158 ScaleGeometry::execute(){
159 
160  if (getGeometry() == NULL) return;
161 
162  int nV = m_geometry->getNVertex();
163  m_displ.resize(nV);
164  m_filter.resize(nV, 1.0);
165 
166 
167  long ID;
168  int idx;
169  liimap mapID = m_geometry->getMapDataInv();
170 
171  //computing centroid
172  darray3E center = m_origin;
173  if (m_meanP){
174  center.fill(0.0);
175  for (auto vertex : m_geometry->getVertices()){
176  ID = vertex.getId();
177  idx = mapID[ID];
178  center += vertex.getCoords() / double(nV);
179  }
180  }
181  for (auto vertex : m_geometry->getVertices()){
182  ID = vertex.getId();
183  idx = mapID[ID];
184 
185  darray3E coords = vertex.getCoords();
186  m_displ[idx] = ( m_scaling*(coords - center) + center ) * m_filter[idx] - coords;
187  }
188 };

193 void
194 ScaleGeometry::apply(){
195 
196  if (getGeometry() == NULL) return;
197  dvecarr3E vertex = getGeometry()->getVertexCoords();
198  long nv = getGeometry()->getNVertex();
199  nv = long(std::min(int(nv), int(m_displ.size())));
200  livector1D & idmap = getGeometry()->getMapData();
201  for (long i=0; i<nv; i++){
202  vertex[i] += m_displ[i];
203  getGeometry()->modifyVertex(vertex[i], idmap[i]);
204  }
205 
206 }

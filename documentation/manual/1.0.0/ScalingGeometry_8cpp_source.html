116 void
117 ScalingGeometry::execute(){
118 
119  if (getGeometry() == NULL) return;
120 
121  int nV = m_geometry->getNVertex();
122  m_displ.resize(nV);
123  m_filter.resize(nV, 1.0);
124 
125 
126  long ID;
127  int idx;
128  liimap mapID = m_geometry->getMapDataInv();
129 
130  //computing centroid
131  darray3E center;
132  center.fill(0.0);
133  for (auto vertex : m_geometry->getVertices()){
134  ID = vertex.getId();
135  idx = mapID[ID];
136  center += vertex.getCoords() / double(nV);
137  }
138 
139  for (auto vertex : m_geometry->getVertices()){
140  ID = vertex.getId();
141  idx = mapID[ID];
142 
143  darray3E coords = vertex.getCoords();
144  m_displ[idx] = ( m_scaling*(coords - center) + center ) * m_filter[idx] - coords;
145  }
146 };

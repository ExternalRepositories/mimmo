247 void
248 StitchGeometry::execute(){
249  if(m_extgeo.empty()) return;
250 
251  std::unique_ptr<MimmoObject> dum(new MimmoObject(m_topo));
252  long nCells = 0;
253  long nVerts = 0;
254 
255 
256  for(auto &obj:m_extgeo){
257  nCells += obj.first->getNCells();
258  nVerts += obj.first->getNVertex();
259  }
260 
261  //reserving memory
262  dum->getPatch()->reserveVertices(nVerts);
263  dum->getPatch()->reserveCells(nCells);
264 
265  //clean maps
266  m_mapCellDivision.clear();
267  m_mapVertDivision.clear();
268 
269  long cV = 0, cC = 0;
270 
271  //start filling your stitched object.
272  //divion maps will be filled coherently
273 
274  {
275  //optional vars;
276  long vId, cId;
277  short PID;
278  bitpit::ElementInfo::Type eltype;
279 
280  for(auto &obj : m_extgeo){
281 
282  //start extracting/reversing vertices of the current obj
283  std::unordered_map<long,long> mapVloc;
284 
285  for(auto & vv : obj.first->getVertices()){
286  vId = vv.getId();
287  dum->addVertex(obj.first->getVertexCoords(vId), cV);
288  //update map;
289  m_mapVertDivision[cV] = std::make_pair(obj.second, vId);
290  mapVloc[vId] = cV;
291  cV++;
292  }
293 
294  //start extracting/reversing cells of the current obj
295  for(auto & cc : obj.first->getCells()){
296  cId = cc.getId();
297  PID = cc.getPID();
298  eltype = cc.getType();
299  //get the local connectivity and update with new vertex numbering;
300  livector1D conn = obj.first->getCellConnectivity(cId);
301  for(auto && ee: conn) ee = mapVloc[ee];
302 
303  dum->addConnectedCell(conn, eltype, PID, cC);
304  //update map;
305  m_mapCellDivision[cC] = std::make_pair(obj.second, cId);
306  cC++;
307  }
308  }
309  }//scope for optional vars;
310 
311  if(m_buildBvTree) dum->buildBvTree();
312  if(m_buildKdTree) dum->buildKdTree();
313 
314  m_patch = std::move(dum);
315 }

323 
324  BITPIT_UNUSED(name);

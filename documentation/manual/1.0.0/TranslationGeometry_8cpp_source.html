147 TranslationGeometry::execute(){
148 
149  if (getGeometry() == NULL) return;
150 
151  int nV = m_geometry->getNVertex();
152  m_displ.resize(nV);
153  m_filter.resize(nV, 1.0);
154 
155  long ID;
156  int idx;
157  liimap mapID = m_geometry->getMapDataInv();
158 
159  for (auto vertex : m_geometry->getVertices()){
160  ID = vertex.getId();
161  idx = mapID[ID];
162 
163 
164  m_displ[idx] = m_alpha*m_direction*m_filter[idx];

172 void
173 TranslationGeometry::apply(){
174 
175  if (getGeometry() == NULL) return;
176  dvecarr3E vertex = getGeometry()->getVertexCoords();
177  long nv = getGeometry()->getNVertex();
178  nv = long(std::min(int(nv), int(m_displ.size())));
179  livector1D & idmap = getGeometry()->getMapData();
180  for (long i=0; i<nv; i++){
181  vertex[i] += m_displ[i];
182  getGeometry()->modifyVertex(vertex[i], idmap[i]);
183  }

30 void example00001() {
31 
32  /* Create IO_CGNS object to import input file. */
33  IOCGNS * cgnsI = new IOCGNS();
34  cgnsI->setRead(true);
35  cgnsI->setReadDir("geodata");
36  cgnsI->setReadFilename("grid");
37 
38  /* Create IO_CGNS object to export output file. */
39  IOCGNS * cgnsO = new IOCGNS();
40  cgnsO->setRead(false);
41  cgnsO->setWriteDir(".");
42  cgnsO->setWriteFilename("iocgns_output_00001");
43 
44  /* Create CGNS PID excrator object to test input file.
45  * Extraction of PID = 1 (Wing wall boundary in input file).
46  */
47  CGNSPidExtractor * cgnsExtr = new CGNSPidExtractor();
48  cgnsExtr->setPID({1});
49  cgnsExtr->setForcedToTriangulate(false);
50  cgnsExtr->setPlotInExecution(true);
51 
52  /* Create PINs. */
53  addPin(cgnsI,cgnsO, PortType::M_GEOM, PortType::M_GEOM);
54  addPin(cgnsI,cgnsExtr, PortType::M_GEOM2, PortType::M_GEOM);
55  addPin(cgnsI,cgnsO, PortType::M_GEOM2, PortType::M_GEOM2);
56  addPin(cgnsI,cgnsO, PortType::M_BCCGNS, PortType::M_BCCGNS);
57 
58  /* Create and execute chain. */
59  Chain ch0;
60  ch0.addObject(cgnsI);
61  ch0.addObject(cgnsExtr);
62  ch0.addObject(cgnsO);
63  ch0.exec(true);
64 
65  /* Destroy objects. */
66  delete cgnsI;
67  delete cgnsExtr;
68  delete cgnsO;
69  cgnsI = NULL;
70  cgnsExtr = NULL;
71  cgnsO = NULL;
72 
73  return;
74 }

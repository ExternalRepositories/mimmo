53 
54  /* Create PINs. */
55  addPin(iovtk,iovtk2, PortType::M_GEOM, PortType::M_GEOM);
56  addPin(iovtk,iovtk2, PortType::M_SCALARFIELD, PortType::M_SCALARFIELD);
57  addPin(iovtk,iovtk2, PortType::M_POLYDATA_, PortType::M_POLYDATA_);
58  addPin(iovtk,iovtk3, PortType::M_GEOM, PortType::M_GEOM);
59  addPin(iovtk,iovtk3, PortType::M_SCALARFIELD, PortType::M_SCALARFIELD);
60 
61  /* Create and execute chain. */
62  Chain ch0;
63  ch0.addObject(iovtk);
64  ch0.addObject(iovtk2);
65  ch0.addObject(iovtk3);
66  ch0.exec(true);

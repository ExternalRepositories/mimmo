464 void
465 BaseManipulation::exec(){
466 
467  map<int, vector<PortIn*> > families;
468  map<int, vector<PortID> > familiesID;
469  for (map<PortID, PortIn*>::iterator i=m_portIn.begin(); i!=m_portIn.end(); i++){
470  if (i->second->isMandatory()){
471  families[i->second->getFamily()].push_back(i->second);
472  familiesID[i->second->getFamily()].push_back(i->first);
473  }
474  }
475  map<int, vector<PortID> >::iterator itID = familiesID.begin();
476  for (map<int, vector<PortIn*> >::iterator i=families.begin(); i!=families.end(); i++){
477  if (i->first == 0){
478  int count = 0;
479  for (auto ip : i->second){
480  std::vector<BaseManipulation*> linked = ip->getLink();
481  if (linked.size() == 0){
482  (*m_log) << "error: " << m_name << " mandatory port " << itID->second[count] << " not linked -> exit! " << std::endl;
483  throw std::runtime_error (m_name + " mandatory port " + std::to_string(itID->second[count]) + " not linked");
484  }
485  count++;
486  }
487  }
488  else{
489  bool check = false;
490  for (auto ip : i->second){
491  std::vector<BaseManipulation*> linked = ip->getLink();
492  check = check || linked.size();
493  }
494  if (!check){
495  (*m_log) << "error: " << m_name << " none of mandatory ports " << itID->second << " linked -> exit! " << std::endl;
496  std::string ports;
497  for (auto p : itID->second)
498  ports += " " + std::to_string(p) + " ";
499  throw std::runtime_error (m_name + " none of mandatory ports " + ports + " linked");
500  }
501  }
502  itID++;
503  }
504 
505  if (m_active) execute();
506 
507  for (map<PortID, PortOut*>::iterator i=m_portOut.begin(); i!=m_portOut.end(); i++){
508  std::vector<BaseManipulation*> linked = i->second->getLink();
509  if (linked.size() > 0){
510  i->second->exec();
511  }
512  }
513 
514  if(isPlotInExecution()) plotOptionalResults();
515  if(isApply()) apply();
516 }

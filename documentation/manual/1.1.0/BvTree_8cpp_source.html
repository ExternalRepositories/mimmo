214 void BvTree::buildTree()
215 {
216  if (m_patch == NULL) return;
217 
218  if (m_nelements == 0 || m_nodes.size() == 0) setup();
219  if (m_nelements == 0) return;
220 
221  int iel = 0;
222  long id;
223 
224  //fill ids and centroid
225  for ( auto & cell : m_patch->getCells() )
226  {
227  id = cell.getId();
228  m_elements[iel].m_label = id;
229  m_elements[iel].m_centroid = m_patch->evalCellCentroid(id);
230  iel++;
231  }
232 
233  //node root
234  BvNode node;
235  node.m_element[0] = 0;
236  node.m_element[1] = m_nelements;
237  node.m_nrange = m_nelements;
238  m_nodes[m_nnodes] = node;
239  if ( m_nelements == 1 )
240  {
241  m_nodes[m_nnodes].m_leaf = true;
242  computeBoundingBox(0);
243  m_nleaf++;
244  m_nnodes++;
245  }
246  else
247  {
248  m_nnodes++;
249  fillTree(0);
250  }
251  decreaseStack();
252 }

258 void BvTree::fillTree(int iparent)
259 {
260 
261 
262  //compute Bounding Box of node
263  computeBoundingBox(iparent);
264 
265 
266  if ( m_nodes[iparent].m_nrange > m_maxsize )
267  {
268 
269  //split parent by mean of centroid in node
270  darray3E meanC = computeMeanPoint(m_nodes[iparent].m_element[0], m_nodes[iparent].m_element[1]);
271 
272  //split by plane normal to maximum dimension of the centroid box
273  darray3E minPC{{m_MAXLIMIT, m_MAXLIMIT, m_MAXLIMIT}};
274  darray3E maxPC{{-m_MAXLIMIT, -m_MAXLIMIT, -m_MAXLIMIT}};
275  int ics = m_nodes[iparent].m_element[0], ice = m_nodes[iparent].m_element[1];
276  for (int ic=ics; ic<ice; ++ic )
277  {
278  for (int j=0; j<m_dim; ++j )
279  {
280  minPC[j] = std::min(minPC[j], m_elements[ic].m_centroid[j]);
281  maxPC[j] = std::max(maxPC[j], m_elements[ic].m_centroid[j]);
282  }
283  }
284 
285  int dir = 0;
286  double maxRange = maxPC[dir] - minPC[dir];
287  for ( int idim=1; idim<m_dim; ++idim )
288  {
289  if ((maxPC[idim] - minPC[idim]) > maxRange)
290  {
291  dir = idim;
292  maxRange = maxPC[dir] - minPC[dir];
293  }
294  }
295 
296  //Sort elements by coordinate dir.
297  std::vector<BvElement>::iterator itstart = m_elements.begin()+m_nodes[iparent].m_element[0];
298  std::vector<BvElement>::iterator itend = m_elements.begin()+(m_nodes[iparent].m_element[1]);
299 
300  //Old algorithm
301  // sort( itstart, itend, compareElements(dir) );
302  // //find first element with coords[dir] > of meanC[dir]
303  // int firstRight = findFirstGreater(iparent, meanC, dir);
304 
305 
306  int firstRight = pseudoSort( itstart, itend, meanC, dir );
307 
308  //insert lchild
309  if ( firstRight > m_nodes[iparent].m_element[0] )
310  {
311  m_nodes[iparent].m_lchild = m_nnodes;
312  if ( (int)m_nodes.size() <= m_nnodes+1) increaseStack();
313  BvNode node;
314  node.m_element[0] = m_nodes[iparent].m_element[0];
315  node.m_element[1] = firstRight;
316  node.m_nrange = node.m_element[1] - node.m_element[0];
317  m_nodes[m_nnodes] = node;
318  if ( node.m_nrange == 1 )
319  {
320  m_nodes[m_nnodes].m_leaf = true;
321  computeBoundingBox(m_nnodes);
322  m_nnodes++;
323  m_nleaf++;
324  }
325  else
326  {
327  m_nnodes++;
328  fillTree(m_nnodes-1);
329  }
330  }
331 
332  //insert rchild
333  if ( firstRight < m_nodes[iparent].m_element[1] )
334  {
335  m_nodes[iparent].m_rchild = m_nnodes;
336  if ( (int)m_nodes.size() <= m_nnodes+1) increaseStack();
337  BvNode node;
338  node.m_element[0] = firstRight;
339  node.m_element[1] = m_nodes[iparent].m_element[1];
340  node.m_nrange = node.m_element[1] - node.m_element[0];
341  m_nodes[m_nnodes] = node;
342  if (node.m_nrange == 1){
343  m_nodes[m_nnodes].m_leaf = true;
344  computeBoundingBox(m_nnodes);
345  m_nnodes++;
346  m_nleaf++;
347  }
348  else{
349  m_nnodes++;
350  fillTree(m_nnodes-1);
351  }
352  }
353 
354  }
355  else
356  {
357  //insert this node as leaf
358  m_nodes[iparent].m_leaf = true;
359  m_nleaf++;
360 
361  }
362

373 {

394 int BvTree::pseudoSort(std::vector<BvElement>::iterator itstart,
395  std::vector<BvElement>::iterator itend, darray3E meanC, int dir){
396 
397  double thres = meanC[dir];
398 
399  itend--;
400  while(itstart != itend){
401 
402  while (itstart->m_centroid[dir] <= thres && itstart != itend){
403  itstart++;
404  }
405  while (itend->m_centroid[dir] > thres && itend != itstart){
406  itend--;
407  }
408 
409  if (itstart != itend){
410  std::iter_swap(itstart, itend);
411  }
412  }
413 
414  return (std::max(0, int(distance(m_elements.begin(), itend))));
415 }

428 int BvTree::findFirstGreater(int inode, darray3E meanC, int dir)
429 {
430  double mean = meanC[dir];
431  int guess = (m_nodes[inode].m_element[1] - m_nodes[inode].m_element[0])/2;
432  int step = guess;
433 
434  while( step > 0 )
435  {
436  step /= 2;
437  guess += step * int(sign(mean - m_elements[guess].m_centroid[dir]));
438  guess = std::max(m_nodes[inode].m_element[0], guess);
439  guess = std::min(m_nodes[inode].m_element[1]-1, guess);
440  }
441 
442  while ( mean < m_elements[guess].m_centroid[dir] && guess > m_nodes[inode].m_element[0] )
443  {
444  guess--;
445  }
446  while ( mean > m_elements[guess].m_centroid[dir] && guess < m_nodes[inode].m_element[1]-1 )
447  {
448  guess++;
449  }
450  return guess;
451 }

462 void BvTree::computeBoundingBox(int inode)
463 {
464  BvNode &node(m_nodes[inode]);
465 
466  int is = node.m_element[0];
467  int ie = node.m_element[1];
468 
469  int nV;
470 
471  node.m_minPoint = {{m_MAXLIMIT, m_MAXLIMIT, m_MAXLIMIT}};
472  node.m_maxPoint = {{-m_MAXLIMIT, -m_MAXLIMIT, -m_MAXLIMIT}};
473 
474  bitpit::Cell cell;
475  darray3E coords;
476 
477  for (int i=is; i<ie; i++ )
478  {
479  cell = m_patch->getCell(m_elements[i].m_label);
480  nV = cell.getVertexCount();
481  for (int iV=0; iV<nV; ++iV )
482  {
483  coords = m_patch->getVertexCoords(cell.getVertex(iV));
484  for (int j=0; j<m_dim; ++j )
485  {
486  node.m_minPoint[j] = std::min(node.m_minPoint[j], coords[j]);
487  node.m_maxPoint[j] = std::max(node.m_maxPoint[j], coords[j]);
488  }
489  }
490  }
491  node.m_minPoint -= m_tol;
492  node.m_maxPoint += m_tol;

527 bool BvTree::SphereBoundingBox(darray3E *P_, BvNode *node_, double r)
528 {
529 
530  darray3E closestP;
531  for ( int i=0; i<m_dim; i++ )
532  {
533  closestP[i] = std::min(std::max((*P_)[i], node_->m_minPoint[i]), node_->m_maxPoint[i]);
534  }
535  double dist = dotProduct(closestP-(*P_), closestP-(*P_));
536 
537  return (dist < r*r);

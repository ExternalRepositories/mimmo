175 void
176 ControlDeformMaxDistance::execute(){
177 
178  MimmoObject * geo = getGeometry();
179  if(geo->isEmpty()) return;
180  if(!(geo->isBvTreeSupported())) return;
181 
182  m_defField.resize(getGeometry()->getNVertex(),darray3E{{0.0,0.0,0.0}});
183  m_violationField.resize(m_defField.size());
184 
185  if(!(geo->isBvTreeBuilt())) geo->buildBvTree();
186 
187  dvecarr3E points = geo->getVertexCoords();
188  points+= m_defField;
189  dvector1D normDef(m_defField.size());
190 
191  int count=0;
192  for(auto & def : m_defField){
193  normDef[count] = norm2(def);
194  ++count;
195  }
196 
197  double dist;
198  double radius ;
199  double rate = 0.05;
200  int kmax = 200;
201  int kiter;
202  bool flag;
203  long id;
204  count = 0;
205 
206  for(auto &p : points){
207  dist =1.0E+18;
208  kiter = 0;
209  flag = true;
210  radius = std::fmax(1.0E-8, normDef[count]);
211  while(flag && kiter < kmax){
212  dist = bvTreeUtils::distance(&p, geo->getBvTree(), id, radius);
213  flag = (dist == 1.0E+18);
214  if(flag) radius *= (1.0+ rate*((double)flag));
215  kiter++;
216  }
217  if(kiter == kmax) dist = m_maxDist - dist;
218  m_violationField[count] = (dist - m_maxDist);
219  count++;
220  }
221 };

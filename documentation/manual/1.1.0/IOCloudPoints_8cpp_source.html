417 void
418 IOCloudPoints::read(){
419 
420  std::unordered_map<long, int> mapP;
421  std::ifstream reading;
422  std::string source = m_dir+"/"+m_filename;
423  reading.open(source.c_str());
424  if(reading.is_open()){
425 
426  m_points.clear();
427  m_labels.clear();
428  m_scalarfield.clear();
429  m_vectorfield.clear();
430 
431  std::string keyword, line;
432  long label;
433  darray3E dtrial;
434  double temp;
435 
436  do{
437  line.clear();
438  keyword.clear();
439  std::getline(reading,line);
440  line = bitpit::utils::trim(line);
441  std::stringstream ss(line);
442  ss>>keyword;
443  keyword = bitpit::utils::trim(keyword);
444  if(keyword == "$POINT") {
445 
446  ss>>label;
447  m_labels.push_back(label);
448 
449  dtrial.fill(0.0);
450  ss>>dtrial[0]>>dtrial[1]>>dtrial[2];
451  m_points.push_back(dtrial);
452  }
453 
454  }while(!reading.eof());
455 
456  int counter = 0;
457  for(auto &lab :m_labels){
458  mapP[lab] = counter;
459  ++counter;
460  }
461 
462  m_scalarfield.resize(m_points.size(),0.0);
463  m_vectorfield.resize(m_points.size(),{{0.0,0.0,0.0}});
464 
465  reading.clear();
466  reading.seekg(0, ios::beg);
467 
468  do{
469  line.clear();
470  keyword.clear();
471  std::getline(reading,line);
472  line = bitpit::utils::trim(line);
473  std::stringstream ss(line);
474  ss>>keyword;
475  keyword = bitpit::utils::trim(keyword);
476  if(keyword == "$SCALARF") {
477 
478  ss>>label>>temp;
479  m_scalarfield[mapP[label]] = temp;
480  }
481  if(keyword == "$VECTORF") {
482 
483  dtrial.fill(0.0);
484  ss>>label>>dtrial[0]>>dtrial[1]>>dtrial[2];
485  m_vectorfield[mapP[label]] = dtrial;
486  }
487 
488  }while(!reading.eof());
489 
490  }else{
491  (*m_log)<<"error of "<<m_name<<" : cannot open "<<m_filename<< " requested. Exiting... "<<std::endl;
492  throw std::runtime_error (m_name + " : cannot open " + m_filename + " requested. Exiting... ");
493  }
494 
495  reading.close();
496 };

501 void
502 IOCloudPoints::write(){
503 
504  //assessing data;
505  int sizelabels = m_labels.size();
506  long maxlabel = 0;
507  for(auto & val: m_labels){
508  maxlabel = std::max(maxlabel,val);
509  }
510 
511  int sizedispl = m_points.size();
512 
513  m_labels.resize(m_points.size(), -1);
514  if(sizelabels < sizedispl){
515  for(int i = sizelabels; i<sizedispl; ++i){
516  m_labels[i] = maxlabel+1;
517  ++maxlabel;
518  }
519  }
520 
521  m_scalarfield.resize(m_points.size(), 0.0);
522  m_vectorfield.resize(m_points.size(), {{0.0,0.0,0.0}});
523 
524  std::string filename;
525  if(m_template){
526  filename = "TEMPLATE_"+m_filename;
527  }else{
528  filename = m_filename;
529  }
530 
531  std::ofstream writing;
532  std::string source = m_dir+"/"+m_filename;
533  writing.open(source.c_str());
534  std::string keyT1 = "{", keyT2 = "}";
535  if(writing.is_open()){
536 
537  int counter = 0;
538  for(auto & dd : m_points){
539  writing<<"$POINT"<<'\t'<<m_labels[counter]<<'\t'<<dd[0]<<'\t'<<dd[1]<<'\t'<<dd[2]<<std::endl;
540  ++counter;
541  }
542  writing<<""<<std::endl;
543 
544  counter = 0;
545  for(auto & dd : m_scalarfield){
546  if(m_template){
547  std::string str1 = keyT1+"s"+std::to_string(m_labels[counter])+keyT2;
548  writing<<"$SCALARF"<<'\t'<<m_labels[counter]<<'\t'<<str1<<std::endl;
549  }else{
550  writing<<"$SCALARF"<<'\t'<<m_labels[counter]<<'\t'<<dd<<std::endl;
551  }
552  ++counter;
553  }
554  writing<<""<<std::endl;
555 
556  counter = 0;
557  for(auto & dd : m_vectorfield){
558  if(m_template){
559  std::string str1 = keyT1+"x"+std::to_string(m_labels[counter])+keyT2;
560  std::string str2 = keyT1+"y"+std::to_string(m_labels[counter])+keyT2;
561  std::string str3 = keyT1+"z"+std::to_string(m_labels[counter])+keyT2;
562 
563  writing<<"$VECTORF"<<'\t'<<m_labels[counter]<<'\t'<<str1<<'\t'<<str2<<'\t'<<str3<<std::endl;
564  }else{
565  writing<<"$VECTORF"<<'\t'<<m_labels[counter]<<'\t'<<dd[0]<<'\t'<<dd[1]<<'\t'<<dd[2]<<std::endl;
566  }
567  ++counter;
568  }
569  writing<<""<<std::endl;
570  }else{
571  (*m_log)<<"error of "<<m_name<<" : cannot open "<<m_filename<< " requested. Exiting... "<<std::endl;
572  throw std::runtime_error (m_name + " : cannot open " + m_filename + " requested. Exiting... ");
573  }
574 
575  writing.close();

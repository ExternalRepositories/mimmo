104 IOConnections_MIMMO::absorbConnections(const bitpit::Config & slotXML, bool debug){
105 
106  if(debug)
107  m_log->setPriority(bitpit::log::NORMAL);
108 
109  if(slotXML.getSectionCount() == 0) {
110  (*m_log)<<"IOConnections::absorbConnections does not found any connection to read in current XML slot."<<std::endl;
111  return;
112  }
113 
114  for( auto & sect : slotXML.getSections()){
115 
116  std::string snd_str;
117  std::string rcv_str;
118  std::string sndP_str;
119  std::string rcvP_str;
120 
121  if(sect.second->hasOption("sender")) snd_str = sect.second->get("sender");
122  if(sect.second->hasOption("receiver")) rcv_str = sect.second->get("receiver");
123  if(sect.second->hasOption("senderPort")) sndP_str = sect.second->get("senderPort");
124  if(sect.second->hasOption("receiverPort")) rcvP_str = sect.second->get("receiverPort");
125 
126  snd_str = bitpit::utils::trim(snd_str);
127  rcv_str = bitpit::utils::trim(rcv_str);
128  sndP_str = bitpit::utils::trim(sndP_str);
129  rcvP_str = bitpit::utils::trim(rcvP_str);
130 
131 
132  auto itSend = m_mapConn.find(snd_str);
133  auto itRece = m_mapConn.find(rcv_str);
134  auto itSPort = m_mapPorts.find(sndP_str);
135  auto itRPort = m_mapPorts.find(rcvP_str);
136 
137  if(itSend == m_mapConn.end() || itRece == m_mapConn.end() || itSPort == m_mapPorts.end() || itRPort == m_mapPorts.end()){
138  (*m_log)<<"---------------------------------------------"<<std::endl;
139  (*m_log)<<"sender: "<<snd_str<<std::endl;
140  (*m_log)<<"receiver: "<<rcv_str<<std::endl;
141  (*m_log)<<"senderPort: "<<sndP_str<<std::endl;
142  (*m_log)<<"receiverPort: "<<rcvP_str<<std::endl;
143  (*m_log)<<"---------------------------------------------"<<std::endl;
144  (*m_log)<<""<<std::endl;
145  (*m_log)<<"IOConnections::absorbConnections does not found any sender/receiver and/or port compatible with API."<<std::endl;
146  (*m_log)<<"Please control connection data entries in your XML file"<<std::endl;
147  (*m_log)<<""<<std::endl;
148  continue;
149  }
150 
151  bool check = pin::addPin(itSend->second, itRece->second, itSPort->second, itRPort->second);
152 
153  if(!check){
154  (*m_log)<<"---------------------------------------------"<<std::endl;
155  (*m_log)<<"sender: "<<snd_str<<std::endl;
156  (*m_log)<<"receiver: "<<rcv_str<<std::endl;
157  (*m_log)<<"senderPort: "<<sndP_str<<std::endl;
158  (*m_log)<<"receiverPort: "<<rcvP_str<<std::endl;
159  (*m_log)<<"---------------------------------------------"<<std::endl;
160  (*m_log)<<""<<std::endl;
161  (*m_log)<<"IOConnections::absorbConnections failed creating connection."<<std::endl;
162  (*m_log)<<""<<std::endl;
163  }
164  else{
165  (*m_log)<<"IOConnections::absorbConnections successfully created connection."<<std::endl;
166  }
167  }
168  m_log->setPriority(bitpit::log::DEBUG);
169 
170  return;
171 };

183 void
184 IOConnections_MIMMO::flushConnections(bitpit::Config & slotXML, bool debug ){
185 
186  int counterGlob = 0;
187  for( auto object : m_mapConn){
188 
189  std::string sender = m_invMapConn[object.second]; //recover sender name
190  auto mapPOut = (object.second)->getPortsOut(); //get its slot of output port
191 
192 
193  for(auto & pOut : mapPOut ){
194 
195  std::string senderPort = m_invMapPorts[pOut.first]; //gets the Port Type of sender
196 
197  std::vector<BaseManipulation * > links = (pOut.second)->getLink(); //get all possible links to this specific port
198  std::vector<short int> lids = (pOut.second)->getPortLink();
199 
200  int counter = 0;
201  for( auto & val : links){
202  std::string receiver = m_invMapConn[val];
203  std::string receiverPort = m_invMapPorts[lids[counter]];
204 
205  std::string title = "conn" + std::to_string(counter+counterGlob);
206  bitpit::Config::Section & conn = slotXML.addSection(title);
207  conn.set("sender", sender);
208  conn.set("senderPort", senderPort);
209  conn.set("receiver", receiver);
210  conn.set("receiverPort", receiverPort);
211 
212  ++counter;
213  }//loop on possible links on the same port
214  counterGlob += counter;
215  }//loop on all the port of the single object
216  }//loop on all instantiated object.
217 
218  if(debug)
219  m_log->setPriority(bitpit::log::NORMAL);
220  (*m_log)<<"IOConnections::flushConnections wrote "<< slotXML.getSectionCount()<< " active connections."<<std::endl;
221  m_log->setPriority(bitpit::log::DEBUG);
222  return;
223 };

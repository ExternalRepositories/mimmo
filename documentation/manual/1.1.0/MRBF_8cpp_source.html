308 ivector1D
309 MRBF::checkDuplicatedNodes(double tol){
310  ivector1D marked;
311  int sizeEff = getTotalNodesCount();
312  if( sizeEff == 0 ) return marked;
313 
314  bvector1D check(sizeEff, false);
315 
316  for(int i=0; i<sizeEff; ++i){
317  for(int j=i+1; j<sizeEff; ++j){
318  double dist = norm2(m_node[j] - m_node[i]);
319  if(!check[j] && dist <= tol){
320  marked.push_back(i);
321  check[j] = true;
322  }
323  }
324  }
325  return(marked);
326 }

462 void
463 MRBF::execute(){
464 
465  MimmoObject * container = getGeometry();
466  if(container->isEmpty() ) return;
467 
468  int size = 0;
469  int sizeF = getDataCount();
470 
471  for (int i=0; i<sizeF; i++){
472 
473  if(m_solver == MRBFSol::NONE) size = m_weight[i].size();
474  else size = m_value[i].size();
475 
476  if(size != getTotalNodesCount()){
477  (*m_log) << "warning: " << getName() << " has displacements of " << i << " field with size (" << size << ") that does not fit number of RBF nodes ("<< getTotalNodesCount() << ")" << std::endl;
478  fitDataToNodes(i);
479  }
480  }
481 
482  double bboxDiag;
483  {
484  darray3E pmin, pmax;
485  container->getPatch()->getBoundingBox(pmin, pmax);
486  bboxDiag= norm2(pmax - pmin);
487  }
488 
489  //Checking supportRadius.
490  double distance = 0.0;
491  if(m_SRRatio <=0.0){ //get maximum weight/value displ and assign support radius a 3 times this value.
492 
493  for(int i=0; i<size; ++i){
494 
495  dvector1D data(sizeF);
496 
497  for(int j=0; j<sizeF; ++j){
498  if(m_solver == MRBFSol::NONE) data[j] = m_weight[j][i];
499  else data[j] = m_value[j][i];
500  }
501 
502  distance = std::max(distance, norm2(data));
503  }
504 
505  distance *=3.0;
506 
507  }else{
508  if (m_supRIsValue){
509  distance = m_SRRatio;
510  }
511  else{
512  distance = m_SRRatio * bboxDiag;
513  }
514  }
515 
516  //TODO remove it (I can't use a support radius 0....why?)
517  if(distance <=1.E-18){ //checkSupportRadius if too small, set it to the semidiagonal value of the geometry AABB
518  distance = 0.5*bboxDiag;
519  }
520 
521  const double radius = distance;
522  RBF::setSupportRadius(radius);
523 
524  if (m_solver == MRBFSol::WHOLE) solve();
525  if (m_solver == MRBFSol::GREEDY) greedy(m_tol);
526 
527  int nv = container->getNVertex();
528  dvecarr3E vertex = container->getVertexCoords();
529 
530  m_displ.resize(nv, darray3E{0,0,0});
531  dvector1D displ;
532  for(int i=0; i<nv; ++i){
533  displ = RBF::evalRBF(vertex[i]);
534  for (int j=0; j<3; j++) m_displ[i][j] = displ[j];
535  }
536 
537  //if m_filter is active;
538  if(m_bfilter){
539  m_filter.resize(nv,1.0);
540  int counter = 0;
541  for (auto && vec : m_displ){
542  vec = vec * m_filter[counter];
543  ++counter;
544  }
545  }

590 void
591 MRBF::plotCloud(std::string directory, std::string filename, int counterFile, bool binary, bool deformed){
592 
593  int nnodes = getTotalNodesCount();
594  nnodes = min(nnodes, int(m_displ.size()));
595  dvecarr3E* nodes_ = getNodes();
596  dvecarr3E nodes(nnodes);
597  dvecarr3E data(nnodes);
598  for(int i=0; i<nnodes; ++i){
599  for(int j=0; j<3; ++j){
600  if(m_solver == MRBFSol::NONE) data[i][j] = m_weight[j][i];
601  else data[i][j] = m_value[j][i];
602  }
603  }
604  if(deformed){
605  for(int i=0; i<nnodes; ++i){
606  nodes[i] = (*nodes_)[i] + data[i];
607  }
608  }else{
609  for(int i=0; i<nnodes; ++i){
610  nodes[i] = (*nodes_)[i];
611  }
612  }
613 
614  bitpit::VTKFormat codex = bitpit::VTKFormat::ASCII;
615  if(binary){codex=bitpit::VTKFormat::APPENDED;}
616 
617  ivector1D conn(nnodes);
618  {
619  int counter = 0;
620  for(auto & val: conn){
621  val = counter;
622  ++counter;
623  }
624  }
625  bitpit::VTKUnstructuredGrid vtk(directory, filename, bitpit::VTKElementType::VERTEX);
626  vtk.setGeomData( bitpit::VTKUnstructuredField::POINTS, nodes) ;
627  vtk.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, conn) ;
628  vtk.setDimensions(conn.size(), nnodes);
629  vtk.setCodex(codex);
630  if(counterFile>=0){vtk.setCounter(counterFile);}
631  vtk.write();
632 };

642  BITPIT_UNUSED(name);

644  std::string input;

646  BaseManipulation::absorbSectionXML(slotXML, name);

671  };
672

674  input = slotXML.get("SupportRadius");
675  double value = -1.0;
676  if(!input.empty()){
677  std::stringstream ss(bitpit::utils::trim(input));
678  ss >> value;

684  input = slotXML.get("SupportRadiusReal");
685  double value = -1.0;
686  if(!input.empty()){

688  ss >> value;

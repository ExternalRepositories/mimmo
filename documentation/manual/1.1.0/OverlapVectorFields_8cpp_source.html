307 OverlapVectorFields::plotData(std::string dir, std::string name, bool flag, int counter, mimmo::MimmoObject * geo){
308 
309  if(m_originals.count(geo) == 0) return;
310  dvecarr3E field;
311  bool flagCalc = (m_results.count(geo) > 0);
312 
313 
314  dvecarr3E points = geo->getVertexCoords();
315  ivector2D connectivity;
316  bitpit::VTKElementType cellType = geo->desumeElement();
317 
318  if (geo->getType() != 3){
319  connectivity = geo->getCompactConnectivity();
320  }
321  else{
322  int np = points.size();
323  connectivity.resize(np);
324  for (int i=0; i<np; i++){
325  connectivity[i].resize(1);
326  connectivity[i][0] = i;
327  }
328  }
329  bitpit::VTKUnstructuredGrid output(dir,name,cellType);
330  output.setGeomData( bitpit::VTKUnstructuredField::POINTS, points);
331  output.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, connectivity);
332  output.setDimensions(connectivity.size(), points.size());
333 
334  if(flagCalc) field = m_results[geo];
335  else field.resize(points.size(), {{0.0,0.0,0.0}});
336 
337  output.addData("vectorfield", bitpit::VTKFieldType::VECTOR, bitpit::VTKLocation::POINT, field);
338 
339  std::vector<long> ids(points.size());
340  long ID;
341  for (auto vertex : geo->getVertices()){
342  ID = vertex.getId();
343  ids[geo->getMapDataInv(ID)] = ID;
344  }
345 
346  output.addData("ID", bitpit::VTKFieldType::SCALAR, bitpit::VTKLocation::POINT, ids);
347 
348  output.setCounter(counter);
349  output.setCodex(bitpit::VTKFormat::APPENDED);
350  if(!flag) output.setCodex(bitpit::VTKFormat::ASCII);
351 
352  output.write();
353 };

376 void
377 OverlapVectorFields::execute(){
378 
379  if(m_originals.empty()) return;
380  m_results.clear();
381  int size, listsize;
382  dvecarr3E temp;
383  dvecarr3E results;
384  for(auto & obj : m_originals){
385 
386  size = obj.first->getPatch()->getVertexCount();
387  //careful resizing
388  results.resize(size, {{0.0,0.0,0.0}});
389  for(auto &vec : obj.second) vec->resize(size,{{0.0,0.0,0.0}});
390 
391  listsize = obj.second.size();
392  temp.resize(listsize);
393 
394  for(int i=0; i<size; ++i){
395  for(int j=0; j<listsize; j++) temp[j] = (*(obj.second[j]))[i];
396  results[i] = overlapFields(temp);
397  }
398 
399  m_results[obj.first] = results;
400  }
401 }

420 darray3E
421 OverlapVectorFields::overlapFields(dvecarr3E & locField){
422  int size = locField.size();
423  if(size < 1) return {{0.0,0.0,0.0}};
424 
425  if(size ==1 )return locField[0];
426  darray3E value;
427  darray3E dir; dir.fill(0.0);
428  double match;
429 
430  switch(m_overlapCriterium){
431  case OverlapMethod::MAX :
432  value = {{0.0,0.0,0.0}};
433  for(auto && loc : locField){
434  dir += loc;
435  }
436 
437  dir /= norm2(dir);
438 
439  match = 1.e-18;
440  for(auto && loc : locField){
441  double dummy = dotProduct(loc, dir);
442  match = std::fmax(match,dummy);
443  }
444 
445  value = match*dir;
446  break;
447 
448  case OverlapMethod::MIN :
449 
450  value = {{0.0,0.0,0.0}};
451  for(auto && loc : locField){
452  dir += loc;
453  }
454 
455  dir /= norm2(dir);
456 
457  match = 1.e18;
458  for(auto && loc : locField){
459  double dummy = dotProduct(loc, dir);
460  match = std::fmin(match,dummy);
461  }
462 
463  value = match*dir;
464  break;
465 
466  case OverlapMethod::AVERAGE :
467  value = {{0.0,0.0,0.0}};
468  for(auto && loc : locField){
469  value += loc/double(size);
470  }
471  break;
472 
473  case OverlapMethod::SUM :
474  value = {{0.0,0.0,0.0}};
475  for(auto && loc : locField){
476  value += loc;
477  }
478  break;
479  default : //never been reached
480  break;
481  }
482 
483  return value;
484 };

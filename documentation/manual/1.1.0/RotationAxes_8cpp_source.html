167 void
168 RotationAxes::execute(){
169 
170  //Rotation of origin
171  m_rotax_origin = {{0,0,0}};
172  m_axes_origin -= m_origin;
173  //rodrigues formula
174  m_rotax_origin = m_axes_origin * cos(m_alpha) +
175  dotProduct(m_direction, m_axes_origin) * (1 - cos(m_alpha)) * m_direction +
176  crossProduct(m_direction, m_axes_origin) * sin(m_alpha);
177 
178  m_rotax_origin += m_origin;
179  m_axes_origin += m_origin;
180 
181  //rotation of axes
182  m_rotax.fill(darray3E{{0,0,0}});
183  //rodrigues formula
184  for (int i=0; i<3; i++){
185  m_rotax[i] = m_axes[i] * cos(m_alpha) +
186  dotProduct(m_direction, m_axes[i]) * (1 - cos(m_alpha)) * m_direction +
187  crossProduct(m_direction, m_axes[i]) * sin(m_alpha);
188  }
189 
190 };

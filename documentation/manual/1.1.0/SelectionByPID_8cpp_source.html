215 livector1D
216 SelectionByPID::extractSelection(){
217  livector1D result;
218  std::set<long> extraction;
219 
220  syncPIDList();
221  shivector1D pids = getActivePID();
222 
223  for(auto && pid : pids){
224  livector1D local = getGeometry()->extractPIDCells(pid);
225  extraction.insert(local.begin(),local.end());
226  }
227 
228  //check if dual selection is triggered
229 
230  if(m_dual){
231  livector1D totID = getGeometry()->getMapCell();
232  result.resize(totID.size() - extraction.size());
233  if(result.size() == 0) return result;
234 
235  std::sort(totID.begin(), totID.end());
236  int counter = 0;
237  auto tot_it = totID.begin();
238  auto cand_it = extraction.begin();
239  while(tot_it != totID.end()){
240  long val = *tot_it;
241  if (cand_it == extraction.end() || val != *cand_it) {
242  result[counter] = val;
243  ++counter;
244  } else {
245  ++cand_it;
246  }
247  ++tot_it;
248  }
249  }else{
250  result.insert(result.end(), extraction.begin(), extraction.end());
251  }
252  return result;
253 };

260 void
261 SelectionByPID::syncPIDList(){
262  if(getGeometry() == NULL) return;
263 
264  if(m_setPID.count(-1) == 0){
265 
266  std::unordered_set<short>::iterator itU;
267  shivector1D negative;
268  for(itU = m_setPID.begin(); itU != m_setPID.end(); ++itU){
269  short value = *itU;
270  if(m_activePID.count(value) > 0) m_activePID[value] = true;
271  else negative.push_back(value);
272  }
273 
274  for(auto &val: negative) m_setPID.erase(val);
275 
276 
277  }else{
278  m_setPID.clear();
279  for(auto &val: m_activePID ){
280  m_setPID.insert(val.first);
281  val.second = true;
282 
283  }
284  }

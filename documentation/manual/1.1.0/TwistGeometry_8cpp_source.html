193 void
194 TwistGeometry::execute(){
195 
196  if (getGeometry() == NULL) return;
197 
198  int nV = m_geometry->getNVertex();
199  m_displ.resize(nV);
200  m_filter.resize(nV, 1.0);
201 
202 
203  darray3E point, rotated;
204  long ID;
205  int idx;
206  liimap mapID = m_geometry->getMapDataInv();
207  darray3E projected;
208  double distance;
209  //double sign;
210  double rot;
211 
212  for (auto vertex : m_geometry->getVertices()){
213  point = vertex.getCoords();
214  ID = vertex.getId();
215  idx = mapID[ID];
216 
217  //signed distance from origin
218  distance = dotProduct((point-m_origin),m_direction);
219 
220  //compute coefficients and constant vectors of rodriguez formula
221  rot = std::min(m_alpha, (std::abs(distance)/m_distance)*m_alpha);
222  if (distance < 0) rot = -rot*int(m_sym);
223  double a = cos(rot);
224  darray3E b = (1 - cos(rot)) * m_direction;
225  double c = sin(rot);
226 
227  //project point on axis (local origin)
228  projected = distance*m_direction + m_origin;
229 
230 
231  point -= projected;
232  //rodrigues formula
233  rotated = a * point +
234  b * dotProduct(m_direction, point) +
235  c * crossProduct(m_direction, point);
236 
237  rotated += projected;
238  point += projected;
239 
240  m_displ[idx] = (rotated-point)*m_filter[idx];
241 
242  }
243 
244 };

250 TwistGeometry::apply(){
251 
252  if (getGeometry() == NULL) return;
253  dvecarr3E vertex = getGeometry()->getVertexCoords();
254  long nv = getGeometry()->getNVertex();
255  nv = long(std::min(int(nv), int(m_displ.size())));
256  livector1D & idmap = getGeometry()->getMapData();
257  for (long i=0; i<nv; i++){
258  vertex[i] += m_displ[i];
259  getGeometry()->modifyVertex(vertex[i], idmap[i]);
260  }

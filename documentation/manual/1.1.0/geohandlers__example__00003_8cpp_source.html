50 void test00003() {
51 
52  /* Creation of mimmo containers.
53  * Input and output MimmoGeometry are instantiated
54  * as two different objects (no loop in chain are permitted).
55  */
56  MimmoGeometry * mimmo0 = new MimmoGeometry();
57  mimmo0->setIOMode(IOMode::CONVERT);
58  mimmo0->setReadDir("geodata");
59  mimmo0->setReadFileType(FileType::STL);
60  mimmo0->setReadFilename("sphere2");
61  mimmo0->setWriteDir("./");
62  mimmo0->setWriteFileType(FileType::STL);
63  mimmo0->setWriteFilename("geohandlers_output_00003.0000");
64 
65  MimmoGeometry * mimmo1 = new MimmoGeometry();
66  mimmo1->setIOMode(IOMode::CONVERT);
67  mimmo1->setReadDir("geodata");
68  mimmo1->setReadFileType(FileType::STL);
69  mimmo1->setReadFilename("plane1");
70  mimmo1->setWriteDir("./");
71  mimmo1->setWriteFileType(FileType::STL);
72  mimmo1->setWriteFilename("geohandlers_output_00003p1.0000");
73 
74  MimmoGeometry * mimmo2 = new MimmoGeometry();
75  mimmo2->setIOMode(IOMode::CONVERT);
76  mimmo2->setReadDir("geodata");
77  mimmo2->setReadFileType(FileType::STL);
78  mimmo2->setReadFilename("plane2");
79  mimmo2->setWriteDir("./");
80  mimmo2->setWriteFileType(FileType::STL);
81  mimmo2->setWriteFilename("geohandlers_output_00003p2.0000");
82 
83  MimmoGeometry * mimmo3 = new MimmoGeometry();
84  mimmo3->setIOMode(IOMode::WRITE);
85  mimmo3->setWriteDir(".");
86  mimmo3->setWriteFileType(FileType::STL);
87  mimmo3->setWriteFilename("geohandlers_output_00003.0001");
88 
89 
90  /* Instantiation of two Selection By Map block.
91  * The planes are used as selection objects with an offset
92  * defined by the user.
93  */
94  SelectionByMapping * mapSel1 = new SelectionByMapping();
95  SelectionByMapping * mapSel2 = new SelectionByMapping();
96  mapSel1->setTolerance(1.0e-01);
97  mapSel1->setPlotInExecution(true);
98  mapSel2->setTolerance(1.0e-01);
99  mapSel2->setPlotInExecution(true);
100 
101  /* Creation of a two points with x coordinates -0.5 and 0.5 (y=z=0.0);
102  */
103  dvecarr3E rbfNodes1(1, {{-0.5, 0.0, 0.0}});
104  dvecarr3E rbfNodes2(1, {{0.5, 0.0, 0.0}});
105 
106  /* Creation of a two displacements for the rbf control points.
107  */
108  dvecarr3E displ1(1, {{-0.25, 0.0, 0.0}});
109  dvecarr3E displ2(1, {{0.25, 0.0, 0.0}});
110 
111  /* Instantiation of two MRBF objects.
112  * Set rbf points and displacements defined above.
113  * Plot Optional results during execution active for MRBF block.
114  */
115  MRBF* mrbf1 = new MRBF();
116  mrbf1->setMode(MRBFSol::NONE);
117  mrbf1->setSupportRadius(0.4);
118  mrbf1->setPlotInExecution(true);
119  mrbf1->setNode(rbfNodes1);
120  mrbf1->setDisplacements(displ1);
121 
122  MRBF* mrbf2 = new MRBF();
123  mrbf2->setMode(MRBFSol::NONE);
124  mrbf2->setSupportRadius(0.4);
125  mrbf2->setPlotInExecution(true);
126  mrbf2->setNode(rbfNodes2);
127  mrbf2->setDisplacements(displ2);
128 
129  /* Create reconstruct vector block and set to recontruct over the whole
130  * input geometry the displacements fields
131  * given by the two rbf blocks on two separate patches.
132  */
133  ReconstructVector* recon = new ReconstructVector();
134 
135 
136  /* Create applier block.
137  * It applies the deformation displacements to the original input geometry.
138  */
139  Apply* applier = new Apply();
140 
141  /* Setup pin connections.
142  */
143  addPin(mimmo0, mapSel1, PortType::M_GEOM, PortType::M_GEOM);
144  addPin(mimmo0, mapSel2, PortType::M_GEOM, PortType::M_GEOM);
145  addPin(mimmo0, applier, PortType::M_GEOM, PortType::M_GEOM);
146  addPin(mimmo1, mapSel1, PortType::M_GEOM, PortType::M_GEOM2);
147  addPin(mimmo2, mapSel2, PortType::M_GEOM, PortType::M_GEOM2);
148  addPin(mapSel1, mrbf1, PortType::M_GEOM, PortType::M_GEOM);
149  addPin(mapSel2, mrbf2, PortType::M_GEOM, PortType::M_GEOM);
150  addPin(mimmo0, recon, PortType::M_GEOM, PortType::M_GEOM);
151  addPin(mrbf1, recon, PortType::M_PAIRVECFIELD, PortType::M_PAIRVECFIELD);
152  addPin(mrbf2, recon, PortType::M_PAIRVECFIELD, PortType::M_PAIRVECFIELD);
153  addPin(recon, applier, PortType::M_GDISPLS, PortType::M_GDISPLS);
154  addPin(applier, mimmo3, PortType::M_GEOM, PortType::M_GEOM);
155 
156 
157  /* Setup execution chain.
158  */
159  Chain ch0;
160  ch0.addObject(mimmo0);
161  ch0.addObject(mimmo1);
162  ch0.addObject(mimmo2);
163  ch0.addObject(mapSel1);
164  ch0.addObject(mapSel2);
165  ch0.addObject(mrbf1);
166  ch0.addObject(mrbf2);
167  ch0.addObject(recon);
168  ch0.addObject(applier);
169  ch0.addObject(mimmo3);
170 
171  /* Execution of chain.
172  * Use debug flag true to to print out the execution steps.
173  */
174  ch0.exec(true);
175 
176  /* Clean up & exit;
177  */
178  delete mimmo0;
179  delete mimmo1;
180  delete mimmo2;
181  delete mimmo3;
182  delete mapSel1;
183  delete mapSel2;
184  delete mrbf1;
185  delete mrbf2;
186  delete recon;
187  delete applier;
188 
189  mimmo0 = NULL;
190  mimmo1 = NULL;
191  mimmo2 = NULL;
192  mimmo3 = NULL;
193  mapSel1 = NULL;
194  mapSel2 = NULL;
195  mrbf1 = NULL;
196  mrbf2 = NULL;
197  recon = NULL;
198  applier = NULL;
199 
200  return;
201 }

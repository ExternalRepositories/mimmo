633  void
634  PabloUniform::getBoundingBox(darray3 & P0, darray3 & P1){
635  darray3 cnode0, cnode1;
636  uint32_t nocts = ParaTree::getNumOctants();
637  uint32_t id = 0;
638  uint8_t nnodes = ParaTree::getNnodes();
639 
640  P0 = ParaTree::getNode(id, 0);
641  P1 = ParaTree::getNode(nocts-1, nnodes-1);
642 
643  for (id=0; id<nocts; id++){
644  cnode0 = ParaTree::getNode(id, 0);
645  cnode1 = ParaTree::getNode(id, nnodes-1);
646  for (int i=0; i<3; i++){
647  P0[i] = min(P0[i], (double)cnode0[i]);
648  P1[i] = max(P1[i], (double)cnode1[i]);
649  }
650  }
651  for (int i=0; i<3; i++){
652  P0[i] = m_origin[i] + m_L * P0[i];
653  P1[i] = m_origin[i] + m_L * P1[i];
654  }
655  };

350 void VTKRectilinearGrid::setDimensions( int n1, int n2, int m1, int m2, int l1, int l2 ){
351 
352  m_dimensions = ( l1 == l2 ) ? 2 :3 ;
353 
354  m_localIndex[0][0] = n1 ;
355  m_localIndex[0][1] = n2 ;
356  m_localIndex[1][0] = m1 ;
357  m_localIndex[1][1] = m2 ;
358  m_localIndex[2][0] = l1 ;
359  m_localIndex[2][1] = l2 ;
360 
361  if(m_procs==1)
362  m_globalIndex = m_localIndex ;
363 
364  m_cells = 1 ;
365  m_points = 1 ;
366 
367  for(int d=0; d<m_dimensions; ++d){
368  m_cells = m_cells * ( m_localIndex[d][1] -m_localIndex[d][0] ) ;
369  m_points = m_points * ( m_localIndex[d][1] -m_localIndex[d][0] +1 ) ;
370  };
371 
372  return ;

537 uint64_t VTKRectilinearGrid::calcFieldEntries( const VTKField &field ){
538 
539  uint64_t entries(0) ;

541 
542  if( name == "x_Coord" ){

549  entries = m_localIndex[2][1] -m_localIndex[2][0] +1 ;
550 
551  } else{
552 
553  VTKLocation location( field.getLocation() ) ;
554  assert( location != VTKLocation::UNDEFINED) ;
555 
556  if( location == VTKLocation::CELL ){
557  entries = m_cells ;
558 
559  } else if( location == VTKLocation::POINT ){
560  entries = m_points ;
561 
562  }
563 
564  VTKFieldType fieldType( field.getFieldType() ) ;
565  assert( fieldType != VTKFieldType::UNDEFINED) ;
566 
567  entries *= static_cast<uint64_t>(fieldType) ;
568

95 double LevelSetCartesian::updateEikonal( double s, double g, const long &I, const std::unordered_map<long,short> &active ){
96 
97  int d;
98  long J;
99  double h2, delta, value, a(0), b(0), c(0);
100 
101  Cell& cell = m_cartesian->getCell(I) ;
102 
103  for( d=0; d<m_cartesian->getDimension(); ++d){ // COMPUTE QUADRATIC FORM COEFFICIENTS FROM UPWIND STENCIL
104 
105  value = levelSetDefaults::VALUE ;
106 
107  // Left neighbor
108  J = cell.getAdjacency( 2*d, 0) ;
109 
110  LSInfo &lsInfo = m_ls[J] ;
111 
112  if( J >= 0 && active.at(J) == 0){
113  value = std::min(s*lsInfo.value, value);
114  };
115 
116 
117  // Right neighbor
118  J = cell.getAdjacency( 2*d+1, 0) ;
119 
120  if( J >= 0 && active.at(J) == 0){
121  value = std::min(s*lsInfo.value, value);
122  };
123 
124 
125  // Update coeffs in the quadratic form
126  if (value < levelSetDefaults::VALUE) {
127  h2 = pow(m_cartesian->getSpacing(d), 2);
128 
129  a += 1.0/h2;
130  b += -2.0 * value/h2;
131  c += std::pow(value, 2)/h2;
132  }
133 
134  };
135 
136 
137  { // SOLVE THE QUADRATIC FORM
138  // Quadratic form determinant
139  delta = pow(b, 2) - 4.0*a*(c - pow(g, 2));
140 
141  // Solution
142  value = -(b - sqrt(delta))/(2.0*a);
143 
144  }
145 
146  return(value);
147 
148 };

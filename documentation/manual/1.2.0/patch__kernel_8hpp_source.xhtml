292  {
293  CellPositionLess(PatchKernel &patch, bool native = true)
294  : m_patch(patch), m_native(native)
295  {
296  }
297 
298  bool operator()(const long &id_1, const long &id_2) const
299  {
300  std::array<double, 3> centroid_1;
301  std::array<double, 3> centroid_2;
302  if (m_native) {
303  centroid_1 = m_patch.evalCellCentroid(id_1);
304  centroid_2 = m_patch.evalCellCentroid(id_2);
305  } else {
306  centroid_1 = m_patch.PatchKernel::evalCellCentroid(id_1);
307  centroid_2 = m_patch.PatchKernel::evalCellCentroid(id_2);
308  }
309 
310  for (int k = 0; k < 3; ++k) {
311  if (std::abs(centroid_1[k] - centroid_2[k]) <= m_patch.getTol()) {
312  continue;
313  }
314 
315  return centroid_1[k] < centroid_2[k];
316  }
317 
318  // If we are here the two cell centroids coincide. It's not
319  // possible to define an order for the two cells.
320  std::ostringstream stream;
321  stream << "It was not possible to define an order for cells " << id_1 << " and " << id_2 << ". ";
322  stream << "The two cells have the same centroid.";
323  throw std::runtime_error (stream.str());
324  }
325 
326  PatchKernel &m_patch;
327  bool m_native;
328  };

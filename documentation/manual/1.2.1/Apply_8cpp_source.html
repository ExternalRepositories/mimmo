114  vertexcoords = vertex.getCoords();

154 void
155 Apply::checkInput(){
156 
157  bool check = m_input.getDataLocation() == mimmo::MPVLocation::POINT;
158  check = check && m_input.getGeometry() == getGeometry();
159  check = check && m_input.completeMissingData({{0.0,0.0,0.0}});
160  if (!check){
161  m_log->setPriority(bitpit::log::Verbosity::DEBUG);
162  (*m_log)<<"Not valid input found in "<<m_name<<". Proceeding with default zero field"<<std::endl;
163  m_log->setPriority(bitpit::log::Verbosity::NORMAL);
164 
165  m_input.clear();
166  m_input.setGeometry(m_geometry);
167  m_input.setDataLocation(mimmo::MPVLocation::POINT);
168  m_input.reserve(getGeometry()->getNVertex());
169  for (const auto & vertex : getGeometry()->getVertices()){
170  m_input.insert(vertex.getId(), {{0.0,0.0,0.0}});
171  }
172  }
173 
174 }

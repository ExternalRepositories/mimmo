502 void
503 BaseManipulation::exec(){
504 
505  if (!MIMMO_EXPERT){
506  std::map<int, vector<PortIn*> > families;
507  std::map<int, vector<PortID> > familiesID;
508  for (std::unordered_map<PortID, PortIn*>::iterator i=m_portIn.begin(); i!=m_portIn.end(); i++){
509  if (i->second->isMandatory()){
510  families[i->second->getFamily()].push_back(i->second);
511  familiesID[i->second->getFamily()].push_back(i->first);
512  }
513  }
514  std::map<int, vector<PortID> >::iterator itID = familiesID.begin();
515  for (std::map<int, vector<PortIn*> >::iterator i=families.begin(); i!=families.end(); i++){
516  if (i->first == 0){
517  int count = 0;
518  for (auto ip : i->second){
519  std::vector<BaseManipulation*> linked = ip->getLink();
520  if (linked.size() == 0){
521  (*m_log) << "error: " << m_name << " mandatory port " << itID->second[count] << " not linked -> exit! " << std::endl;
522  throw std::runtime_error (m_name + " mandatory port " + itID->second[count] + " not linked");
523  }
524  count++;
525  }
526  }
527  else{
528  bool check = false;
529  for (auto ip : i->second){
530  std::vector<BaseManipulation*> linked = ip->getLink();
531  check = check || linked.size();
532  }
533  if (!check){
534  (*m_log) << "error: " << m_name << " none of mandatory ports " << itID->second << " linked -> exit! " << std::endl;
535  std::string ports;
536  for (auto p : itID->second)
537  ports += " " + p + " ";
538  throw std::runtime_error (m_name + " none of mandatory ports " + ports + " linked");
539  }
540  }
541  itID++;
542  }
543  }
544 
545  if (m_active) execute();
546 
547  for (std::unordered_map<PortID, PortOut*>::iterator i=m_portOut.begin(); i!=m_portOut.end(); i++){
548  std::vector<BaseManipulation*> linked = i->second->getLink();
549  if (linked.size() > 0){
550  i->second->exec();
551  }
552  }
553 
554  if(isPlotInExecution()) plotOptionalResults();
555  if(isApply()) apply();
556 }

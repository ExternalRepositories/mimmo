234 void
235 BendGeometry::execute(){
236 
237  if (getGeometry() == NULL){
238  throw std::runtime_error (m_name + " : NULL pointer to linked geometry");
239  }
240  if (getGeometry()->isEmpty()){
241  throw std::runtime_error (m_name + " : empty linked geometry");
242  }
243 
244  //check coherence of degrees and coeffs;
245  for(int i=0; i<3; ++i){
246  for(int j=0; j<3; ++j){
247  m_coeffs[i][j].resize(m_degree[i][j]+1, 0.0);
248  }
249  }
250 
251  checkFilter();
252 
253  m_displ.clear();
254  m_displ.setDataLocation(mimmo::MPVLocation::POINT);
255  m_displ.reserve(getGeometry()->getNVertex());
256  m_displ.setGeometry(getGeometry());
257 
258  long ID;
259  darray3E value;
260  darray3E point, point0;
261  for (const auto & vertex : m_geometry->getVertices()){
262  point = vertex.getCoords();
263  if (m_local){
264  point0 = point;
265  point = toLocalCoord(point);
266  }
267  ID = vertex.getId();
268  value.fill(0.0);
269  for (int j=0; j<3; j++){
270  for (int z=0; z<3; z++){
271  if (m_degree[j][z] > 0){
272  for (int k=0; k<(int)m_degree[j][z]+1; k++){
273  value[j] += pow(point[z],(double)k)*m_coeffs[j][z][k]*m_filter[ID];
274  }
275  }
276  }
277  }
278  if (m_local){
279  point += value;
280  point = toGlobalCoord(point);
281  value = point - point0;
282  }
283  m_displ.insert(ID, value);
284  }
285 };

311 BendGeometry::checkFilter(){
312  bool check = m_filter.getDataLocation() == mimmo::MPVLocation::POINT;
313  check = check && m_filter.completeMissingData(0.0);
314  check = check && m_filter.getGeometry() == getGeometry();
315 
316  if (!check){
317  m_log->setPriority(bitpit::log::Verbosity::DEBUG);
318  (*m_log)<<"Not valid filter found in "<<m_name<<". Proceeding with default unitary field"<<std::endl;
319  m_log->setPriority(bitpit::log::Verbosity::NORMAL);
320 
321  m_filter.clear();
322  m_filter.setGeometry(m_geometry);
323  m_filter.setDataLocation(mimmo::MPVLocation::POINT);
324  m_filter.reserve(getGeometry()->getNVertex());
325  for (const auto & vertex : getGeometry()->getVertices()){
326  m_filter.insert(vertex.getId(), 1.0);
327  }
328  }

335 darray3E
336 BendGeometry::toLocalCoord(darray3E point){
337  darray3E work, work2;
338  //unapply origin translation
339  work = point - m_origin;
340  //apply change to local sdr transformation
341  dmatrix33E transp = linearalgebra::transpose(m_system);
342  linearalgebra::matmul(work, transp, work2);
343  return(work2);
344 };

352 darray3E BendGeometry::toGlobalCoord(darray3E point){
353 
354  darray3E work, work2;
355  //unscale your local point
356  for(int i =0; i<3; ++i){
357  work[i] = point[i];
358  }
359 
360  //unapply change to local sdr transformation
361  linearalgebra::matmul(work, m_system, work2);
362 
363  //unapply origin translation
364  work = work2 + m_origin;
365  return(work);
366 };

376 BendGeometry::absorbSectionXML(const bitpit::Config::Section & slotXML, std::string name){
377 
378  BITPIT_UNUSED(name);
379 
380  std::string input;
381 
382  BaseManipulation::absorbSectionXML(slotXML, name);
383 
384  if(slotXML.hasSection("DegreesMatrix")){
385  auto & subslot = slotXML.getSection("DegreesMatrix");
386  umatrix33E temp;
387  temp.fill({{0,0,0}});
388 
389  if(subslot.hasOption("xDispl")){
390  input = subslot.get("xDispl");
391  std::stringstream ss(bitpit::utils::string::trim(input));
392  for(int i=0; i<3; ++i) ss>>temp[0][i];
393  }
394 
395  if(subslot.hasOption("yDispl")){
396  input = subslot.get("yDispl");
397  std::stringstream ss(bitpit::utils::string::trim(input));
398  for(int i=0; i<3; ++i) ss>>temp[1][i];
399  }
400 
401  if(subslot.hasOption("zDispl")){
402  input = subslot.get("zDispl");
403  std::stringstream ss(bitpit::utils::string::trim(input));
404  for(int i=0; i<3; ++i) ss>>temp[2][i];
405  }
406 
407  setDegree(temp);
408  };
409 
410  if(slotXML.hasSection("PolyCoefficients")){
411  auto & subslot = slotXML.getSection("PolyCoefficients");
412  dmat33Evec temp = getCoeffs();
413  std::string rootPoly = "Poly";
414  std::string locPoly;
415  int ik,jk;
416  for (int k=0; k<9; ++k){
417  locPoly = rootPoly + std::to_string(k);
418  if(subslot.hasOption(locPoly)){
419  input = subslot.get(locPoly);
420  std::stringstream ss(bitpit::utils::string::trim(input));
421  ik = (int)(k/3);
422  jk = k%3;
423  for(auto & val: temp[ik][jk]) ss>>val;
424  }
425  }
426  setCoeffs(temp);
427  };
428 
429  if(slotXML.hasOption("Origin")){
430  std::string input = slotXML.get("Origin");
431  input = bitpit::utils::string::trim(input);
432  darray3E temp = {{0.0,0.0,0.0}};
433  if(!input.empty()){
434  std::stringstream ss(input);
435  for(auto &val : temp) ss>>val;
436  }
437  setOrigin(temp);
438  };
439 
440  if(slotXML.hasSection("RefSystem")){
441  const bitpit::Config::Section & rfXML = slotXML.getSection("RefSystem");
442  std::string rootAxis = "axis";
443  std::string axis;
444  dmatrix33E temp;
445  temp[0].fill(0.0); temp[0][0] = 1.0;
446  temp[1].fill(0.0); temp[1][1] = 1.0;
447  temp[2].fill(0.0); temp[2][2] = 1.0;
448  for(int i=0; i<3; ++i){
449  axis = rootAxis + std::to_string(i);
450  std::string input = rfXML.get(axis);
451  input = bitpit::utils::string::trim(input);
452  if(!input.empty()){
453  std::stringstream ss(input);
454  for(auto &val : temp[i]) ss>>val;
455  }
456  }
457  setRefSystem(temp);
458  };
459 };

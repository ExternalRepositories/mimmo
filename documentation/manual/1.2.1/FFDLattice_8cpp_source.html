752 darray3E
753 FFDLattice::nurbsEvaluator(darray3E & pointOr){
754 
755  darray3E point = transfToLocal(pointOr);
756  darray3E scaling = getShape()->getScaling();
757 
758  ivector1D knotInterval(3,0);
759  dvector2D BSbasis(3);
760  dvector1D valH(4,0), temp1(4,0),temp2(4,0), zeros(4,0);
761  int uind, vind, wind, index;
762 
763  dvecarr3E* displ = getDisplacements();
764 
765  int i0 = m_mapdeg[0];
766  int i1 = m_mapdeg[1];
767  int i2 = m_mapdeg[2];
768 
769  iarray3E mappedIndex;
770 
771  for(int i=0; i<3; i++){
772  knotInterval[i] = getKnotInterval(point[i],i);
773  BSbasis[i] = basisITS0(knotInterval[i], i, point[i]);
774  }
775 
776  uind = knotInterval[i0] - m_deg[i0];
777  vind = knotInterval[i1] - m_deg[i1];
778  wind = knotInterval[i2] - m_deg[i2];
779 
780  for(int i=0; i<=m_deg[i0]; ++i){
781 
782  mappedIndex[0] = uind+i;
783  temp1 = zeros;
784 
785  for(int j=0; j<=m_deg[i1]; ++j){
786 
787  mappedIndex[1] = vind+j;
788  temp2 = zeros;
789 
790  for(int k=0; k<=m_deg[i2]; ++k){
791 
792  mappedIndex[2] = wind+k;
793  index = accessMapNodes(mappedIndex[0],mappedIndex[1],mappedIndex[2]);
794 
795  for(int intv=0; intv<3; ++intv){
796  temp2[intv] += BSbasis[i2][k]*m_weights[m_intMapDOF[index]]*(*displ)[m_intMapDOF[index]][intv];
797  }
798  temp2[3] += BSbasis[i2][k]*m_weights[index];
799  }
800  for(int intv=0; intv<4; ++intv){
801  temp1[intv] += BSbasis[i1][j]*temp2[intv];
802  }
803 
804  }
805  for(int intv=0; intv<4; ++intv){
806  valH[intv] += BSbasis[i0][i]*temp1[intv];
807  }
808  }
809 
810  darray3E outres;
811  if(isDisplGlobal()){
812  for(int i=0; i<3; ++i){
813  outres[i] = valH[i]/valH[3];
814  }
815 
816  }else{
817  //summing scaled displ in local ref frame;
818  for(int i=0; i<3; ++i){
819  point[i] += valH[i]/(valH[3]*scaling[i]);
820  }
821 
822  //get final displ in global ref frame:
823  outres = transfToGlobal(point) - pointOr;
824  }
825  return(outres);
826 
827 };

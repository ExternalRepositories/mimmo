278 void
279 Lattice::resizeMapDof(){
280  //reallocate your displacement node
281  m_intMapDOF.clear();
282  m_intMapDOF.resize((m_nx+1)*(m_ny+1)*(m_nz+1), -1);
283  ivector1D::iterator itMapBegin = m_intMapDOF.begin();
284  ivector1D::iterator itMap = itMapBegin;
285  ivector1D::iterator itMapEnd = m_intMapDOF.end();
286  bvector1D info;
287  m_np = reduceDimToDOF(m_nx+1,m_ny+1,m_nz+1, info);
288 
289  //set m_intMapDOF
290 
291  int target;
292  int index;
293  ivector1D dummy;
294 
295  int i0,i1,i2;
296  switch(getShapeType()){
297 
298  case ShapeType::CYLINDER :
299  target=0;
300  while(itMap != itMapEnd){
301 
302  *itMap = target;
303  index = std::distance(itMapBegin, itMap);
304  accessPointIndex(index,i0,i1,i2);
305 
306  if(info[0] && i0 == 0){
307  for(int k=0; k<=m_ny;++k){
308  m_intMapDOF[accessPointIndex(i0,k,i2)] = target;
309  }
310  }
311  if(info[1] && i1 == 0){
312  m_intMapDOF[accessPointIndex(i0,m_ny,i2)] = target;
313  }
314 
315  itMap = find(m_intMapDOF.begin(), itMapEnd,-1);
316  target++;
317  }
318  break;
319 
320  case ShapeType::SPHERE :
321 
322  target = 0;
323  while(itMap != itMapEnd){
324 
325  *itMap = target;
326  index = std::distance(itMapBegin, itMap);
327  accessPointIndex(index,i0,i1,i2);
328 
329  if(info[0] && i0 == 0){
330  for(int k1=0; k1<=m_ny;++k1){
331  for(int k2=0; k2<=m_nz; ++k2){
332  m_intMapDOF[accessPointIndex(i0,k1,k2)] = target;
333  }
334  }
335  }
336 
337  if(info[1] && i1 == 0){
338  m_intMapDOF[accessPointIndex(i0,m_ny-1,i2)] = target;
339  }
340 
341  if(info[2] && i2 == 0){
342  for(int k1=0; k1<=m_ny; ++k1){
343  m_intMapDOF[accessPointIndex(i0,k1,i2)] = target;
344  }
345  }
346 
347  if(info[3] && i2 == (m_nz)){
348  for(int k1=0; k1<=m_ny;++k1){
349  m_intMapDOF[accessPointIndex(i0,k1,i2)] = target;
350  }
351  }
352 
353  itMap = find(m_intMapDOF.begin(), itMapEnd,-1);
354  target++;
355  }
356  break;
357 
358 
359  case ShapeType::CUBE :
360  target = 0;
361  while(itMap != itMapEnd){
362 
363  *itMap = target;
364  itMap = find(m_intMapDOF.begin(), itMapEnd,-1);
365  target++;
366  }
367  break;
368 
369  default: //doing nothing
370  break;
371  }//end switch

543 int
544 Lattice::reduceDimToDOF(int nx, int ny, int nz, bvector1D & info){
545 
546  int delta = 0;
547  double dval;
548  switch(getShapeType()){
549 
550  case ShapeType::CYLINDER :
551  delta += nz;
552  nx--;
553  if(getCoordType(1) == CoordType::PERIODIC) ny--;
554 
555  info.push_back(true);
556  info.push_back(getCoordType(1) == CoordType::PERIODIC);
557  break;
558 
559  case ShapeType::SPHERE :
560  delta ++;
561  nx--;
562  if(getCoordType(1) == CoordType::PERIODIC) ny--;
563  dval = getInfLimits()[2];
564  if(dval == 0.0) {
565  nz--;
566  delta += nx;
567  }
568  if((dval + getLocalSpan()[2]) == M_PI){
569  nz--;
570  delta += nx;
571  }
572 
573  info.push_back(true);
574  info.push_back(getCoordType(1) == CoordType::PERIODIC);
575  info.push_back(dval==0.0);
576  info.push_back((dval + getLocalSpan()[2]) == M_PI);
577  break;
578 
579  default:
580  //doing nothing
581  break;
582  }
583 
584  int result = nx*ny*nz + delta;
585  return(result);
586 };

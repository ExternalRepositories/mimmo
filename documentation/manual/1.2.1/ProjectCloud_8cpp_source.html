135 void
136 ProjectCloud::execute(){
137 
138  if (getGeometry() == NULL){
139  throw std::runtime_error (m_name + " : NULL pointer to linked geometry");
140  }
141  if (getGeometry()->isEmpty()){
142  throw std::runtime_error (m_name + " : empty linked geometry");
143  }
144 
145  if(!getGeometry()->isSkdTreeSync()) getGeometry()->buildSkdTree();
146 
147  //project points on surface.
148  int counter = 0;
149  m_proj.resize(m_points.size());
150  for(auto &val : m_points){
151  m_proj[counter]= skdTreeUtils::projectPoint(&val, getGeometry()->getSkdTree());
152  ++counter;
153  }
154  return;
155 };

161 void
162 ProjectCloud::plotOptionalResults(){
163  bitpit::VTKFormat codex = bitpit::VTKFormat::APPENDED;
164 
165  int size = m_proj.size();
166  ivector1D conn(size);
167  for(int i=0; i<size; i++){
168  conn[i] = i;
169  }
170  std::string dir = "./";
171  std::string file = m_name + "_" + std::to_string(getId());
172 
173  bitpit::VTKUnstructuredGrid vtk(dir, file, bitpit::VTKElementType::VERTEX);
174  vtk.setGeomData( bitpit::VTKUnstructuredField::POINTS, m_proj) ;
175  vtk.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, conn) ;
176  vtk.setDimensions(size, size);
177  vtk.setCodex(codex);
178 
179  vtk.write();
180 }

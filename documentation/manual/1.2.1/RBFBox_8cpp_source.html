194 void
195 RBFBox::plot(std::string directory, std::string filename,int counter, bool binary){
196 
197 
198  dvecarr3E activeP(8);
199 
200  activeP[0] = - 0.5 * m_span;
201  activeP[6] = 0.5 * m_span;
202 
203  activeP[1] = activeP[0]; activeP[1][0] += m_span[0];
204  activeP[3] = activeP[0]; activeP[3][1] += m_span[1];
205  activeP[2] = activeP[6]; activeP[2][2] += -1.0*m_span[2];
206 
207  activeP[7] = activeP[6]; activeP[7][0] += -1.0*m_span[0];
208  activeP[5] = activeP[6]; activeP[5][1] += -1.0*m_span[1];
209  activeP[4] = activeP[0]; activeP[4][2] += m_span[2];
210 
211  darray3E temp;
212  dmatrix33E trasp = bitpit::linearalgebra::transpose(m_axes);
213  for(auto &val : activeP){
214 
215 
216  for(int i=0; i<3; ++i){
217  temp[i] = dotProduct(val, trasp[i]);
218  }
219  val = temp + m_origin;
220  }
221 
222  ivector2D activeConn(1);
223  for(int i=0; i<8; ++i) activeConn[0].push_back(i);
224 
225  bitpit::VTKFormat codex = bitpit::VTKFormat::ASCII;
226  if(binary){codex=bitpit::VTKFormat::APPENDED;}
227  bitpit::VTKElementType elDM = bitpit::VTKElementType::HEXAHEDRON;
228  bitpit::VTKUnstructuredGrid vtk(directory, filename, elDM);
229  vtk.setGeomData( bitpit::VTKUnstructuredField::POINTS, activeP) ;
230  vtk.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, activeConn) ;
231  vtk.setDimensions(1, 8);
232  vtk.setCodex(codex);
233  if(counter>=0){vtk.setCounter(counter);}
234 
235  vtk.write();
236 };

243 void
244 RBFBox::execute(){
245 
246  int np = m_nodes.size();
247 
248  darray3E min, max;
249  min.fill(1.0e+18);
250  max.fill(-1.0e+18);
251 
252  for(int i=0; i<np; i++){
253  for(int j=0; j<3; j++){
254  if (m_nodes[i][j] - m_suppR*1.01 < min[j]) min[j] = m_nodes[i][j];
255  if (m_nodes[i][j] + m_suppR*1.01 > max[j]) max[j] = m_nodes[i][j];
256  }
257  }
258 
259  m_span = max - min;
260  m_origin = min + m_span * 0.5;
261 
262 };

187 ReconstructVector::removeData(MimmoObject * patch){
188  //in m_subpatch remove progressively all pierced vector elements in last position
189  //which links towards a geometry of type patch.
190  while(m_subpatch.back().getGeometry() == patch){
191  m_subpatch.pop_back();
192  }
193 
194  //start searching from the begin on element linking patch
195  std::vector<dmpvecarr3E>::iterator it = m_subpatch.begin();
196 
197  while(it != m_subpatch.end()){
198  if (it->getGeometry() == patch){
199  *it = m_subpatch.back();
200  m_subpatch.pop_back();
201  //be always sure the last element is clean and does not link
202  // the patch
203  while(m_subpatch.back().getGeometry() == patch){
204  m_subpatch.pop_back();
205  }
206  }
207  ++it; //forward the iterator.
208  }

238 ReconstructVector::plotData(std::string dir, std::string name, bool flag){
239 
240  if(getGeometry() == NULL) return;
241  if(getGeometry()->isEmpty()) return;
242  if(!m_result.completeMissingData({{0.0,0.0,0.0}})) return;
243 
244  bitpit::VTKLocation loc = bitpit::VTKLocation::POINT;
245  if(m_loc == MPVLocation::CELL){
246  loc = bitpit::VTKLocation::CELL;
247  }
248 
249  dvecarr3E field = m_result.getDataAsVector();
250 
251  if (getGeometry()->getType() != 3){
252  getGeometry()->getPatch()->getVTK().addData("vectorfield", bitpit::VTKFieldType::VECTOR, loc, field);
253  getGeometry()->getPatch()->getVTK().setDataCodex(bitpit::VTKFormat::APPENDED);
254  if(!flag) getGeometry()->getPatch()->getVTK().setDataCodex(bitpit::VTKFormat::ASCII);
255  getGeometry()->getPatch()->write(dir+"/"+name);
256  getGeometry()->getPatch()->getVTK().removeData("vectorfield");
257  }else{
258  liimap mapData;
259  dvecarr3E points = getGeometry()->getVertexCoords(&mapData);
260  ivector2D connectivity;
261  int np = points.size();
262  connectivity.resize(np);
263  for (int i=0; i<np; i++){
264  connectivity[i].resize(1);
265  connectivity[i][0] = i;
266  }
267  bitpit::VTKUnstructuredGrid output(dir,name,bitpit::VTKElementType::VERTEX);
268  output.setGeomData( bitpit::VTKUnstructuredField::POINTS, points);
269  output.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, connectivity);
270  output.setDimensions(connectivity.size(), points.size());
271  output.addData("vectorfield", bitpit::VTKFieldType::VECTOR, loc, field);
272  output.setDataCodex(bitpit::VTKFormat::APPENDED);
273  if(!flag) output.setDataCodex(bitpit::VTKFormat::ASCII);
274  output.write();
275  }

333 void
334 ReconstructVector::execute(){
335 
336  if(getGeometry() == NULL){
337  throw std::runtime_error(m_name + "NULL pointer to linked geometry found");
338  }
339 
340  if(getGeometry()->isEmpty()){
341  throw std::runtime_error(m_name + "empty linked geometry found");
342  }
343 
344  //Overlap fields
345  m_result.clear();
346  m_result.setGeometry(getGeometry());
347  m_result.setDataLocation(m_loc);
348 
349  m_subresults.clear();
350 
351  std::unordered_set<long> idsTarget;
352  {
353  livector1D ids = idsGeoDataLocation(getGeometry());
354  idsTarget.insert(ids.begin(), ids.end());
355  }
356 
357  bitpit::PiercedVector<int> counter;
358  for (int i=0; i<getNData(); i++){
359  dmpvecarr3E* pv = &m_subpatch[i];
360  livector1D ids = idsGeoDataLocation(pv->getGeometry());
361  for (auto ID: ids){
362  if(idsTarget.count(ID)==0) continue;
363 
364  if (!m_result.exists(ID)){
365  m_result.insert(ID, (*pv)[ID]);
366  counter.insert(ID, 1);
367  }
368  else{
369  overlapFields(ID, (*pv)[ID]);
370  counter[ID]++;
371  }
372  }
373  }
374 
375  if (m_result.isEmpty()){
376  (*m_log)<<"Warning in "<<m_name<<". Resulting reconstructed field is empty.This is could be caused by unrelated fields linked geometry and target geometry"<<std::endl;
377  throw std::runtime_error(m_name + "empty field reconstructed in class execution.");
378  }
379 
380  if (m_overlapCriterium == OverlapMethod::AVERAGE){
381  long int ID;
382  auto itend = m_result.end();
383  for (auto it=m_result.begin(); it!=itend; ++it){
384  ID = it.getId();
385  for (int j=0; j<3; j++)
386  (*it)[j] = (*it)[j] / double(counter[ID]);
387  }
388  }
389 
390  //Update field on whole geometry
391  darray3E zero = {{0.0,0.0,0.0}};
392  m_result.completeMissingData(zero);
393 
394  //Create subresults
395  m_subresults.resize(getNData());
396  for (int i=0; i<getNData(); i++){
397  dmpvecarr3E* pv = &m_subpatch[i];
398  m_subresults[i].setGeometry(pv->getGeometry());
399  m_subresults[i].setDataLocation(m_loc);
400  livector1D ids = idsGeoDataLocation(pv->getGeometry());
401  for (auto ID : ids){
402  if(idsTarget.count(ID)>0){
403  m_subresults[i].insert(ID, m_result[ID]);
404  }
405  }
406  }
407 }

429 void
430 ReconstructVector::overlapFields(long int ID, darray3E & locField){
431 
432  switch(m_overlapCriterium){
433  case OverlapMethod::MAX :
434 
435  {
436  double actual = norm2(m_result[ID]);
437  double dummy = norm2(locField);
438 
439  if (actual < dummy)
440  m_result[ID] = locField;
441  }
442  break;
443 
444  case OverlapMethod::MIN :
445  {
446  double actual = norm2(m_result[ID]);
447  double dummy = norm2(locField);
448 
449  if (actual > dummy)
450  m_result[ID] = locField;
451  }
452  break;
453 
454  case OverlapMethod::AVERAGE :
455 
456  m_result[ID] += locField;
457  break;
458 
459  case OverlapMethod::SUM :
460 
461  m_result[ID] += locField;
462  break;
463 
464  default : //never been reached
465  break;
466  }

497 void ReconstructVector::absorbSectionXML(const bitpit::Config::Section & slotXML, std::string name){
498 
499  BITPIT_UNUSED(name);
500 
501  if(slotXML.hasOption("DataLocation")){
502  std::string input = slotXML.get("DataLocation");
503  input = bitpit::utils::string::trim(input);
504  int temp = -1;
505  if(!input.empty()){
506  std::stringstream ss(input);
507  ss>>temp;
508  }
509  if(int(m_loc) != temp){
510  (*m_log)<<"Error absorbing DataLocation in "<<m_name<<". Class and read locations mismatch"<<std::endl;
511  if (temp == 0) (*m_log)<<"XML DataLocation in "<<m_name<<" is set to 0-UNDEFINED"<<std::endl;
512  if (temp == 3) (*m_log)<<"XML DataLocation in "<<m_name<<" is set to 3-INTERFACE, not supported for now."<<std::endl;
513  throw std::runtime_error (m_name + " : xml absorbing failed.");
514  }
515  }
516 
517  //start absorbing
518  BaseManipulation::absorbSectionXML(slotXML, name);
519 
520  if(slotXML.hasOption("OverlapCriterium")){
521  std::string input = slotXML.get("OverlapCriterium");
522  input = bitpit::utils::string::trim(input);
523  int value = 4;
524  if(!input.empty()){
525  std::stringstream ss(input);
526  ss >> value;
527  value = std::min(std::max(1, value),4);
528  }
529  setOverlapCriterium(value);
530  }

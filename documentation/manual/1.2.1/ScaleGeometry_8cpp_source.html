157 void
158 ScaleGeometry::execute(){
159 
160  if (getGeometry() == NULL){
161  throw std::runtime_error (m_name + " : NULL pointer to linked geometry");
162  }
163  if (getGeometry()->isEmpty()){
164  throw std::runtime_error (m_name + " : empty linked geometry");
165  }
166 
167  checkFilter();
168 
169  m_displ.clear();
170  m_displ.setDataLocation(mimmo::MPVLocation::POINT);
171  m_displ.reserve(getGeometry()->getNVertex());
172  m_displ.setGeometry(getGeometry());
173 
174  long ID;
175  darray3E value;
176  //computing centroid
177  int nV = m_geometry->getNVertex();
178  darray3E center = m_origin;
179  if (m_meanP){
180  center.fill(0.0);
181  for (const auto & vertex : m_geometry->getVertices()){
182  center += vertex.getCoords();
183  }
184  center /=double(nV);
185  }
186  for (const auto & vertex : m_geometry->getVertices()){
187  ID = vertex.getId();
188 
189  darray3E coords = vertex.getCoords();
190  value = ( m_scaling*(coords - center) + center ) * m_filter[ID] - coords;
191  m_displ.insert(ID, value);
192  }
193 };

199 ScaleGeometry::apply(){
200 
201  if (getGeometry() == NULL) return;
202  if (getGeometry()->isEmpty() || m_displ.isEmpty()) return;
203  darray3E vertexcoords;
204  long int ID;
205  for (const auto & vertex : m_geometry->getVertices()){
206  vertexcoords = vertex.getCoords();
207  ID = vertex.getId();
208  vertexcoords += m_displ[ID];
209  getGeometry()->modifyVertex(vertexcoords, ID);
210  }

218 void
219 ScaleGeometry::checkFilter(){
220  bool check = m_filter.getDataLocation() == mimmo::MPVLocation::POINT;
221  check = check && m_filter.completeMissingData(0.0);
222  check = check && m_filter.getGeometry() == getGeometry();
223 
224  if (!check){
225  m_log->setPriority(bitpit::log::Verbosity::DEBUG);
226  (*m_log)<<"Not valid filter found in "<<m_name<<". Proceeding with default unitary field"<<std::endl;
227  m_log->setPriority(bitpit::log::Verbosity::NORMAL);
228 
229  m_filter.clear();
230  m_filter.setGeometry(m_geometry);
231  m_filter.setDataLocation(mimmo::MPVLocation::POINT);
232  m_filter.reserve(getGeometry()->getNVertex());
233  for (const auto & vertex : getGeometry()->getVertices()){
234  m_filter.insert(vertex.getId(), 1.0);
235  }
236  }

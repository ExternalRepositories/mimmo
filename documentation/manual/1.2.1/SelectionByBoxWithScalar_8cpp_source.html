153 void
154 SelectionByBoxWithScalar::execute(){
155 
156  SelectionByBox::execute();
157 
158  if(m_field.isEmpty()){
159  (*m_log)<<"warning in "<<m_name<<" : empty scalar field found"<<std::endl;
160  return;
161  }
162  //check m_field in input if coherent with the linked geometry
163  if(m_field.getGeometry() != getGeometry()) {
164  throw std::runtime_error(m_name+" : linked scalar field is not referred to target geometry");
165  }
166  if(getPatch()->getType()==3 && m_field.getDataLocation()!= MPVLocation::POINT){
167  (*m_log)<<"warning in "<<m_name<<" : Attempting to extract a non POINT located field on a Point Cloud target geometry. Do Nothing."<<std::endl;
168  return;
169  }
170 
171  ExtractScalarField * extractorField = new ExtractScalarField();
172  extractorField->setGeometry(getPatch());
173  extractorField->setMode(ExtractMode::ID);
174  extractorField->setField(m_field);
175  extractorField->execute();
176  m_field = extractorField->getExtractedField();
177  delete extractorField;
178 }

185 void
186 SelectionByBoxWithScalar::plotOptionalResults(){
187  if(getPatch() == NULL) return;
188  if(getPatch()->isEmpty()) return;
189  if(m_field.getGeometry() != getPatch()) return;
190 
191  bitpit::VTKLocation loc = bitpit::VTKLocation::UNDEFINED;
192  switch(m_field.getDataLocation()){
193  case MPVLocation::POINT :
194  loc = bitpit::VTKLocation::POINT;
195  break;
196  case MPVLocation::CELL :
197  loc = bitpit::VTKLocation::CELL;
198  break;
199  default:
200  (*m_log)<<"warning: Undefined Reference Location in plotOptionalResults of "<<m_name<<std::endl;
201  (*m_log)<<"Interface or Undefined locations are not supported in VTU writing." <<std::endl;
202  break;
203  }
204  if(loc == bitpit::VTKLocation::UNDEFINED) return;
205 
206  if(getPatch()->getType()==3 && m_field.getDataLocation()!= MPVLocation::POINT){
207  (*m_log)<<"warning in "<<m_name<<" : Attempting to plot a non POINT located field on a Point Cloud target geometry. Do Nothing."<<std::endl;
208  return;
209  }
210 
211  std::string dir = m_outputPlot;
212  std::string name = m_name + "_Patch."+ std::to_string(getId());
213 
214  //check size of field and adjust missing values to zero for writing purposes only.
215  dmpvector1D field_supp = m_field;
216  bool checkField = field_supp.completeMissingData(0.0);
217 
218  if(getPatch()->getType() != 3){
219  if(checkField){
220  dvector1D field = field_supp.getDataAsVector();
221  getPatch()->getPatch()->getVTK().addData("field", bitpit::VTKFieldType::SCALAR, loc, field);
222  }
223  std::string totpath = dir +"/"+name;
224  getPatch()->getPatch()->write(totpath);
225  if(checkField) getPatch()->getPatch()->getVTK().removeData("field");
226  }else{
227  liimap mapDataInv;
228  dvecarr3E points = getPatch()->getVertexCoords(&mapDataInv);
229  ivector2D connectivity;
230  bitpit::VTKElementType cellType = bitpit::VTKElementType::VERTEX;
231 
232  int np = points.size();
233  connectivity.resize(np);
234  for (int i=0; i<np; i++){
235  connectivity[i].resize(1);
236  connectivity[i][0] = i;
237 
238  }
239  bitpit::VTKUnstructuredGrid output(dir,name,cellType);
240  output.setGeomData( bitpit::VTKUnstructuredField::POINTS, points);
241  output.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, connectivity);
242  output.setDimensions(connectivity.size(), points.size());
243  output.setCodex(bitpit::VTKFormat::APPENDED);
244  if(checkField){
245  dvector1D field = field_supp.getDataAsVector();
246  output.addData("field", bitpit::VTKFieldType::SCALAR, loc, field);
247  }
248  output.write();
249  }
250 }

80 double signedDistance(std::array<double,3> *P_, bitpit::PatchSkdTree *bvtree_, long &id, std::array<double,3> &n, double &r)
81 {
82 
83  double h;
84  //static_cast<bitpit::SurfaceSkdTree*>(bvtree_)->findPointClosestCell(*P_, r, &id, &h);
85  findPointClosestCell(*P_, *bvtree_, r, &id, &h);
86  const bitpit::PatchKernel &patch = (bvtree_->getPatch());
87  const bitpit::SurfUnstructured &spatch = static_cast<const bitpit::SurfUnstructured &>(patch);
88 
89  //signed distance only for 2D element patches(quads, pixels, triangles or segments)
90  const bitpit::Cell & cell = spatch.getCell(id);
91  bitpit::ConstProxyVector<long> vertIds = cell.getVertexIds();
92  dvecarr3E VS(vertIds.size());
93  int count = 0;
94  for (const auto & iV: vertIds){
95  VS[count] = spatch.getVertexCoords(iV);
96  ++count;
97  }
98 
99  darray3E xP = {{0.0,0.0,0.0}};
100  darray3E normal= {{0.0,0.0,0.0}};
101 
102  if ( vertIds.size() == 3 ){ //TRIANGLE
103  darray3E lambda;
104  h = bitpit::CGElem::distancePointTriangle((*P_), VS[0], VS[1], VS[2],lambda);
105  int count = 0;
106  for(const auto &val: lambda){
107  normal += val * spatch.evalVertexNormal(id,count) ;
108  xP += val * VS[count];
109  ++count;
110  }
111  }else if ( vertIds.size() == 2 ){ //LINE/SEGMENT
112  darray2E lambda;
113  h = bitpit::CGElem::distancePointSegment((*P_), VS[0], VS[1], lambda);
114  int count = 0;
115  for(const auto &val: lambda){
116  normal += val * spatch.evalVertexNormal(id,count) ;
117  xP += val * VS[count];
118  ++count;
119  }
120  }else{ //GENERAL POLYGON
121  std::vector<double> lambda;
122  h = bitpit::CGElem::distancePointPolygon((*P_), VS,lambda);
123  int count = 0;
124  for(const auto &val: lambda){
125  normal += val * spatch.evalVertexNormal(id,count) ;
126  xP += val * VS[count];
127  ++count;
128  }
129  }
130 
131  double s = sign( dotProduct(normal, (*P_) - xP) );
132  h = s * h;
133  //pseudo-normal (direction P and xP closest point on triangle)
134  n = s * ((*P_) - xP);
135  double normX = norm2(n);
136  if(normX < 1.E-15){
137  n = normal/norm2(normal);
138  }else{
139  n /= norm2(n);
140  }
141  return h;
142 
143 }

202 void extractTarget(bitpit::PatchSkdTree *target, std::vector<const bitpit::SkdNode*> leafSelection, std::vector<long> &extracted, double tol, int next){
203 
204  bool check = false;
205  std::vector<const bitpit::SkdNode*> tocheck;
206  for (int i=0; i<(int)leafSelection.size(); i++){
207  if (bitpit::CGElem::intersectBoxBox(leafSelection[i]->getBoxMin()-tol,
208  leafSelection[i]->getBoxMax()+tol,
209  target->getNode(next).getBoxMin(),
210  target->getNode(next).getBoxMax() ) ){
211  check = true;
212  tocheck.push_back(leafSelection[i]);
213  }
214  }
215 
216  leafSelection.clear();
217  leafSelection = tocheck;
218 
219  if (check){
220  bitpit::SkdNode node = target->getNode(next);
221  if (node.isLeaf()){
222  std::vector<long> cellids = node.getCells();
223  extracted.insert(extracted.end(),cellids.begin(), cellids.end());
224  }
225  else{
226  for (int i = bitpit::SkdNode::CHILD_BEGIN; i != bitpit::SkdNode::CHILD_END; ++i) {
227  bitpit::SkdNode::ChildLocation childLocation = static_cast<bitpit::SkdNode::ChildLocation>(i);
228  if(node.getChildId(childLocation) != bitpit::SkdNode::NULL_ID) {
229  extractTarget(target, leafSelection, extracted, tol, node.getChildId(childLocation));
230  }
231  }
232  }
233  }

250 darray3E projectPoint( std::array<double,3> *P_, bitpit::PatchSkdTree *bvtree_, double r )
251 {
252 
253  long id;
254  darray3E normal;
255 
256  double dist = 1.0e+18;
257  while (std::abs(dist) >= 1.0e+18){
258  //use method sphere by default
259  dist = signedDistance(P_, bvtree_, id, normal, r);
260  r *= 1.5;
261  }
262 
263  darray3E projP = (*P_) - dist*normal;
264 
265  return (projP);

285 long findPointClosestCell(const std::array<double, 3> &point, bitpit::PatchSkdTree &tree, double maxDistance,long *id, double *distance)
286 {
287  // Initialize the cell id
288  *id = Cell::NULL_ID;
289 
290  // Initialize the distance with an estimate
291  //
292  // The real distance will be lesser than or equal to the estimate.
293  std::size_t rootId = 0;
294  const SkdNode &root = tree.getNode(rootId);
295  *distance = std::min(root.evalPointMaxDistance(point), maxDistance);
296 
297  // Get a list of candidates nodes
298  //
299  std::vector<std::size_t> m_candidateIds;
300  std::vector<double > m_candidateMinDistances;
301 
302  std::vector<std::size_t> nodeStack;
303  nodeStack.push_back(rootId);
304  while (!nodeStack.empty()) {
305  std::size_t nodeId = nodeStack.back();
306  const SkdNode &node = tree.getNode(nodeId);
307  nodeStack.pop_back();
308 
309  // Do not consider nodes with a minimum distance greater than
310  // the distance estimate
311  double nodeMinDistance = node.evalPointMinDistance(point);
312  if (nodeMinDistance > *distance) {
313  continue;
314  }
315 
316  // Update the distance estimate
317  //
318  // The real distance will be lesser than or equal to the
319  // estimate.
320  double nodeMaxDistance = node.evalPointMaxDistance(point);
321  *distance = std::min(nodeMaxDistance, *distance);
322 
323  // If the node is a leaf add it to the candidates, otherwise
324  // add its children to the stack.
325  bool isLeaf = true;
326  for (int i = SkdNode::CHILD_BEGIN; i != SkdNode::CHILD_END; ++i) {
327  SkdNode::ChildLocation childLocation = static_cast<SkdNode::ChildLocation>(i);
328  std::size_t childId = node.getChildId(childLocation);
329  if (childId != SkdNode::NULL_ID) {
330  isLeaf = false;
331  nodeStack.push_back(childId);
332  }
333  }
334 
335  if (isLeaf) {
336  m_candidateIds.push_back(nodeId);
337  m_candidateMinDistances.push_back(nodeMinDistance);
338  }
339  }
340 
341  // Process the candidates and find the closest cell
342  long nDistanceEvaluations = 0;
343  for (std::size_t k = 0; k < m_candidateIds.size(); ++k) {
344  // Do not consider nodes with a minimum distance greater than
345  // the distance estimate
346  if (m_candidateMinDistances[k] > *distance) {
347  continue;
348  }
349 
350  // Evaluate the min distance between all cells in the candidate node
351  std::size_t nodeId = m_candidateIds[k];
352  const SkdNode &node = tree.getNode(nodeId);
353 
354  bitpit::ElementType eltype ;
355  double cellDistance = maxDistance;
356 
357  for(const auto & cellId : node.getCells() ){
358  eltype = tree.getPatch().getCell(cellId).getType();
359  auto vList = tree.getPatch().getCell(cellId).getVertexIds();
360  dvecarr3E vertCoords(vList.size());
361  int counterList = 0;
362  for(const auto & idV : vList){
363  vertCoords[counterList] = tree.getPatch().getVertexCoords(idV);
364  ++counterList;
365  }
366  switch(eltype){
367  case ElementType::LINE:
368  cellDistance = CGElem::distancePointSegment(point, vertCoords[0], vertCoords[1]);
369  break;
370  case ElementType::TRIANGLE:
371  cellDistance = CGElem::distancePointTriangle(point, vertCoords[0], vertCoords[1], vertCoords[2]);
372  break;
373  case ElementType::PIXEL:
374  case ElementType::QUAD:
375  case ElementType::POLYGON:
376  cellDistance = mimmoCGUtils::distancePointPolygon(point, vertCoords);
377  break;
378  default:
379  throw std::runtime_error("Not supported cell type");
380  break;
381  }
382  if (cellDistance < *distance) {
383  *id = cellId;
384  *distance = cellDistance;
385  }
386  ++nDistanceEvaluations;
387  }
388  }
389 
390  // If no closest cell was found set the distance to the maximum
391  // representable distance.
392  if (*id == Cell::NULL_ID) {
393  *distance = std::numeric_limits<double>::max();
394  }
395  return nDistanceEvaluations;
396 }

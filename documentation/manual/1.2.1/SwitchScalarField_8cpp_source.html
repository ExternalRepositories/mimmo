166 void
167 SwitchScalarField::plotOptionalResults(){
168 
169  if (m_result.size() == 0 || getGeometry() == NULL) return;
170  if (getGeometry()->isEmpty()) return;
171 
172  bitpit::VTKLocation loc = bitpit::VTKLocation::UNDEFINED;
173  switch(m_loc){
174  case MPVLocation::POINT :
175  loc = bitpit::VTKLocation::POINT;
176  break;
177  case MPVLocation::CELL :
178  loc = bitpit::VTKLocation::CELL;
179  break;
180  default:
181  (*m_log)<<"Warning: Undefined Reference Location in plotOptionalResults of "<<m_name<<std::endl;
182  (*m_log)<<"Interface locations are not supported in VTU writing." <<std::endl;
183  break;
184  }
185 
186  if(loc == bitpit::VTKLocation::UNDEFINED) return;
187 
188  //check size of field and adjust missing values to zero for writing purposes only.
189  dmpvector1D field_supp = m_result;
190  if(!field_supp.completeMissingData(0.0)) return;
191  dvector1D field = field_supp.getDataAsVector();
192 
193  if(getGeometry()->getType() != 3){
194  getGeometry()->getPatch()->getVTK().addData("field", bitpit::VTKFieldType::SCALAR, loc, field);
195  getGeometry()->getPatch()->write(m_outputPlot+"/"+m_name+std::to_string(getId()));
196  getGeometry()->getPatch()->getVTK().removeData("field");
197  }else{
198  if(loc == bitpit::VTKLocation::CELL){
199  (*m_log)<<"Warning: Attempt writing Cell data field on cloud point in plotOptionalResults of "<<m_name<<std::endl;
200  return;
201  }
202  liimap mapDataInv;
203  dvecarr3E points = getGeometry()->getVertexCoords(&mapDataInv);
204  int size = points.size();
205  ivector2D connectivity(size, ivector1D(1));
206  for(int i=0; i<size; ++i) connectivity[i][0]=i;
207  bitpit::VTKUnstructuredGrid output(m_outputPlot,m_name+std::to_string(getId()), bitpit::VTKElementType::VERTEX);
208  output.setGeomData( bitpit::VTKUnstructuredField::POINTS, points);
209  output.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, connectivity);
210  output.setDimensions(connectivity.size(), points.size());
211  output.addData("field", bitpit::VTKFieldType::SCALAR, loc, field);
212  output.setCodex(bitpit::VTKFormat::APPENDED);
213  output.write();
214  }

223 bool
224 SwitchScalarField::mswitch(){
225 
226  if (getGeometry() == NULL) return false;
227 
228  m_result.clear();
229 
230  //Extract by link to geometry
231  for (const auto & field : m_fields){
232  if (field.getGeometry() == getGeometry()){
233  m_result = field;
234  //geometry and location are copied from field.
235  return true;
236  }
237  }
238 
239  //Extract by geometric mapping if active and if no positive match is found.
240  if (m_mapping && getGeometry()->getType() != 3){
241 
242  m_result.setGeometry(getGeometry());
243  m_result.setDataLocation(m_loc);
244 
245  ExtractScalarField * ef = new ExtractScalarField();
246  ef->setGeometry(getGeometry());
247  ef->setMode(ExtractMode::MAPPING);
248  ef->setTolerance(m_tol);
249 
250  //create map for overlapping ids purpose;
251  std::unordered_map<long, int> idRepetition;
252 
253  for (const auto & field : m_fields){
254  ef->setField(field);
255  bool check = ef->extract();
256  if(!check) continue;
257 
258  auto temp = ef->getExtractedField();
259  dmpvector1D::iterator itB;
260  auto itE = temp.end();
261  long id;
262  for(itB = temp.begin(); itB != itE; ++itB){
263  id = itB.getId();
264  if(!m_result.exists(id)){
265  m_result.insert(id, *itB);
266  }else{
267  m_result[id] += *itB;
268  if(idRepetition.count(id)>0){
269  ++idRepetition[id];
270  }else{
271  idRepetition[id] = 2;
272  }
273  }
274  }
275  }
276 
277  //resolve overlapping ids by averaging correspondent value;
278 
279  for(auto &itval : idRepetition){
280  m_result[itval.first] /= double(itval.second);
281  }
282 
283  delete ef;
284  }
285 
286  if (m_result.isEmpty()) return false;
287  return true;
288 }

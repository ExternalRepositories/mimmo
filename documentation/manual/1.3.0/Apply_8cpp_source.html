205 void
206 Apply::checkInput(){
207 
208  bool check = false;
209  if (m_input.size()){
210  check = m_input.getDataLocation() == mimmo::MPVLocation::POINT;
211  check = check && m_input.getGeometry() == m_geometry;
212  check = check && m_input.completeMissingData({{0.0,0.0,0.0}});
213  }
214  if (m_scalarinput.size()){
215  check = m_scalarinput.getDataLocation() == mimmo::MPVLocation::POINT;
216  check = check && m_geometry->getType() == 1;
217  check = check && m_scalarinput.getGeometry() == m_geometry;
218  check = check && m_scalarinput.completeMissingData(0.0);
219  if (check){
220  bitpit::SurfaceKernel* skernel = static_cast<bitpit::SurfaceKernel*>(m_geometry->getPatch());
221  bitpit::PiercedVector<darray3E> vNormals;
222  bitpit::ConstProxyVector<long> verts;
223  std::size_t size;
224  long idN;
225  for(const auto & cell: m_geometry->getCells()){
226  verts = cell.getVertexIds();
227  size = verts.size();
228  for(std::size_t i=0; i<size; ++i){
229  idN = verts[i];
230  if(!vNormals.exists(idN)){
231  vNormals.insert(idN, skernel->evalVertexNormal(cell.getId(), i));
232  }
233  }
234  }
235  m_input.clear();
236  for(const auto & vertex: m_geometry->getVertices()){
237  m_input.setDataLocation(mimmo::MPVLocation::POINT);
238  m_input.setGeometry(m_geometry);
239  long id = vertex.getId();
240  m_input.insert(id, darray3E(m_scalarinput[id]*vNormals[id]));
241  }
242  }
243  }
244  if (!check){
245  m_log->setPriority(bitpit::log::Verbosity::DEBUG);
246  (*m_log)<<"Not valid input found in "<<m_name<<". Proceeding with default zero field"<<std::endl;
247  m_log->setPriority(bitpit::log::Verbosity::NORMAL);
248  m_input.clear();
249  m_input.setGeometry(m_geometry);
250  m_input.setDataLocation(mimmo::MPVLocation::POINT);
251  m_input.reserve(m_geometry->getNVertices());
252  for (const auto & vertex : m_geometry->getVertices()){
253  m_input.insert(vertex.getId(), {{0.0,0.0,0.0}});
254  }
255  }
256 };
257 
258 }

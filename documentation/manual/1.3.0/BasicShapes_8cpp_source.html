142 void BasicShape::setRefSystem(darray3E axis0, darray3E axis1, darray3E axis2){
143 
144  double a = norm2(axis0);
145  double b = norm2(axis1);
146  double c = norm2(axis2);
147 
148  bool check = ( a>0.0 && b >0.0 && c > 0.0);
149  if (! check){
150  assert(false && "one or more 0-vector passed as arguments in BasicShape::SetRefSystem method");
151  return;
152  }
153 
154  axis0 /= a;
155  axis1 /= b;
156  axis2 /= c;
157 
158  double tol = 1.0e-3, val;
159  val = std::abs(dotProduct(axis0,axis1)) + std::abs(dotProduct(axis1,axis2)) + std::abs(dotProduct(axis0,axis2));
160 
161  if(val > tol) {
162  assert(false && "not orthogonal axes passed as arguments in BasicShape::SetRefSystem method");
163  return;
164  }
165 
166  m_sdr[0] = axis0;
167  m_sdr[1] = axis1;
168  m_sdr[2] = axis2;
169 
170  m_sdr_inverse = inverse(m_sdr);
171 }

178 void BasicShape::setRefSystem(int label, darray3E axis){
179 
180  if(label <0 || label >2 ){
181  assert(false && "no correct label passed as argument in BasicShape::setRefSystem");
182  return;
183  }
184 
185  double a = norm2(axis);
186  if (!(a > 0.0) ){
187  assert(false && "0-vector passed as axis argument in BasicShape::SetRefSystem method");
188  return;
189  }
190 
191  m_sdr[label] = axis/a;
192  dvecarr3E point_mat(3,darray3E{0,0,0});
193  point_mat[0][0] = point_mat[1][1]= point_mat[2][2]=1.0;
194 
195  int next_label = (label + 1)%3;
196  int fin_label = (label + 2)%3;
197 
198  double pj = dotProduct(point_mat[next_label], m_sdr[label]);
199  m_sdr[next_label] = point_mat[next_label] - pj*m_sdr[label];
200  m_sdr[next_label] = m_sdr[next_label]/norm2(m_sdr[next_label]);
201 
202  m_sdr[fin_label] = crossProduct(m_sdr[label],m_sdr[next_label]);
203  m_sdr[fin_label] = m_sdr[fin_label]/norm2(m_sdr[fin_label]);
204 
205  m_sdr_inverse = inverse(m_sdr);
206 }

626 void BasicShape::searchKdTreeMatches(bitpit::KdTree<3,bitpit::Vertex,long> & tree, livector1D & result, bool squeeze ){
627 
628  //1st step get data
629  std::vector<int> candidates;
630  std::vector< std::pair<int, int> > nodeStackLoc; //cointains touple with kdNode id and level label.
631 
632  nodeStackLoc.push_back(std::make_pair(0, 0));
633  while(!nodeStackLoc.empty()){
634  std::pair<int, int> toupleId = nodeStackLoc.back();
635  bitpit::KdNode<bitpit::Vertex, long> & target = tree.nodes[toupleId.first];
636  nodeStackLoc.pop_back();
637 
638  //add children to the stack, if node has any of them.
639  uint32_t check = intersectShapePlane((toupleId.second)%3, target.object_->getCoords());
640  if((check%2 == 0) && (target.lchild_ >= 0)){
641  nodeStackLoc.push_back(std::make_pair(target.lchild_, (toupleId.second) + 1));
642  }
643  if((check > 0) && (target.rchild_ >= 0)){
644  nodeStackLoc.push_back(std::make_pair(target.rchild_, (toupleId.second) + 1));
645  }
646 
647  //push node as candidate if is in the raw bounding box of the shape
648  if (bitpit::CGElem::intersectPointBox(target.object_->getCoords(), m_bbox[0], m_bbox[1], 3)){
649  candidates.push_back(toupleId.first);
650  }
651  }
652 
653  result.clear();
654  result.reserve(candidates.size());
655  for (const auto & idCand : candidates){
656  bitpit::KdNode<bitpit::Vertex, long> & target = tree.nodes[idCand];
657  if(isPointIncluded(target.object_->getCoords())){
658  result.push_back(target.label);
659  }
660  }
661  if (squeeze)
662  result.shrink_to_fit();
663 };

674 void BasicShape::searchBvTreeMatches(bitpit::PatchSkdTree & tree, bitpit::PatchKernel * geo, livector1D & result, bool squeeze){
675 
676  std::size_t rootId = 0;
677 
678  std::vector<std::size_t> toBeCandidates;
679  toBeCandidates.reserve(geo->getCellCount());
680 
681  std::vector<size_t> nodeStack;
682  nodeStack.push_back(rootId);
683  while(!nodeStack.empty()){
684  std::size_t nodeId = nodeStack.back();
685  const bitpit::SkdNode & node = tree.getNode(nodeId);
686  nodeStack.pop_back();
687 
688  //second step: if the current node AABB does not intersect or does not completely contains the Shape, then thrown it away and continue.
689  if(!intersectShapeAABBox(node.getBoxMin(), node.getBoxMax()) ){
690  continue;
691  }
692 
693  //now the only option is to visit node's children. If node is not leaf, add children to the stack,
694  //otherwise add current node as a possible candidate in shape inclusion.
695  bool isLeaf = true;
696  for (int i = bitpit::SkdNode::CHILD_BEGIN; i != bitpit::SkdNode::CHILD_END; ++i) {
697  bitpit::SkdNode::ChildLocation childLocation = static_cast<bitpit::SkdNode::ChildLocation>(i);
698  std::size_t childId = node.getChildId(childLocation);
699  if (childId != bitpit::SkdNode::NULL_ID) {
700  isLeaf = false;
701  nodeStack.push_back(childId);
702  }
703  }
704  if (isLeaf) {
705  toBeCandidates.push_back(nodeId);
706  }
707  }
708 
709  result.clear();
710  result.reserve(toBeCandidates.size());
711  for (const auto & idCand : toBeCandidates){
712  const bitpit::SkdNode &node = tree.getNode(idCand);
713  std::vector<long> cellids = node.getCells();
714  for(long id : cellids){
715  if(isSimplexIncluded(geo, id)){
716  result.push_back(id);
717  }
718  }
719  }
720  if (squeeze)
721  result.shrink_to_fit();
722 };

769 dmatrix33E BasicShape::inverse(const dmatrix33E & mat){
770  dmatrix33E out;
771 
772  double det = mat[0][0] * (mat[1][1]*mat[2][2] - mat[2][1]*mat[1][2]) -
773  mat[0][1] * (mat[1][0]*mat[2][2] - mat[2][0]*mat[1][2]) +
774  mat[0][2] * (mat[1][0]*mat[2][1] - mat[2][0]*mat[1][1]);
775 
776  out[0][0] = (mat[1][1]*mat[2][2] - mat[2][1]*mat[1][2])/det;
777  out[0][1] = (mat[0][2]*mat[2][1] - mat[2][2]*mat[0][1])/det;
778  out[0][2] = (mat[0][1]*mat[1][2] - mat[1][1]*mat[0][2])/det;
779  out[1][0] = (mat[1][2]*mat[2][0] - mat[2][2]*mat[1][0])/det;
780  out[1][1] = (mat[0][0]*mat[2][2] - mat[2][0]*mat[0][2])/det;
781  out[1][2] = (mat[0][2]*mat[1][0] - mat[1][2]*mat[0][0])/det;
782  out[2][0] = (mat[1][0]*mat[2][1] - mat[2][0]*mat[1][1])/det;
783  out[2][1] = (mat[0][1]*mat[2][0] - mat[2][1]*mat[0][0])/det;
784  out[2][2] = (mat[0][0]*mat[1][1] - mat[1][0]*mat[0][1])/det;
785 
786  return out;
787 }

868 darray3E Cube::toWorldCoord(const darray3E &point){
869 
870  darray3E work, work2;
871 
872  //unscale your local point
873  for(int i =0; i<3; ++i){
874  work[i] = point[i]*m_scaling[i];
875  }
876 
877  //return to local xyz system
878  // -> cube, doing nothing
879 
880  //unapply change to local sdr transformation
881  for(int i=0; i<3; ++i){
882  work2[i] = dotProduct(work, m_sdr_inverse[i]);
883  }
884 
885  //unapply origin translation
886  return(work2 + m_origin);
887 };

895 darray3E Cube::toLocalCoord(const darray3E &point){
896 
897  darray3E work, work2;
898 
899  //unapply origin translation
900  work = point - m_origin;
901 
902  //apply change to local sdr transformation
903  for(int i=0; i<3; ++i){
904  work2[i] = dotProduct(work, m_sdr[i]);
905  }
906 
907  //get to proper local system
908  // -> cube, doing nothing
909 
910  //scale your local point
911  for(int i =0; i<3; ++i){
912  work[i] = work2[i]/m_scaling[i];
913  }
914 
915  return(work);
916 
917 };

289 void
290 ControlDeformExtSurface::execute(){
291 
292  MimmoObject * geo = getGeometry();
293  if(geo == NULL){
294  (*m_log)<<m_name + " : NULL pointer to linked geometry found"<<std::endl;
295  throw std::runtime_error(m_name + "NULL pointer to linked geometry found");
296 
297  }
298  if(geo->isEmpty()){
299  (*m_log)<<m_name + " : empty linked geometry found"<<std::endl;
300  return;
301  }
302 
303  bool check = m_defField.getGeometry() == geo;
304  check = check && m_defField.getDataLocation()== MPVLocation::POINT;
305  check = check && m_defField.completeMissingData({{0.0,0.0,0.0}});
306  if(!check){
307  (*m_log)<<"Error in "<<m_name<<": Unsuitable deformation field linked"<<std::endl;
308  throw std::runtime_error (m_name + " : Unsuitable deformation field linked.");
309  }
310 
311 
312  int nDFS = m_defField.size();
313  m_violationField.clear();
314 
315  long int ID;
316  for (const auto & v : geo->getVertices()){
317  ID = v.getId();
318  m_violationField.insert(ID, -1.0e+18);
319  }
320 
321  dvector1D violationField;
322  violationField.resize(nDFS);
323 
324  dvecarr3E points = geo->getVerticesCoords();
325  dvecarr3E pointsOR = points;
326 
327  //evaluate deformable geometry barycenter************************
328  darray3E geoBary = {{0.0,0.0,0.0}};
329  double distBary = 0.0;
330 
331  for(const auto & p: points) geoBary += p;
332 
333  geoBary /= (double)geo->getNVertices();
334 
335  for(const auto & p: points){
336  distBary= std::fmax(distBary,norm2(p - geoBary));
337  }
338  //***************************************************************
339 
340  //adding deformation to points **********************************
341  int count = 0;
342  for (const auto & v : geo->getVertices()){
343  ID = v.getId();
344  points[count] +=m_defField[ID];
345  ++count;
346  }
347 
348  //***************************************************************
349 
350  //read external surfaces*****************************************
351  std::vector<std::unique_ptr<MimmoGeometry> > extgeo;
352  dvector1D tols;
353  readGeometries(extgeo, tols);
354  //***************************************************************
355 
356  if(extgeo.size() < 1) {
357 // throw std::runtime_error (m_name + " : read of all external geometries failed. Empty valid constraint geometries list.");
358  return;
359  }
360 
361  //evaluating bounding box of deformation*************************
362  darray3E bbMinDef = {{0,0,0}}, bbMaxDef={{0,0,0}};
363  darray3E bbMin, bbMax;
364  {
365  int count2 = 0;
366  for(const auto & p : points){
367  if(count2 == 0) {
368  bbMinDef = p;
369  bbMaxDef = p;
370  }else{
371  for(int i=0; i<3; ++i ){
372  bbMinDef[i] = std::fmin(bbMinDef[i], p[i]);
373  bbMaxDef[i] = std::fmax(bbMaxDef[i], p[i]);
374  }
375  }
376  count2++;
377  }
378  count2=0;
379  for(const auto & p : pointsOR){
380  for(int i=0; i<3; ++i ){
381  bbMinDef[i] = std::fmin(bbMinDef[i], p[i]);
382  bbMaxDef[i] = std::fmax(bbMaxDef[i], p[i]);
383  }
384  count2++;
385  }
386  }
387  //***************************************************************
388  int counterExtGeo =0;
389  // start examining all external geometries***********************
390  for(const auto &gg : extgeo){
391 
392  //check constraints properties ******************************
393  MimmoObject * local = gg->getGeometry();
394  if(!(local->isSkdTreeSync())) local->buildSkdTree();
395  bool checkOpen = local->isClosedLoop();
396 
397  double dist;
398  double radius, radius_old;
399  long id;
400  darray3E normal;
401  int count;
402  //************************************************************
403 
404  //add local constraints to the bounding box compute **********
405  bbMin = bbMinDef;
406  bbMax = bbMaxDef;
407 
408  for(const auto & p : local->getVerticesCoords()){
409  for(int i=0; i<3; ++i ){
410  bbMin[i] = std::fmin(bbMin[i], p[i]);
411  bbMax[i] = std::fmax(bbMax[i], p[i]);
412  }
413  }
414  //*************************************************************
415 
416  //Evaluate background grid dimension for Level set ************
417  //calculate dh as a tot part of bb diagonal *******************
418 
419  darray3E span = bbMax - bbMin;
420  bbMin += -1.0*(0.05*span);
421  span *= 1.1;
422  double dh = norm2(span)/(double)m_cellBackground;
423  iarray3E dim;
424  for(int i=0; i<3; ++i){
425  dim[i] = (int)(span[i]/dh + 0.5);
426  dim[i] = std::max(2, dim[i]);
427  }
428  //*************************************************************
429 
430  //check if its more convenient evaluate distances with a direct BvTree
431  //evaluation of distance on target deformation points or using a background grid +
432  //interpolation.
433  double nReq = double(dim[0]+1)*double(dim[1]+1)*double(dim[2]+1);
434  double nAva = 0.8*nDFS;
435 
436  if( nReq > nAva){
437 
438  //going to use direct evaluation.
439 
440  radius = distBary;
441  radius_old = radius;
442  dvector1D refsigns(nDFS, 1.0);
443 
444  //get the actual sign of distance of the undeformed cloud w.r.t constraints
445  if(checkOpen){
446  count = 0;
447  for(auto &p : pointsOR){
448  dist = evaluateSignedDistance(p, local, id, normal, radius);
449  if(radius < 1.0E-12) radius = radius_old;
450  else radius_old = radius;
451  if(dist < 0.0) refsigns[count] = -1.0;
452  count++;
453  }
454  }else{
455  dist = evaluateSignedDistance(geoBary, local, id, normal, radius);
456 
457  if(dist < 0.0) refsigns *= -1.0;
458  if(radius <1.0E-12) radius = radius_old;
459  }
460 
461  //evaluate distance of the deformation cloud w.r.t. constraints
462  count = 0;
463  for(auto &p : points){
464  dist = evaluateSignedDistance(p, local, id, normal, radius);
465  if(radius < 1.0E-12) radius = radius_old;
466  else radius_old = radius;
467  violationField[count] = -1.0*refsigns[count]*dist;
468  count++;
469  }
470 
471  }else{
472 
473  //going to use background grid to evaluate distances
474 
475  //instantiate a VolCartesian;
476  bitpit::VolCartesian * mesh = new bitpit::VolCartesian(3,bbMin,span, dim);
477  mesh->update();
478 
479  //calculate distance on point of background grid.
480  dvecarr3E background_points(mesh->getVertexCount());
481  int count = 0;
482  for(const auto & v: mesh->getVertices()){
483  background_points[count] = v.getCoords();
484  count++;
485  }
486  dvector1D background_LS(background_points.size());
487 
488  radius = 0.5*norm2(span);
489  radius_old = radius;
490  count = 0;
491  for(auto &p : background_points){
492  dist = evaluateSignedDistance(p, local, id, normal, radius);
493  if(radius < 1.0E-12) radius = radius_old;
494  else radius_old = radius;
495  background_LS[count] = dist;
496  count++;
497  }
498 
499  //interpolate on background to obtain distances
500 
501  //evaluate sign of distances of the undeformed cloud w.r.t to constraints.
502  dvector1D refsigns(nDFS, 1.0);
503  if(checkOpen){
504  count= 0;
505  for(auto & p : pointsOR){
506  ivector1D stencil;
507  dvector1D weights;
508  int res = mesh->linearVertexInterpolation(p, stencil,weights);
509 
510  dist = 0.0;
511  for(int j=0; j<res; ++j){
512  dist += background_LS[stencil[j]]*weights[j];
513  }
514 
515  if(dist < 0.0) refsigns[count] = -1.0;
516  ++count;
517  }
518  }else{
519  dist = evaluateSignedDistance(geoBary, local, id, normal, radius);
520  if(dist< 0) refsigns *= -1.0;
521  if(radius < 1.0E-12) radius = radius_old;
522  }
523 
524  //evaluate violation field
525  count = 0;
526  for(auto & p : points){
527  ivector1D stencil;
528  dvector1D weights;
529 
530  int res = mesh->linearVertexInterpolation(p, stencil,weights);
531 
532  violationField[count] = 0.0;
533  for(int j=0; j<res; ++j){
534  violationField[count] += background_LS[stencil[j]]*weights[j];
535  }
536 
537  violationField[count] *= -1.0*refsigns[count];
538  ++count;
539 
540  }
541  delete mesh;
542  mesh = NULL;
543  }//end if else
544 
545  int ii = 0;
546  for(auto & val : m_violationField){
547  val = std::fmax(val, (violationField[ii] + tols[counterExtGeo]));
548  ++ii;
549  }
550  ++counterExtGeo;
551  }
552 
553  m_violationField.setGeometry(getGeometry());
554  m_violationField.setDataLocation(MPVLocation::POINT);
555 
556  writeLog();
557 
558 };

739 double
740 ControlDeformExtSurface::evaluateSignedDistance(darray3E &point, mimmo::MimmoObject * geo, long & id, darray3E & normal, double &initRadius){
741 
742  double dist = 1.0E+18;
743  double rate = 0.02;
744  int kmax = 1000;
745  int kiter = 0;
746  bool flag = true;
747 
748  if(!geo->isSkdTreeSync()) geo->buildSkdTree();
749 
750  while(flag && kiter < kmax){
751  dist = skdTreeUtils::signedDistance(&point, geo->getSkdTree(), id, normal, initRadius);
752  flag = (dist == 1.0E+18);
753  if(flag) initRadius *= (1.0+ rate*((double)flag));
754  kiter++;
755  }
756 
757  return dist;
758 }

764 void

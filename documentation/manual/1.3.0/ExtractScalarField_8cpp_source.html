152 void
153 ExtractScalarField::plotOptionalResults(){
154 
155  if (m_result.size() == 0 || m_result.getGeometry() == NULL) return;
156 
157  bitpit::VTKLocation loc = bitpit::VTKLocation::UNDEFINED;
158  switch(m_result.getDataLocation()){
159  case MPVLocation::POINT :
160  loc = bitpit::VTKLocation::POINT;
161  break;
162  case MPVLocation::CELL :
163  loc = bitpit::VTKLocation::CELL;
164  break;
165  default:
166  (*m_log)<<"warning: Undefined Reference Location in plotOptionalResults of "<<m_name<<std::endl;
167  (*m_log)<<"Interface or Undefined locations are not supported in VTU writing." <<std::endl;
168  break;
169  }
170  if(loc == bitpit::VTKLocation::UNDEFINED){
171  throw std::runtime_error("Undefined data location");
172  };
173 
174  //check size of field and adjust missing values to zero for writing purposes only.
175  dmpvector1D field_supp = m_result;
176  if(!field_supp.completeMissingData(0.0)) return;
177  dvector1D field = field_supp.getDataAsVector();
178 
179 
180  if(m_result.getGeometry()->getType() != 3){
181  m_result.getGeometry()->getPatch()->getVTK().setDirectory(m_outputPlot+"/");
182  m_result.getGeometry()->getPatch()->getVTK().setName(m_name+std::to_string(getId()));
183  m_result.getGeometry()->getPatch()->getVTK().addData("field", bitpit::VTKFieldType::SCALAR, loc, field);
184  m_result.getGeometry()->getPatch()->write();
185  m_result.getGeometry()->getPatch()->getVTK().removeData("field");
186  }else{
187  if(loc == bitpit::VTKLocation::CELL){
188  (*m_log)<<"Warning: Attempt writing Cell data field on cloud of points in plotOptionalResults of "<<m_name<<std::endl;
189  throw std::runtime_error("Attempt writing Cell data field on cloud of points");
190  }
191 
192  liimap mapDataInv;
193  dvecarr3E points = m_result.getGeometry()->getVerticesCoords(&mapDataInv);
194  int size = points.size();
195  ivector2D connectivity(size, ivector1D(1));
196  for(int i=0; i<size; ++i) connectivity[i][0]=i;
197  bitpit::VTKUnstructuredGrid output(m_outputPlot,m_name+std::to_string(getId()),bitpit::VTKElementType::VERTEX);
198  output.setGeomData( bitpit::VTKUnstructuredField::POINTS, points);
199  output.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, connectivity);
200  output.setDimensions(connectivity.size(), points.size());
201  output.addData("field", bitpit::VTKFieldType::SCALAR, loc, field);
202  output.setCodex(bitpit::VTKFormat::APPENDED);
203  output.write();
204  }
205 }

211 bool
212 ExtractScalarField::extract(){
213 
214  if (getGeometry() == NULL || m_field.getGeometry() == NULL) return false;
215  //checking internal ids coherence of the field.
216  if(!m_field.checkDataIdsCoherence()) return false;
217 
218  mimmo::MPVLocation refLoc = m_field.getDataLocation();
219  if(refLoc == mimmo::MPVLocation::UNDEFINED) refLoc = mimmo::MPVLocation::POINT;
220 
221  m_result.clear();
222  m_result.setDataLocation(refLoc);
223 
224  switch(m_mode){
225  case ExtractMode::ID :
226  extractID(refLoc);
227  m_result.setGeometry(getGeometry());
228  break;
229 
230  case ExtractMode::PID :
231  extractPID(refLoc);
232  m_result.setGeometry(m_field.getGeometry());
233  break;
234 
235  case ExtractMode::MAPPING :
236  extractMapping(refLoc);
237  m_result.setGeometry(getGeometry());
238  break;
239 
240  default :
241  assert(false && "unrecognized field location");
242  break;
243  }
244 
245  if (m_result.isEmpty()) return false;
246 
247  return true;
248 
249 }

256 void ExtractScalarField::extractID(mimmo::MPVLocation loc){
257 
258  switch(loc){
259  case mimmo::MPVLocation::POINT:
260  for (const auto & ID : getGeometry()->getVertices().getIds()){
261  if (m_field.exists(ID)){
262  m_result.insert(ID, m_field[ID]);
263  }
264  }
265  break;
266  case mimmo::MPVLocation::CELL:
267  for (const auto & ID : getGeometry()->getCells().getIds()){
268  if (m_field.exists(ID)){
269  m_result.insert(ID, m_field[ID]);
270  }
271  }
272  break;
273  case mimmo::MPVLocation::INTERFACE:
274  if(!getGeometry()->areInterfacesBuilt()) getGeometry()->buildInterfaces();
275  for (const auto & ID : getGeometry()->getInterfaces().getIds()){
276  if (m_field.exists(ID)){
277  m_result.insert(ID, m_field[ID]);
278  }
279  }
280  break;
281  default:
282  //do nothing
283  break;
284  }

292 void ExtractScalarField::extractPID(mimmo::MPVLocation loc){
293  //Extract by PIDs
294  livector1D commonPID;
295  {
296  std::unordered_set<long> pidTarget = getGeometry()->getPIDTypeList();
297  std::unordered_set<long> pidLinked = m_field.getGeometry()->getPIDTypeList();
298  std::set<long> unionPID(pidTarget.begin(), pidTarget.end());
299  unionPID.insert(pidLinked.begin(), pidLinked.end());
300  for(auto val: unionPID){
301  if(pidLinked.count(val) && pidTarget.count(val)){
302  commonPID.push_back(val);
303  }
304  }
305  }
306  livector1D cellExtracted = m_field.getGeometry()->extractPIDCells(commonPID);
307 
308  switch(loc){
309  case mimmo::MPVLocation::POINT:
310  {
311  livector1D vertExtracted = m_field.getGeometry()->getVertexFromCellList(cellExtracted);
312  for (const auto & ID : vertExtracted){
313  if (m_field.exists(ID)){
314  m_result.insert(ID, m_field[ID]);
315  }
316  }
317  }
318  break;
319  case mimmo::MPVLocation::CELL:
320  for (const auto & ID : cellExtracted){
321  if (m_field.exists(ID)){
322  m_result.insert(ID, m_field[ID]);
323  }
324  }
325  break;
326  case mimmo::MPVLocation::INTERFACE:
327  {
328  livector1D interfExtracted = m_field.getGeometry()->getInterfaceFromCellList(cellExtracted);
329  for (const auto & ID : interfExtracted){
330  if (m_field.exists(ID)){
331  m_result.insert(ID, m_field[ID]);
332  }
333  }
334  }
335  break;
336  default:
337  //do nothing
338  break;
339  }

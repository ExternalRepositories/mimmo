133 void
134 ExtractVectorField::plotOptionalResults(){
135 
136  if (m_result.size() == 0 || m_result.getGeometry() == NULL) return;
137 
138  bitpit::VTKLocation loc = bitpit::VTKLocation::UNDEFINED;
139  switch(m_result.getDataLocation()){
140  case MPVLocation::POINT :
141  loc = bitpit::VTKLocation::POINT;
142  break;
143  case MPVLocation::CELL :
144  loc = bitpit::VTKLocation::CELL;
145  break;
146  default:
147  (*m_log)<<"Warning: Undefined Reference Location in plotOptionalResults of "<<m_name<<std::endl;
148  (*m_log)<<"Interface or Undefined locations are not supported in VTU writing." <<std::endl;
149  break;
150  }
151  if(loc == bitpit::VTKLocation::UNDEFINED) return;
152 
153  //check size of field and adjust missing values to zero for writing purposes only.
154  dmpvecarr3E field_supp = m_result;
155  if(!field_supp.completeMissingData({{0.0,0.0,0.0}})) return;
156  dvecarr3E field = field_supp.getDataAsVector();
157 
158 
159  if(m_result.getGeometry()->getType() != 3){
160  m_result.getGeometry()->getPatch()->getVTK().addData("field", bitpit::VTKFieldType::VECTOR, loc, field);
161  m_result.getGeometry()->getPatch()->write(m_outputPlot+"/"+m_name+std::to_string(getId()));
162  m_result.getGeometry()->getPatch()->getVTK().removeData("field");
163  }else{
164  if(loc == bitpit::VTKLocation::CELL){
165  (*m_log)<<"Warning: Attempt writing Cell data field on cloud of points in plotOptionalResults of "<<m_name<<std::endl;
166  return;
167  }
168  liimap mapDataInv;
169  dvecarr3E points = m_result.getGeometry()->getVertexCoords(&mapDataInv);
170  int size = points.size();
171  ivector2D connectivity(size, ivector1D(1));
172  for(int i=0; i<size; ++i) connectivity[i][0]=i;
173  bitpit::VTKUnstructuredGrid output(m_outputPlot,m_name+std::to_string(getId()),bitpit::VTKElementType::VERTEX);
174  output.setGeomData( bitpit::VTKUnstructuredField::POINTS, points);
175  output.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, connectivity);
176  output.setDimensions(connectivity.size(), points.size());
177  output.addData("field", bitpit::VTKFieldType::VECTOR, loc, field);
178  output.setCodex(bitpit::VTKFormat::APPENDED);
179  output.write();
180  }
181 }

188 bool
189 ExtractVectorField::extract(){
190 
191  if (getGeometry() == NULL || m_field.getGeometry() == NULL) return false;
192  //checking internal ids coherence of the field.
193  if(!m_field.checkDataIdsCoherence()) return false;
194 
195  mimmo::MPVLocation refLoc = m_field.getDataLocation();
196  if(refLoc == mimmo::MPVLocation::UNDEFINED) refLoc = mimmo::MPVLocation::POINT;
197 
198  m_result.clear();
199  m_result.setDataLocation(refLoc);
200 
201  switch(m_mode){
202  case ExtractMode::ID :
203  extractID(refLoc);
204  m_result.setGeometry(getGeometry());
205  break;
206 
207  case ExtractMode::PID :
208  extractPID(refLoc);
209  m_result.setGeometry(m_field.getGeometry());
210  break;
211 
212  case ExtractMode::MAPPING :
213  extractMapping(refLoc);
214  m_result.setGeometry(getGeometry());
215  break;
216 
217  default :
218  assert(false && "unrecognized field location");
219  break;
220  }
221 
222  if (m_result.isEmpty()) return false;
223 
224  return true;
225 
226 }

233 void ExtractVectorField::extractID(mimmo::MPVLocation loc){
234 
235  switch(loc){
236  case mimmo::MPVLocation::POINT:
237  for (const auto & ID : getGeometry()->getVertices().getIds()){
238  if (m_field.exists(ID)){
239  m_result.insert(ID, m_field[ID]);
240  }
241  }
242  break;
243  case mimmo::MPVLocation::CELL:
244  for (const auto & ID : getGeometry()->getCells().getIds()){
245  if (m_field.exists(ID)){
246  m_result.insert(ID, m_field[ID]);
247  }
248  }
249  break;
250  case mimmo::MPVLocation::INTERFACE:
251  if(!getGeometry()->areInterfacesBuilt()) getGeometry()->buildInterfaces();
252  for (const auto & ID : getGeometry()->getInterfaces().getIds()){
253  if (m_field.exists(ID)){
254  m_result.insert(ID, m_field[ID]);
255  }
256  }
257  break;
258  default:
259  break;
260  }

268 void ExtractVectorField::extractPID(mimmo::MPVLocation loc){
269  //Extract by PIDs
270  shivector1D commonPID;
271  {
272  std::unordered_set<short> pidTarget = getGeometry()->getPIDTypeList();
273  std::unordered_set<short> pidLinked = m_field.getGeometry()->getPIDTypeList();
274  std::set<short> unionPID(pidTarget.begin(), pidTarget.end());
275  unionPID.insert(pidLinked.begin(), pidLinked.end());
276  for(auto val: unionPID){
277  if(pidLinked.count(val) && pidTarget.count(val)){
278  commonPID.push_back(val);
279  }
280  }
281  }
282  livector1D cellExtracted = m_field.getGeometry()->extractPIDCells(commonPID);
283 
284  switch(loc){
285  case mimmo::MPVLocation::POINT:
286  {
287  livector1D vertExtracted = m_field.getGeometry()->getVertexFromCellList(cellExtracted);
288  for (const auto & ID : vertExtracted){
289  if (m_field.exists(ID)){
290  m_result.insert(ID, m_field[ID]);
291  }
292  }
293  }
294  break;
295  case mimmo::MPVLocation::CELL:
296  for (const auto & ID : cellExtracted){
297  if (m_field.exists(ID)){
298  m_result.insert(ID, m_field[ID]);
299  }
300  }
301  break;
302  case mimmo::MPVLocation::INTERFACE:
303  {
304  livector1D interfExtracted = m_field.getGeometry()->getInterfaceFromCellList(cellExtracted);
305  for (const auto & ID : interfExtracted){
306  if (m_field.exists(ID)){
307  m_result.insert(ID, m_field[ID]);
308  }
309  }
310  }
311  break;
312  default:
313  break;
314  }

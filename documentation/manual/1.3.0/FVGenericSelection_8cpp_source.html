211 void
212 FVGenericSelection::execute(){
213  if(m_geometry == NULL || m_bndgeometry == NULL) {
214  throw std::runtime_error (m_name + " : NULL pointer to target bulk/boundary geometry found or both");
215  }
216  if(m_geometry->isEmpty() || m_bndgeometry->isEmpty() ){
217  throw std::runtime_error (m_name + " : empty bulk/boundary geometry linked");
218  };
219 
220  if(!checkCoherenceBulkBoundary()){
221  throw std::runtime_error (m_name + " : connected bulk and boundary geometries are not coherent");
222  }
223 
224  m_volpatch.reset(nullptr);
225  m_bndpatch.reset(nullptr);
226 
227  livector1D extractedVol;
228  short maxPID = 0;
229  std::unordered_map<short,livector1D> extractedBndVert;
230  {
231  std::unordered_map<short,livector1D> extractedBnd;
232 
233  extractSelection(extractedVol, extractedBnd);
234 
235  for (const auto & touple: extractedBnd){
236  extractedBndVert[touple.first] = m_bndgeometry->getVertexFromCellList(touple.second);
237  maxPID = std::max(touple.first, maxPID);
238  }
239  }
240 
241  if(extractedVol.empty()) {
242  throw std::runtime_error (m_name + " : empty selection performed. check block set-up");
243  }
244 
245  /*Create subpatches.*/
246  int topovol=2, topobnd=1;
247  if(m_topo == 2){
248  topovol = 1;
249  topobnd = 4;
250  }
251 
252  std::unique_ptr<MimmoObject> tempVol(new MimmoObject(topovol));
253  std::unique_ptr<MimmoObject> tempBnd(new MimmoObject(topobnd));
254 
255  {
256  livector1D TT;
257  bitpit::ElementType eltype;
258  short PID;
259 
260  livector1D vertExtracted = m_geometry->getVertexFromCellList(extractedVol);
261  for(const auto & idV : vertExtracted){
262  tempVol->addVertex(m_geometry->getVertexCoords(idV), idV);
263  }
264 
265  for(const auto & idCell : extractedVol){
266  bitpit::Cell & cell = m_geometry->getPatch()->getCell(idCell);
267  eltype = cell.getType();
268  PID = (short)cell.getPID();
269  TT = m_geometry->getCellConnectivity(idCell);
270  tempVol->addConnectedCell(TT,eltype, PID, idCell);
271  TT.clear();
272  }
273 
274  tempVol->buildInterfaces();
275  }
276 
277  //now create boundary mesh of selection.
278  {
279  std::set<long> boundaryInterfaces;
280  std::set<long> boundaryVertices;
281  bitpit::ConstProxyVector<long> vcount;
282  for(const auto & interf : tempVol->getInterfaces()){
283  if(interf.isBorder()){
284  boundaryInterfaces.insert(interf.getId());
285  vcount = interf.getVertexIds();
286  boundaryVertices.insert(vcount.begin(), vcount.end());
287  }
288  }
289 
290  //fill new boundary
291  bitpit::PiercedVector<bitpit::Interface> & bulkInterf = tempVol->getInterfaces();
292 
293  for(const auto & idV: boundaryVertices){
294  tempBnd->addVertex(tempVol->getVertexCoords(idV),idV);
295  }
296 
297  short PID = maxPID+1;
298  bitpit::ElementType eltype;
299  livector1D conn;
300  for(const auto & idI: boundaryInterfaces){
301 
302  int size = bulkInterf[idI].getConnectSize();
303  conn.resize(size);
304  long * cc = bulkInterf[idI].getConnect();
305  for(int i=0; i<size; ++i){
306  conn[i] = cc[i];
307  }
308  eltype = bulkInterf[idI].getType();
309  tempBnd->addConnectedCell(conn, eltype, PID, idI);
310  }
311  }
312 
313  //post-processing boundary (assign existent PID if any).
314  for(auto & touple : extractedBndVert){
315  livector1D cellids = tempBnd->getCellFromVertexList(touple.second, true);
316  for(const long & cid: cellids){
317  tempBnd->setPIDCell(cid, touple.first);
318  }
319  }
320 
321  m_volpatch = std::move(tempVol);
322  m_bndpatch = std::move(tempBnd);
323 };

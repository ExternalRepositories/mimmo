173 livector1D
174 GenericSelection::constrainedBoundary(){
175 
176  if(getGeometry() == NULL || getPatch() == NULL) return livector1D(0);
177  if(getGeometry()->isEmpty() || getPatch()->isEmpty()) return livector1D(0);
178  std::unordered_map<long, std::set<int> > survivors;
179 
180  auto daughterBCells = getPatch()->extractBoundaryFaceCellID();
181  auto motherBCells = getGeometry()->extractBoundaryFaceCellID();
182 
183  //save all those daughter boundary faces not included in mother alse
184  for(const auto & sT : daughterBCells){
185  if(motherBCells.count(sT.first) > 0){
186  for(const auto & val: sT.second){
187  if(motherBCells[sT.first].count(val) > 0) continue;
188  survivors[sT.first].insert(val);
189  }
190  }else{
191  survivors[sT.first] = daughterBCells[sT.first];
192  }
193  }
194  motherBCells.clear();
195  daughterBCells.clear();
196 
197  //get vertex of the cleaned daughter boundary.
198  std::set<long> containerVert;
199  for(const auto & sT :survivors){
200  const bitpit::Cell & cell = getPatch()->getPatch()->getCell(sT.first);
201  for(const auto & val: sT.second){
202  bitpit::ConstProxyVector<long> faceVertIds = cell.getFaceVertexIds(val);
203  containerVert.insert(faceVertIds.begin(), faceVertIds.end());
204  }
205  }
206  //move up in livector1D container.
207  livector1D result;
208  result.reserve(containerVert.size());
209  result.insert(result.end(), containerVert.begin(), containerVert.end());
210 
211  return result;
212 };

219 void
220 GenericSelection::execute(){
221  if(getGeometry() == NULL) {
222  throw std::runtime_error (m_name + " : NULL pointer to target geometry found");
223  }
224  if(getGeometry()->isEmpty()){
225  throw std::runtime_error (m_name + " : empty geometry linked");
226  };
227 
228  m_subpatch.reset(nullptr);
229 
230  livector1D extracted = extractSelection();
231 
232  if(extracted.empty()) {
233  throw std::runtime_error (m_name + " : empty selection performed. check block set-up");
234  }
235 
236  /*Create subpatch.*/
237  std::unique_ptr<MimmoObject> temp(new MimmoObject(m_topo));
238 
239 
240  livector1D TT;
241  bitpit::ElementType eltype;
242  short PID;
243  if (m_topo != 3){
244 
245  livector1D vertExtracted = getGeometry()->getVertexFromCellList(extracted);
246  for(const auto & idV : vertExtracted){
247  temp->addVertex(getGeometry()->getVertexCoords(idV), idV);
248  }
249 
250  for(const auto & idCell : extracted){
251 
252  bitpit::Cell & cell = getGeometry()->getPatch()->getCell(idCell);
253  eltype = cell.getType();
254  PID = (short)cell.getPID();
255  TT = getGeometry()->getCellConnectivity(idCell);
256  temp->addConnectedCell(TT,eltype, PID, idCell);
257  TT.clear();
258  }
259  }else{
260  for(const auto & idV : extracted){
261  temp->addVertex(getGeometry()->getVertexCoords(idV),idV);
262  }
263  }
264 
265  auto originalmap = getGeometry()->getPIDTypeListWNames();
266  auto currentPIDmap = temp->getPIDTypeList();
267  for(const auto & val: currentPIDmap){
268  temp->setPIDName(val, originalmap[val]);
269  }
270  m_subpatch = std::move(temp);
271 };

278 GenericSelection::plotOptionalResults(){
279  if(getPatch() == NULL) return;
280  if(getPatch()->isEmpty()) return;
281 
282  std::string dir = m_outputPlot;
283  std::string name = m_name + "_Patch."+ std::to_string(getId());
284 
285 
286  if (m_topo != 3){
287 
288  std::string fullpath = dir + "/" + name ;
289  getPatch()->getPatch()->write(fullpath);
290 
291  }else{
292 
293  liimap mapDataInv;
294  dvecarr3E points = getPatch()->getVertexCoords(&mapDataInv);
295  ivector2D connectivity;
296  bitpit::VTKElementType cellType = bitpit::VTKElementType::VERTEX;
297 
298  int np = points.size();
299  connectivity.resize(np);
300  for (int i=0; i<np; i++){
301  connectivity[i].resize(1);
302  connectivity[i][0] = i;
303  }
304 
305  bitpit::VTKUnstructuredGrid output(dir,name,cellType);
306  output.setGeomData( bitpit::VTKUnstructuredField::POINTS, points);
307  output.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, connectivity);
308  output.setDimensions(connectivity.size(), points.size());
309  output.setCodex(bitpit::VTKFormat::APPENDED);
310  output.write();
311  }
312 }

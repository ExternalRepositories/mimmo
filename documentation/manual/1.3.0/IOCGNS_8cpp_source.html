384 void
385 IOCGNS::execute(){
386 
387  std::string target = m_dir+"/"+m_filename+".cgns";
388 
389  switch(m_mode){
390  case IOCGNS_Mode::READ:
391  if(!read(target)){
392  (*m_log) << "Error IOCGNS Reading mode: file corrupted or not found : "<< target << std::endl;
393  (*m_log) << " " << std::endl;
394  throw std::runtime_error ("file corrupted or not found : " + target);
395  }
396  writeInfoFile();
397  break;
398  case IOCGNS_Mode::DUMP:
399  {
400  int archiveVersion = 1;
401  std::string header(m_name);
402  std::string filedump = m_dir+"/"+m_filename;
403  #if MIMMO_ENABLE_MPI
404  bitpit::OBinaryArchive binaryWriter(filedump, "dump", archiveVersion, header, m_rank);
405  #else
406  bitpit::OBinaryArchive binaryWriter(filedump, "dump", archiveVersion, header);
407  #endif
408  if(!dump(binaryWriter.getStream())){
409  (*m_log) << "Error IOCGNS Dumping mode: impossible to write on : "<< filedump << std::endl;
410  (*m_log) << " " << std::endl;
411  throw std::runtime_error ("impossible to write dumpfile : " + filedump);
412  }
413  binaryWriter.close();
414  }
415  break;
416  case IOCGNS_Mode::RESTORE:
417  {
418  std::string filedump = m_dir+"/"+m_filename;
419  #if MIMMO_ENABLE_MPI
420  bitpit::IBinaryArchive binaryReader(filedump,"dump", m_rank);
421  #else
422  bitpit::IBinaryArchive binaryReader(filedump,"dump");
423  #endif
424 
425  m_surfmesh_not = nullptr;
426  m_geometry = nullptr;
427  m_volmesh = std::unique_ptr<MimmoObject>(new MimmoObject(2));
428  m_surfmesh = std::unique_ptr<MimmoObject>(new MimmoObject(1));
429 
430  if(!restore(binaryReader.getStream())){
431  (*m_log) << "Error IOCGNS Restoring mode: impossible to restore from dump : "<< filedump << std::endl;
432  (*m_log) << " " << std::endl;
433  throw std::runtime_error ("impossible to read dumpfile : " + filedump);
434  }
435  binaryReader.close();
436  }
437  break;
438  case IOCGNS_Mode::WRITE:
439  if(!write(target)){
440  (*m_log) << "Error IOCGNS Writing mode: impossible to write on : "<< target << std::endl;
441  (*m_log) << " " << std::endl;
442  throw std::runtime_error ("impossible to write : " + target);
443  }
444  writeInfoFile();
445 
446  break;
447  }

505 bool
506 IOCGNS::read(const std::string & file){
507 
508  m_volmesh = std::unique_ptr<MimmoObject>(new MimmoObject(2));
509  m_surfmesh = std::unique_ptr<MimmoObject>(new MimmoObject(1));
510  std::unique_ptr<MimmoObject> patchVol(new MimmoObject(2));
511 
512 #if MIMMO_ENABLE_MPI
513  if(m_rank == 0){
514 #endif
515  std::string error_string = "read CGNS grid: " + file;
516 
517  std::vector<long> nVertices;
518  std::vector<long> nCells;
519  // long nCells, nBoundVertices;
520  std::vector<int> nCoords; // how many coordinates 3 - x y z
521  std::vector< std::array< std::vector<double>,3 > > zonecoords; //node coordinates for each zone
522  std::vector<int> nSections; // haw many grouped sections of homogeneous cell elements
523  std::vector< std::vector<CGNS_ENUMT(ElementType_t)> > orderedConns; //orderedConnectivities - retain element type at index i, for each zone.
524  std::vector<std::unordered_map<int, ivector1D> > conns; //connectivity to be absorbed for each zone
525  std::vector<int> nBcs; // number of bc for each zone.
526  std::vector<std::unordered_map<int, ivector1D >> bcLists; //list of elements composing bcs, for each zone.
527  std::vector<std::unordered_map<int, std::string >> bcNames; // list of names of bcs, for each zone
528  std::vector<std::unordered_map<int, CGNS_ENUMT(BCType_t) >> bcTypes; // list of types of bcs, for each zone
529  std::vector<std::unordered_map<int, bool > > bcOnElements;
530 
531  //FIRST STEP ABSORB INFO from FILE//
532 
533  //Open cgns file
534  int indexfile;
535  if(cg_open(file.c_str(), CG_MODE_READ, &indexfile) != CG_OK){
536  return false;
537  }
538 
539  //Read number of bases
540  int nbases;
541  if(cg_nbases(indexfile, &nbases) != CG_OK){
542  return false;
543  }
544  if(nbases > 1){
545  (*m_log) << m_name << " more than one base found in grid file -> only the first zone will be read" << std::endl;
546  }
547 
548  //Read name of basis and physical dimension
549  char basename[33];
550  int physdim, celldim;
551  if(cg_base_read(indexfile, 1, basename, &celldim, &physdim) != CG_OK){
552  return false;
553  }
554  if(celldim != 3 || physdim !=3){
555  //Only volume mesh supported
556  return false;
557  }
558 
559  //Read number of zone in basis, check if only one.
560  int nzones;
561  if(cg_nzones(indexfile, 1, &nzones)!=CG_OK){
562  return false;
563  }
564 
565  //Read type mesh for each zones
566 
567  CGNS_ENUMT(ZoneType_t) zoneType[nzones];
568  int index_dim[nzones];
569  std::vector<std::vector<cgsize_t>> sizeG(nzones, std::vector<cgsize_t>(3));
570  std::vector<std::string> zonenames(nzones);
571  nVertices.resize(nzones);
572  nCells.resize(nzones);
573  nCoords.resize(nzones);
574  zonecoords.resize(nzones);
575  nSections.resize(nzones);
576  conns.resize(nzones);
577  orderedConns.resize(nzones);
578  nBcs.resize(nzones);
579  bcLists.resize(nzones);
580  bcNames.resize(nzones);
581  bcTypes.resize(nzones);
582  bcOnElements.resize(nzones);
583 
584  for(int indexzone=0; indexzone<nzones; ++indexzone){
585  if(cg_zone_type(indexfile,1,indexzone+1, &zoneType[indexzone])!= CG_OK){
586  return false;
587  }
588  if(cg_index_dim(indexfile,1,indexzone+1, &index_dim[indexzone])!= CG_OK){
589  return false;
590  }
591  if(zoneType[indexzone] != CGNS_ENUMT(ZoneType_t)::CGNS_ENUMV(Unstructured) || index_dim[indexzone] != 1){
592  //Only unstructured mesh supported for now (index_dim == 1 for unstructured)
593  return false;
594  }
595  //Read size of zone (n nodes, n cells, n boundary nodes)
596  char fzz[33];
597  if(cg_zone_read(indexfile,1,indexzone+1, fzz, sizeG[indexzone].data()) != CG_OK ){
598  return false;
599  }
600  zonenames[indexzone] = std::string(fzz);
601 
602  nVertices[indexzone]= sizeG[indexzone][0];
603  nCells[indexzone]= sizeG[indexzone][1];
604  //Read Vertices.
605  if(cg_ncoords(indexfile,1,indexzone+1, &nCoords[indexzone])!= CG_OK){
606  return false;
607  }
608 
609 
610  for(int i = 0; i < nCoords[indexzone]; ++i){
611  cgsize_t startIndex = 1;
612  cgsize_t finishIndex = sizeG[indexzone][0];
613  CGNS_ENUMT(DataType_t) datatype;
614  char name[33];
615  zonecoords[indexzone][i].resize(nVertices[indexzone]);
616  if(cg_coord_info(indexfile,1,indexzone+1,i+1, &datatype, name)!=CG_OK){
617  return false;
618  }
619  if(cg_coord_read(indexfile,1,indexzone+1,name, datatype, &startIndex, &finishIndex, zonecoords[indexzone][i].data() )!=CG_OK){
620  return false;
621  }
622  }
623 
624  //Read connectivities.
625  //They are read starting from 1, fortran style. When matching up w/ coords
626  //vector positions remember to diminish conn value of 1.
627  //Read number of sections of connectivity structure
628 
629  if(cg_nsections(indexfile,1,indexzone+1, &nSections[indexzone])!= CG_OK){
630  return false;
631  }
632 
633  for(int sec = 0; sec < nSections[indexzone]; ++sec){
634 
635  //Read section sec
636  char elementname[33];
637  CGNS_ENUMT(ElementType_t) type;
638  cgsize_t eBeg, eEnd;
639  int enBdry;
640  int parent_flag;
641  std::vector<cgsize_t> connlocal;
642  cgsize_t size;
643 
644  //Read elements name, type and range
645  if(cg_section_read(indexfile,1,indexzone+1, sec+1, elementname, &type, &eBeg, &eEnd,&enBdry, & parent_flag)!= CG_OK){
646  return false;
647  }
648 
649  //Read size of connectivity data
650  if(cg_ElementDataSize(indexfile,1,indexzone+1, sec+1,&size)!= CG_OK){
651  return false;
652  }
653 
654  connlocal.resize((size_t) size);
655  if(cg_elements_read(indexfile,1,indexzone+1,sec+1, connlocal.data(),nullptr) !=CG_OK){
656  return false;
657  }
658 
659  conns[indexzone][sec].resize(connlocal.size());
660  int count=0;
661  for(const auto &val: connlocal){
662  conns[indexzone][sec][count] = (int)val-1; //from fortran to c indexing.
663  ++count;
664  }
665  orderedConns[indexzone].push_back(type);
666  }
667 
668 
669  //explore superficial boundary conditions definition, for boundary surface extraction.
670  if(cg_nbocos(indexfile,1,indexzone+1, &nBcs[indexzone])!= CG_OK){
671  return false;
672  }
673 
674  //Everything marked as bc for me it's a wall.
675  for(int indexbc=0; indexbc<nBcs[indexzone]; ++indexbc){
676 
677  char name[33];
678  CGNS_ENUMT(BCType_t) bocotype;
679  CGNS_ENUMT(PointSetType_t) ptset_type;
680  std::vector<cgsize_t> nBCElements(2);
681  int normalIndex;
682  cgsize_t normalListSize;
683  CGNS_ENUMT(DataType_t) normalDataType;
684  int ndataset;
685  //CGNS_ENUMT(GridLocation_t) location;
686 
687  //reading information of the boundary. What i need here is
688  // the name (for sewing betwenn zones after) and the ptset_type-nBCElements.
689  if(cg_boco_info(indexfile,1,indexzone+1,indexbc+1, name, &bocotype, &ptset_type, nBCElements.data(),
690  &normalIndex, &normalListSize, &normalDataType, &ndataset) != CG_OK){
691  return false;
692  }
693  GridLocation_t bclocation;
694  if(cg_boco_gridlocation_read(indexfile,1,indexzone+1,indexbc+1, &bclocation) != CG_OK){
695  return false;
696  }
697 
698  bcNames[indexzone][indexbc] = std::string(name);
699  bcTypes[indexzone][indexbc] = bocotype;
700 
701  bool validSetType = true;
702  std::vector<cgsize_t> localbclist;
703 
704  switch(ptset_type){
705  case CGNS_ENUMV(PointList):
706  case CGNS_ENUMV(ElementList):
707  {
708  localbclist.resize((size_t) nBCElements[0]);
709  if(cg_boco_read(indexfile,1,indexzone+1,indexbc+1, localbclist.data(), nullptr )!= CG_OK){
710  return false;
711  }
712  }
713  break;
714  case CGNS_ENUMV(PointRange):
715  case CGNS_ENUMV(ElementRange):
716  {
717  cgsize_t dim = nBCElements[1] - nBCElements[0] + 1;
718  localbclist.reserve((size_t) dim);
719  for (cgsize_t idx = nBCElements[0]; idx <= nBCElements[1]; idx++){
720  localbclist.push_back(idx);
721  }
722  }
723  break;
724  default:
725  validSetType = false;
726  break;
727  }
728  if(!validSetType){
729  (*m_log)<<"IOCGNS reader cannot support BC PointSetType_t different from PointList, PointRange, ElementList and ElementRange.Aborting"<<std::endl;
730  return false;
731  }
732 
733  bool flag = ( ptset_type == CGNS_ENUMV(ElementList) );
734  flag = flag || (ptset_type == CGNS_ENUMV(ElementRange) );
735  flag = flag || ( (ptset_type == CGNS_ENUMV(PointList) || ptset_type == CGNS_ENUMV(PointRange) )
736  && bclocation == CGNS_ENUMV(FaceCenter) ) ;
737  bcOnElements[indexzone].insert(std::make_pair(indexbc,flag));
738  bcLists[indexzone][indexbc].resize(localbclist.size());
739  int count=0;
740  for(const auto &val: localbclist){
741  bcLists[indexzone][indexbc][count] = (int)val-1;//from fortran to c style
742  ++count;

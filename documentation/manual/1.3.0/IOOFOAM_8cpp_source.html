292 livector1D
293 IOOFOAM::mapEleVConnectivity(const livector1D & FoamConn, const bitpit::ElementType & eltype){
294  livector1D conn;
295 
296  switch(eltype){
297  case bitpit::ElementType::TETRA:
298  conn.resize(4);
299  conn[0] = FoamConn[2];
300  conn[1] = FoamConn[1];
301  conn[2] = FoamConn[3];
302  conn[3] = FoamConn[0];
303  break;
304  case bitpit::ElementType::HEXAHEDRON:
305  conn.resize(8);
306  conn[0] = FoamConn[0];
307  conn[1] = FoamConn[3];
308  conn[2] = FoamConn[7];
309  conn[3] = FoamConn[4];
310  conn[4] = FoamConn[1];
311  conn[5] = FoamConn[2];
312  conn[6] = FoamConn[6];
313  conn[7] = FoamConn[5];
314  break;
315  case bitpit::ElementType::WEDGE:
316  conn.resize(6);
317  conn[0] = FoamConn[0];
318  conn[1] = FoamConn[2];
319  conn[2] = FoamConn[1];
320  conn[3] = FoamConn[3];
321  conn[4] = FoamConn[5];
322  conn[5] = FoamConn[4];
323  break;
324  case bitpit::ElementType::PYRAMID:
325  conn.resize(5);
326  conn[0] = FoamConn[3];
327  conn[1] = FoamConn[2];
328  conn[2] = FoamConn[1];
329  conn[3] = FoamConn[0];
330  conn[4] = FoamConn[4];
331  break;
332  default:
333  //do nothing
334  break;
335  }
336 
337  return conn;
338 }

344 bool
345 IOOFOAM::read(){
346 
347  Foam::Time *foamRunTime = 0;
348  Foam::fvMesh *foamMesh = 0;
349 
350  //read mesh from OpenFoam case directory
351  foamUtilsNative::initializeCase(m_path.c_str(), &foamRunTime, &foamMesh);
352 
353  //prepare my bulk geometry container
354  std::unique_ptr<bitpit::PatchKernel> mesh(new mimmo::MimmoVolUnstructured(3));
355  mesh->reserveVertices(std::size_t(foamMesh->nPoints()));
356  mesh->reserveCells(std::size_t(foamMesh->nCells()));
357 
358  //start absorbing mesh nodes/points.
359  Foam::pointField nodes = foamMesh->points();
360  darray3E coords;
361  forAll(nodes, in){
362  for (int k = 0; k < 3; k++) {
363  coords[k] = nodes[in][k];
364  }
365  mesh->addVertex(coords, long(in));
366  }
367 
368  //absorbing cells.
369  const Foam::cellList & cells = foamMesh->cells();
370  const Foam::cellShapeList & cellShapes = foamMesh->cellShapes();
371  const Foam::faceList & faces = foamMesh->faces();
372  const Foam::labelList & faceOwner = foamMesh->faceOwner();
373  const Foam::labelList & faceNeighbour = foamMesh->faceNeighbour();
374 
375  Foam::label sizeNeighbours = faceNeighbour.size();
376 
377  std::string eleshape;
378  bitpit::ElementType eltype;
379  long iDC;
380  short PID = 0;
381  livector1D conn, temp;
382  livector2D adjacency;
383 
384  forAll(cells, iC){
385 
386  iDC = long(iC);
387  eleshape = std::string(cellShapes[iC].model().name());
388  //first step verify the model in twin cellShapes list.
389  conn.clear();
390  adjacency.clear();
391  adjacency.resize(std::size_t(cells[iC].size()), livector1D(1, bitpit::Cell::NULL_ID));
392 
393  if(m_OFE_supp.count(eleshape) > 0){
394 
395  eltype = m_OFE_supp[eleshape];
396  temp.resize(cellShapes[iC].size());
397  forAll(cellShapes[iC], loc){
398  temp[loc] = (long)cellShapes[iC][loc];
399  }
400  conn = mapEleVConnectivity(temp, eltype);
401 
402  auto ordFaceList = cellShapes[iC].meshFaces(faces, cells[iC]);
403  Foam::label refFace;
404  forAll(ordFaceList, ofcount){
405  refFace = ordFaceList[ofcount];
406  if(refFace >= sizeNeighbours) continue;
407  if(iC == faceOwner[refFace]){
408  adjacency[int(ofcount)][0] = faceNeighbour[refFace];
409  }else{
410  adjacency[int(ofcount)][0] = faceOwner[refFace];
411  }
412  }
413 
414  }else{
415 
416  eltype = bitpit::ElementType::POLYHEDRON;
417  //manually calculate connectivity.
418  conn.push_back((long)cells[iC].size()); //total number of faces on the top.
419  forAll(cells[iC], locC){
420  Foam::label iFace = cells[iC][locC];
421  long faceNVertex = (long)faces[iFace].size();
422  temp.resize(faceNVertex);
423  forAll(faces[iFace], locF){
424  temp[locF] = (long)faces[iFace][locF];
425  }
426 
427  conn.push_back(faceNVertex);
428  if(iFace >= sizeNeighbours) {
429  //border face, normal outwards, take as it is
430  conn.insert(conn.end(), temp.begin(), temp.end());
431  continue;
432  }
433  bool normalIsOut;
434  //recover right adjacency and check if face normal pointing outwards.
435  //OpenFoam policy wants the face normal between cell pointing towards
436  // the cell with greater id.
437  if(iC == faceOwner[iFace]){
438  adjacency[int(locC)][0] = faceNeighbour[iFace];
439  normalIsOut = faceNeighbour[iFace] > iC;
440  }else{
441  adjacency[int(locC)][0] = faceOwner[iFace];
442  normalIsOut = faceOwner[iFace] > iC;
443  }
444 
445  if(normalIsOut){
446  conn.insert(conn.end(), temp.begin(), temp.end());
447  }else{
448  conn.insert(conn.end(), temp.rbegin(), temp.rend());
449  }
450  }
451  }
452  bitpit::PatchKernel::CellIterator it = mesh->addCell(eltype, true, conn, iDC);
453  it->setPID(int(PID));
454  it->setAdjacencies(adjacency);
455  }
456  //build as bitpit the Interfaces -> adjacency is automatically recoverd from openFoam info.
457  mesh->buildInterfaces();
458 
459  //create OpenFoam faces - bitpit Interfaces local map to detect boundaries
460  std::unordered_map<long,long> mapBitOF_faces;
461 
462  forAll(cells, iC){
463  iDC = long(iC);
464  eleshape = std::string(cellShapes[iC].model().name());
465  Foam::labelList ofFaceList;
466  if(m_OFE_supp.count(eleshape) > 0){
467  eltype = m_OFE_supp[eleshape];
468  ofFaceList = cellShapes[iC].meshFaces(faces, cells[iC]);
469  }else{
470  eltype = bitpit::ElementType::POLYHEDRON;
471  ofFaceList = cells[iC];
472  }
473  long * bitFaceList = mesh->getCell(iC).getInterfaces();
474  std::size_t sizeFList = mesh->getCell(iC).getInterfaceCount();
475  for(std::size_t i = 0; i<sizeFList; ++i){
476  mapBitOF_faces[ofFaceList[i]] = bitFaceList[i];
477  }
478  }
479 
480  //finally store bulk mesh in the internal bulk member of the class (from MimmoFvMesh);
481  m_bulk = std::move(std::unique_ptr<MimmoObject>(new MimmoObject(2, mesh)));
482  m_internalBulk = true;
483  m_bulkext = NULL;
484 
485  //from MimmoFvMesh protected utilities, create the raw boundary mesh, storing it in m_boundary internal member.
486  //PID will be every where 0. Need to be compiled to align with patch division of foamBoundaryMesh.
487  //every cell of the boundary mesh will have the same id of the border interfaces of the bulk.
488  createBoundaryMesh();
489 
490  //Once OFoam faces and bitpit Interfaces link is set,
491  //Extract boundary patch info from foamBoundary and
492  // get boundary patch division automatically pidding the class boundary mesh.
493 
494  const Foam::fvBoundaryMesh &foamBMesh = foamMesh->boundary();
495  long startIndex;
496  long endIndex;
497 
498  forAll(foamBMesh, iBoundary){
499  PID = short(iBoundary+1);
500  startIndex = foamBMesh[iBoundary].start();
501  endIndex = startIndex + long(foamBMesh[iBoundary].size());
502  for(long ind=startIndex; ind<endIndex; ++ind){
503  m_boundary->setPIDCell(mapBitOF_faces[ind], PID);
504  }
505 
506  }
507  m_boundary->resyncPID();
508  //minimo sindacale fatto.
509  return true;
510 
511 }

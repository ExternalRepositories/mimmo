469 bool
470 IOOFOAM::read(){
471 
472  Foam::Time *foamRunTime = 0;
473  Foam::fvMesh *foamMesh = 0;
474 
475  //read mesh from OpenFoam case directory
476  foamUtilsNative::initializeCase(m_path.c_str(), &foamRunTime, &foamMesh);
477 
478  //prepare my bulk geometry container
479  std::unique_ptr<bitpit::PatchKernel> mesh(new mimmo::MimmoVolUnstructured(3));
480  mesh->reserveVertices(std::size_t(foamMesh->nPoints()));
481  mesh->reserveCells(std::size_t(foamMesh->nCells()));
482 
483  //start absorbing mesh nodes/points.
484  Foam::pointField nodes = foamMesh->points();
485  darray3E coords;
486  forAll(nodes, in){
487  for (int k = 0; k < 3; k++) {
488  coords[k] = nodes[in][k];
489  }
490  mesh->addVertex(coords, long(in));
491  }
492 
493  //absorbing cells.
494  const Foam::cellList & cells = foamMesh->cells();
495  const Foam::cellShapeList & cellShapes = foamMesh->cellShapes();
496  const Foam::faceList & faces = foamMesh->faces();
497  const Foam::labelList & faceOwner = foamMesh->faceOwner();
498  const Foam::labelList & faceNeighbour = foamMesh->faceNeighbour();
499 
500  Foam::label sizeNeighbours = faceNeighbour.size();
501 
502  std::string eleshape;
503  bitpit::ElementType eltype;
504  long iDC;
505  long PID = 0;
506  livector1D conn, temp;
507 
508  forAll(cells, iC){
509 
510  iDC = long(iC);
511  eleshape = std::string(cellShapes[iC].model().name());
512  //first step verify the model in twin cellShapes list.
513  conn.clear();
514 
515  if(m_OFE_supp.count(eleshape) > 0){
516 
517  eltype = m_OFE_supp[eleshape];
518  temp.resize(cellShapes[iC].size());
519  forAll(cellShapes[iC], loc){
520  temp[loc] = (long)cellShapes[iC][loc];
521  }
522  conn = foamUtilsNative::mapEleVConnectivity(temp, eltype);
523 
524  }else{
525 
526  eltype = bitpit::ElementType::POLYHEDRON;
527  //manually calculate connectivity.
528  conn.push_back((long)cells[iC].size()); //total number of faces on the top.
529  forAll(cells[iC], locC){
530  Foam::label iFace = cells[iC][locC];
531  long faceNVertex = (long)faces[iFace].size();
532  temp.resize(faceNVertex);
533  forAll(faces[iFace], locF){
534  temp[locF] = (long)faces[iFace][locF];
535  }
536 
537  conn.push_back(faceNVertex);
538  if(iFace >= sizeNeighbours) {
539  //border face, normal outwards, take as it is
540  conn.insert(conn.end(), temp.begin(), temp.end());
541  continue;
542  }
543  bool normalIsOut;
544  //recover right adjacency and check if face normal pointing outwards.
545  //OpenFoam policy wants the face normal between cell pointing towards
546  // the cell with greater id.
547  if(iC == faceOwner[iFace]){
548  normalIsOut = faceNeighbour[iFace] > iC;
549  }else{
550  normalIsOut = faceOwner[iFace] > iC;
551  }
552 
553  if(normalIsOut){
554  conn.insert(conn.end(), temp.begin(), temp.end());
555  }else{
556  conn.insert(conn.end(), temp.rbegin(), temp.rend());
557  }
558  }
559  }
560  bitpit::PatchKernel::CellIterator it = mesh->addCell(eltype, conn, iDC);
561  it->setPID(int(PID));
562  }
563 
564  mesh->buildAdjacencies();
565  mesh->buildInterfaces();
566 
567  bitpit::PiercedVector<bitpit::Cell> & bitCells = mesh->getCells();
568  bitpit::PiercedVector<bitpit::Interface> & bitInterfaces = mesh->getInterfaces();
569 
570  forAll(faces, iOF){
571 
572  std::vector<long> vListOF(faces[iOF].size());
573  forAll(faces[iOF], index){
574  vListOF[index] = long(faces[iOF][index]);
575  }
576  std::sort(vListOF.begin(), vListOF.end());
577 
578  long iDC = long(faceOwner[iOF]);
579  long * bitFaceList = bitCells[iDC].getInterfaces();
580  std::size_t sizeFList = bitCells[iDC].getInterfaceCount();
581 
582  long iBIT = bitpit::Interface::NULL_ID;
583  std::size_t j(0);
584  while(iBIT < 0 && j<sizeFList){
585  long * vconn = bitInterfaces[bitFaceList[j]].getConnect();
586  std::size_t vconnsize = bitInterfaces[bitFaceList[j]].getConnectSize();
587  std::vector<long> vListBIT(vconn, vconn+vconnsize);
588  std::sort(vListBIT.begin(), vListBIT.end());
589 
590  if(std::equal(vListBIT.begin(), vListBIT.end(), vListOF.begin()) ){
591  iBIT = bitFaceList[j];
592  }else{
593  ++j;
594  }
595  }
596  m_OFbitpitmapfaces.insert(std::make_pair(long(iOF),iBIT) );
597  }
598 
599  //finally store bulk mesh in the internal bulk member of the class (from MimmoFvMesh);
600  m_bulk = std::move(std::unique_ptr<MimmoObject>(new MimmoObject(2, mesh)));
601  m_internalBulk = true;
602  m_bulkext = NULL;
603 
604  //from MimmoFvMesh protected utilities, create the raw boundary mesh, storing it in m_boundary internal member.
605  //PID will be every where 0. Need to be compiled to align with patch division of foamBoundaryMesh.
606  //every cell of the boundary mesh will have the same id of the border interfaces of the bulk.
607  createBoundaryMesh();
608 
609  //Once OFoam faces and bitpit Interfaces link is set,
610  //Extract boundary patch info from foamBoundary and
611  // get boundary patch division automatically pidding the class boundary mesh.
612 
613  const Foam::fvBoundaryMesh &foamBMesh = foamMesh->boundary();
614  long startIndex;
615  long endIndex;
616 
617  forAll(foamBMesh, iBoundary){
618  PID = long(iBoundary+1);
619  startIndex = foamBMesh[iBoundary].patch().start();
620  endIndex = startIndex + long(foamBMesh[iBoundary].patch().size());
621  for(long ind=startIndex; ind<endIndex; ++ind){
622  m_boundary->setPIDCell(m_OFbitpitmapfaces[ind], PID);
623  }
624  m_boundary->setPIDName(PID,foamBMesh[iBoundary].name().c_str());
625  }
626  m_boundary->resyncPID();
627  //minimo sindacale fatto.
628  return true;
629 
630 }

810 bool
811 IOOFOAMScalarField::read(){
812 
813  //read mesh from OpenFoam case directory (initialize)
814  Foam::Time *foamRunTime = 0;
815  Foam::fvMesh *foamMesh = 0;
816  foamUtilsNative::initializeCase(m_path.c_str(), &foamRunTime, &foamMesh);
817 
818 
819  if ( getGeometry() != nullptr){
820  std::size_t size;
821  std::vector<double> field;
822  foamUtilsNative::readScalarField(m_path.c_str(), m_fieldname.c_str(), -1, size, field);
823  m_field.clear();
824  m_field.reserve(size);
825 
826  auto itfield = field.begin();
827  for (bitpit::Cell & cell : getGeometry()->getCells()){
828  m_field.insert(cell.getId(), *itfield);
829  itfield++;
830  }
831  m_field.setGeometry(getGeometry());
832  //Data on cells
833  m_field.setDataLocation(2);
834 
835  //TODO ADD CELL TO POINT INTERPOLATION
836 
837  }
838 
839  //One field stored.
840  if ( getBoundaryGeometry() != nullptr ){
841 
842  const Foam::fvBoundaryMesh &foamBMesh = foamMesh->boundary();
843  long startIndex;
844 
845  std::unordered_set<long> pids = getBoundaryGeometry()->getPIDTypeList();
846  dmpvector1D boundaryFieldOnFace;
847  for (long pid : pids){
848  if (pid > 0){
849  std::size_t size = 0;
850  std::vector<double> field;
851  foamUtilsNative::readScalarField(m_path.c_str(), m_fieldname.c_str(), pid-1, size, field);
852  boundaryFieldOnFace.reserve(boundaryFieldOnFace.size() + size);
853  if (size > 0){
854  long iBoundary = pid-1;
855  startIndex = foamBMesh[iBoundary].start();
856  long ind = startIndex;
857  for (double val : field){
858  boundaryFieldOnFace.insert(m_OFbitpitmapfaces[ind], val);
859  ind++;
860  }
861  }
862  else{
863  for (bitpit::Cell cell : getBoundaryGeometry()->getCells()){
864  if (cell.getPID() == pid)

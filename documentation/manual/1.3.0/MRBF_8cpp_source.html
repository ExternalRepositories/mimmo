309 ivector1D
310 MRBF::checkDuplicatedNodes(double tol){
311  ivector1D marked;
312  int sizeEff = getTotalNodesCount();
313  if( sizeEff == 0 ) return marked;
314 
315  bvector1D check(sizeEff, false);
316 
317  for(int i=0; i<sizeEff; ++i){
318  for(int j=i+1; j<sizeEff; ++j){
319  double dist = norm2(m_node[j] - m_node[i]);
320  if(!check[j] && dist <= tol){
321  marked.push_back(i);
322  check[j] = true;
323  }
324  }
325  }
326  return(marked);
327 }

517 void
518 MRBF::execute(){
519 
520  MimmoObject * container = getGeometry();
521  if(container == NULL){
522  (*m_log)<<m_name + " : NULL pointer to linked geometry found"<<std::endl;
523  throw std::runtime_error(m_name + "NULL pointer to linked geometry found");
524  }
525 
526  if(container->isEmpty()){
527  (*m_log)<<m_name + " : empty linked geometry found"<<std::endl;
528  }
529 
530 
531  m_displ.clear();
532  m_displ.setDataLocation(mimmo::MPVLocation::POINT);
533  m_displ.reserve(getGeometry()->getNVertices());
534  m_displ.setGeometry(getGeometry());
535 
536 
537  int size = 0;
538  int sizeF = getDataCount();
539 
540  for (int i=0; i<sizeF; i++){
541 
542  if(m_solver == MRBFSol::NONE) size = m_weight[i].size();
543  else size = m_value[i].size();
544 
545  if(size != getTotalNodesCount()){
546  (*m_log) << "warning: " << getName() << " has displacements of " << i << " field with size (" << size << ") that does not fit number of RBF nodes ("<< getTotalNodesCount() << ")" << std::endl;
547  fitDataToNodes(i);
548  }
549  }
550 
551  double bboxDiag;
552  {
553  darray3E pmin, pmax;
554  container->getPatch()->getBoundingBox(pmin, pmax);
555  bboxDiag= norm2(pmax - pmin);
556  }
557 
558  //Checking supportRadius.
559  double distance = 0.0;
560  if(m_SRRatio <=0.0){ //get maximum weight/value displ and assign support radius a 3 times this value.
561 
562  for(int i=0; i<size; ++i){
563 
564  dvector1D data(sizeF);
565 
566  for(int j=0; j<sizeF; ++j){
567  if(m_solver == MRBFSol::NONE) data[j] = m_weight[j][i];
568  else data[j] = m_value[j][i];
569  }
570 
571  distance = std::max(distance, norm2(data));
572  }
573 
574  distance *=3.0;
575 
576  }else{
577  if (m_supRIsValue){
578  distance = m_SRRatio;
579  }
580  else{
581  distance = m_SRRatio * bboxDiag;
582  }
583  }
584 
585  //TODO remove it (I can't use a support radius 0....why?)
586  if(distance <=1.E-18){ //checkSupportRadius if too small, set it to the semidiagonal value of the geometry AABB
587  distance = 0.5*bboxDiag;
588  }
589 
590  const double radius = distance;
591  RBF::setSupportRadius(radius);
592 
593 
594  if (m_solver == MRBFSol::WHOLE) solve();
595  if (m_solver == MRBFSol::GREEDY) greedy(m_tol);
596 
597 
598  dvector1D displ;
599  darray3E adispl;
600  for(const auto & vertex : container->getVertices()){
601  displ = bitpit::RBF::evalRBF(vertex.getCoords());
602  for (int j=0; j<3; ++j)
603  adispl[j] = displ[j];
604  m_displ.insert(vertex.getId(), adispl);
605  }
606 
607  //if m_filter is active;
608  if(m_bfilter){
609 
610  checkFilter();
611 
612  long int ID;
613  for (const auto & vertex : m_geometry->getVertices()){
614  ID = vertex.getId();
615  m_displ[ID] = m_displ[ID] * m_filter[ID];
616  }
617  }
618 
619 };

643

647  m_filter.reserve(getGeometry()->getNVertices());
648  for (const auto & vertex : getGeometry()->getVertices()){
649  m_filter.insert(vertex.getId(), 1.0);
650  }
651  }

677 void
678 MRBF::plotCloud(std::string directory, std::string filename, int counterFile, bool binary, bool deformed){
679 
680  int nnodes = getTotalNodesCount();
681  nnodes = std::min(nnodes, int(m_displ.size()));
682  dvecarr3E* nodes_ = getNodes();
683  dvecarr3E nodes(nnodes);
684  dvecarr3E data(nnodes);
685  for(int i=0; i<nnodes; ++i){
686  for(int j=0; j<3; ++j){
687  if(m_solver == MRBFSol::NONE) data[i][j] = m_weight[j][i];
688  else data[i][j] = m_value[j][i];
689  }
690  }
691  if(deformed){
692  for(int i=0; i<nnodes; ++i){
693  nodes[i] = (*nodes_)[i] + data[i];
694  }
695  }else{
696  for(int i=0; i<nnodes; ++i){
697  nodes[i] = (*nodes_)[i];
698  }
699  }
700 
701  bitpit::VTKFormat codex = bitpit::VTKFormat::ASCII;
702  if(binary){codex=bitpit::VTKFormat::APPENDED;}
703 
704  ivector1D conn(nnodes);
705  {
706  int counter = 0;
707  for(auto & val: conn){
708  val = counter;
709  ++counter;
710  }
711  }
712  bitpit::VTKUnstructuredGrid vtk(directory, filename, bitpit::VTKElementType::VERTEX);
713  vtk.setGeomData( bitpit::VTKUnstructuredField::POINTS, nodes) ;
714  vtk.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, conn) ;
715  vtk.setDimensions(conn.size(), nnodes);
716  //use connectivity as node labels also;
717  vtk.addData("labels",bitpit::VTKFieldType::SCALAR, bitpit::VTKLocation::POINT, conn) ;
718  vtk.setCodex(codex);
719  if(counterFile>=0){vtk.setCounter(counterFile);}
720  vtk.write();
721 };

733  std::string input;

747  value = 104;

763  int value = 0;
764  if(!input.empty()){
765  std::stringstream ss(bitpit::utils::string::trim(input));
766  ss >> value;

769  }
770  setMode(value);

774  input = slotXML.get("SupportRadius");
775  double value = -1.0;

777  std::stringstream ss(bitpit::utils::string::trim(input));
778  ss >> value;

784  input = slotXML.get("SupportRadiusReal");

787  std::stringstream ss(bitpit::utils::string::trim(input));
788  ss >> value;
789  setSupportRadiusValue(value);

793  m_tol = 1.0E-6;
794  if(slotXML.hasOption("Tolerance")){
795  input = slotXML.get("Tolerance");
796  input = bitpit::utils::string::trim(input);
797  double value = m_tol;
798  if(!input.empty()){
799  std::stringstream ss(bitpit::utils::string::trim(input));

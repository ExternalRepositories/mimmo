43 double distancePointPolygon(const darray3E & p,const dvecarr3E & vertCoords){
44 
45  //calculate barycenter
46  darray3E barycenter = {{0.0,0.0,0.0}};
47  for(const auto & vv : vertCoords){
48  barycenter += vv;
49  }
50  barycenter /= double(vertCoords.size());
51 
52  //split it in subtriangles - fixed barycenter, run 2-by-2 along vertCoords to get the other two vertices
53  double distance(std::numeric_limits<double>::max());
54  std::size_t sizeV = vertCoords.size();
55  for(std::size_t i=0; i<sizeV-1; ++i){
56  distance = std::min(distance, bitpit::CGElem::distancePointTriangle(p, barycenter, vertCoords[i], vertCoords[i+1]));
57  }
58  return distance;
59 }

68 bool isPointInsideSegment(const darray3E & p, const darray3E & V0,const darray3E & V1 ){
69  darray3E Pv = p - V0;
70  darray3E segment = V1 - V0;
71  double normSegment = norm2(segment);
72  if(normSegment > std::numeric_limits<double>::min()) segment /= normSegment;
73  double V = dotProduct(Pv, segment);
74  if(norm2(Pv - V*segment)> 1.E-12) return false;
75  return ( V >=0.0 && V <= normSegment);
76 }

85 bool isPointInsideTriangle(const darray3E & p, const darray3E & V0,const darray3E & V1,const darray3E & V2){
86 
87  darray3E normal = crossProduct(V1 - V0,V2 - V0);
88  double normNormal = norm2(normal);
89  if(normNormal > std::numeric_limits<double>::min()) normal /= normNormal;
90  else return false;
91 
92  if(std::abs(dotProduct((p-V0), normal)) > std::numeric_limits<double>::min()){ return false;}
93 
94  double a1 =CGElem::areaTriangle(p,V0,V1)/(0.5*normNormal);
95  double a2 =CGElem::areaTriangle(p,V1,V2)/(0.5*normNormal);
96  double a3 =CGElem::areaTriangle(p,V2,V0)/(0.5*normNormal);
97  return ( (a1+a2+a3 -1.0) <=1.0E-12 );
98 }

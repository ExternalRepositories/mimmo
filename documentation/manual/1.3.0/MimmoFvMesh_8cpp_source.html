73 MimmoFvMesh::MimmoFvMesh(std::unique_ptr<MimmoObject> & bulk, std::unique_ptr<MimmoObject> &boundary){
74  //check couples first

85  m_bulk = std::move(bulk);
86  m_bulkext = NULL;
87 
88  m_internalBoundary = true;

242 void MimmoFvMesh::createBoundaryMesh(){
243 
244  MimmoObject * bulk = getGeometry();
245  if (!bulk->areInterfacesBuilt()) bulk->buildInterfaces();
246 
247  std::set<long> boundaryInterfaces;
248  std::set<long> boundaryVertices;
249  bitpit::ConstProxyVector<long> vcount;
250  for(const auto & interf : bulk->getInterfaces()){
251  if(interf.isBorder()){
252  boundaryInterfaces.insert(interf.getId());
253  vcount = interf.getVertexIds();
254  boundaryVertices.insert(vcount.begin(), vcount.end());
255  }
256  }
257 
258  //fill new boundary
259  int type = int(bulk->getType() == 2) + 4*int(bulk->getType() == 1);
260  std::unique_ptr<MimmoObject> temp(new MimmoObject(type));
261 
262  bitpit::PiercedVector<bitpit::Interface> & bulkInterf = bulk->getInterfaces();
263 
264  for(const auto & idV: boundaryVertices){
265  temp->addVertex(bulk->getVertexCoords(idV),idV);
266  }
267 
268 
269  long PID = 0;
270  bitpit::ElementType eltype;
271  livector1D conn;
272  for(const auto & idI: boundaryInterfaces){
273 
274  int size = bulkInterf[idI].getConnectSize();
275  conn.resize(size);
276  long * cc = bulkInterf[idI].getConnect();
277  for(int i=0; i<size; ++i){
278  conn[i] = cc[i];
279  }
280  eltype = bulkInterf[idI].getType();
281  PID = long(bulkInterf[idI].getPID());
282 
283  temp->addConnectedCell(conn, eltype, PID, idI);
284  }
285 
286  m_boundary = std::move(temp);
287  m_internalBoundary = true;
288  m_boundaryext = NULL;
289 }

304 bool MimmoFvMesh::checkMeshCoherence(){
305 
306  MimmoObject * bulk = getGeometry();
307  if(bulk == NULL) return false;
308  if(bulk->isEmpty()) return false;
309 
310  MimmoObject * boundary = getBoundaryGeometry();
311 
312  bool checkBoundaries = true;
313  if(boundary == NULL) checkBoundaries = false;
314  else if(boundary->isEmpty()) checkBoundaries = false;
315 
316  if(checkBoundaries){
317  if(!bulk->areInterfacesBuilt()) return false;
318  std::string key = std::to_string(bulk->getType()) + std::to_string(boundary->getType());
319  if(key !="21" && key !="14") return false;
320 
321  std::set<long> idBorderInterf;
322  for(const auto & interf : bulk->getInterfaces()){
323  if(interf.isBorder()) idBorderInterf.insert(interf.getId());
324  };
325  std::vector<long> idInterf;
326  idInterf.insert(idInterf.end(), idBorderInterf.begin(), idBorderInterf.end());
327  std::vector<long> idCell = boundary->getCells().getIds(true);
328 
329  return idInterf == idCell;
330  }else{
331  createBoundaryMesh();
332  return true;
333  }
334 }

527 bool
528 MimmoGeometry::write(){
529 
530 // if (isEmpty()) return false;
531 
532  //adjusting with reference pid;
533  {
534  auto locpids = getGeometry()->getCompactPID();
535  auto pidsMap = getGeometry()->getPIDTypeListWNames();
536  for( auto & val: locpids) val+=m_refPID;
537  setPID(locpids);
538  auto pids = getGeometry()->getPIDTypeList();
539  for( auto & val: pids) getGeometry()->setPIDName(val, pidsMap[val - m_refPID]);
540  //action completed, reset m_refPID to zero.
541  m_refPID = 0;
542  }
543  //writing
544  switch(FileType::_from_integral(m_winfo.ftype)){
545 
546  case FileType::STL :
547  //Export STL
548  {
549  auto pidsMap = getGeometry()->getPIDTypeListWNames();
550  std::unordered_map<int, std::string> mpp;
551  for(const auto & touple : pidsMap){
552  mpp[touple.first] = touple.second;
553  }
554 #if MIMMO_ENABLE_MPI
555  std::string name = (m_winfo.fdir+"/"+m_winfo.fname+"."+std::to_string(m_rank)+".stl");
556 #else
557  std::string name = (m_winfo.fdir+"/"+m_winfo.fname+".stl");
558 #endif
559  dynamic_cast<bitpit::SurfUnstructured*>(getGeometry()->getPatch())->exportSTL(name, m_codex, m_multiSolidSTL, false, &mpp);
560  return true;
561  }
562  break;
563 
564 
565  case FileType::SURFVTU :
566  case FileType::VOLVTU :
567  case FileType::CURVEVTU:
568  //Export Surface/Volume/3DCurve VTU
569  {
570  if(!m_codex){
571  VTUFlushStreamerASCII streamer;
572  VTUGridWriterASCII vtkascii(streamer, *(getGeometry()->getPatch()) );
573  vtkascii.write(m_winfo.fdir+"/", m_winfo.fname);
574  }
575  else{
576  getGeometry()->getPatch()->getVTK().setCodex(bitpit::VTKFormat::APPENDED);
577  getGeometry()->getPatch()->getVTK().setDirectory(m_winfo.fdir+"/");
578  getGeometry()->getPatch()->getVTK().setName(m_winfo.fname);
579  getGeometry()->getPatch()->write();
580  }
581  return true;
582  }
583  break;
584 
585  case FileType::NAS :
586  //Export Nastran file
587  {
588  liimap mapDataInv;
589  dvecarr3E points = getGeometry()->getVerticesCoords(&mapDataInv);
590  livector1D pointsID = getGeometry()->getVertices().getIds(false);
591  livector2D connectivity = getGeometry()->getCompactConnectivity(mapDataInv);
592  livector1D elementsID = getGeometry()->getCells().getIds(false);
593  NastranInterface nastran;
594  nastran.setWFormat(m_wformat);
595  livector1D pids = getGeometry()->getCompactPID();
596  std::unordered_set<long> pidsset = getGeometry()->getPIDTypeList();
597 #if MIMMO_ENABLE_MPI
598  std::string namefile = m_winfo.fname+"."+std::to_string(m_rank);
599 #else
600  std::string namefile = m_winfo.fname;
601 #endif
602  if (pids.size() == connectivity.size()){
603  nastran.write(m_winfo.fdir,namefile,points, pointsID, connectivity,elementsID, &pids, &pidsset);
604  }else{
605  nastran.write(m_winfo.fdir,namefile,points, pointsID, connectivity, elementsID);
606  }
607  return true;
608  }
609  break;
610 
611  case FileType::OFP : //Export ascii OpenFOAM point cloud
612  {
613 
614  dvecarr3E points = getGeometry()->getVerticesCoords();
615 #if MIMMO_ENABLE_MPI
616  std::string namefile = m_winfo.fname+"."+std::to_string(m_rank);
617 #else
618  std::string namefile = m_winfo.fname;
619 #endif
620  writeOFP(m_winfo.fdir, namefile, points);
621  return true;
622  }
623  break;
624 
625  case FileType::PCVTU :
626  //Export Point Cloud VTU
627  {
628  dvecarr3E points = getGeometry()->getVerticesCoords();
629  ivector2D connectivity(points.size(), ivector1D(1,0));
630  int counter = 0;
631  for(auto & c:connectivity){
632  c[0] = counter;
633  counter++;
634  }
635  bitpit::VTKUnstructuredGrid vtk(m_winfo.fdir, m_winfo.fname, bitpit::VTKElementType::VERTEX);
636  vtk.setGeomData( bitpit::VTKUnstructuredField::POINTS, points) ;
637  vtk.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, connectivity) ;
638  vtk.setDimensions(connectivity.size(), points.size());
639  if(!m_codex) vtk.setCodex(bitpit::VTKFormat::ASCII);
640  else vtk.setCodex(bitpit::VTKFormat::APPENDED);
641  vtk.write() ;
642  return true;
643  }
644  break;
645 
646  case FileType::MIMMO :
647  //Export in mimmo (bitpit) dump format
648  {
649  int archiveVersion = 1;
650  std::string header = m_name;
651  std::string filename = (m_winfo.fdir+"/"+m_winfo.fname+".geomimmo");
652 #if MIMMO_ENABLE_MPI
653  bitpit::OBinaryArchive binaryWriter(filename, archiveVersion, header, m_rank);
654 #else
655  bitpit::OBinaryArchive binaryWriter(filename, archiveVersion, header);
656 #endif
657  getGeometry()->dump(binaryWriter.getStream());
658  binaryWriter.close();
659  return true;
660  }
661  break;
662 
663  default: //never been reached
664  break;
665  }
666 
667  return false;
668 };

674 bool
675 MimmoGeometry::read(){
676 
677  if(!m_isInternal) return false;
678 
679  switch(FileType::_from_integral(m_rinfo.ftype)){
680 
681  //Import STL
682  case FileType::STL :
683  {
684  setGeometry(1);
685  std::string name;
686 #if MIMMO_ENABLE_MPI
687  if (m_rank == 0) {
688 #endif
689  {
690  std::ifstream infile(m_rinfo.fdir+"/"+m_rinfo.fname+".stl");
691  bool check = infile.good();
692  name = m_rinfo.fdir+"/"+m_rinfo.fname+".stl";
693  if (!check){
694  infile.open(m_rinfo.fdir+m_rinfo.fname+".STL");
695  check = infile.good();
696  name = m_rinfo.fdir+"/"+m_rinfo.fname+".STL";
697  if (!check) return false;
698  }
699  }
700 
701  std::ifstream in(name);
702  std::string ss, sstype;
703  std::getline(in,ss);
704  std::stringstream ins;
705  ins << ss;
706  ins >> sstype;
707  bool binary = true;
708  if (sstype == "solid" || sstype == "SOLID") binary = false;
709  in.close();
710 
711  std::unordered_map<int,std::string> mapPIDSolid;
712  dynamic_cast<bitpit::SurfUnstructured*>(getGeometry()->getPatch())->importSTL(name, binary, 0, false, &mapPIDSolid);
713 
714  getGeometry()->cleanGeometry();
715 
716  //count PID if multi-solid
717  auto & mapset = getGeometry()->getPIDTypeList();
718  for(const auto & cell : getGeometry()->getCells() ){
719  mapset.insert(cell.getPID());
720  }
721  std::unordered_map<long, std::string> & mapWNames = getGeometry()->getPIDTypeListWNames();
722  for(const auto & val :mapset ){
723  mapWNames[val] = mapPIDSolid[val];
724  }
725 #if MIMMO_ENABLE_MPI
726  }
727 #endif
728  }
729  break;
730 
731  case FileType::SURFVTU :
732  //Import Surface VTU meshes
733  {
734  setGeometry(1);
735 #if MIMMO_ENABLE_MPI
736  if (m_rank == 0) {
737 #endif
738  std::ifstream infile(m_rinfo.fdir+"/"+m_rinfo.fname+".vtu");
739  bool check = infile.good();
740  if (!check) return false;
741 
742  VTUGridStreamer vtustreamer;
743  VTUGridReader input(m_rinfo.fdir, m_rinfo.fname, vtustreamer, *(getGeometry()->getPatch()));
744 
745  input.read() ;
746 
747  getGeometry()->resyncPID();
748 #if MIMMO_ENABLE_MPI
749  }
750 #endif
751  }
752  break;
753 
754  case FileType::VOLVTU :
755  //Import Volume VTU meshes
756  {
757  setGeometry(2);
758 #if MIMMO_ENABLE_MPI
759  if (m_rank == 0) {
760 #endif
761  std::ifstream infile(m_rinfo.fdir+"/"+m_rinfo.fname+".vtu");
762  bool check = infile.good();
763  if (!check) return false;
764 
765  VTUGridStreamer vtustreamer;
766  VTUGridReader input(m_rinfo.fdir, m_rinfo.fname, vtustreamer, *(getGeometry()->getPatch()));
767  input.read() ;
768 
769  getGeometry()->resyncPID();
770 #if MIMMO_ENABLE_MPI
771  }
772 #endif
773  }
774  break;
775 
776  case FileType::NAS :
777  //Import Surface NAS
778  {
779  setGeometry(1);
780 #if MIMMO_ENABLE_MPI
781  if (m_rank == 0) {
782 #endif
783  std::ifstream infile(m_rinfo.fdir+"/"+m_rinfo.fname+".nas");
784  bool check = infile.good();
785  if (!check) return false;
786  infile.close();
787 
788  dvecarr3E Ipoints ;
789  livector2D Iconnectivity ;
790  livector1D pointsID;
791  livector1D cellsID;
792 
793  NastranInterface nastran;
794  nastran.setWFormat(m_wformat);
795 
796  livector1D pids;
797  nastran.read(m_rinfo.fdir, m_rinfo.fname, Ipoints, pointsID, Iconnectivity, cellsID, pids );
798 
799  bitpit::ElementType eltype;
800 
801  int sizeV, sizeC;
802  sizeV = Ipoints.size();
803  sizeC = Iconnectivity.size();
804  m_intgeo->getPatch()->reserveVertices(sizeV);
805  m_intgeo->getPatch()->reserveCells(sizeC);
806 
807  int counter = 0;
808  for(const auto & vv : Ipoints){
809  m_intgeo->addVertex(vv, pointsID[counter]);
810  ++counter;
811  }
812  counter = 0;
813  for(const auto & cc : Iconnectivity) {
814  eltype = bitpit::ElementType::UNDEFINED;
815  std::size_t ccsize = cc.size();
816  if(ccsize == 1) eltype = bitpit::ElementType::VERTEX;
817  if(ccsize == 3) eltype = bitpit::ElementType::TRIANGLE;
818  if(ccsize == 4) eltype = bitpit::ElementType::QUAD;
819  m_intgeo->addConnectedCell(cc, eltype, cellsID[counter]);
820  ++counter;
821  }
822  m_intgeo->setPID(pids);
823 #if MIMMO_ENABLE_MPI
824  }
825 #endif
826  }
827  break;
828 
829  //Import ascii OpenFOAM point cloud
830  case FileType::OFP :
831  {
832  setGeometry(3);
833 #if MIMMO_ENABLE_MPI
834  if (m_rank == 0) {
835 #endif
836  std::ifstream infile(m_rinfo.fdir+"/"+m_rinfo.fname);
837  bool check = infile.good();
838  if (!check) return false;
839  infile.close();
840 
841  dvecarr3E Ipoints;
842  readOFP(m_rinfo.fdir, m_rinfo.fname, Ipoints);
843 
844  int sizeV = Ipoints.size();
845  m_intgeo->getPatch()->reserveVertices(sizeV);
846 
847  for(const auto & vv : Ipoints) m_intgeo->addVertex(vv);
848 #if MIMMO_ENABLE_MPI
849  }
850 #endif
851  }
852  break;
853 
854  //Import point cloud from vtu
855  case FileType::PCVTU :
856  {
857  setGeometry(3);
858 #if MIMMO_ENABLE_MPI
859  if (m_rank == 0) {
860 #endif
861  std::ifstream infile(m_rinfo.fdir+"/"+m_rinfo.fname+".vtu");
862  bool check = infile.good();
863  if (!check) return false;
864 
865  VTUPointCloudStreamer vtustreamer;
866  VTUGridReader input(m_rinfo.fdir, m_rinfo.fname, vtustreamer, *(getGeometry()->getPatch()), bitpit::VTKElementType::VERTEX);
867  input.read() ;
868 #if MIMMO_ENABLE_MPI
869  }
870 #endif
871  }
872  break;
873

256  template<class Type>
257  void writeValue (Type& value, std::ofstream& os){
258 
259  int offset = 5;
260  if (wformat == Long) offset = 13;
261 
262  std::stringstream manip;
263  manip << value;
264  std::string manips = manip.str();
265  std::string mantissa, expon;
266  int pos = manips.find("E");
267  if (pos < manips.size()){
268  mantissa = manips.substr(0,std::min(offset,pos));
269  expon = manips.substr(pos+1,manips.size());
270  manips = mantissa + expon;
271  }
272  pos = manips.find("e");
273  if (pos < manips.size()){
274  mantissa = manips.substr(0,std::min(offset,pos));
275  expon = manips.substr(pos+1,manips.size());
276  manips = mantissa + expon;
277  }
278  manips = manips.substr(0, offset+3);
279 
280  switch (wformat)
281  {
282  case Short:
283  {
284  os << std::setw(8) << manips;
285  break;
286  }
287  case Long:
288  {
289  os << std::setw(16) << manips;
290  break;
291  }
292  }
293  }
294 };

1416 void MimmoObject::updatePointGhostExchangeInfo()
1417 {
1418 
1419  // Clear targets
1420  m_pointGhostExchangeTargets.clear();
1421 
1422  // Clear sources
1423  m_pointGhostExchangeSources.clear();
1424 
1425  // Clear shared
1426  m_pointGhostExchangeShared.clear();
1427 
1428  //Clear interior points structure
1429  m_isPointInterior.clear();
1430 
1431  //Fill interior points structure
1432  //Initialize as interior all the local nodes
1433  for (long id : getVertices().getIds()){
1434  m_isPointInterior[id] = true;
1435  }
1436 
1437  //Start update structure if partitioned
1438  if (getPatch()->isPartitioned()){
1439 
1440  //Fill the nodes of the targets
1441  for (const auto &entry : m_patch->getGhostExchangeTargets()) {
1442  int ghostRank = entry.first;
1443  std::vector<long> ghostIds = entry.second;
1444  std::unordered_set<long> ghostVertices;
1445  for (long cellId : ghostIds){
1446  for (long vertexId : m_patch->getCell(cellId).getVertexIds()){
1447  ghostVertices.insert(vertexId);
1448  }
1449  }
1450  m_pointGhostExchangeTargets[ghostRank].assign(ghostVertices.begin(), ghostVertices.end());
1451  }
1452 
1453  //Fill the nodes of the sources
1454  for (const auto &entry : m_patch->getGhostExchangeSources()) {
1455  int recvRank = entry.first;
1456  std::vector<long> localIds = entry.second;
1457  std::unordered_set<long> localVertices;
1458  for (long cellId : localIds){
1459  for (long vertexId : m_patch->getCell(cellId).getVertexIds()){
1460  localVertices.insert(vertexId);
1461  }
1462  }
1463  m_pointGhostExchangeSources[recvRank].assign(localVertices.begin(), localVertices.end());
1464  }
1465 
1466  //Note. All the processes will receive the data on the shared points from all the processes,
1467  // but they will save the data received by the lowest process that shares those nodes.
1468 
1469  //Erase common vertices
1470  for (auto & sourceEntry : m_pointGhostExchangeSources){
1471  int recvRank = sourceEntry.first;
1472  if (m_pointGhostExchangeTargets.count(recvRank)){
1473  std::size_t sourceLast = sourceEntry.second.size()-1;
1474  auto & sourceVertices = sourceEntry.second;
1475  std::size_t targetLast = m_pointGhostExchangeTargets[recvRank].size()-1;
1476  auto & targetVertices = m_pointGhostExchangeTargets[recvRank];
1477  std::vector<long>::iterator itsourceend = sourceVertices.end();
1478  for (std::vector<long>::iterator itsource=sourceVertices.begin(); itsource!=itsourceend; itsource++){
1479  auto iter = std::find(targetVertices.begin(), targetVertices.end(), *itsource);
1480  if (iter != targetVertices.end()){
1481  //insert shared point
1482  m_pointGhostExchangeShared[recvRank].push_back(*itsource);
1483 
1484  //Maintain shared points only in the sources of the lower rank and in the target of the greater one.
1485  //Note. Some nodes will be repeated if shared by several processes. During the communications only the
1486  //the data given by the lowest sender will be saved.
1487  if (m_rank < recvRank){
1488  //The nodes are not repeated so the target vertices don't need to be resized during loop
1489  *iter = targetVertices[targetLast];
1490  targetLast--;
1491  }
1492  else if (m_rank > recvRank){
1493  *itsource = *(sourceVertices.end()-1);
1494  sourceLast--;
1495  sourceVertices.resize(sourceLast+1);
1496  itsourceend = sourceVertices.end();
1497  itsource--;
1498  }
1499  }
1500  }
1501  targetVertices.resize(targetLast+1);
1502  }
1503  }
1504 
1505  // Sort the targets
1506  for (auto &entry : m_pointGhostExchangeTargets) {
1507  std::vector<long> &rankTargets = entry.second;
1508  std::sort(rankTargets.begin(), rankTargets.end(), VertexPositionLess(*this));
1509  }
1510 
1511  // Sort the sources
1512  for (auto &entry : m_pointGhostExchangeSources) {
1513  std::vector<long> &rankSources = entry.second;
1514  std::sort(rankSources.begin(), rankSources.end(), VertexPositionLess(*this));
1515  }
1516 
1517  // Sort the shared
1518  for (auto &entry : m_pointGhostExchangeShared) {
1519  std::vector<long> &rankShared = entry.second;
1520  std::sort(rankShared.begin(), rankShared.end(), VertexPositionLess(*this));
1521  }
1522 
1523 
1524  //Correct target ghost points
1525  for (auto &entry : m_pointGhostExchangeTargets) {
1526  std::vector<long> &rankTargets = entry.second;
1527  for (long id : rankTargets){
1528  m_isPointInterior.at(id) = false;
1529  }
1530  }
1531 
1532  //The owner of a shared point is the lower rank
1533  for (auto &entry : m_pointGhostExchangeShared){
1534  int sharingRank = entry.first;
1535  //Correct shared points only if shared with a lower rank
1536  if (sharingRank < m_rank){
1537  std::vector<long> &sharedPoints = entry.second;
1538  for (long id : sharedPoints){
1539  m_isPointInterior.at(id) = false;
1540  }
1541  }
1542  }
1543 
1544  }//end update if partitioned
1545 
1546 
1547  //Update n interior vertices
1548  m_ninteriorvertices = 0;
1549  for (auto &entry : m_isPointInterior){
1550  if (entry.second)
1551  m_ninteriorvertices++;
1552  }
1553

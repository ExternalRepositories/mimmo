344  struct VertexPositionLess
345  {
349  VertexPositionLess(const MimmoObject &object)
350  : m_object(object)
351  {
352  }
354  virtual ~VertexPositionLess() = default;
355 
365  bool operator()(const long &id_1, const long &id_2) const
366  {
367  std::array<double, 3> vertex_1 = m_object.getVertexCoords(id_1);
368  std::array<double, 3> vertex_2 = m_object.getVertexCoords(id_2);
369 
370  for (int k = 0; k < 3; ++k) {
371  if (std::abs(vertex_1[k] - vertex_2[k]) <= 1.0E-12) {
372  continue;
373  }
374  return vertex_1[k] < vertex_2[k];
375  }
376 
377  // If we are here the two vertices coincide. It's not
378  // possible to define an order for the two vertices.
379  std::ostringstream stream;
380  stream << "It was not possible to define an order for vertices " << id_1 << " and " << id_2 << ". ";
381  stream << "The two vertices have the same coordinates.";
382  throw std::runtime_error (stream.str());
383  }
384 
385  const MimmoObject &m_object;
386  };

168 template<typename mpv_t>
169 std::vector<mpv_t>
170 MimmoPiercedVector<mpv_t>::getInternalDataAsVector(bool ordered, bool squeeze){
171  if(getGeometry() == NULL) return std::vector<mpv_t>(0);
172  std::vector<mpv_t> result;
173  livector1D ids;
174  std::unordered_set<long> ids_;
175  switch (getDataLocation())
176  {
177  case MPVLocation::CELL:
178  result.reserve(getGeometry()->getPatch()->getInternalCount());
179  ids = getGeometryIds(ordered);
180  for(const auto val: ids){
181  if(this->exists(val) && getGeometry()->getPatch()->getCell(val).isInterior()){
182  result.push_back((*this)[val]);
183  }
184  }
185  break;
186  case MPVLocation::POINT:
187  result.reserve(getGeometry()->getPatch()->getVertexCount());
188  for (auto & cell : getGeometry()->getCells()){
189  if (cell.isInterior()){
190  for (long id : cell.getVertexIds())
191  ids_.insert(id);
192  }
193  }
194  ids = getGeometryIds(ordered);
195  for(const auto val: ids){
196  if(this->exists(val) && ids_.count(val)){
197  result.push_back((*this)[val]);
198  }
199  }
200  break;
201  case MPVLocation::INTERFACE:
202  result.reserve(getGeometry()->getPatch()->getInterfaceCount());
203  //Interfaces ghost/ghost are not stored in bitpit::PathcKernel, so use all the interfaces in the geometry
204  ids = getGeometryIds(ordered);
205  for(const auto val: ids){
206  if(this->exists(val)){
207  result.push_back((*this)[val]);
208  }
209  }
210  break;
211  default:
212  break;
213  }
214  if (squeeze)
215  result.shrink_to_fit();
216  return result;
217 }

297 template<typename mpv_t>
298 bool
299 MimmoPiercedVector<mpv_t>::checkDataSizeCoherence(){
300  if(getGeometry()==NULL) return false;
301  bool check = false;
302  switch(m_loc){
303  case MPVLocation::CELL:
304  check = (this->size()==m_geometry->getPatch()->getCells().size());
305  break;
306  case MPVLocation::INTERFACE:
307  {
308  size_t sizeInterfaces = m_geometry->getPatch()->getInterfaces().size();
309  check = (this->size()==sizeInterfaces);
310  if(sizeInterfaces == 0){
311  (*m_log)<<"Warning: Asked Data Size Coherence in MimmoPiercedVector for INTERFACES, but linked geometry may not have them built."<<std::endl;
312  }
313  }
314  break;
315  case MPVLocation::POINT:
316  check = (this->size()==m_geometry->getPatch()->getVertices().size());
317  break;
318  default:
319  check=false;
320  (*m_log)<<"NO suitable location data found to perform data size coherence check"<<std::endl;
321  break;
322  }
323  return check;
324 }

334 template<typename mpv_t>
335 bool
336 MimmoPiercedVector<mpv_t>::checkDataIdsCoherence(){
337  if(getGeometry()==NULL) return false;
338  if (this->isEmpty()) return true;
339  auto ids = this->getIds();
340  bool check = true;
341  switch(m_loc){
342  case MPVLocation::CELL:
343  {
344  auto vcell = m_geometry->getPatch()->getCells();
345  for(auto el : ids){
346  check =check && vcell.exists(el);
347  }
348  }
349  break;
350  case MPVLocation::INTERFACE:
351  {
352  size_t sizeInterfaces = m_geometry->getPatch()->getInterfaces().size();
353  if(sizeInterfaces == 0){
354  (*m_log)<<"Warning: Asked Data Ids Coherence in MimmoPiercedVector for INTERFACES, but linked geometry may not have them built."<<std::endl;
355  }
356  auto vint = m_geometry->getPatch()->getInterfaces();
357  for(auto el : ids){
358  check =check && vint.exists(el);
359  }
360  }
361  break;
362  case MPVLocation::POINT:
363  {
364  auto vvert = m_geometry->getPatch()->getVertices();
365  for(auto el : ids){
366  check =check && vvert.exists(el);
367  }
368  }
369  break;
370  default:
371  check = false;
372  (*m_log)<<"NO suitable location data found to perform ids coherence check"<<std::endl;
373  break;
374  }
375  return check;
376 }

587 template<typename mpv_t>
588 MimmoPiercedVector<mpv_t> MimmoPiercedVector<mpv_t>::pointDataToCellData(double p){
589  MimmoObject* geo = this->getGeometry();
590  MimmoPiercedVector<mpv_t> cellData(geo, MPVLocation::CELL);
591  MimmoPiercedVector<double> sumWeights(geo, MPVLocation::POINT);
592  for (bitpit::Cell & cell : geo->getCells()){
593  long idcell = cell.getId();
594  std::array<double,3> center = geo->getPatch()->evalCellCentroid(idcell);
595  mpv_t data{};
596  bool init = false;
597  for (long idvertex : cell.getVertexIds()){
598  std::array<double,3> point = geo->getPatch()->getVertexCoords(idvertex);
599  double weight = 1. / std::pow(norm2(center-point),p);
600  if (!init){
601  data = this->at(idvertex)*weight;
602  sumWeights.insert(idcell, weight);
603  init = true;
604  }
605  else{
606  data = data + this->at(idvertex)*weight;
607  sumWeights[idcell] = sumWeights[idcell] + weight;
608  }
609  }
610  data = data / sumWeights[idcell];
611  cellData.insert(idcell, data);
612  }
613  return cellData;
614 };

624 MimmoPiercedVector<mpv_t> MimmoPiercedVector<mpv_t>::cellDataToPointData(double p){
625  MimmoObject* geo = this->getGeometry();
626  MimmoPiercedVector<mpv_t> pointData(geo, MPVLocation::POINT);
627  MimmoPiercedVector<int> countCells(geo, MPVLocation::POINT);
628  MimmoPiercedVector<double> sumWeights(geo, MPVLocation::POINT);
629  for (bitpit::Cell & cell : geo->getCells()){
630  long idcell = cell.getId();
631  if (this->exists(idcell)){
632  std::array<double,3> center = geo->getPatch()->evalCellCentroid(idcell);
633  for (long idvertex : cell.getVertexIds()){
634  std::array<double,3> point = geo->getPatch()->getVertexCoords(idvertex);
635  double weight = 1. / std::pow(norm2(center-point),p);
636  if (!pointData.exists(idvertex)){
637  pointData.insert(idvertex, this->at(idcell)*weight);
638  sumWeights.insert(idvertex, weight);
639  }
640  else{
641  pointData[idvertex] = pointData[idvertex] + this->at(idcell)*weight;
642  sumWeights[idvertex] = sumWeights[idvertex] + weight;
643  }
644  }
645  }
646  }
647  for (bitpit::Vertex & vertex : geo->getVertices()){
648  long idvertex = vertex.getId();
649  if (pointData.exists(idvertex))
650  pointData[idvertex] = pointData[idvertex] / sumWeights[idvertex];
651  }
652  return pointData;

665 template<typename mpv_t>
666 MimmoPiercedVector<mpv_t> MimmoPiercedVector<mpv_t>::cellDataToPointData(const MimmoPiercedVector<mpv_t> & cellGradientsX, const MimmoPiercedVector<mpv_t> & cellGradientsY, const MimmoPiercedVector<mpv_t> & cellGradientsZ, bool maximum){
667  double p = 3.;
668  if (maximum){
669  p = 0.;
670  }
671  MimmoObject* geo = this->getGeometry();
672  MimmoPiercedVector<mpv_t> pointData(geo, MPVLocation::POINT);
673  MimmoPiercedVector<double> sumWeights(geo, MPVLocation::POINT);
674  for (bitpit::Cell & cell : geo->getCells()){
675  long idcell = cell.getId();
676  if (this->exists(idcell)){
677  double volume = geo->evalCellVolume(idcell);
678  double weight = 1. / std::pow(volume,p);
679  std::array<double,3> center = geo->getPatch()->evalCellCentroid(idcell);
680  for (long idvertex : cell.getVertexIds()){
681  std::array<double,3> point = geo->getPatch()->getVertexCoords(idvertex);
682  std::array<mpv_t,3> grad({cellGradientsX.at(idcell), cellGradientsY.at(idcell), cellGradientsZ.at(idcell)});
683  mpv_t value;
684  value = this->at(idcell) + this->at(idcell)*(grad[0]*(point-center)[0]);
685  value = value + this->at(idcell)*(grad[1]*(point-center)[1]);
686  value = value + this->at(idcell)*(grad[2]*(point-center)[2]);
687  value = value * weight;
688  if (!pointData.exists(idvertex)){
689  pointData.insert(idvertex, value);
690  sumWeights.insert(idvertex, weight);
691  }
692  else{
693  if (maximum){
694  if (norm2(value) > norm2(pointData[idvertex]))
695  pointData[idvertex] = value;
696  }
697  else{
698  pointData[idvertex] = pointData[idvertex] + value;
699  sumWeights[idvertex] = sumWeights[idvertex] + weight;
700  }
701  }
702  }
703  }
704  }
705  if (!maximum){
706  for (bitpit::Vertex & vertex : geo->getVertices()){
707  long idvertex = vertex.getId();
708  if (pointData.exists(idvertex)){
709  pointData[idvertex] = pointData[idvertex] / sumWeights[idvertex];
710  }
711  }
712  }
713  return pointData;
714 };

800 template<typename mpv_t>
801 void
802 MimmoPiercedVector<mpv_t>::squeezeOutExcept(const std::vector<long int> & list, bool keepOrder){
803 
804  if(keepOrder){
805  // erase elements in the list then squeeze the piercedvector.
806  std::unordered_set<long> maplist(list.begin(), list.end());
807  MimmoPiercedVector<mpv_t> result (m_geometry, m_loc);
808  result.reserve(maplist.size());
809  for(auto it = this->begin(); it != this->end(); ++it){
810  if(maplist.count(it.getId()) > 0){
811  result.insert(it.getId(), *it);
812  }
813  }
814  this->swap(result);
815  }else{
816  // if the order does not matter, create a new mpv and swap with the current
817  MimmoPiercedVector<mpv_t> result (m_geometry, m_loc);
818  for(long id: list){
819  if(this->exists(id)){
820  result.insert(id, this->at(id));
821  }
822  }
823  this->swap(result);
824  }
825 }

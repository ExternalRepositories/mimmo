249

263  darray3E temp;
264  dmatrix33E inv = inverse(m_axes);
265  for(auto &val : activeP){

270  }

293 void
294 OBBox::execute(){
295 
296  darray3E pmin, pmax;
297  pmin.fill(1.e18);
298  pmax.fill(-1.e18);
299  double val;
300  dmatrix33E eye;
301  {
302  int count = 0;
303  for(auto & local: eye) {
304  local.fill(0);
305  local[count] = 1.0;
306  m_axes[count] = local;
307  ++count;
308  }
309  }
310  // check if the list is empty.
311  if(m_listgeo.empty()){
312  m_span.fill(0.0);
313  m_origin.fill(0.0);
314  return;
315  };
316 
317  std::unordered_map<MimmoObject*, int>::iterator itB = m_listgeo.begin();
318  //if one geometry at least is a cloud point, solve all them as cloud points.
319  bool allCloud = false;
320  while(itB != m_listgeo.end() && !allCloud){
321  allCloud = (itB->second == 3 || itB->second == 4);
322  itB++;
323  }
324 
325  if(!m_forceAABB){
326 
327  dmatrix33E covariance;
328  darray3E spectrum;
329 
330  if(allCloud){
331  covariance = evaluatePointsCovarianceMatrix(getGeometries());
332  }else{
333  covariance = evaluateElementsCovarianceMatrix(getGeometries());
334  }
335  m_axes = eigenVectors(covariance, spectrum);
336  adjustBasis(m_axes, spectrum);
337  }
338 
339 
340  for(auto ptr : getGeometries()){
341  for(auto & vert: ptr->getVertices()){
342  darray3E coord = vert.getCoords();
343  for(int i=0;i<3; ++i){
344  val = dotProduct(coord, m_axes[i]);
345  pmin[i] = std::fmin(pmin[i], val);
346  pmax[i] = std::fmax(pmax[i], val);
347  }
348  }
349  }
350 
351  dmatrix33E inv = inverse(m_axes);
352  m_span = pmax - pmin;
353  //check if one of the span goes to 0;
354  double avg_span = 0.0;
355  for(auto & val: m_span) avg_span+=val;
356  avg_span /= 3.0;
357 
358  for(auto &val : m_span) {
359  val = std::fmax(val, 1.E-04*avg_span);
360  }
361 
362  darray3E originLoc = 0.5*(pmin+pmax);
363  for(int i=0; i<3; ++i){
364  m_origin[i] = dotProduct(originLoc, inv[i]);
365  }
366 
367 
368  if (m_writeInfo){
369 
370  std::ofstream out;
371  out.open(m_outputPlot+"/"+m_name+std::to_string(getId())+"_INFO.dat");
372  if(out.is_open()){
373  out<<"OBBox "<<std::to_string(getId())<<" info:"<<std::endl;
374  out<<std::endl;
375  out<<std::endl;
376  out<<"Origin: "<<std::scientific<<m_origin<<std::endl;
377  out<<std::endl;
378  out<<"Axis 0: "<<std::scientific<<m_axes[0]<<std::endl;
379  out<<"Axis 1: "<<std::scientific<<m_axes[1]<<std::endl;
380  out<<"Axis 2: "<<std::scientific<<m_axes[2]<<std::endl;
381  out<<std::endl;
382  out<<"Span: "<<std::scientific<<m_span<<std::endl;
383 
384  out.close();
385  }
386  }
387 };

456 dmatrix33E
457 OBBox::evaluatePointsCovarianceMatrix(std::vector<MimmoObject*> list){
458 
459  //evaluate the mass center first;
460  darray3E masscenter = {{0.0,0.0,0.0}};
461  int countVert = 0;
462  for(auto geo: list){
463  for(auto & vert: geo->getVertices()){
464  masscenter += vert.getCoords();
465  ++countVert;
466  }
467  }
468 
469  masscenter /= double(countVert);
470 
471  //assembly covariance matrix node by node.
472  dmatrix33E loc;
473  dmatrix33E covariance;
474  for(auto & val:covariance) val.fill(0.0);
475 
476  darray3E temp;
477  for(auto geo: list){
478  for(auto & val:loc) val.fill(0.0);
479  //evaluate local contributes to covariance
480  for(auto & vert: geo->getVertices()){
481  temp = vert.getCoords() - masscenter;
482  for(int j=0; j<3; ++j){
483  for(int k=j; k<3; ++k){
484  loc[j][k] += temp[j]*temp[k];
485  }
486  }
487  }
488  //store temporarely in covContributes
489  int counter = 0;
490  for(auto & val: covariance){
491  val += loc[counter];
492  ++counter;
493  }
494  }
495 
496  int counter = 0;
497  for(auto & val: covariance){
498  val /= double(countVert);
499  ++counter;
500  }
501 
502  covariance[1][0] = covariance[0][1];
503  covariance[2][0] = covariance[0][2];
504  covariance[2][1] = covariance[1][2];
505 
506  return covariance;
507 };

516 dmatrix33E
517 OBBox::evaluateElementsCovarianceMatrix(std::vector<MimmoObject*> list){
518 
519  //You need to evaluate the mass center and the 3x3 matrix of the total moments
520  //obtained as the sum of each element moments.
521  // Non triangular cells are approximated as a representative triangle formed by
522  // the first set of 3 non-aligned vertices.
523 
524 
525  darray3E masscenter = {{0.0,0.0,0.0}};
526  //assembly moments matrix cell by cell.
527  dmatrix33E moments;
528  for(auto & val:moments) val.fill(0.0);
529 
530  darray3E p,q,r, centroid;
531  double areatri, areatot(0.0);
532 
533  for(auto geo: list){
534  for(auto & cell: geo->getCells()){
535 
536  std::array<long,3> vids = get3RepPoints(cell.getId(), geo);
537  if(vids[0] < 0){
538  continue;
539  }
540  p = geo->getVertexCoords(vids[0]);
541  q = geo->getVertexCoords(vids[1]);
542  r = geo->getVertexCoords(vids[2]);
543 
544  areatri = 0.5*norm2(crossProduct(q-p, r-p));
545  centroid = (p+q+r)/3.0;
546  masscenter += areatri*centroid;
547  areatot += areatri;
548 
549  //assemby moments
550  // on-diagonal terms
551  moments[0][0] += areatri*(9.0*centroid[0]*centroid[0] + p[0]*p[0] + q[0]*q[0] + r[0]*r[0])/12.0;
552  moments[1][1] += areatri*(9.0*centroid[1]*centroid[1] + p[1]*p[1] + q[1]*q[1] + r[1]*r[1])/12.0;
553  moments[2][2] += areatri*(9.0*centroid[2]*centroid[2] + p[2]*p[2] + q[2]*q[2] + r[2]*r[2])/12.0;
554 
555  // off-diagonal terms
556  moments[0][1] += areatri*(9.0*centroid[0]*centroid[1] + p[0]*p[1] + q[0]*q[1] + r[0]*r[1])/12.0;
557  moments[0][2] += areatri*(9.0*centroid[0]*centroid[2] + p[0]*p[2] + q[0]*q[2] + r[0]*r[2])/12.0;
558  moments[1][2] += areatri*(9.0*centroid[1]*centroid[2] + p[1]*p[2] + q[1]*q[2] + r[1]*r[2])/12.0;
559  }
560  }// end of cell by cell loop.
561 
562  //normalize masscenter
563  masscenter /= areatot;
564 
565  // moments are symmetric
566  moments[1][0] = moments[0][1];
567  moments[2][0] = moments[0][2];
568  moments[2][1] = moments[1][2];
569 
570  //get final covariance matrix
571  dmatrix33E covariance;
572  for(int i=0; i<3; ++i ){
573  for(int j=0; j<3; ++j ){
574  covariance[i][j] = moments[i][j]/areatot - masscenter[i]*masscenter[j];
575  }
576  }
577  return covariance;
578 };

587 dmatrix33E
588 OBBox::eigenVectors( dmatrix33E & matrix, darray3E & eigenvalues){
589 
590  dmatrix33E result;
591  double * a = new double [9];
592  double * s = new double[3];
593 
594  int k=0;
595  for(int i=0; i<3; i++){
596  for(int j=0; j<3; j++){
597  //transpose
598  a[k] = matrix[j][i];
599  k++;
600  }
601  }
602 
603  LAPACKE_dsyev( LAPACK_COL_MAJOR, 'V','U', 3, a, 3, s);
604  //eigenvec are returned in a column wise, and with eigenvals s in ascending order.
605 
606  //rearrange the eigvec
607  for (int i=0; i<9; i++){
608  result[i/3][i%3] = a[i];
609  }
610  for(int i=0; i<3; ++i) {
611  result[i] /= norm2(result[i]);
612  eigenvalues[i] = s[i];
613  }
614 
615  //change them in descending order.
616  std::swap(result[0], result[2]);
617  std::swap(eigenvalues[0], eigenvalues[2]);
618 
619  // clean allocations.
620  delete [] a; a = NULL;
621  delete [] s; s = NULL;
622 
623  return result;
624 }

635 void
636 OBBox::adjustBasis(dmatrix33E & eigVec, darray3E & eigVal){
637 
638  darray3E diff;
639  double tol = std::numeric_limits<double>::min();
640 
641  diff[0] = std::abs(eigVal[1]-eigVal[0]);
642  diff[1] = std::abs(eigVal[2]-eigVal[1]);
643  diff[2] = std::abs(eigVal[0]-eigVal[2]);
644 
645  //3 real quasi coincident eigenval-> return fundamental axis.
646  if(diff[0]<=tol && diff[1]<=tol && diff[2]<= tol){
647  int counter=0;
648  for(auto & val : eigVec){
649  val.fill(0.0);
650  val[counter] = 1.0;
651  ++counter;
652  }
653  return;
654  }
655 
656  // 3 real distinct eigenval
657  if(diff[0]>tol && diff[1]>tol && diff[2]>tol){
658  //you need a real reference triad here. Check it out
659  if(dotProduct(eigVec[2], crossProduct(eigVec[0], eigVec[1])) < 0.0){
660  eigVec[2] *= -1.0;
661  };
662  return;
663  };
664 
665  int guess = 0, third =1, stable=2;
666 
667  if(diff[1] <= tol) {
668  guess = 1;
669  third = 2;
670  stable= 0;
671  }
672 
673  eigVec[third] = crossProduct(eigVec[stable],eigVec[guess]);
674  double normv = norm2(eigVec[third]);
675  if(normv > std::numeric_limits<double>::min()){
676  eigVec[third] /= normv;
677  }
678 
679  //you need a real reference triad here. Check it out
680  if(dotProduct(eigVec[2], crossProduct(eigVec[0], eigVec[1])) < 0.0){
681  eigVec[2] *= -1.0;
682  };
683 
684 };

708 dmatrix33E OBBox::inverse(const dmatrix33E & mat){
709  dmatrix33E out;
710 
711  double det = mat[0][0] * (mat[1][1]*mat[2][2] - mat[2][1]*mat[1][2]) -
712  mat[0][1] * (mat[1][0]*mat[2][2] - mat[2][0]*mat[1][2]) +
713  mat[0][2] * (mat[1][0]*mat[2][1] - mat[2][0]*mat[1][1]);
714 
715  out[0][0] = (mat[1][1]*mat[2][2] - mat[2][1]*mat[1][2])/det;
716  out[0][1] = (mat[0][2]*mat[2][1] - mat[2][2]*mat[0][1])/det;
717  out[0][2] = (mat[0][1]*mat[1][2] - mat[1][1]*mat[0][2])/det;
718  out[1][0] = (mat[1][2]*mat[2][0] - mat[2][2]*mat[1][0])/det;
719  out[1][1] = (mat[0][0]*mat[2][2] - mat[2][0]*mat[0][2])/det;
720  out[1][2] = (mat[0][2]*mat[1][0] - mat[1][2]*mat[0][0])/det;
721  out[2][0] = (mat[1][0]*mat[2][1] - mat[2][0]*mat[1][1])/det;
722  out[2][1] = (mat[0][1]*mat[2][0] - mat[2][1]*mat[0][0])/det;
723  out[2][2] = (mat[0][0]*mat[1][1] - mat[1][0]*mat[0][1])/det;
724 
725 
726  return out;
727 }

737 std::array<long,3> OBBox::get3RepPoints(long cellID, MimmoObject * geo){
738  if(!geo) return{{-1,-1,-1}};
739  bitpit::Cell & cell = geo->getCells().at(cellID);
740  std::array<long,3> result;
741 
742  switch(cell.getType()){
743  case bitpit::ElementType::TRIANGLE:
744  {
745  long * conn = cell.getConnect();
746  result[0] = conn[0];
747  result[1] = conn[1];
748  result[2] = conn[2];
749  }
750  break;
751  case bitpit::ElementType::QUAD:
752  case bitpit::ElementType::POLYGON:
753  {
754  bitpit::ConstProxyVector<long> vertids = cell.getVertexIds();
755  std::size_t ssz= vertids.size();
756 
757  //find a set of 3 points whose cross product is non zero, if you can.
758  darray3E p0,s1,s2;
759  result[0] = vertids[0];
760  p0 = geo->getVertexCoords(result[0]);
761 
762  std::size_t cand = 2;
763  bool found = false;
764

298 Proj3DCurveOnSurface::projection(){
299 
300  std::unique_ptr<MimmoObject> dum(new MimmoObject(4));
301 
302  dvecarr3E points;
303  std::unordered_map<long, std::array<long,2> > connectivity;
304  int furtherCells = fillPreliminaryStructure(points, connectivity);
305  refineObject(points, connectivity, furtherCells);
306 
307 
308  //...and projecting them onto target surface
309  if(!getGeometry()->isSkdTreeSync()) getGeometry()->buildSkdTree();
310  int counter=0;
311  dvecarr3E projs(points.size());
312  for(auto &val : points){
313  projs[counter]= skdTreeUtils::projectPoint(&val, getGeometry()->getSkdTree());
314  ++counter;
315  }
316 
317  dum->getPatch()->reserveVertices(points.size());
318  dum->getPatch()->reserveCells(connectivity.size());
319 
320  //start filling connectivity of your object.
321  bitpit::ElementType eltype = bitpit::ElementType::LINE;
322 
323  for(auto & con: connectivity){
324  livector1D con2(2);
325  con2[0] = con.second[0];
326  con2[1] = con.second[1];
327  dum->addConnectedCell(con2, eltype);
328  }
329 
330  //storing the projected points in the MimmoObject:
331  long idS = 0;
332  for(const auto & vv : projs){
333  dum->addVertex(vv, idS);
334  ++idS;
335  }
336 
337  dum->cleanGeometry();
338  m_patch = std::move(dum);
339 };

351 int
352 Proj3DCurveOnSurface::fillPreliminaryStructure(dvecarr3E & points, std::unordered_map<long, std::array<long,2> > &connectivity){
353  int nVerts, nCells, res;
354  long id;
355  liimap mapDataInv;
356  if (m_cobj) {
357  nVerts = m_cobj->getNVertices();
358  mapDataInv = m_cobj->getMapDataInv();
359  if(m_cobj->getType() == 4){
360  nCells = m_cobj->getNCells();
361  res = std::max(0, (m_nC - nCells));
362  auto conns= m_cobj->getCompactConnectivity(mapDataInv);
363  id=0;
364  for(auto & val: conns){
365  for(int i=0; i<2; ++i){
366  connectivity[id][i] = val[i];
367  }
368  ++id;
369  }
370  }else{
371  nCells = nVerts-1+(int)m_closed;
372  res = std::max(0, (m_nC - nVerts));
373  for(int i=0; i<(nCells -(int)m_closed); ++i){
374  connectivity[i][0] = i;
375  connectivity[i][1] = i+1;
376  }
377  if(m_closed){
378  connectivity[nCells-1][0] = nCells-1;
379  connectivity[nCells-1][1] = 0;
380  }
381  }
382  points = m_cobj->getVerticesCoords();
383 
384  }else{
385  nVerts = (int) m_cpoints.size();
386  nCells = nVerts-1+(int)m_closed;
387  res = std::max(0, (m_nC - nVerts));
388  points = m_cpoints;
389  for(int i=0; i<(nCells -(int)m_closed); ++i){
390  connectivity[i][0] = i;
391  connectivity[i][1] = i+1;
392  }
393  if(m_closed){
394  connectivity[nCells-1][0] = nCells-1;
395  connectivity[nCells-1][1] = 0;
396  }
397  }
398 
399  return (res);
400 }

410 void
411 Proj3DCurveOnSurface::refineObject(dvecarr3E & points, std::unordered_map<long, std::array<long,2> > &connectivity, int fCells){
412  if ( fCells == 0) return;
413 
414  greatDist mycomp;
415 
416  double dist;
417  int idV = (int)points.size() -1;
418  long idC = 0;
419  std::vector<std::pair<double, long>> values;
420  values.reserve(int(points.size()) + fCells);
421  for (auto &conn : connectivity){
422  dist = norm2(points[conn.second[1]] - points[conn.second[0]]);
423  values.push_back(std::make_pair(dist, conn.first));
424  idC = std::max(idC, conn.first);
425  }
426 
427  //prep your priority queue
428  std::priority_queue<std::pair<double,long>, std::vector<std::pair<double,long>>, greatDist > pQue(mycomp, values);
429 
430  int counterCell = 0;
431  darray3E point;
432  while (counterCell < fCells){
433 
434  //extract the longest segment
435  auto pp = pQue.top();
436  pQue.pop();
437 
438  idV++;
439  idC++;
440  //divide it by two and create a new point
441  point = 0.5*(points[connectivity[pp.second][1]] + points[connectivity[pp.second][0]]);
442  points.push_back(point);
443  //update connectivity
444  connectivity[idC]= {{ idV, connectivity[pp.second][1] }};
445  connectivity[pp.second][1] = idV;
446 
447  //update the priority queue
448  pQue.push(std::make_pair(0.5*pp.first, pp.second));
449  pQue.push(std::make_pair(0.5*pp.first, idC));
450 
451  //adding a new cell to the count;
452  ++counterCell;
453  }
454 };

461 template<std::size_t NCOMP>
462 void
463 PropagateField<NCOMP>::computeDumpingFunction(){
464 
465  bitpit::PatchKernel * patch_ = getGeometry()->getPatch();
466  double dist;
467  long ID;
468 
469  /* Maxdist should be the maximum distance between
470  * boundaries with zero values and
471  * boundaries with values different from zero.
472  */
473  //TODO compute it
474  const double maxd(m_radius);
475  m_dumping.clear();
476 
477  for (auto const & vertex : patch_->getVertices()){
478  ID = vertex.getId();
479  m_dumping.insert(ID, 1.0);
480  }
481 
482  if (m_dumpingActive && m_decayFactor > 1.0e-12){
483 
484  //MODULATING DUMPING WITH DISTANCE
485  MimmoObject * dumptarget= m_dsurface;
486  if(m_dsurface == NULL) dumptarget = m_bsurface;
487 
488  bitpit::PiercedVector<double> distFactor;
489  getGeometry()->getVerticesNarrowBandToExtSurface(*dumptarget, maxd, distFactor);
490 
491  double distanceMax = std::pow((maxd/m_plateau), m_decayFactor);
492  for(auto it = distFactor.begin(); it !=distFactor.end(); ++it){
493  if(*it < m_plateau){
494  (*it) = 1.0;
495  }else{
496  (*it) = (std::pow(maxd/(*it), m_decayFactor) -1.0) / (distanceMax -1.0);
497  }
498  }
499 
500  //evaluating volume on each vertex
501  bitpit::PiercedVectorStorage<double> volFactor;
502  volFactor.setStaticKernel(&distFactor); //, bitpit::PiercedSyncMaster::SyncMode::SYNC_MODE_DISABLED);
503  volFactor.fill(1.0);
504 
505  if(m_dumpingType == 1){
506  //evaluating cell volumes
507  double volmax = 0.0, volmin=1.0E18;
508  livector1D cellList = getGeometry()->getCellFromVertexList(distFactor.getIds(), false);
509  bitpit::PiercedVector<double> volumes;
510  volumes.reserve(cellList.size());
511  //evaluate volumes on each cell and save it
512 
513  for(const auto & idC: cellList){
514  auto it = volumes.insert(idC, getGeometry()->evalCellVolume(idC));
515  if(*it <= 0.0){
516  throw std::runtime_error("Detected cells with negative volume");
517  }
518  volmin = std::min(volmin,*it);
519  volmax = std::max(volmax,*it);
520  }
521 
522  //pass on vertices assigning the min value of volume for each vertices.
523  for(auto & id : cellList){
524  bitpit::ConstProxyVector<long> vids = patch_->getCell(id).getVertexIds();
525  for(auto & idV : vids){
526  if(distFactor.exists(idV)) volFactor[idV] = std::min(volFactor[idV], volumes[id]);
527  }
528  }
529 
530  //evaluate the volume normalized function
531  for(auto it = distFactor.begin(); it !=distFactor.end(); ++it){
532  std::size_t pos = it.getRawIndex();
533  volFactor.rawAt(pos) = std::pow(1.0 + (volmax -volmin)/volFactor.rawAt(pos), distFactor.rawAt(pos));
534  }
535  }
536 
537 
538  //get an average of distance and volume functions.
539  for(auto it=distFactor.begin(); it!=distFactor.end(); ++it){
540  long id = it.getId();
541  double val;
542  if(m_dumpingType == 1){
543  val = volFactor[id];
544  }else{
545  val = (distanceMax - 1.0)*distFactor[id] + 1.0;
546  }
547  m_dumping[id] = val;
548  }
549  }
550 }

562 template<std::size_t NCOMP>
563 void
564 PropagateField<NCOMP>::solveSmoothing(int nstep,
565  ivector2D &stencils,
566  dvector2D &weights,
567  dvector1D &rhs,
568  liimap & dataInv,
569  MimmoPiercedVector<std::array<double, NCOMP> > & field)
570 {
571  long ID;
572  int ind;
573  field.clear();
574 
575  //initialize field
576  field.clear();
577  for (auto vertex : getGeometry()->getVertices()){
578  ID = vertex.getId();
579  field.insert(ID, std::array<double, NCOMP>({}));
580  }
581 
582  //found maxval in rhs for normalization purposes
583  double maxval = 0.0;
584  for (const auto & val : rhs){
585  maxval = std::max(maxval, val);
586  }
587 
588  dvector1D guess = rhs;
589 
590  (*m_log)<< m_name <<" starts field propagation."<<std::endl;
591 
592  for (int istep = 0; istep < nstep; istep++){
593 
594  if (!m_convergence) (*m_log)<<m_name << " smoothing step : " << istep+1 << " / " << nstep <<std::endl;
595 
596  double maxdiff = 0.0;
597  for (int locId=0; locId<NCOMP*m_np; ++locId){
598 
599  double delta = 0.0;
600  std::size_t sizeStencil = stencils[locId].size();
601  for (std::size_t i=0; i<sizeStencil; ++i){
602  delta += -1.0*guess[stencils[locId][i]]*weights[locId][i];
603  }
604  delta += rhs[locId];
605  guess[locId] += delta;
606 
607  if (m_convergence){
608  maxdiff = std::max(maxdiff, delta);
609  }
610  }//end for vertex
611 
612  if (m_convergence) (*m_log)<< m_name<<" residual : " << maxdiff <<std::endl;
613 
614  //convergence
615  if (m_convergence){
616  if (maxdiff <= m_tol)
617  istep = nstep;
618  else{
619  nstep = istep+2;
620  }
621  }
622  }// end step
623 
624  (*m_log)<< m_name<<" ends field propagation."<<std::endl;
625 
626  for (auto vertex : getGeometry()->getVertices()){
627  for (int icomp=0; icomp<NCOMP; ++icomp ){
628  ID = vertex.getId();
629  ind = dataInv[ID];
630  field[ID][icomp] = guess[ind + icomp*m_np];
631  }
632  }
633 
634  field.setDataLocation(MPVLocation::POINT);
635  field.setGeometry(getGeometry());
636 }

649 template<std::size_t NCOMP>
650 void
651 PropagateField<NCOMP>::solveLaplace(ivector2D &stencils,
652  dvector2D &weights,
653  dvector1D &rhs,
654  liimap &dataInv,
655  MimmoPiercedVector<std::array<double, NCOMP> > & field)
656 {
657  //initialize field
658  field.clear();
659  for (auto vertex : getGeometry()->getVertices()){
660  long int ID = vertex.getId();
661  field.insert(ID, std::array<double, NCOMP>({}));
662  }
663 
664  // Create the system for solving the pressure
665  m_solver = std::unique_ptr<mimmo::SystemSolver>(new mimmo::SystemSolver(false));
666 
667  // Initialize the system
668  KSPOptions &solverOptions = m_solver->getKSPOptions();
669  solverOptions.nullspace = false;
670  solverOptions.rtol = m_tol;
671  solverOptions.subrtol = m_tol;
672 
673 #if ENABLE_MPI==1
674  m_solver->initialize(stencils, weights, rhs, ghosts);
675 #else
676  m_solver->initialize(stencils, weights, rhs);
677 #endif
678 
679  // Solve the system
680  m_solver->solve();
681 
682  // Get the solution
683  const double *solution = m_solver->getSolutionRawReadPtr();
684 
685  long ID;
686  int ind;
687  for (auto vertex : getGeometry()->getVertices()){
688  for (int icomp=0; icomp<NCOMP; ++icomp ){
689  ID = vertex.getId();
690  ind = dataInv[ID];
691  field[ID][icomp] = solution[ind + icomp*m_np];
692  }
693  }
694 
695  // Clear the solver
696  m_solver->clear();
697  field.setDataLocation(MPVLocation::POINT);
698  field.setGeometry(getGeometry());
699 }

712 template<std::size_t NCOMP>
713 void
714 PropagateField<NCOMP>::computeStencils(liimap &dataInv,
715  ivector2D &stencils,
716  dvector2D &weights)
717 {
718  MimmoPiercedVector<livector1D> conn;
719  MimmoPiercedVector<dvector1D> wgt;
720 
721  computeRingConnectivity(conn);
722  computeRingWeights(conn,wgt);
723 
724  stencils.clear();
725  weights.clear();
726  stencils.resize(NCOMP*m_np);
727  weights.resize(NCOMP*m_np);
728 
729  long ID;
730  int ind;
731  //Create stencils
732  for (auto vertex : getGeometry()->getVertices()){
733  ID = vertex.getId();
734  ind = dataInv[ID];
735  //bulk evaluation
736  for(int comp=0; comp<NCOMP; ++comp){
737  for (const long & IDN : conn[ID]){
738  stencils[ind+comp*m_np].push_back(dataInv[IDN]+comp*m_np);
739  }
740  stencils[ind+comp*m_np].push_back(ind+comp*m_np);
741  weights[ind+comp*m_np] = -1.0*wgt[ID];
742  weights[ind+comp*m_np].push_back(1.0);
743  }
744  }
745 
746 }

752 template<std::size_t NCOMP>
753 void PropagateField<NCOMP>::computeRingConnectivity(MimmoPiercedVector<livector1D> & conn){
754  //DOESN'T WORK FOR POINTS CLOUD OR CURVES
755 
756  bitpit::PatchKernel * patch_ = getGeometry()->getPatch();
757

776  // }
777  // else{
778  econn = cell.getEdgeConnect(ie);
779  // }
780  if (!visitedge[std::pair<long, long>(econn[0],econn[1])]){

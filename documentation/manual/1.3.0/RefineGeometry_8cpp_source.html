175 void
176 RefineGeometry::execute(){
177 
178  if(getGeometry() == nullptr){
179  (*m_log)<<m_name + " : no geometry to refine found"<<std::endl;
180  return;
181  }
182 
183  if (getGeometry()->getType() != 1){
184  (*m_log)<<m_name + " : data structure type different from Surface "<<std::endl;
185  throw std::runtime_error (m_name + " : data structure type different from Surface");
186  }
187 
188  if (m_type == RefineType::TERNARY){
189  for (int i=0; i<m_refinements; i++)
190  ternaryRefine();
191  }
192 
193  if (m_steps>0)
194  smoothing();
195 
196 }

287 void
288 RefineGeometry::ternaryRefine()
289 {
290  //TERNARY REFINEMENT
291 
292  // Ternary refinement force the geometry to be a triangulation by placing a new vertex
293  // on the mean point of each cell (the cells have to be convex)
294 
295  // For Now this method works only in SERIAL mode.
296 #if MIMMO_ENABLE_MPI
297  if (m_nprocs > 1){
298  //TODO provide implementation to deal with insertion/deletion of vertices and cells in parallel
299  (*m_log)<< "WARNING " <<m_name <<" : is not available yet in parallel process."<<std::endl;
300  }
301 #else
302 
303  MimmoObject * geometry = getGeometry();
304  long maxID, newID, newVertID;
305  const auto orderedCellID = geometry->getCells().getIds(true);
306  maxID = orderedCellID[(int)orderedCellID.size()-1];
307  newID = maxID+1;
308  {
309  const auto orderedVertID = geometry->getVertices().getIds(true);
310  newVertID = orderedVertID[(int)orderedVertID.size()-1] +1;
311  }
312 
313  bitpit::ElementType eletype;
314  bitpit::ElementType eletri = bitpit::ElementType::TRIANGLE;
315  livector1D connTriangle(3);
316 
317  for(const auto &idcell : orderedCellID){
318 
319  livector1D conn = geometry->getCellConnectivity(idcell);
320  eletype = geometry->getPatch()->getCell(idcell).getType();
321  long pid = geometry->getPatch()->getCell(idcell).getPID();
322 
323  switch (eletype){
324  case bitpit::ElementType::TRIANGLE:
325  case bitpit::ElementType::PIXEL:
326  case bitpit::ElementType::QUAD:
327  {
328  std::size_t startIndex = 0;
329  std::size_t nnewTri = conn.size() - startIndex;
330  //calculate barycenter and add it as new vertex
331  darray3E barycenter = geometry->getPatch()->evalCellCentroid(idcell);
332  geometry->addVertex(barycenter, newVertID);
333  //delete current polygon
334  geometry->getPatch()->deleteCell(idcell);
335  //insert new triangles from polygon subdivision
336  for(std::size_t i=0; i<nnewTri; ++i){
337  connTriangle[0] = newVertID;
338  connTriangle[1] = conn[ startIndex + std::size_t( i % nnewTri) ];
339  connTriangle[2] = conn[ startIndex + std::size_t( (i+1) % nnewTri ) ];
340  geometry->addConnectedCell(connTriangle, eletri, pid, newID);
341  ++newID;
342  }
343  //increment label of vertices
344  ++newVertID;
345  }
346  break;
347  case bitpit::ElementType::POLYGON:
348  {
349  std::size_t startIndex = 1;
350  std::size_t nnewTri = conn.size() - startIndex;
351  //calculate barycenter and add it as new vertex
352  darray3E barycenter = geometry->getPatch()->evalCellCentroid(idcell);
353  geometry->addVertex(barycenter, newVertID);
354  //delete current polygon
355  geometry->getPatch()->deleteCell(idcell);
356  //insert new triangles from polygon subdivision
357  for(std::size_t i=0; i<nnewTri; ++i){
358  connTriangle[0] = newVertID;
359  connTriangle[1] = conn[ startIndex + std::size_t( i % nnewTri) ];
360  connTriangle[2] = conn[ startIndex + std::size_t( (i+1) % nnewTri ) ];
361  geometry->addConnectedCell(connTriangle, eletri, pid, newID);
362  ++newID;
363  }
364  //increment label of vertices
365  ++newVertID;
366  }
367  break;
368  default:
369  throw std::runtime_error("unrecognized cell type in 3D surface mesh of CGNSPidExtractor");
370  break;
371  }
372  }
373 #endif
374 }

381 void
382 RefineGeometry::smoothing()
383 {
384  MimmoObject * geometry = getGeometry();
385  if (!(geometry->areAdjacenciesBuilt()))
386  geometry->buildAdjacencies();
387 
388  if (!(geometry->isPointConnectivitySync()))
389  geometry->buildPointConnectivity();
390 
391 
392  double lambda = 0.6;
393  double kappa = -0.603*lambda;
394 
395  //TODO WORK ON FEATURES PRESERVING (PRE-PROCESSING IDENTIFICATION OF FEATURES AND REINTRODUCTION OF NORMAL BASED FORMULA?)
396  //Threshold angle for features preserving
397  double angledeg = 10; /*degrees*/
398  double anglerad = angledeg * M_PI / 180.;
399 
400  //Compute vertex normals
401  std::unordered_map<long, std::array<double, 3>> normals;
402  if (angledeg > 0.){
403  for (const bitpit::Cell & cell : geometry->getCells()){
404  int iv = 0;
405  for (long idvertex : cell.getVertexIds()){
406  if (!(normals.count(idvertex)))
407  normals[idvertex] = static_cast<bitpit::SurfaceKernel*>(geometry->getPatch())->evalVertexNormal(cell.getId(), iv);
408  iv++;
409  }
410  }
411  }
412 
413  if (!(geometry->isPointConnectivitySync()))
414  geometry->buildPointConnectivity();
415 
416  for (int istep=0; istep < m_steps; istep++){
417 
418 // //Compute vertex normals
419 // std::unordered_map<long, std::array<double, 3>> normals;
420 // if (angledeg > 0.){
421 // for (const bitpit::Cell & cell : geometry->getCells()){
422 // int iv = 0;
423 // for (long idvertex : cell.getVertexIds()){
424 // if (!(normals.count(idvertex)))
425 // normals[idvertex] = static_cast<bitpit::SurfaceKernel*>(geometry->getPatch())->evalVertexNormal(cell.getId(), iv);
426 // iv++;
427 // }
428 // }
429 // }
430 
431 // std::cout << "smoothing steps " << istep+1 << " of " << m_steps << std::endl;
432 
433  {
434  // First sub-step (positive) of laplacian smoothing
435  std::unordered_map<long, std::array<double,3>> newcoordinates;
436  std::unordered_set<long> pointconnectivity;
437  std::array<double,3> newcoords, oldcoords, neighcoords;
438  double weight, sumweights;
439  newcoordinates.reserve(geometry->getNVertices());
440  //compute new coordinates
441  for (long id : geometry->getVertices().getIds()){
442 
443  oldcoords = geometry->getVertexCoords(id);
444  pointconnectivity = geometry->getPointConnectivity(id);
445  newcoords = std::array<double,3>{{0.,0.,0.}};
446 
447  sumweights = 0.;
448  for (long idneigh : pointconnectivity){
449  neighcoords = geometry->getVertexCoords(idneigh);
450  std::array<double,3> distancevector = (neighcoords-oldcoords);
451  weight = 1.;
452 
453  if (angledeg > 0.){
454  //features preserving
455  weight *= double(std::abs(dotProduct(distancevector/norm2(distancevector), normals[id])) < std::cos(M_PI/2. - anglerad));
456  //weight *= double(std::abs(dotProduct(normals[idneigh], normals[id])) > std::cos(anglerad));
457  }
458 
459  sumweights += weight;
460  newcoords += lambda*weight*distancevector;
461  }
462  if (sumweights > 0.)
463  newcoords /= sumweights;
464  newcoords += oldcoords;
465  newcoordinates[id] = newcoords;
466  }
467  //Set new coordinates
468  for (long id : geometry->getVertices().getIds()){
469  geometry->modifyVertex(newcoordinates[id], id);
470  }
471 
472 // //TODO DEBUG WRITE
473 // std::string name = "geometry.a."+std::to_string(istep);
474 // geometry->getPatch()->write(name);
475  }
476 
477  {
478  //Second sub-step (negative) of laplacian anti-smoothing
479  std::unordered_map<long, std::array<double,3>> newcoordinates;
480  std::unordered_set<long> pointconnectivity;
481  std::array<double,3> newcoords, oldcoords, neighcoords;
482  double weight, sumweights;
483  newcoordinates.reserve(geometry->getNVertices());
484  //compute new coordinates
485  for (long id : geometry->getVertices().getIds()){
486 
487  oldcoords = geometry->getVertexCoords(id);
488  pointconnectivity = geometry->getPointConnectivity(id);
489  newcoords = std::array<double,3>{{0.,0.,0.}};
490 
491  sumweights = 0.;
492  for (long idneigh : pointconnectivity){
493  neighcoords = geometry->getVertexCoords(idneigh);
494  std::array<double,3> distancevector = (neighcoords-oldcoords);
495  weight = 1.;
496 
497  if (angledeg > 0.){
498  //features preserving
499  weight *= double(std::abs(dotProduct(distancevector/norm2(distancevector), normals[id])) < std::cos(M_PI/2. - anglerad));
500  //weight *= double(std::abs(dotProduct(normals[idneigh], normals[id])) > std::cos(anglerad));
501  }
502 
503  sumweights += weight;
504  newcoords += kappa*weight*(neighcoords-oldcoords);
505  }
506  if (sumweights > 0.)
507  newcoords /= sumweights;
508  newcoords += oldcoords;
509  newcoordinates[id] = newcoords;
510  }
511  //Set new coordinates
512  for (long id : geometry->getVertices().getIds()){
513  geometry->modifyVertex(newcoordinates[id], id);
514  }
515 
516 // //TODO DEBUG WRITE

520  }

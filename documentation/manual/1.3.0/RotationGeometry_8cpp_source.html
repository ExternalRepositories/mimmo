174 void
175 RotationGeometry::execute(){
176 
177  if (getGeometry() == NULL){
178  throw std::runtime_error (m_name + " : NULL pointer to linked geometry");
179  }
180  if (getGeometry()->isEmpty()){
181  throw std::runtime_error (m_name + " : empty linked geometry");
182  }
183 
184  checkFilter();
185 
186  m_displ.clear();
187  m_displ.setDataLocation(mimmo::MPVLocation::POINT);
188  m_displ.reserve(getGeometry()->getNVertex());
189  m_displ.setGeometry(getGeometry());
190 
191  //compute coefficients and constant vectors of rodriguez formula
192  double a = cos(m_alpha);
193  darray3E b = (1 - cos(m_alpha)) * m_direction;
194  double c = sin(m_alpha);
195 
196  darray3E point, rotated;
197  long ID;
198  darray3E value;
199  for (const auto & vertex : m_geometry->getVertices()){
200  point = vertex.getCoords();
201  ID = vertex.getId();
202 
203  point -= m_origin;
204  //rodrigues formula
205  rotated = a * point +
206  b * dotProduct(m_direction, point) +
207  c * crossProduct(m_direction, point);
208 
209  rotated += m_origin;
210  point += m_origin;
211 
212  value = (rotated-point)*m_filter[ID];
213  m_displ.insert(ID, value);
214 
215  }
216 };

241 void
242 RotationGeometry::checkFilter(){
243  bool check = m_filter.getDataLocation() == mimmo::MPVLocation::POINT;
244  check = check && m_filter.completeMissingData(0.0);
245  check = check && m_filter.getGeometry() == getGeometry();
246 
247  if (!check){
248  m_log->setPriority(bitpit::log::Verbosity::DEBUG);
249  (*m_log)<<"Not valid filter found in "<<m_name<<". Proceeding with default unitary field"<<std::endl;
250  m_log->setPriority(bitpit::log::Verbosity::NORMAL);
251 
252  m_filter.clear();
253  m_filter.setGeometry(m_geometry);
254  m_filter.setDataLocation(mimmo::MPVLocation::POINT);
255  m_filter.reserve(getGeometry()->getNVertex());
256  for (const auto & vertex : getGeometry()->getVertices()){
257  m_filter.insert(vertex.getId(), 1.0);
258  }
259  }

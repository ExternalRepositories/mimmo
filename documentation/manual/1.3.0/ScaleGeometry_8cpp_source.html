158 void
159 ScaleGeometry::execute(){
160 
161  if(getGeometry() == NULL){
162  (*m_log)<<m_name + " : NULL pointer to linked geometry found"<<std::endl;
163  throw std::runtime_error(m_name + "NULL pointer to linked geometry found");
164  }
165 
166  if(getGeometry()->isEmpty()){
167  (*m_log)<<m_name + " : empty linked geometry found"<<std::endl;
168  }
169 
170  checkFilter();
171  m_displ.clear();
172  m_displ.setDataLocation(mimmo::MPVLocation::POINT);
173  m_displ.reserve(getGeometry()->getNVertices());
174  m_displ.setGeometry(getGeometry());
175 
176 
177  long ID;
178  darray3E value;
179  //computing centroid
180  int nV = m_geometry->getNVertices();
181  darray3E center = m_origin;
182  if (m_meanP){
183  center.fill(0.0);
184  for (const auto & vertex : m_geometry->getVertices()){
185  center += vertex.getCoords();
186  }
187  center /=double(nV);
188  }
189  for (const auto & vertex : m_geometry->getVertices()){
190  ID = vertex.getId();
191 
192  darray3E coords = vertex.getCoords();
193  value = (( m_scaling*(coords - center) + center ) - coords) * m_filter[ID] ;
194  m_displ.insert(ID, value);
195  }
196 };

210 void
211 ScaleGeometry::checkFilter(){
212  bool check = m_filter.getDataLocation() == mimmo::MPVLocation::POINT;
213  check = check && m_filter.completeMissingData(0.0);
214  check = check && m_filter.getGeometry() == getGeometry();
215 
216  if (!check){
217  m_log->setPriority(bitpit::log::Verbosity::DEBUG);
218  (*m_log)<<"Not valid filter found in "<<m_name<<". Proceeding with default unitary field"<<std::endl;
219  m_log->setPriority(bitpit::log::Verbosity::NORMAL);
220 
221  m_filter.clear();
222  m_filter.setGeometry(m_geometry);
223  m_filter.setDataLocation(mimmo::MPVLocation::POINT);
224  m_filter.reserve(getGeometry()->getNVertices());
225  for (const auto & vertex : getGeometry()->getVertices()){
226  m_filter.insert(vertex.getId(), 1.0);
227  }
228  }
229 }

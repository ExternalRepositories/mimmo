325 livector1D
326 SelectionByMapping::extractSelection(){
327 
328  if(!(getGeometry()->isSkdTreeSync())) getGeometry()->buildSkdTree();
329  std::set<long> cellList;
330 
331  for (auto && file : m_geolist){
332  livector1D list = getProximity(file);
333  cellList.insert(list.begin(), list.end());
334  }
335 
336  for (auto mimmo : m_mimmolist){
337  livector1D list = getProximity(mimmo);
338  cellList.insert(list.begin(), list.end());
339  }
340 
341 /* check if dual */
342  livector1D result;
343  if(m_dual){
344  livector1D totID = getGeometry()->getCellsIds();
345  result.resize(totID.size() - cellList.size());
346  if(result.size() == 0) return result;
347 
348  std::sort(totID.begin(), totID.end());
349  int counter = 0;
350  auto tot_it = totID.begin();
351  auto cand_it = cellList.begin();
352  while(tot_it != totID.end()){
353  long val = *tot_it;
354  if (cand_it == cellList.end() || val != *cand_it) {
355  result[counter] = val;
356  ++counter;
357  } else {
358  ++cand_it;
359  }
360  ++tot_it;
361  }
362  }else{
363  result.insert(result.end(), cellList.begin(), cellList.end());
364  }
365  return result;
366 };

374 SelectionByMapping::getProximity(std::pair<std::string, int> val){
375 
376  svector1D info = extractInfo(val.first);
377 
378  MimmoGeometry * geo = new MimmoGeometry();
379  geo->setIOMode(IOMode::READ);
380  geo->setDir(info[0]);
381  geo->setFilename(info[1]);
382  geo->setFileType(val.second);
383  geo->setBuildSkdTree(true);
384  geo->execute();
385 
386  if(geo->getGeometry()->getNVertex() == 0 || geo->getGeometry()->getNCells() == 0 || geo->getGeometry()->getType()==3 ){
387  m_log->setPriority(bitpit::log::NORMAL);
388  (*m_log)<< m_name << " failed to read or unsuitable geometry in SelectionByMapping::getProximity"<<std::endl;
389  m_log->setPriority(bitpit::log::DEBUG);
390  return livector1D();
391  }
392  livector1D result = mimmo::skdTreeUtils::selectByPatch(geo->getGeometry()->getSkdTree(), getGeometry()->getSkdTree(), m_tolerance);
393  delete geo;
394  geo=NULL;
395 
396  return result;

451 SelectionByMapping::absorbSectionXML(const bitpit::Config::Section & slotXML, std::string name){
452 
453  BITPIT_UNUSED(name);
454 
455  //checking topology
456  if(slotXML.hasOption("Topology")){
457  std::string input = slotXML.get("Topology");
458  input = bitpit::utils::string::trim(input);
459  int temp = -1;
460  if(!input.empty()){
461  std::stringstream ss(input);
462  ss>>temp;
463  }
464  if(m_topo != temp) {
465  throw std::runtime_error (m_name + " : xml absorbing failed.");
466  }
467  }
468 
469  BaseManipulation::absorbSectionXML(slotXML, name);
470  //start absorbing
471  if(slotXML.hasOption("Dual")){
472  std::string input = slotXML.get("Dual");
473  input = bitpit::utils::string::trim(input);
474  bool value = false;
475  if(!input.empty()){
476  std::stringstream ss(input);
477  ss >> value;
478  }
479  setDual(value);
480  }
481 
482  if(slotXML.hasOption("Tolerance")){
483  std::string input = slotXML.get("Tolerance");
484  input = bitpit::utils::string::trim(input);
485  double temp = 1.0E-8;
486  if(!input.empty()){
487  std::stringstream ss(input);
488  ss>>temp;
489  }
490  setTolerance(temp);
491 
492  }
493 
494  std::unordered_map<std::string, int> mapp;
495  if(slotXML.hasSection("Files")){
496 
497  const bitpit::Config::Section & filesXML = slotXML.getSection("Files");
498 
499  for(auto & subfile : filesXML.getSections()){
500  std::string path;
501  std::string tag;
502 
503  if(subfile.second->hasOption("fullpath")) {
504  path = subfile.second->get("fullpath");
505  path = bitpit::utils::string::trim(path);
506  }
507  if(subfile.second->hasOption("tag")){
508  tag = subfile.second->get("tag");
509  tag = bitpit::utils::string::trim(tag);
510  //check tag;
511  auto maybe_tag = FileType::_from_string_nothrow(tag.c_str());
512  if(!maybe_tag) tag.clear();
513  else tag = maybe_tag->_to_string();
514  }
515 
516  if(!path.empty() && !tag.empty()){
517  mapp[path] = (int) FileType::_from_string(tag.c_str());
518  }
519  }
520 
521  setFiles(mapp);
522 
523  }

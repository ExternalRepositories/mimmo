84 double signedDistance(std::array<double,3> *P_, bitpit::PatchSkdTree *bvtree_, long &id, std::array<double,3> &n, double &r)
85 {
86 
87  double h = 1.E+18;
88 
89  if(!bvtree_ ){
90  throw std::runtime_error("Invalid use of skdTreeUtils::signedDistance method: a void tree is detected.");
91  }
92  const bitpit::SurfUnstructured *spatch = dynamic_cast<const bitpit::SurfUnstructured*>(&(bvtree_->getPatch()));
93  if(!spatch){
94  throw std::runtime_error("Invalid use of skdTreeUtils::signedDistance method: a not surface patch tree is detected.");
95  }
96 
97  static_cast<bitpit::SurfaceSkdTree*>(bvtree_)->findPointClosestCell(*P_, r, &id, &h);
98 
99  //signed distance only for 2D element patches(quads, pixels, triangles or segments)
100  if (id != bitpit::Cell::NULL_ID){
101  const bitpit::Cell & cell = spatch->getCell(id);
102  bitpit::ConstProxyVector<long> vertIds = cell.getVertexIds();
103  dvecarr3E VS(vertIds.size());
104  int count = 0;
105  for (const auto & iV: vertIds){
106  VS[count] = spatch->getVertexCoords(iV);
107  ++count;
108  }
109 
110  darray3E xP = {{0.0,0.0,0.0}};
111  darray3E normal= {{0.0,0.0,0.0}};
112 
113  if ( vertIds.size() == 3 ){ //TRIANGLE
114  darray3E lambda;
115  h = bitpit::CGElem::distancePointTriangle((*P_), VS[0], VS[1], VS[2],lambda);
116  int count = 0;
117  for(const auto &val: lambda){
118  normal += val * spatch->evalVertexNormal(id,count) ;
119  xP += val * VS[count];
120  ++count;
121  }
122  }else if ( vertIds.size() == 2 ){ //LINE/SEGMENT
123  darray2E lambda;
124  h = bitpit::CGElem::distancePointSegment((*P_), VS[0], VS[1], lambda);
125  int count = 0;
126  for(const auto &val: lambda){
127  normal += val * spatch->evalVertexNormal(id,count) ;
128  xP += val * VS[count];
129  ++count;
130  }
131  }else{ //GENERAL POLYGON
132  std::vector<double> lambda;
133  h = bitpit::CGElem::distancePointPolygon((*P_), VS,lambda);
134  int count = 0;
135  for(const auto &val: lambda){
136  normal += val * spatch->evalVertexNormal(id,count) ;
137  xP += val * VS[count];
138  ++count;
139  }
140  }
141 
142  double s = sign( dotProduct(normal, (*P_) - xP) );
143  if(s == 0.0) s =1.0;
144  h = s * h;
145  //pseudo-normal (direction P and xP closest point on triangle)
146  n = s * ((*P_) - xP);
147  double normX = norm2(n);
148  if(normX < 1.E-15){
149  n = normal/norm2(normal);
150  }else{
151  n /= norm2(n);
152  }
153  }//end if not id null
154  return h;
155 
156 }

168 std::vector<long> selectByPatch(bitpit::PatchSkdTree *selection, bitpit::PatchSkdTree *target, double tol){
169 
170  int nleafs = selection->getLeafCount();
171  int nnodess = selection->getNodeCount();
172  std::vector<const bitpit::SkdNode*> leafSelection(nleafs);
173  int count = 0;
174  for (int i=0; i<nnodess; i++){
175  if (selection->getNode(i).isLeaf()){
176  if (bitpit::CGElem::intersectBoxBox(selection->getNode(i).getBoxMin()-tol,
177  selection->getNode(i).getBoxMax()+tol,
178  target->getNode(0).getBoxMin(),
179  target->getNode(0).getBoxMax() ) ){
180  leafSelection[count] = &(selection->getNode(i));
181  count++;
182 
183  }
184  }
185  }
186  leafSelection.resize(count);
187 
188 
189  std::vector<long> extracted;
190  extractTarget(target, leafSelection, extracted, tol);
191 
192  return extracted;

211 void extractTarget(bitpit::PatchSkdTree *target, const std::vector<const bitpit::SkdNode*> & leafSelection, std::vector<long> &extracted, double tol){
212 
213  if(leafSelection.empty()) return;
214  std::size_t rootId =0;
215 
216  std::vector<std::size_t> candidates;
217  std::vector<std::pair< std::size_t, std::vector<const bitpit::SkdNode*> > > nodeStack;
218 
219  std::vector<const bitpit::SkdNode*> tocheck;
220  for (int i=0; i<(int)leafSelection.size(); i++){
221  if (bitpit::CGElem::intersectBoxBox(leafSelection[i]->getBoxMin()-tol,
222  leafSelection[i]->getBoxMax()+tol,
223  target->getNode(rootId).getBoxMin(),
224  target->getNode(rootId).getBoxMax() ) )
225  {
226  tocheck.push_back(leafSelection[i]);
227  }
228  }
229  nodeStack.push_back(std::make_pair(rootId, tocheck) );
230 
231  while(!nodeStack.empty()){
232 
233  std::pair<std::size_t, std::vector<const bitpit::SkdNode*> > touple = nodeStack.back();
234  const bitpit::SkdNode & node = target->getNode(touple.first);
235  nodeStack.pop_back();
236 
237 
238  bool isLeaf = true;
239  for (int i = bitpit::SkdNode::CHILD_BEGIN; i != bitpit::SkdNode::CHILD_END; ++i) {
240  bitpit::SkdNode::ChildLocation childLocation = static_cast<bitpit::SkdNode::ChildLocation>(i);
241  std::size_t childId = node.getChildId(childLocation);
242  if (childId != bitpit::SkdNode::NULL_ID) {
243  isLeaf = false;
244  tocheck.clear();
245  for (int i=0; i<(int)touple.second.size(); i++){
246  if (bitpit::CGElem::intersectBoxBox(touple.second[i]->getBoxMin()-tol,
247  touple.second[i]->getBoxMax()+tol,
248  target->getNode(childId).getBoxMin(),
249  target->getNode(childId).getBoxMax() ) )
250  {
251  tocheck.push_back(touple.second[i]);
252  }
253  }
254  if(!tocheck.empty()) nodeStack.push_back(std::make_pair(childId, tocheck) );
255  }
256  }
257 
258  if (isLeaf) {
259  candidates.push_back(touple.first);
260  }
261  }
262 
263  std::unordered_set<long> cellExtracted;
264  for(const auto & nodeId: candidates){
265  const bitpit::SkdNode & node = target->getNode(nodeId);
266  std::vector<long> cellids = node.getCells();
267  cellExtracted.insert(cellids.begin(), cellids.end());
268  }
269 
270  extracted.insert(extracted.end(), cellExtracted.begin(), cellExtracted.end());
271 }

286 darray3E projectPoint( std::array<double,3> *P_, bitpit::PatchSkdTree *bvtree_, double r )
287 {
288 
289  long id;
290  darray3E normal;
291 
292  r = std::max(r, bvtree_->getPatch().getTol());
293 
294  double dist = 1.0e+18;
295  while (std::abs(dist) >= 1.0e+18){
296  //use method sphere by default
297  dist = signedDistance(P_, bvtree_, id, normal, r);
298  r *= 1.5;
299  }
300 
301  darray3E projP = (*P_) - dist*normal;
302 
303  return (projP);
304 }

314 long locatePointOnPatch(const std::array<double, 3> &point, bitpit::PatchSkdTree &tree)
315 {
316  if(!dynamic_cast<const bitpit::SurfUnstructured*>(&(tree.getPatch()))){
317  throw std::runtime_error("Invalid use of skdTreeUtils::locatePointOnPatch method: a non surface patch or void patch was detected.");
318  }
319  // Initialize the cell id
320  long id = bitpit::Cell::NULL_ID;
321 
322  // Initialize the distance with an estimate
323  //
324  // The real distance will be lesser than or equal to the estimate.
325  std::size_t rootId = 0;
326  const bitpit::SkdNode &root = tree.getNode(rootId);
327  double distance = root.evalPointMaxDistance(point);
328 
329  // Get a list of candidates nodes
330  //
331  std::vector<std::size_t> m_candidateIds;
332 
333  std::vector<std::size_t> nodeStack;
334  nodeStack.push_back(rootId);
335  while (!nodeStack.empty()) {
336  std::size_t nodeId = nodeStack.back();
337  const bitpit::SkdNode &node = tree.getNode(nodeId);
338  nodeStack.pop_back();
339 
340  // Do not consider nodes with a minimum distance greater than
341  // the distance estimate
342  double nodeMinDistance = node.evalPointMinDistance(point);
343  if (nodeMinDistance > distance) {
344  continue;
345  }
346 
347  // Update the distance estimate
348  //
349  // The real distance will be lesser than or equal to the
350  // estimate.
351  double nodeMaxDistance = node.evalPointMaxDistance(point);
352  distance = std::min(nodeMaxDistance, distance);
353 
354  // If the node is a leaf add it to the candidates, otherwise
355  // add its children to the stack.
356  bool isLeaf = true;
357  for (int i = bitpit::SkdNode::CHILD_BEGIN; i != bitpit::SkdNode::CHILD_END; ++i) {
358  bitpit::SkdNode::ChildLocation childLocation = static_cast<bitpit::SkdNode::ChildLocation>(i);
359  std::size_t childId = node.getChildId(childLocation);
360  if (childId != bitpit::SkdNode::NULL_ID) {
361  isLeaf = false;
362  nodeStack.push_back(childId);
363  }
364  }
365 
366  if (isLeaf) {
367  m_candidateIds.push_back(nodeId);
368  }
369  }
370 
371  // Process the candidates and find which cell contains the point
372  for (std::size_t k = 0; k < m_candidateIds.size(); ++k) {
373 
374  // Evaluate the min distance between all cells in the candidate node
375  std::size_t nodeId = m_candidateIds[k];
376  const bitpit::SkdNode &node = tree.getNode(nodeId);
377 
378  bitpit::ElementType eltype ;
379  bool checkBelong;
380  for(const auto & cellId : node.getCells() ){
381  eltype = tree.getPatch().getCell(cellId).getType();
382  auto vList = tree.getPatch().getCell(cellId).getVertexIds();
383  dvecarr3E vertCoords(vList.size());
384  int counterList = 0;
385  for(const auto & idV : vList){
386  vertCoords[counterList] = tree.getPatch().getVertexCoords(idV);
387  ++counterList;
388  }
389  switch(eltype){
390  case bitpit::ElementType::LINE:
391  checkBelong = mimmoCGUtils::isPointInsideSegment(point, vertCoords[0], vertCoords[1]);
392  break;
393  case bitpit::ElementType::TRIANGLE:
394  checkBelong = mimmoCGUtils::isPointInsideTriangle(point, vertCoords[0], vertCoords[1], vertCoords[2]);
395  break;
396  case bitpit::ElementType::PIXEL:
397  case bitpit::ElementType::QUAD:
398  case bitpit::ElementType::POLYGON:
399  checkBelong = mimmoCGUtils::isPointInsidePolygon(point, vertCoords);
400  break;
401  default:
402  throw std::runtime_error("Not supported cell type");
403  break;
404  }
405  if (checkBelong) {
406  id = cellId;
407  }
408  }
409  }
410  return id;
411 }

323 void
324 SpecularPoints::execute(){
325 
326  /* If an implicit definition is not present it has to be computed
327  * by using origin and normal.
328  */
329  if (!m_implicit) setPlane(m_origin, m_normal);
330 
331  darray3E norm;
332  double offset;
333  for(int i=0; i<3; ++i) norm[i] = m_plane[i];
334  offset = m_plane[3];
335  double normPlane = norm2(norm);
336  if(normPlane <= std::numeric_limits<double>::min() || m_points.empty()){
337  (*m_log)<< "Warning: "<<m_name <<" : no valid plane normal or empty list of points to be mirrored"<<std::endl;
338  return;
339  }
340  norm /= normPlane;
341  bool project = false;
342  if (getGeometry() != NULL){
343  if (!getGeometry()->isEmpty()) project = true;
344  }
345 
346  //choosing margin for plane offset
347  double margin = 1.e-12;
348  if(project && getGeometry()->getType() == 1){
349  double aTot = 0.0;
350  int cellSize = getGeometry()->getNCells();
351  bitpit::SurfaceKernel * tri = static_cast<bitpit::SurfaceKernel * >(getGeometry()->getPatch());
352  for(const auto &cell: tri->getCells()){
353  aTot += tri->evalCellArea(cell.getId());
354  }
355 
356  if(aTot > 0.0) {margin = 1.2*std::pow(aTot/((double)cellSize),0.5);}
357  }
358  // THIS CLASS CANNOT WORK YET WITH PROJECTION ON VOLUME MESHES. THIS PORTION IS USELESS RIGHT NOW.
359  // }else if(project && getGeometry()->getType() == 2){
360  // double vTot = 0.0;
361  // int cellSize = getGeometry()->getNCells();
362  // bitpit::VolumeKernel * tetra = static_cast<bitpit::VolumeKernel * >(getGeometry()->getPatch());
363  // for(const auto &cell: tetra->getCells()){
364  // vTot += tetra->evalCellVolume(cell.getId());
365  // }
366  //
367  // if(vTot > 0.0) {margin = 1.2*std::pow(vTot/((double)cellSize),0.5);}
368  // }
369 
370 
371  double sig = (1.0 - 2.0*((int)m_insideout));
372  double distance;
373 
374  //mirroring.
375  m_scalar.resize(m_points.size());
376  m_vector.resize(m_points.size());
377 
378  int counterProj = m_points.size();
379  int counterData = 0;
380 
381  m_proj = m_points;
382  m_scalarMirrored = m_scalar;
383  m_vectorMirrored = m_vector;
384  m_labelsMirrored = m_labels;
385 
386  long maxLabel(0);
387  if(!m_labels.empty()){
388  maxLabel = *(std::max_element(m_labels.begin(),m_labels.end()));
389  }
390 
391  m_proj.resize(2*counterProj);
392  m_scalarMirrored.resize(2*counterProj);
393  m_vectorMirrored.resize(2*counterProj);
394  m_labelsMirrored.resize(2*counterProj);
395 
396  for(const auto &val: m_points){
397  distance = sig*(dotProduct(norm, val) + offset);
398  if(distance > margin || m_force){
399  m_proj[counterProj] = val - 2.0*distance*sig*norm;
400  m_scalarMirrored[counterProj] = m_scalar[counterData];
401  m_vectorMirrored[counterProj] = m_vector[counterData] -2.0*dotProduct(m_vector[counterData], sig*norm)*sig*norm;
402  m_labelsMirrored[counterProj] = m_labels[counterData] + maxLabel + 1;
403  counterProj++;
404  }
405  counterData++;
406  }
407 
408  m_proj.resize(counterProj);
409  m_scalarMirrored.resize(counterProj);
410  m_vectorMirrored.resize(counterProj);
411  m_labelsMirrored.resize(counterProj);
412 
413  if(project){
414  if(!getGeometry()->isSkdTreeSync()) getGeometry()->buildSkdTree();
415 
416  //project points on surface.
417  int counter = 0;
418  for(auto &val : m_proj){
419  m_proj[counter]= skdTreeUtils::projectPoint(&val, getGeometry()->getSkdTree());
420  ++counter;
421  }
422  }

505 SpecularPoints::flushSectionXML(bitpit::Config::Section & slotXML, std::string name){
506 
507  BITPIT_UNUSED(name);
508 
509  BaseManipulation::flushSectionXML(slotXML, name);
510 
511  int value = m_force;
512  slotXML.set("Force", std::to_string(value));
513 
514  value = m_insideout;
515  slotXML.set("InsideOut", std::to_string(value));
516 
517 
518  {
519  darray4E org = getPlane();
520  darray3E normal;
521  darray3E point = {{0.0,0.0,0.0}};
522  int imax = -1;
523  double dum = 0.0;
524  for(int i=0; i<3; ++i) {
525  normal[i] =org[i];
526  if(abs(normal[i]) > dum) {
527  imax = i;
528  dum = abs(normal[i]);
529  }
530  }
531  if(imax != -1) point[imax] = -1.0*org[3]/normal[imax];
532 
533  std::stringstream ss1, ss2;
534  ss1<<std::scientific<<point[0]<<'\t'<<point[1]<<'\t'<<point[2];
535  ss2<<std::scientific<<normal[0]<<'\t'<<normal[1]<<'\t'<<normal[2];
536 
537  bitpit::Config::Section & planeXML = slotXML.addSection("Plane");
538  planeXML.set("Point",ss1.str());
539  planeXML.set("Normal",ss2.str());
540  }

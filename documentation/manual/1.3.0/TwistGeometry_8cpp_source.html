194 void
195 TwistGeometry::execute(){
196 
197  if (getGeometry() == NULL){
198  throw std::runtime_error (m_name + " : NULL pointer to linked geometry");
199  }
200  if (getGeometry()->isEmpty()){
201  throw std::runtime_error (m_name + " : empty linked geometry");
202  }
203 
204  checkFilter();
205 
206  m_displ.clear();
207  m_displ.setDataLocation(mimmo::MPVLocation::POINT);
208  m_displ.reserve(getGeometry()->getNVertex());
209  m_displ.setGeometry(getGeometry());
210 
211  darray3E point, rotated;
212  long ID;
213  darray3E projected;
214  double distance;
215  //double sign;
216  double rot;
217  darray3E value;
218 
219  for (const auto & vertex : m_geometry->getVertices()){
220  point = vertex.getCoords();
221  ID = vertex.getId();
222 
223  //signed distance from origin
224  distance = dotProduct((point-m_origin),m_direction);
225 
226  //compute coefficients and constant vectors of rodriguez formula
227  rot = std::min(m_alpha, (std::abs(distance)/m_distance)*m_alpha);
228  if (distance < 0) rot = -rot*int(m_sym);
229  double a = cos(rot);
230  darray3E b = (1 - cos(rot)) * m_direction;
231  double c = sin(rot);
232 
233  //project point on axis (local origin)
234  projected = distance*m_direction + m_origin;
235 
236 
237  point -= projected;
238  //rodrigues formula
239  rotated = a * point +
240  b * dotProduct(m_direction, point) +
241  c * crossProduct(m_direction, point);
242 
243  rotated += projected;
244  point += projected;
245 
246  value = (rotated-point)*m_filter[ID];
247  m_displ.insert(ID, value);
248 
249  }
250 };

258  if (getGeometry() == NULL) return;

276 TwistGeometry::checkFilter(){
277 
278  bool check = m_filter.getDataLocation() == mimmo::MPVLocation::POINT;
279  check = check && m_filter.completeMissingData(0.0);
280  check = check && m_filter.getGeometry() == getGeometry();
281 
282  if (!check){
283  m_log->setPriority(bitpit::log::Verbosity::DEBUG);
284  (*m_log)<<"Not valid filter found in "<<m_name<<". Proceeding with default unitary field"<<std::endl;
285  m_log->setPriority(bitpit::log::Verbosity::NORMAL);
286 
287  m_filter.clear();
288  m_filter.setGeometry(m_geometry);
289  m_filter.setDataLocation(mimmo::MPVLocation::POINT);
290  m_filter.reserve(getGeometry()->getNVertex());
291  for (const auto & vertex : getGeometry()->getVertices()){
292  m_filter.insert(vertex.getId(), 1.0);
293  }
294  }

109 void test00001() {
110 
111  //create a surface polygonal mesh with texture MIO pidded as PID=1
112  std::unique_ptr<mimmo::MimmoObject> geo = createMIOMesh();
113 
114  mimmo::setExpertMode(true);
115 
116  /*
117  * extract texture MIO with SelectionByPID
118  */
119  std::unique_ptr<mimmo::SelectionByPID> sel(new mimmo::SelectionByPID());
120  sel->setName("PIDExtraction");
121  sel->setGeometry(geo.get());
122  sel->setPID(1);
123  sel->setDual(false);
124  sel->setPlotInExecution(true);
125 
126  /*
127  * isolate M from MIO using a plane clipping
128  */
129  std::unique_ptr<mimmo::ClipGeometry> clip(new mimmo::ClipGeometry());
130  clip->setName("PlaneClipping");
131  clip->setOrigin({{1.1,0.0,0.0}});
132  clip->setNormal({{1.0,0.0,0.0}});
133  clip->setInsideOut(true);
134  clip->setPlotInExecution(true);
135 
136  /*
137  * triangulate the M polygonal tessellation
138  */
139  std::unique_ptr<mimmo::SurfaceTriangulator> triang(new mimmo::SurfaceTriangulator());
140  triang->setName("TriangulateSurface");
141  triang->setPlotInExecution(true);
142 
143 
144  /* Setup pin connections.
145  */
146  mimmo::pin::addPin(sel.get(), clip.get(), M_GEOM, M_GEOM);
147  mimmo::pin::addPin(clip.get(), triang.get(), M_GEOM, M_GEOM);
148 
149  /* Setup execution chain.
150  */
151  mimmo::Chain ch0;
152  ch0.addObject(sel.get());
153  ch0.addObject(clip.get());
154  ch0.addObject(triang.get());
155 
156  /* Execution of chain.
157  * Use debug flag true to to print out the execution steps.
158  */
159  ch0.exec(true);
160 
161  return;
162 }

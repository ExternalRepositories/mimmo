48 void example00001() {
49 
50  /* Create IO_CGNS object to import input file. */
51  IOCGNS * cgnsI = new IOCGNS();
52  cgnsI->setRead(true);
53  cgnsI->setReadDir("geodata");
54  cgnsI->setReadFilename("grid");
55 
56  /* Create IO_CGNS object to export output file. */
57  IOCGNS * cgnsO = new IOCGNS();
58  cgnsO->setRead(false);
59  cgnsO->setWriteDir(".");
60  cgnsO->setWriteFilename("iocgns_output_00001");
61 
62  /* Create CGNS PID extractor object to test input file.
63  * Extraction of PID = 1,2,3 (All boundaries in input file).
64  */
65  CGNSPidExtractor * cgnsExtr = new CGNSPidExtractor();
66  cgnsExtr->setPID({1, 2, 3});
67  cgnsExtr->setForcedToTriangulate(false);
68  cgnsExtr->setPlotInExecution(true);
69 
70  /* Create CGNS PID extractor object to test input file.
71  * Extraction of PID = 1,2 (Wing wall and outer part boundaries
72  * where imposing Dirichlet conditions).
73  */
74  CGNSPidExtractor * cgnsDirichlet = new CGNSPidExtractor();
75  cgnsDirichlet->setPID({1, 2});
76  cgnsDirichlet->setForcedToTriangulate(false);
77  cgnsDirichlet->setPlotInExecution(true);
78 
79  /* Create CGNS PID extractor object to test input file.
80  * Extraction of PID = 3 (Simmetry plane
81  * where imposing Slip/impermeability conditions).
82  */
83  CGNSPidExtractor * cgnsSlip = new CGNSPidExtractor();
84  cgnsSlip->setPID({3});
85  cgnsSlip->setForcedToTriangulate(false);
86  cgnsSlip->setPlotInExecution(true);
87 
88  /* Instantiation of a Selection By Box block.
89  * Setup of span and origin of cube.
90  */
91  SelectionByBox * boxSel = new SelectionByBox();
92  boxSel->setOrigin({{700., 800., 0.}});
93  boxSel->setSpan(1500.,1600.,100.);
94  boxSel->setPlotInExecution(true);
95 
96  /* Creation of rotation block.
97  */
98  RotationGeometry* rotation = new RotationGeometry();
99  rotation->setDirection(darray3E{0.1,1.,0.});
100  rotation->setRotation((M_PI/12));
101 
102  /* Create reconstruct vector block and set to reconstruct rotation
103  * displacement field over the whole Dirichlet surface geometry
104  */
105  ReconstructVector* recon = new ReconstructVector();
106 
107  /* Create propagate vector block and set to propagate over the whole
108  * input volume geometry the displacements field.
109  */
110  PropagateVectorField* prop = new PropagateVectorField();
111  prop->setWeightConstant(3.0);
112  //prop->setSmoothingSteps(250);
113  prop->setPlotInExecution(true);
114  prop->setConvergence(true);
115  prop->setTolerance(1.0e-07);
116  prop->setDumping(true);
117  prop->setDumpingInnerDistance(1000);
118  prop->setDumpingOuterDistance(3000);
119  prop->setSolver(true);
120 
121  /* Create propagate vector block and set to propagate over the whole
122  * input volume geometry the displacements field.
123  */
124  ExtractVectorField* extrF = new ExtractVectorField();
125  extrF->setMode(1);
126  extrF->setPlotInExecution(true);
127 
128  /* Create applier block.
129  * It applies the deformation displacements
130  * to the selected input volume geometry.
131  */
132  Apply* applier = new Apply();
133 
134  /* Create applier block.
135  * It applies the deformation displacements
136  * to the selected input surface geometry.
137  */
138  Apply* applierS = new Apply();
139 
140  /* Create PINs. */
141  addPin(cgnsI, cgnsExtr, M_GEOM2, M_GEOM);
142  addPin(cgnsI, cgnsDirichlet, M_GEOM2, M_GEOM);
143  addPin(cgnsI, cgnsSlip, M_GEOM2, M_GEOM);
144 
145  addPin(cgnsDirichlet, boxSel, M_GEOM, M_GEOM);
146  addPin(boxSel, rotation, M_GEOM, M_GEOM);
147  addPin(rotation, recon, M_GDISPLS, M_VECTORFIELD);
148  addPin(cgnsDirichlet, recon, M_GEOM, M_GEOM);
149 
150  addPin(cgnsI, prop, M_GEOM, M_GEOM);
151  addPin(cgnsDirichlet, prop, M_GEOM, M_GEOM2);
152  addPin(cgnsSlip, prop, M_GEOM, M_GEOM4);
153  addPin(boxSel, prop, M_GEOM, M_GEOM3);
154 
155  addPin(recon, prop, M_VECTORFIELD, M_GDISPLS);
156 
157  addPin(prop, applier, M_GDISPLS, M_GDISPLS);
158  addPin(cgnsI, applier, M_GEOM, M_GEOM);
159 
160  addPin(cgnsExtr, extrF, M_GEOM, M_GEOM);
161  addPin(prop, extrF, M_GDISPLS, M_VECTORFIELD);
162  addPin(extrF, applierS, M_VECTORFIELD, M_GDISPLS);
163  addPin(cgnsExtr, applierS, M_GEOM, M_GEOM);
164 
165  addPin(applier, cgnsO, M_GEOM, M_GEOM);
166  addPin(applierS, cgnsO, M_GEOM, M_GEOM2);
167  addPin(cgnsI, cgnsO, M_BCCGNS, M_BCCGNS);
168 
169  /* Create and execute chain. */
170  Chain ch0;
171  ch0.addObject(cgnsI);
172  ch0.addObject(cgnsExtr);
173  ch0.addObject(cgnsDirichlet);
174  ch0.addObject(cgnsSlip);
175  ch0.addObject(boxSel);
176  ch0.addObject(rotation);
177  ch0.addObject(recon);
178  ch0.addObject(prop);
179  ch0.addObject(extrF);
180  ch0.addObject(applier);
181  ch0.addObject(applierS);
182  ch0.addObject(cgnsO);
183  ch0.exec(true);
184 
185  cgnsO->getGeometry()->getPatch()->write("bulk");
186  cgnsO->getSurfaceBoundary()->getPatch()->write("boundary");
187  /* Destroy objects. */
188  delete cgnsI;
189  delete cgnsExtr;
190  delete cgnsDirichlet;
191  delete cgnsSlip;
192  delete cgnsO;
193  delete boxSel;
194  delete rotation;
195  delete recon;
196  delete prop;
197  delete extrF;
198  delete applier;
199  delete applierS;
200 
201  return;
202 }

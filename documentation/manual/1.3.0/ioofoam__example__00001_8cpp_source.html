48 void OFOAM_manip() {
49 
50  mimmo::IOOFOAM * reader = new mimmo::IOOFOAM(IOOFMode::READ);
51  reader->setDir("geodata/OFOAM");
52 
53  mimmo::FFDLattice * ffd = new mimmo::FFDLattice();
54  ffd->setShape(mimmo::ShapeType::CUBE);
55 
56  darray3E origin = {{0.367, 0., 0.}};
57  darray3E span = {{0.2, 0.1, 0.1}};
58  dvecarr3E displ(12,{{0.0,0.0,0.0}});
59  displ[4][1] = 0.045;
60  displ[5][1] = 0.045;
61  displ[6][1] = -0.045;
62  displ[7][1] = -0.045;
63 
64  ffd->setOrigin(origin);
65  ffd->setSpan(span);
66  ffd->setDimension(ivector1D({{3,2,2}}));
67  ffd->setDegrees(iarray3E({{2,2,2}}));
68  ffd->setDisplacements(displ);
69  ffd->setPlotInExecution(true);
70 
71  mimmo::Apply * applier = new mimmo::Apply();
72 
73  mimmo::IOOFOAM * writer = new mimmo::IOOFOAM(IOOFMode::WRITEPOINTSONLY);
74  writer->setOverwrite(false);
75 
76  mimmo::pin::addPin(reader, ffd, M_GEOM, M_GEOM);
77  mimmo::pin::addPin(reader, applier, M_GEOM, M_GEOM);
78  mimmo::pin::addPin(ffd, applier, M_GDISPLS, M_GDISPLS);
79  mimmo::pin::addPin(applier, writer, M_GEOM, M_GEOM);
80 
81  mimmo::Chain c0;
82  c0.addObject(reader);
83  c0.addObject(ffd);
84  c0.addObject(applier);
85  c0.addObject(writer);
86  c0.exec(true);
87 
88  delete reader;
89  delete ffd;
90  delete applier;
91  delete writer;
92 }

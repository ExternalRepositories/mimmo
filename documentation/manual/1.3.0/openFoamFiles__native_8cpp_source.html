420

428

495 bool writePointsOnCase(const char *rootPath, std::vector<std::array<double,3> > & points, bool overwriteStart)
496 {
497  // Initialize case
498  Foam::Time *foamRunTime = 0;
499  Foam::fvMesh *foamMesh = 0;
500 
501  initializeCase(rootPath, &foamRunTime, &foamMesh);
502 
503  functionObjectList & objfunctions = foamRunTime->functionObjects();
504  objfunctions.off();
505 
506  if(std::size_t(foamMesh->points().size()) != points.size()){
507  return false;
508  }
509 
510  Foam::pointField movedPoints(foamMesh->points());
511 
512  forAll(movedPoints, i){
513  for(Foam::label j=0; j<3; ++j){
514  movedPoints[i][j] = points[i][j];
515  }
516  }
517 
518  foamMesh->movePoints(movedPoints);
519 
520  if(overwriteStart){
521 #if OPENFOAM_OLDVER
522  foamMesh->writeObjects(Foam::IOstream::streamFormat::BINARY, foamRunTime->writeVersion(), foamRunTime->writeCompression());
523 #else
524  foamMesh->writeObject(Foam::IOstream::streamFormat::BINARY, foamRunTime->writeVersion(), foamRunTime->writeCompression(), true);
525 #endif
526  }else{
527 
528  Foam::dimensionedScalar fakeEndTime
529  (
530  "fakeEndTime",
531  dimensionSet(0,0,1,0,0,0,0),
532  scalar(1)
533  );
534  foamRunTime->setEndTime(foamRunTime->startTime()+fakeEndTime);
535 
536  while(foamRunTime->loop()){
537 #if OPENFOAM_OLDVER
538  foamMesh->writeObjects(Foam::IOstream::streamFormat::BINARY, foamRunTime->writeVersion(), foamRunTime->writeCompression());
539 #else
540  foamMesh->writeObject(Foam::IOstream::streamFormat::BINARY, foamRunTime->writeVersion(), foamRunTime->writeCompression(), true);
541 #endif
542  }
543 
544  }
545  return true;
546 }

575  conn[5] = FoamConn[2];
576  conn[6] = FoamConn[6];
577  conn[7] = FoamConn[5];
578  break;
579  case bitpit::ElementType::WEDGE:
580  conn.resize(6);
581  conn[0] = FoamConn[0];
582  conn[1] = FoamConn[2];
583  conn[2] = FoamConn[1];
584  conn[3] = FoamConn[3];
585  conn[4] = FoamConn[5];
586  conn[5] = FoamConn[4];
587  break;
588  case bitpit::ElementType::PYRAMID:
589  conn.resize(5);
590  conn[0] = FoamConn[3];
591  conn[1] = FoamConn[2];
592  conn[2] = FoamConn[1];
593  conn[3] = FoamConn[0];
594  conn[4] = FoamConn[4];
595  break;
596  default:
597  //do nothing
598  break;
599  }
600 
601  return conn;

609 bool interpolateFaceToPoint(dmpvector1D & facefield, dmpvector1D & pointfield){
610 
611  if (facefield.getGeometry()->getType() != 1){
612  throw std::runtime_error ("Error interpolating data on surface mesh field: geometry is not a surface mesh");
613 
614  }
615 
616  dmpvector1D sumAngles;
617  pointfield.clear();
618  for (bitpit::Vertex & vertex : facefield.getGeometry()->getVertices()){
619  long idV = vertex.getId();
620  sumAngles.insert(idV, 0.);
621  pointfield.insert(idV, 0.);
622  }
623 
624  for (bitpit::Cell & cell : facefield.getGeometry()->getCells()){
625 
626  //Cell id
627  long id = cell.getId();
628 
629  // Cell data
630  double cellData = facefield[id];
631 
632  // Get vertex id
633  bitpit::ConstProxyVector<long> cellVertexIds = cell.getVertexIds();
634 
635  for (int vertexLocalId=0; vertexLocalId<cell.getVertexCount(); vertexLocalId++){
636 
637  // Cell angle at vertex
638  double cellVertexAngle = static_cast<bitpit::SurfaceKernel*>(facefield.getGeometry()->getPatch())->evalAngleAtVertex(id, vertexLocalId);
639 
640  //Update data vertex and sum angles
641  pointfield[cellVertexIds[vertexLocalId]] += cellData * cellVertexAngle;
642  sumAngles[cellVertexIds[vertexLocalId]] += cellVertexAngle;
643 
644  }
645 
646  }
647 
648  //Normalize data with sum angles
649  for (bitpit::Vertex & vertex : facefield.getGeometry()->getVertices()){
650  long idV = vertex.getId();
651  pointfield[idV] /= sumAngles[idV];
652  }
653 
654  //Assign geometry and data location to pointfield
655  pointfield.setGeometry(facefield.getGeometry());
656  pointfield.setDataLocation(1);
657 
658  return true;
659 
660 }

407 void initializeCase(const char *rootPath, Foam::Time **foamRunTime_retPtr, Foam::fvMesh **foamMesh_retPtr)
408 {
409  // Arguments and root case
410  int nArguments = 3;
411  char **arguments = new char*[nArguments];
412 
413  char executable[] = "foamReader";
414  arguments[0] = executable;
415 
416  char caseOption[] = "-case";
417  arguments[1] = caseOption;
418 
419  char caseRoot[strlen(rootPath) + 1];
420  strcpy(caseRoot, rootPath);
421  arguments[2] = caseRoot;
422 
423 
424  static Foam::argList *foamArgs = 0;
425  if (!foamArgs) {
426  foamArgs = new Foam::argList(nArguments, arguments);
427  if (!foamArgs->checkRootCase()) {
428  Foam::FatalError.exit();
429  }
430  }
431 
432  // Time
433  static Foam::Time *foamRunTime = 0;
434  if (!foamRunTime) {
435  foamRunTime = new Foam::Time(Foam::Time::controlDictName, *foamArgs);
436  }
437 
438  *foamRunTime_retPtr = foamRunTime;
439 
440  // Mesh
441  static Foam::fvMesh *foamMesh = 0;
442  if (!foamMesh) {
443  foamMesh = new Foam::fvMesh(
444  Foam::IOobject(
445  Foam::fvMesh::defaultRegion,
446  foamRunTime->timeName(),
447  *foamRunTime,
448  Foam::IOobject::MUST_READ
449  )
450  );
451  }
452 
453  *foamMesh_retPtr = foamMesh;

498 bool writePointsOnCase(const char *rootPath, std::vector<std::array<double,3> > & points, bool overwriteStart)
499 {
500  // Initialize case
501  Foam::Time *foamRunTime = 0;
502  Foam::fvMesh *foamMesh = 0;
503 
504  initializeCase(rootPath, &foamRunTime, &foamMesh);
505 
506  functionObjectList & objfunctions = foamRunTime->functionObjects();
507  objfunctions.off();
508 
509  if(std::size_t(foamMesh->points().size()) != points.size()){
510  return false;
511  }
512 
513  Foam::pointField movedPoints(foamMesh->points());
514 
515  forAll(movedPoints, i){
516  for(Foam::label j=0; j<3; ++j){
517  movedPoints[i][j] = points[i][j];
518  }
519  }
520 
521  foamMesh->movePoints(movedPoints);
522 
523  if(overwriteStart){
524 #if OPENFOAM_OLDVER
525  foamMesh->writeObjects(Foam::IOstream::streamFormat::BINARY, foamRunTime->writeVersion(), foamRunTime->writeCompression());
526 #else
527  foamMesh->writeObject(Foam::IOstream::streamFormat::BINARY, foamRunTime->writeVersion(), foamRunTime->writeCompression(), true);
528 #endif
529  }else{
530 
531  Foam::dimensionedScalar fakeEndTime
532  (
533  "fakeEndTime",
534  dimensionSet(0,0,1,0,0,0,0),
535  scalar(1)
536  );
537  foamRunTime->setEndTime(foamRunTime->startTime()+fakeEndTime);
538 
539  while(foamRunTime->loop()){
540 #if OPENFOAM_OLDVER
541  foamMesh->writeObjects(Foam::IOstream::streamFormat::BINARY, foamRunTime->writeVersion(), foamRunTime->writeCompression());
542 #else
543  foamMesh->writeObject(Foam::IOstream::streamFormat::BINARY, foamRunTime->writeVersion(), foamRunTime->writeCompression(), true);
544 #endif
545  }
546 
547  }
548  return true;
549 }

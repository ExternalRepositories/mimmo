206 void SystemSolver::solve()
207 {
208  // Sum matrix terms to the RHS
209  int nRows;
210  VecGetLocalSize(m_rhs, &nRows);
211 
212  const PetscScalar *raw_solution;
213  VecGetArrayRead(m_solution, &raw_solution);
214 
215  PetscScalar *raw_rhs;
216  VecGetArray(m_rhs, &raw_rhs);
217  for (int i = 0; i < nRows; ++i) {
218 // raw_rhs[i] -= m_A_rhs.at(i);
219  raw_rhs[i] = m_A_rhs.at(i);
220  }
221  VecRestoreArray(m_rhs, &raw_rhs);
222 
223 
224  VecRestoreArrayRead(m_solution, &raw_solution);
225 
226  // Reorder the vectors
227  if (getPivotType() != PIVOT_NONE) {
228  vectorsReorder(PETSC_FALSE);
229  }
230 
231  // Solve the system
232  m_KSPStatus.error = KSPSolve(m_KSP, m_rhs, m_solution);
233 
234  // Set solver info
235  if (m_KSPStatus.error == 0) {
236  KSPGetIterationNumber(m_KSP, &m_KSPStatus.its);
237  KSPGetConvergedReason(m_KSP, &m_KSPStatus.convergence);
238  } else {
239  m_KSPStatus.its = -1;
240  m_KSPStatus.convergence = KSP_DIVERGED_BREAKDOWN;
241  }
242 
243  // Reorder the vectors
244  if (getPivotType() != PIVOT_NONE) {
245  vectorsReorder(PETSC_TRUE);
246  }
247 }

274 void SystemSolver::matrixInit(localivector2D &stencils)
275 {
276  long nRows = stencils.size();
277 
278  // Evaluate the offset for the numbering on this partition
279  m_rowGlobalIdOffset = 0;
280 
281 #if ENABLE_MPI == 1
282 
283  int nProcessors;
284  MPI_Comm_size(m_communicator, &nProcessors);
285  if (nProcessors > 1) {
286  std::vector<long> nGlobalRows(nProcessors);
287  MPI_Allgather(&nRows, 1, MPI_LONG, nGlobalRows.data(), 1, MPI_LONG, m_communicator);
288 
289  int rank;
290  MPI_Comm_rank(m_communicator, &rank);
291  for (int i = 0; i < rank; ++i) {
292  m_rowGlobalIdOffset += nGlobalRows[i];
293  }
294  }
295 
296 #endif
297 
298  // Evaluate the number of non-zero elements
299  //
300  // For each row we count the number of local non-zero elements (d_nnz) and
301  // the number of non-zero elements that belong to other processors (o_nnz)
302  std::vector<int> d_nnz(nRows);
303  std::vector<int> o_nnz(nRows);
304 
305 #if ENABLE_MPI == 1
306  long firstRowGlobalId = m_rowGlobalIdOffset;
307  long lastRowGlobalId = firstRowGlobalId + stencils.size() - 1;
308 #endif
309  int row = 0;
310  for (auto &stencilEntry : stencils) {
311 
312  int nWeights = stencilEntry.size();
313 
314  d_nnz[row] = 0;
315  o_nnz[row] = 0;
316  for (int k = 0; k < nWeights; ++k) {
317 #if ENABLE_MPI == 1
318  const StencilScalar::WeightInfo &weight = *(weights.data() + k);
319  auto &columnGlobalId = weight.id;
320  if (columnGlobalId >= firstRowGlobalId && columnGlobalId <= lastRowGlobalId) {
321  ++d_nnz[row];
322  } else {
323  ++o_nnz[row];
324  }
325 #else
326  ++d_nnz[row];
327 #endif
328  }
329  ++row;
330  }
331 
332  // Create the matrix
333 #if ENABLE_MPI == 1
334  MatCreateAIJ(m_communicator, nRows, nRows, PETSC_DETERMINE, PETSC_DETERMINE, 0, d_nnz.data(), 0, o_nnz.data(), &m_A);
335 #else
336  MatCreateSeqAIJ(PETSC_COMM_SELF, nRows, nRows, 0, d_nnz.data(), &m_A);
337 #endif
338 }

346 void SystemSolver::matrixFill(localivector2D &stencils, localdvector2D &weights, localdvector1D &rhs)
347 {
348  // Determine buffer space needed for column indices of any one row
349  int maxRowNZ = 0;
350  for (auto &stencilEntry : stencils) {
351  maxRowNZ = std::max(int(stencilEntry.size()), maxRowNZ);
352  }
353 
354  // Create the matrix
355  std::vector<PetscInt> rowNZGlobalIds(maxRowNZ);
356  std::vector<PetscScalar> rowNZValues(maxRowNZ);
357 
358  int count = 0;
359  for (auto &stencilEntry : stencils) {
360 
361  const PetscInt rowGlobalId = count + m_rowGlobalIdOffset;
362  const long row = count;
363 
364  long nRowNZ = stencilEntry.size();
365  for (int k = 0; k < nRowNZ; ++k) {
366  rowNZGlobalIds[k] = stencilEntry[k];
367  rowNZValues[k] = weights[row][k];
368  }
369 
370  MatSetValues(m_A, 1, &rowGlobalId, nRowNZ, rowNZGlobalIds.data(), rowNZValues.data(), INSERT_VALUES);
371  double val = rhs[row];
372  m_A_rhs.insert({row, val});
373 
374  ++count;
375  }
376 
377  // Let petsc build the matrix
378  MatAssemblyBegin(m_A, MAT_FINAL_ASSEMBLY);
379  MatAssemblyEnd(m_A, MAT_FINAL_ASSEMBLY);
380 }

684 void SystemSolver::KSPInit()
685 {
686  int nProcessors;
687 #if ENABLE_MPI == 1
688  MPI_Comm_size(m_communicator, &nProcessors);
689 #else
690  nProcessors = 1;
691 #endif
692 
693  KSPCreate(m_communicator, &m_KSP);
694 
695  KSPSetOperators(m_KSP, m_A, m_A);
696 
697  if (m_KSPOptions.nullspace) {
698  MatNullSpace nullspace;
699  MatNullSpaceCreate(m_communicator, PETSC_TRUE, 0, NULL, &nullspace);
700  MatSetNullSpace(m_A, nullspace);
701  MatNullSpaceDestroy(&nullspace);
702  }
703 
704  KSPSetType(m_KSP, KSPFGMRES);
705  KSPGMRESSetRestart(m_KSP, m_KSPOptions.restart);
706  KSPSetTolerances(m_KSP, m_KSPOptions.rtol, PETSC_DEFAULT, 1e10, m_KSPOptions.maxits);
707  KSPSetInitialGuessNonzero(m_KSP, PETSC_TRUE);
708 
709  PC preconditioner;
710  KSPGetPC(m_KSP, &preconditioner);
711  if (nProcessors > 1) {
712  PCSetType(preconditioner, PCASM);
713  PCASMSetOverlap(preconditioner, m_KSPOptions.overlap);
714  } else {
715  PCSetType(preconditioner, PCILU);
716  PCFactorSetLevels(preconditioner, m_KSPOptions.levels);
717  }
718  KSPSetFromOptions(m_KSP);
719  KSPSetUp(m_KSP);
720 
721  // Set ASM sub block preconditioners
722  if (nProcessors > 1) {
723  KSP *subksp;
724  PC subpc;
725  PetscInt nlocal, first;
726  PCASMGetSubKSP(preconditioner, &nlocal, &first, &subksp);
727  for (PetscInt i = 0; i < nlocal; ++i) {
728  KSPGetPC(subksp[i], &subpc);
729  PCSetType(subpc, PCILU);
730  PCFactorSetLevels(subpc, m_KSPOptions.sublevels);
731  KSPSetTolerances(subksp[i], m_KSPOptions.subrtol, PETSC_DEFAULT, PETSC_DEFAULT, PETSC_DEFAULT);
732  }
733  }
734 }

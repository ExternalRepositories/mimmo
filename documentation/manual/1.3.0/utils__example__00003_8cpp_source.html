47 void test00003() {
48 
49  mimmo::setExpertMode(true); // avoid control of unlinked ports.
50 
51  /* Creation of mimmo containers.
52  */
53  mimmo::MimmoGeometry * mimmo0 = new mimmo::MimmoGeometry();
54  mimmo0->setName("converterGeometry");
55  mimmo0->setIOMode(IOMode::CONVERT);
56  mimmo0->setReadDir("geodata");
57  mimmo0->setReadFileType(FileType::STL);
58  mimmo0->setReadFilename("sphere2");
59  mimmo0->setWriteDir(".");
60  mimmo0->setWriteFileType(FileType::SURFVTU);
61  mimmo0->setWriteFilename("utils_mesh_00003.0000");
62 
63  //create 4 rbf points on a y-plane at {{0.0,-0.6,0.0}}
64  dvecarr3E rbfpoints(4, {{0.0,-0.6,0.0}});
65  rbfpoints[1][0] = -0.1;
66  rbfpoints[2][0] = 0.1; rbfpoints[2][2] = 0.12;
67  rbfpoints[3][0] = -0.03; rbfpoints[3][2] = -0.1;
68 
69  //declare a support Radius for them
70  double suppR = 0.2;
71 
72  //create a set of displacements for rbf points
73  dvecarr3E rbfdispls(4, {{0.0,-0.1,0.0}});
74  rbfdispls[1][0] = -0.15;
75  rbfdispls[2][0] = 0.08; rbfdispls[2][2] = 0.2;
76  rbfdispls[3][0] = -0.2; rbfdispls[3][2] = -0.09;
77 
78  /*
79  Project onto surface and mirror these rbf points with their displacements attached
80  w.r.t. the y plane crossing the origin
81  */
82  mimmo::SpecularPoints * spec = new mimmo::SpecularPoints();
83  spec->setName("MirroringRBF");
84  spec->setCoords(rbfpoints);
85  spec->setVectorData(rbfdispls);
86  spec->setOrigin({{0.0,0.0,0.0}});
87  spec->setNormal({{0.0,1.0,0.0}});
88  spec->setInsideOut(true);
89  spec->setForce(false);
90  spec->setPlotInExecution(true);
91 
92  /*
93  Calculate AABB of the Mirrored RBF set accounting for the current support
94  radius chosen. Mirrored points are passed through port connection.
95  */
96  mimmo::RBFBox * rbfbox = new mimmo::RBFBox();
97  rbfbox->setName("AABB_RBFBox");
98  rbfbox->setSupportRadius(suppR);
99  rbfbox->setPlotInExecution(true);
100  /*
101  Use RBFBox to provide a sub-selection of the original geometry.
102  Origin and span of RBFBox AABB are provided through ports.
103  */
104  mimmo::SelectionByBox * sel = new mimmo::SelectionByBox();
105  sel->setName("RBFBoxSelection");
106  sel->setDual(false);
107  sel->setPlotInExecution(true);
108 
109  /*
110  Get deformation on the box selected surface with the current RBF mirrored set,
111  but not apply it yet.
112  */
113  mimmo::MRBF * manip = new mimmo::MRBF();
114  manip->setName("RBFManipulator");
115  manip->setSupportRadiusValue(suppR);
116 
117  /*
118  Reconstruct the deformation field on the whole body
119  */
120  mimmo::ReconstructVector * recon = new mimmo::ReconstructVector();
121  recon->setName("ReconstructDeformation");
122  recon->setPlotInExecution(true);
123 
124  /*
125  Verify actual geometry deformation vs the original geometry
126  level-set isolevel at d=0.14. Get the max penetration value on file.
127  Original geometry and its to-check deformation are passed through ports
128  */
129  mimmo::ControlDeformMaxDistance * isolevelCheck = new mimmo::ControlDeformMaxDistance();
130  isolevelCheck->setName("IsoLevelCheckCollision");
131  isolevelCheck->setLimitDistance(0.25);
132  isolevelCheck->setPlotInExecution(true);
133 
134  /*
135  Apply deformation to the original geometry
136  */
137  mimmo::Apply * applier = new mimmo::Apply();
138 
139 
140  /* Setup pin connections.
141  */
142  // original geometry passed to spec
143  mimmo::pin::addPin(mimmo0, spec, M_GEOM, M_GEOM);
144 
145  // spec passing projected and mirrored rbf points to rbfbox
146  mimmo::pin::addPin(spec, rbfbox, M_COORDS, M_COORDS);
147 
148  // sel receiving original geometry from mimmo0, and origin and span
149  // of bounding box from rbfbox
150  mimmo::pin::addPin(mimmo0, sel, M_GEOM, M_GEOM);
151  mimmo::pin::addPin(rbfbox, sel, M_POINT, M_POINT);
152  mimmo::pin::addPin(rbfbox, sel, M_SPAN, M_SPAN);
153 
154  // manip get as working geometry the selected patch of sel.
155  // rbf set of points and displacements are provided by spec
156  mimmo::pin::addPin(sel, manip, M_GEOM, M_GEOM);
157  mimmo::pin::addPin(spec, manip, M_COORDS, M_COORDS);
158  mimmo::pin::addPin(spec, manip, M_DISPLS, M_DISPLS);
159 
160  // recon reconstruct the deformation on the whole geometry
161  // taking it from mimmo0, while the deformation is provided from manip
162  //(on a selected subportion)
163  mimmo::pin::addPin(mimmo0, recon, M_GEOM, M_GEOM);
164  mimmo::pin::addPin(manip, recon, M_GDISPLS, M_VECTORFIELD);
165 
166  // pass original geometry and reconstructed deformation to isolevelCheck
167  mimmo::pin::addPin(mimmo0, isolevelCheck, M_GEOM, M_GEOM);
168  mimmo::pin::addPin(recon, isolevelCheck, M_VECTORFIELD, M_GDISPLS);
169 
170  // pass original geometry and reconstructed deformation to applier
171  mimmo::pin::addPin(mimmo0, applier, M_GEOM, M_GEOM);
172  mimmo::pin::addPin(recon, applier, M_VECTORFIELD, M_GDISPLS);
173 
174 
175  /* Setup execution chain.
176  */
177  mimmo::Chain ch0;
178  ch0.addObject(mimmo0);
179  ch0.addObject(spec);
180  ch0.addObject(rbfbox);
181  ch0.addObject(sel);
182  ch0.addObject(manip);
183  ch0.addObject(recon);
184  ch0.addObject(isolevelCheck);
185  ch0.addObject(applier);
186 
187  /* Execution of chain.
188  * Use debug flag false to avoid to print out the execution steps on console.
189  */
190  ch0.exec(true);
191 
192  //Last step write deformed original geometry in vtu
193  mimmo0->getGeometry()->getPatch()->write("utils_mesh_00003.0001");
194 
195  /* Clean up & exit;
196  */
197  delete mimmo0;
198  delete spec;
199  delete rbfbox;
200  delete sel;
201  delete manip;
202  delete recon;
203  delete isolevelCheck;
204  delete applier;
205 
206  return;
207 
208 }
